43 double PhasePressure ([[maybe_unused]] double Temperature, const Tensor<double,1>& ChemicalPotential) const override
44 {
45 const double& K = BulkModulus[0];
46 const double& Vm0 = ReferenceMolarVolume[0];
47 const double& mu = ChemicalPotential({0});
48 const double& mu0 = ReferenceChemicalPotential[0];
49 const double& p0 = ReferencePressure[0];
50
51 assert(K*Vm0 + mu0 - mu > 0 && "Invalid input parameters!");
52
53 return p0 - K*std::log((K*Vm0 + mu0 - mu)/(K*Vm0));
54 }
55 double PhaseConcentration ([[maybe_unused]] double Temperature, double ChemicalPotential, size_t comp) const override
56 {
