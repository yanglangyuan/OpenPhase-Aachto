227 double norm(void) const
228 {
229 double tmp = storage[0] * storage[0]
230 + storage[1] * storage[1]
231 + storage[2] * storage[2]
232 + storage[3] * storage[3] * 2.0
233 + storage[4] * storage[4] * 2.0
234 + storage[5] * storage[5] * 2.0;
235 return sqrt(tmp);
236 }

238 double double_contract(const vStress& Bstress) const // "double-dot product"
239 {
240 double tmp =
241 storage[0] * Bstress[0] +
242 storage[1] * Bstress[1] +
243 storage[2] * Bstress[2] +
244 2.0*storage[3] * Bstress[3] +
245 2.0*storage[4] * Bstress[4] +
246 2.0*storage[5] * Bstress[5];
247 return tmp;
248 }

263 storage[0]*storage[3]*storage[3] -

266 storage[3]*storage[4]*storage[5]*2.0;

269 dVector3 invariants(void) const
270 {
271 const double I1 = trace();
272 const double I2 = 0.5*(trace()*trace() -
273 (storage[0]*storage[0] +
274 storage[1]*storage[1] +
275 storage[2]*storage[2] +
276 storage[3]*storage[3]*2 +
277 storage[4]*storage[4]*2 +
278 storage[5]*storage[5]*2));
279 const double I3 = determinant();
280 return dVector3({I1,I2,I3});
281 }

283 double Mises(void) const
284 {
285 double vMises = 0.0;
286 vMises = (storage[0]-storage[1])*(storage[0]-storage[1]) +
287 (storage[1]-storage[2])*(storage[1]-storage[2]) +
288 (storage[2]-storage[0])*(storage[2]-storage[0]) +
289 6.0*(storage[3]*storage[3]+storage[4]*storage[4]+storage[5]*storage[5]);
290
291 return sqrt(0.5*vMises);
292 }

294 vStress rotated(const dMatrix3x3& RotationMatrix) const
295 {
296 double In[3][3];
297 double Out[3][3];
298
299 In[0][0] = storage[0];
300 In[0][1] = storage[5];
301 In[0][2] = storage[4];
302 In[1][0] = storage[5];
303 In[1][1] = storage[1];
304 In[1][2] = storage[3];
305 In[2][0] = storage[4];
306 In[2][1] = storage[3];
307 In[2][2] = storage[2];
308
309 for(int p = 0; p < 3; ++p)
310 for(int q = 0; q < 3; ++q)
311 {
312 Out[p][q] = 0;
313 for(int i = 0; i < 3; ++i)
314 for(int j = 0; j < 3; ++j)
315 {
316 Out[p][q] += RotationMatrix(p,i)*In[i][j]*RotationMatrix(q,j);
317 }
318 }
319 vStress OUT;
320
321 OUT[0] = Out[0][0];
322 OUT[5] = Out[0][1];
323 OUT[4] = Out[0][2];
324 OUT[1] = Out[1][1];
325 OUT[3] = Out[1][2];
326 OUT[2] = Out[2][2];
327
328 return OUT;
329 }

331 vStress& rotate(const dMatrix3x3& RotationMatrix)

336 In[0][0] = storage[0];
337 In[0][1] = storage[5];
338 In[0][2] = storage[4];
339 In[1][0] = storage[5];
340 In[1][1] = storage[1];
341 In[1][2] = storage[3];
342 In[2][0] = storage[4];
343 In[2][1] = storage[3];
344 In[2][2] = storage[2];
345
346 for(int p = 0; p < 3; ++p)
347 for(int q = 0; q < 3; ++q)
348 {
349 Out[p][q] = 0;
350 for(int i = 0; i < 3; ++i)
351 for(int j = 0; j < 3; ++j)
352 {
353 Out[p][q] += RotationMatrix(p,i)*In[i][j]*RotationMatrix(q,j);
354 }
355 }
356
357 storage[0] = Out[0][0];
358 storage[5] = Out[0][1];
359 storage[4] = Out[0][2];
360 storage[1] = Out[1][1];
361 storage[3] = Out[1][2];
362 storage[2] = Out[2][2];
363
364 return *this;
365 }
