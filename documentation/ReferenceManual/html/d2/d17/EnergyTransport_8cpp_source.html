41void EnergyTransport::ReadInput(string InputFile)
42{
43 ConsoleOutput::WriteBlankLine();
44 ConsoleOutput::WriteLineInsert("EnergyTransport");
45 ConsoleOutput::WriteStandard("Source", InputFile);
46 std::fstream inp(InputFile, std::ios::in);
47 if (!inp)
48 {
49 std::stringstream message;
50 message << "File \"" << InputFile << "\" could not be opened";
51 throw std::runtime_error(message.str());

53 std::stringstream inp_data;
54 inp_data << inp.rdbuf();
55 inp.close();
56 int moduleLocation = FileInterface::FindModuleLocation(inp_data, "EnergyTransport");
57 Pr = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("Pr"), false, 0.71);
58 Cp = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("Cp"), false, 1005.0);
59 Mw = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("Mw"), false, 0.02896);
60 AdvectionUpwind = FileInterface::ReadParameterB(inp_data, moduleLocation, std::string("AdvectionUpwind"), false, false);
61 AdvectionCentral = FileInterface::ReadParameterB(inp_data, moduleLocation, std::string("AdvectionCentral"), false, false);
62 AdvectionVanLeer = FileInterface::ReadParameterB(inp_data, moduleLocation, std::string("AdvectionVanLeer"), false, true);
63 Conjugate = FileInterface::ReadParameterB(inp_data, moduleLocation, std::string("Conjugate"), false, false);
64 Ts0 = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("Ts0"), false, 300.0);
65 Cps = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("Cps"), false, 900.0);
66 Rhos = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("Rhos"), false, 5000.0);
67 Ks = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("Ks"), false, 237);
68 T0 = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("T0"), false, 300.0);
69 ATstar = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("ATstar"), false, 100.0);
70 TempSolid = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("TempSolid"), false, 300.0);
71 TempSolidCold = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("TempSolidCold"), false, 300.0);
72 TMu0 = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("TMu0"), false, 273.0);
73 SMu0 = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("SMu0"), false, 110.5);
74 Mu0 = FileInterface::ReadParameterD(inp_data, moduleLocation, std::string("Mu0"), false, 1.68e-05);

122void EnergyTransport::SetFreeBCNX(PhaseField& Phase)
123{
124 int iEnd = Grid.Nx;
125 double Temp=T0; // Temporary variable for extrapolated temperature
126 // Loop over the extrapolated boundary nodes
127 if(Grid.OffsetX+Grid.Nx==Grid.TotalNx)
128 for (int i = iEnd; i < Grid.Nx + Tx.Bcells(); ++i)
129 {
130 for (int j = 0; j< Grid.Ny ; ++j)
131 {
132 for (int k = 0; k< Grid.Nz ; ++k)
133 {
134 if (BCOrder==0)
135 {
136 Temp = Tx(iEnd - 1, j, k);
137 }
138 else if (BCOrder==1)
139 {
140 Temp = 2.0 * Tx(iEnd - 1, 0, k) - Tx(iEnd - 2, 0, k);
141 }
142 else if (BCOrder==2)
143 {
144 Temp = Tx(iEnd - 1, j, k) + (Tx(iEnd - 1, j, k) - Tx(iEnd - 2, j, k)) +
145 0.5 * (Tx(iEnd - 1, j, k) - 2.0 * Tx(iEnd - 2, j, k) + Tx(iEnd - 3, j, k));
146 }
147 else if (BCOrder==3)
148 {
149 Temp = Lagrange_polynomial4(0.0, 1.0, 2.0, 3.0, 4.0,
150 Tx(iEnd - 4, j, k), Tx(iEnd - 3, j, k),
151 Tx(iEnd - 2, j, k), Tx(iEnd - 1, j, k));
152 }
153 Tx(i, j, k) = Temp;
154 }
155 }
156 }

179void EnergyTransport::CalculateGhostPointsConjugateHeatTransfer(PhaseField& Phase, FlowSolverLBM& FL, SolidBody& SB)
180{
181 size_t Gasidx = 0;
182 size_t Solididx = 1;
183 double ep=1e-8;
184 double D = SB.nDist;
185
186 for(size_t idx = 0; idx < Phase.FieldsProperties.size(); idx++)
187 if(Phase.FieldsProperties[idx].State == AggregateStates::Gas)
188 {
189 Gasidx = idx;
190 }
191 int iBC=Grid.dNx*2.0;
192 int jBC=Grid.dNy*2.0;
193 int kBC=Grid.dNz*2.0;
194
195 int idi=Grid.dNx;
196 int jdi=Grid.dNy;
197 int kdi=Grid.dNz;
198
199 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Tx,0,)
200 {
201 bool Set_TsOut=false;
202 if(!FL.Obstacle(i,j,k))
203 for(int ii = -iBC; ii <= iBC; ii++)
204 for(int jj = -jBC; jj <= jBC; jj++)
205 for(int kk = -kBC; kk <= kBC; kk++)
206 {
207 if(FL.Obstacle(i+ii,j+jj,k+kk))
208 {
209 if(!Set_TsOut)
210 {
211 dVector3 Norm{0.0,0.0,0.0};
212 for (auto it = Phase.Fields(i,j,k).cbegin(); it != Phase.Fields(i,j,k).cend(); ++it)
213 {
214 if(Phase.FieldsProperties[it->index].State == AggregateStates::Solid)
215 {
216 Solididx=it->index;
217 NodeAB<dVector3,dVector3> Normals=Phase.Normals(i,j,k);
218 Norm=Normals.get_asym1(Solididx, Gasidx);
219 }
220 }
221 double Phif = Phase.Fractions(i,j,k,{Gasidx});
222 double X = fabs(Phase.Grid.iWidth/Pi * asin(1.0-2.0*Phif));
223 dVector3 Xf = {double(i), double(j), double(k)};
224 dVector3 Xb = Xf - Norm*X;
225 dVector3 Xs = Xb - Norm*D;
226 double ic = Xs.getX();
227 double jc = Xs.getY();
228 double kc = Xs.getZ();
229 double i0 = floor(ic);
230 double i1 = i0 + 1.0;
231 double j0 = floor(jc);
232 double j1 = j0 + 1.0;
233 double k0 = floor(kc);
234 double k1 = k0 + 1.0;
235
236 double Tsi = SB.TrilinearInterPolation(ic,jc,kc,
237 Ts(i0*idi,j0*jdi,k0*kdi),Ts(i1*idi,j0*jdi,k0*kdi),
238 Ts(i0*idi,j1*jdi,k0*kdi),Ts(i1*idi,j1*jdi,k0*kdi),
239 Ts(i0*idi,j0*jdi,k1*kdi),Ts(i1*idi,j0*jdi,k1*kdi),
240 Ts(i0*idi,j1*jdi,k1*kdi),Ts(i1*idi,j1*jdi,k1*kdi));
241 double Ksi = SB.TrilinearInterPolation(ic,jc,kc,
242 K_Solid(i0*idi,j0*jdi,k0*kdi),K_Solid(i1*idi,j0*jdi,k0*kdi),
243 K_Solid(i0*idi,j1*jdi,k0*kdi),K_Solid(i1*idi,j1*jdi,k0*kdi),
244 K_Solid(i0*idi,j0*jdi,k1*kdi),K_Solid(i1*idi,j0*jdi,k1*kdi),
245 K_Solid(i0*idi,j1*jdi,k1*kdi),K_Solid(i1*idi,j1*jdi,k1*kdi));
246 double Tf = Tx(i,j,k);
247 double Kf = K_Mixture(i,j,k);
248
249 double Tb = (Kf/X*Tf+Ksi/D*Tsi)/(Kf/X+Ksi/D);
250 Ts (i,j,k) = Tb * (1.0+X/D) - X/D*Tsi;
251 Set_TsOut=true;
252 }
253 }
254 }
255
256 bool Set_TfOut=false;
257 if(FL.Obstacle(i,j,k))
258 for(int ii = -iBC; ii <= iBC; ii++)
259 for(int jj = -jBC; jj <= jBC; jj++)
260 for(int kk = -kBC; kk <= kBC; kk++)
261 {
262 if(!FL.Obstacle(i+ii,j+jj,k+kk))
263 {
264 if(!Set_TfOut)
265 {
266 dVector3 Norm{0.0,0.0,0.0};
267 for (auto it = Phase.Fields(i,j,k).cbegin(); it != Phase.Fields(i,j,k).cend(); ++it)
268 {
269 if(Phase.FieldsProperties[it->index].State == AggregateStates::Solid)
270 {
271 Solididx=it->index;
272 NodeAB<dVector3,dVector3> Normals=Phase.Normals(i,j,k);
273 Norm=Normals.get_asym1(Solididx, Gasidx);
274 }
275 }
276 double Phif = Phase.Fractions(i,j,k,{Gasidx});
277 double X = fabs(Phase.Grid.iWidth/Pi * asin(1.0-2.0*Phif));
278 dVector3 Xs = {double(i), double(j), double(k)};
279 dVector3 Xb = Xs + Norm*X;
280 dVector3 Xf = Xb + Norm*D;
281 double ic = Xf.getX();
282 double jc = Xf.getY();
283 double kc = Xf.getZ();
284 double i0 = floor(ic);
285 double i1 = i0 + 1.0;
286 double j0 = floor(jc);
287 double j1 = j0 + 1.0;
288 double k0 = floor(kc);
289 double k1 = k0 + 1.0;
290
291 double Tfi = SB.TrilinearInterPolation(ic,jc,kc,
292 Tx(i0*idi,j0*jdi,k0*kdi),Tx(i1*idi,j0*jdi,k0*kdi),
293 Tx(i0*idi,j1*jdi,k0*kdi),Tx(i1*idi,j1*jdi,k0*kdi),
294 Tx(i0*idi,j0*jdi,k1*kdi),Tx(i1*idi,j0*jdi,k1*kdi),
295 Tx(i0*idi,j1*jdi,k1*kdi),Tx(i1*idi,j1*jdi,k1*kdi));
296 double Kfi = SB.TrilinearInterPolation(ic,jc,kc,
297 K_Mixture(i0*idi,j0*jdi,k0*kdi),K_Mixture(i1*idi,j0*jdi,k0*kdi),
298 K_Mixture(i0*idi,j1*jdi,k0*kdi),K_Mixture(i1*idi,j1*jdi,k0*kdi),
299 K_Mixture(i0*idi,j0*jdi,k1*kdi),K_Mixture(i1*idi,j0*jdi,k1*kdi),
300 K_Mixture(i0*idi,j1*jdi,k1*kdi),K_Mixture(i1*idi,j1*jdi,k1*kdi));
301 double Tsi = Ts(i,j,k);
302 double Ksi = K_Solid(i,j,k);
303 if(X<ep)
304 {
305 Tx (i,j,k) = Tsi;
306 }
307 else
308 {
309 double Tb = (Kfi/D*Tfi+Ksi/X*Tsi)/(Kfi/D+Ksi/X);
310 Tx(i,j,k) = Tb * (1.0+X/D) - X/D * Tfi;
311 }
312 Set_TfOut=true;
313 }
314 }
315 }
316 }
317 OMP_PARALLEL_STORAGE_LOOP_END
318}

320void EnergyTransport::CalculateGhostPoints(PhaseField& Phase, FlowSolverLBM& FL, SolidBody& SB)
321{
322 size_t Gasidx = 0;
323 size_t Solididx = 1;
324 double Tb=TempSolid;
325 double Qb=HeatFlux;
326 double ep=1e-8;
327 double D = SB.nDist;
328 for(size_t idx = 0; idx < Phase.FieldsProperties.size(); idx++)
329 if(Phase.FieldsProperties[idx].State == AggregateStates::Gas)
330 {
331 Gasidx = idx;
332 }
333 int iBC=Grid.dNx*2.0;
334 int jBC=Grid.dNy*2.0;
335 int kBC=Grid.dNz*2.0;
336 int idi=Grid.dNx;
337 int jdi=Grid.dNy;
338 int kdi=Grid.dNz;
339
340 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Tx,0,)
341 {
342 bool Set_Tfout=false;
343 if(FL.Obstacle(i,j,k))
344 for(int ii = -iBC; ii <= iBC; ii++)
345 for(int jj = -jBC; jj <= jBC; jj++)
346 for(int kk = -kBC; kk <= kBC; kk++)
347 {
348 if(!Set_Tfout)
349 {
350 if(!FL.Obstacle(i+ii,j+jj,k+kk))
351 {
352 dVector3 Norm{0.0,0.0,0.0};
353 for (auto it = Phase.Fields(i,j,k).cbegin(); it != Phase.Fields(i,j,k).cend(); ++it)
354 {
355 if(Phase.FieldsProperties[it->index].State == AggregateStates::Solid)
356 {
357 Solididx=it->index;
358 NodeAB<dVector3,dVector3> Normals=Phase.Normals(i,j,k);
359 Norm=Normals.get_asym1(Solididx, Gasidx);
360 }
361 }
362
363 double Phif = Phase.Fractions(i,j,k,{Gasidx});
364 double X = fabs(Phase.Grid.iWidth/Pi * asin(1.0-2.0*Phif));
365
366 dVector3 Xs = {double(i), double(j), double(k)};
367 dVector3 Xb = Xs + Norm*X;
368 dVector3 Xf = Xb + Norm*SB.nDist;
369
370 double ic = Xf.getX();
371 double jc = Xf.getY();
372 double kc = Xf.getZ();
373
374 double i0 = floor(ic);
375 double i1 = i0 + 1.0;
376 double j0 = floor(jc);
377 double j1 = j0 + 1.0;
378 double k0 = floor(kc);
379 double k1 = k0 + 1.0;
380
381 double Tf = SB.TrilinearInterPolation(ic,jc,kc,
382 Tx(i0*idi,j0*jdi,k0*kdi),Tx(i1*idi,j0*jdi,k0*kdi),
383 Tx(i0*idi,j1*jdi,k0*kdi),Tx(i1*idi,j1*jdi,k0*kdi),
384 Tx(i0*idi,j0*jdi,k1*kdi),Tx(i1*idi,j0*jdi,k1*kdi),
385 Tx(i0*idi,j1*jdi,k1*kdi),Tx(i1*idi,j1*jdi,k1*kdi));
386 double Kf = SB.TrilinearInterPolation(ic,jc,kc,
387 K_Mixture(i0*idi,j0*jdi,k0*kdi),K_Mixture(i1*idi,j0*jdi,k0*kdi),
388 K_Mixture(i0*idi,j1*jdi,k0*kdi),K_Mixture(i1*idi,j1*jdi,k0*kdi),
389 K_Mixture(i0*idi,j0*jdi,k1*kdi),K_Mixture(i1*idi,j0*jdi,k1*kdi),
390 K_Mixture(i0*idi,j1*jdi,k1*kdi),K_Mixture(i1*idi,j1*jdi,k1*kdi));
391 if(IF_ConstTemp[Solididx])
392 {
393 Tb = SurfaceTemp[Solididx];
394 if(X<ep)
395 {
396 Tx(i,j,k) = Tb;
397 }
398 else
399 {
400 Tx(i,j,k) = Tb * (1.0+X/D) - X/D * Tf;
401 }
402 }
403 else if(IF_ConstFlux[Solididx])
404 {
405 Qb = SurfaceFlux[Solididx];
406 Tx(i,j,k) = Tf-(X+D)*Qb/Kf;
407 }
408 Set_Tfout = true;
409 }
410 }
411 }
412 }
413 OMP_PARALLEL_STORAGE_LOOP_END

421void EnergyTransport::CalculateThermodynamicPressure(FlowSolverLBM& FL, double P0, double T0)
422{
423 double SumTinv=0.0;
424 double fcells=0.0;
425 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Tx,0,)
426 {
427 if(!FL.Obstacle(i,j,k))
428 {
429 SumTinv += 1.0/Tx(i,j,k);
430 fcells++;
431 }
432 }
433 OMP_PARALLEL_STORAGE_LOOP_END
434 #ifdef MPI_PARALLEL
435 OP_MPI_Allreduce(OP_MPI_IN_PLACE, &SumTinv, 1, OP_MPI_DOUBLE, OP_MPI_SUM, OP_MPI_COMM_WORLD);
436 OP_MPI_Allreduce(OP_MPI_IN_PLACE, &fcells, 1, OP_MPI_DOUBLE, OP_MPI_SUM, OP_MPI_COMM_WORLD);
437 #endif
438 FL.Pth = P0/T0*fcells/SumTinv;
439}

441void EnergyTransport::CalculateDiffusion(PhaseField& Phase, FlowSolverLBM& FL, double dt)
442{
443 size_t MixtureComp=0;
444 std::vector<int> dir(3);
445 double dx = Grid.dx;
446 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Tx,0,)
447 {
448 if(!FL.Obstacle(i,j,k))
449 {
450 double Tc = TxOld(i,j,k);
451 double Rhoc = FL.DensityWetting(i,j,k,{MixtureComp});
452 double cpc = Cp_Mixture(i,j,k);
453 double ktc = K_Mixture(i,j,k);
454 for (int direction=0; direction<3; ++direction)
455 {
456 dir[0] = (direction == 0) ? 1*Grid.dNx : 0;
457 dir[1] = (direction == 1) ? 1*Grid.dNy : 0;
458 dir[2] = (direction == 2) ? 1*Grid.dNz : 0;
459
460 int ip=i+dir[0];
461 int jp=j+dir[1];
462 int kp=k+dir[2];
463 int im=i-dir[0];
464 int jm=j-dir[1];
465 int km=k-dir[2];
466
467 double Tp = TxOld(ip,jp,kp);
468 double Tm = TxOld(im,jm,km);
469 double ktp = K_Mixture(ip,jp,kp);
470 double ktm = K_Mixture(im,jm,km);
471
472 Tx(i, j, k) += dt/(Rhoc*cpc) * (ktc * (Tp+Tm-2.0*Tc)/dx/dx + (ktp-ktm)/dx/2.0 * (Tp-Tm)/dx/2.0);
473 }
474 }
475 }
476 OMP_PARALLEL_STORAGE_LOOP_END
477}

479void EnergyTransport::CalculatePhaseFieldCoupling(PhaseField& Phase, FlowSolverLBM& FL, double hT, double dt)
480{
481 double Ts=TempSolid;
482 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Tx,0,)
483 if(!FL.Obstacle(i,j,k))
484 {
485 double SF=0;
486 for (auto it = Phase.Fields(i,j,k).cbegin(); it != Phase.Fields(i,j,k).cend(); ++it)
487 {
488 size_t PhaseIdx = Phase.FieldsProperties[it->index].Phase;
489 if(Phase.FieldsProperties[PhaseIdx].State == AggregateStates::Solid)
490 {
491 int PFIdx=it->index;
492 Ts = SurfaceTemp[PFIdx];
493 SF += it->value;
494 }
495 }
496 double ST = hT * K_Mixture(i,j,k)* pow(SF,2)*(1.0-SF) / pow(Phase.Grid.iWidth*Grid.dx,2.0) * (TxOld(i,j,k)-Ts);
497 Tx(i,j,k) -= dt / FL.DensityWetting(i,j,k,{0}) / Cp_Mixture(i,j,k) * ST;
498 }
499 OMP_PARALLEL_STORAGE_LOOP_END
500}

502void EnergyTransport::CalculateAdvection(FlowSolverLBM& FL, FlowMixture& FM, double dt)
503{
504 std::vector<int> dir(3);
505 double eps=1e-10;
506 double dx = Grid.dx;
507 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Tx,0,)
508 {
509 if(!FL.Obstacle(i,j,k))
510 {
511 double Tc = TxOld(i,j,k);
512 for (int direction=0; direction<3; ++direction)
513 {
514 dir[0] = (direction == 0)?1*Grid.dNx:0;
515 dir[1] = (direction == 1)?1*Grid.dNy:0;
516 dir[2] = (direction == 2)?1*Grid.dNz:0;
517 int ip=i+dir[0];
518 int jp=j+dir[1];
519 int kp=k+dir[2];
520 int im=i-dir[0];
521 int jm=j-dir[1];
522 int km=k-dir[2];
523
524 double Tp = TxOld(ip,jp,kp);
525 double Tm = TxOld(im,jm,km);
526 double Vc = FM.V_Mixture(i,j,k)[direction];
527
528 if(AdvectionUpwind)
529 {
530 if(Vc>=0)
531 {
532 Tx(i,j,k) += -dt/dx * Vc * (Tc-Tm);
533 }
534 else
535 {
536 Tx(i,j,k) += -dt/dx * Vc * (Tp-Tc);
537 }
538 }
539 else if(AdvectionVanLeer)
540 {
541 double Thp = (Tc+Tp)/2.0;
542 double Thm = (Tc+Tm)/2.0;
543 if(Vc>=0)
544 {
545 int imm=i-2.0*dir[0];
546 int jmm=j-2.0*dir[1];
547 int kmm=k-2.0*dir[2];
548 double Tmm = TxOld(imm,jmm,kmm);
549 double r = (Tc-Tm)/(Tp-Tc+eps);
550 double phi = (r+fabs(r))/(1.0+fabs(r));
551 Thp = Tc+0.5*phi*(Tp-Tc);
552 double rm = (Tm-Tmm)/(Tc-Tm+eps);
553 double phim=(rm+fabs(rm))/(1.0+fabs(rm));
554 Thm=Tm+0.5*phim*(Tc-Tm);
555 Tx(i,j,k) += -dt*Vc*(Thp-Thm)/dx;
556 }
557 else
558 {
559 int ipp=i+2.0*dir[0];
560 int jpp=j+2.0*dir[1];
561 int kpp=k+2.0*dir[2];
562 double Tpp = TxOld(ipp,jpp,kpp);
563 double r = (Tpp-Tp)/(Tp-Tc+eps);
564 double phi=(r+fabs(r))/(1.0+fabs(r));
565 Thp=Tp-0.5*phi*(Tp-Tc);
566 double rm = (Tp-Tc)/(Tc-Tm+eps);
567 double phim=(rm+fabs(rm))/(1.0+fabs(rm));
568 Thm=Tc-0.5*phim*(Tc-Tm);
569 Tx(i,j,k) += -dt*Vc*(Thp-Thm)/dx;
570 }
571 }
572 else if (AdvectionCentral)
573 {
574 Tx(i,j,k) += -dt/dx * Vc * (Tp-Tm)/2.0;
575 }
576 }
577 }
578 }
579 OMP_PARALLEL_STORAGE_LOOP_END
580}

582void EnergyTransport::CalculateSolidDiffusion(PhaseField& Phase, FlowSolverLBM& FL, double dt)
583{
584 std::vector<int> dir(3);
585 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Ts,0,)
586 {
587 if(FL.Obstacle(i,j,k))
588 {
589 for (int direction=0; direction<3; ++direction)
590 {
591 dir[0] = (direction == 0) ? 1*Grid.dNx : 0;
592 dir[1] = (direction == 1) ? 1*Grid.dNy : 0;
593 dir[2] = (direction == 2) ? 1*Grid.dNz : 0;
594 int ip=i+dir[0];
595 int jp=j+dir[1];
596 int kp=k+dir[2];
597 int im=i-dir[0];
598 int jm=j-dir[1];
599 int km=k-dir[2];
600 double dx = Grid.dx;
601 double Tc = TsOld(i,j,k);
602 double Tp = TsOld(ip,jp,kp);
603 double Tm = TsOld(im,jm,km);
604 double Rhoc = Density_Solid(i,j,k);
605 double cpc = Cp_Solid(i,j,k);
606 double ktc = K_Solid(i,j,k);
607 Ts(i, j, k) += dt/(Rhoc*cpc) * (ktc * (Tp+Tm-2.0*Tc)/dx/dx);
608 }
609 }
610 }
611 OMP_PARALLEL_STORAGE_LOOP_END
612}

637{
