42double NucleationParameters::SetSeedRadius()
43{
44 double radius = 0.0;
45
46 switch(Distribution)
47 {
48 case NucleiSizeDistributions::Normal:
49 {
50 radius = 0.5*SizeDistributionNormal(SizeGenerator);
51 break;
52 }
53 case NucleiSizeDistributions::Cauchy:
54 {
55 radius = 0.5*SizeDistributionCauchy(SizeGenerator);
56 break;
57 }
58 case NucleiSizeDistributions::Uniform:
59 {
60 radius = 0.5*SizeDistributionUniform(SizeGenerator);
61 break;
62 }
63 case NucleiSizeDistributions::FixedRadius:
64 case NucleiSizeDistributions::None:
65 {
66 radius = SeedRadius;
67 break;
68 }
69 default:
70 {
71 radius = 0.0;
72 break;
73 }
74 }
75
76 if (radius == 0.0)
77 {
78 ConsoleOutput::WriteWarning("Seed radius has been set to zero. Check input Parameters!", "NucleationParametersEXP", "SetSeedRadius");
79 }
80 return radius;
81}

127Quaternion NucleationParameters::SetSeedOrientation()
128{
129 Quaternion loc_orientation;
130
131 int active_dimensions = (PositionDistributionX.b() != 0)
132 + (PositionDistributionY.b() != 0)
133 + (PositionDistributionZ.b() != 0);
134
135 if(OrientationMode == NucleiOrientationModes::Random)
136 {
137 switch(active_dimensions)
138 {
139 case 1: // No rotation in 1D
140 {
141 loc_orientation.set(1.0, 0.0, 0.0, 0.0);
142 break;
143 }
144 case 2: // Only in-plain rotation in 2D
145 {
146 double a1 = 0.0;
147 double a2 = 0.0;
148 double a3 = 0.0;
149
150 if(PositionDistributionX.b() == 0) a1 = OrientationDistributionA(OrientationGenerator1);
151 if(PositionDistributionY.b() == 0) a2 = OrientationDistributionA(OrientationGenerator2);
152 if(PositionDistributionZ.b() == 0) a3 = OrientationDistributionA(OrientationGenerator3);
153
154 EulerAngles ph1({a1,a2,a3},XYZ);
155
156 loc_orientation = ph1.getQuaternion().normalized();
157 break;
158 }
159 case 3: // Full rotation freedom in 3D
160 {
161 double u1 = OrientationDistributionQ(OrientationGenerator1);
162 double u2 = OrientationDistributionQ(OrientationGenerator2);
163 double u3 = OrientationDistributionQ(OrientationGenerator3);
164
165 Quaternion locQuaternion;
166 locQuaternion.set(sqrt(1.0-u1)*sin(2.0*Pi*u2),
167 sqrt(1.0-u1)*cos(2.0*Pi*u2),
168 sqrt(u1)*sin(2.0*Pi*u3),
169 sqrt(u1)*cos(2.0*Pi*u3));
170
171 loc_orientation = locQuaternion.normalized();
172 break;
173 }
174 default: // No rotation in zero dimensions.
175 {
176 loc_orientation.set(1.0, 0.0, 0.0, 0.0);
177 break;
178 }
179 }
180 }
181 return loc_orientation;
182}

184bool NucleationParameters::CheckLocation(const PhaseField& Phase,
185 const iVector3 loc_position) const
186{
187 bool enable_seed = false;
188 if (Phase.Grid.PositionInLocalBounds(loc_position) and
189 PositionNotShielded(loc_position))
190 {
191 int i = loc_position.get_x() - Phase.Grid.OffsetX;
192 int j = loc_position.get_y() - Phase.Grid.OffsetY;
193 int k = loc_position.get_z() - Phase.Grid.OffsetZ;
194
195 switch (LocationMode)
196 {
197 case NucleiLocationModes::XBottom:
198 case NucleiLocationModes::XTop:
199 case NucleiLocationModes::YBottom:
200 case NucleiLocationModes::YTop:
201 case NucleiLocationModes::ZBottom:
202 case NucleiLocationModes::ZTop:
203 case NucleiLocationModes::BulkAndInterfaces:
204 {
205 if(Phase.Fractions(i,j,k,{MatrixPhase}) > 0.25)
206 {
207 enable_seed = true;
208 }
209 break;
210 }
211 case NucleiLocationModes::Bulk:
212 {
213 if( !Phase.Fields(i,j,k).interface() and
214 (Phase.Fractions(i,j,k,{MatrixPhase}) == 1.0))
215 {
216 enable_seed = true;
217 }
218 break;
219 }
220 case NucleiLocationModes::BulkAndGrainBoundaries:
221 {
222 if(Phase.Fractions(i,j,k,{MatrixPhase}) == 1.0)
223 {
224 enable_seed = true;
225 }
226 break;
227 }
228 case NucleiLocationModes::GrainBoundaries:
229 {
230 if( Phase.Fields(i,j,k).interface() and
231 (Phase.Fractions(i,j,k,{MatrixPhase}) == 1.0))
232 {
233 enable_seed = true;
234 }
235 break;
236 }
237 case NucleiLocationModes::PhaseBoundaries:
238 {
239 if( Phase.Fields(i,j,k).interface() and
240 (Phase.Fractions(i,j,k,{MatrixPhase}) < 1.0) and
241 (Phase.Fractions(i,j,k,{MatrixPhase}) > 0.25))
242 {
243 enable_seed = true;
244 }
245 break;
246 }
247 case NucleiLocationModes::Junctions:
248 {
249 if((Phase.Fractions(i,j,k,{MatrixPhase}) == 1.0) and
250 (Phase.Fields(i,j,k).size() > 2))
251 {
252 enable_seed = true;
253 }
254 break;
255 }
256 case NucleiLocationModes::Interfaces:
257 {
258 if( Phase.Fields(i,j,k).interface() and
259 (Phase.Fractions(i,j,k,{MatrixPhase}) > 0.25))
260 {
261 enable_seed = true;
262 }
263 break;
264 }
265 }
266 }
267 return enable_seed;
268}

270bool NucleationParameters::PositionNotShielded(const iVector3 position) const
271{
272 bool my_return = true;
273 double ShieldingRadiusSquare = ShieldingRadius*ShieldingRadius;
274
275 long int TotalNx = PositionDistributionX.b();
276 long int TotalNy = PositionDistributionY.b();
277 long int TotalNz = PositionDistributionZ.b();
278
279 for (auto it = GeneratedNuclei.begin(); it != GeneratedNuclei.end(); ++it)
280 {
281 int x = position.get_x();
282 int y = position.get_y();
283 int z = position.get_z();
284 const double xdis = std::min(std::fabs(x - it->position[0]), std::min( std::fabs(x - it->position[0] + TotalNx), std::fabs(x - it->position[0] - TotalNx)));
285 const double ydis = std::min(std::fabs(y - it->position[1]), std::min( std::fabs(y - it->position[1] + TotalNy), std::fabs(y - it->position[1] - TotalNy)));
286 const double zdis = std::min(std::fabs(z - it->position[2]), std::min( std::fabs(z - it->position[2] + TotalNz), std::fabs(z - it->position[2] - TotalNz)));
287 const double distanceSquare = xdis*xdis + ydis*ydis + zdis*zdis;
288 if(distanceSquare < ShieldingRadiusSquare)
289 {
290 my_return = false;
291 }
292 }
293 return my_return;
294}

308

315
316 for(size_t n = 0; n < Nvariants.size(); n++)

321 Parameters.Allocate(Nphases, Nphases);
322
323 NucleateEvery = 1;
324 NumberOfAttempts = 100;
325 NucleiPlanted = false;
326
327 for(size_t n = 0; n < Nphases; n++)
328 for(size_t m = 0; m < Nphases; m++)

334 Parameters(n, m).Nsites = 0;
335 Parameters(n, m).Nseeds = 0;
336 Parameters(n, m).Density = 0;
337 Parameters(n, m).PositionDistributionX.param(uniform_int_distribution<int>::param_type(0, Grid.TotalNx - 1));
338 Parameters(n, m).PositionDistributionY.param(uniform_int_distribution<int>::param_type(0, Grid.TotalNy - 1));

371void Nucleation::ReadInput(std::stringstream& inp)
372{
373 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
374
375 if(moduleLocation == -1)
376 {
377 ConsoleOutput::WriteWarning("Input for module \"" + thisclassname + "\" is not found in the input file! Nucleation is disabled!",thisclassname, "ReadInput()");
378 }
379 else
380 {
381 const long int RandomNumberSeedInput = FileInterface::ReadParameterI(inp, moduleLocation, "RandomNumberSeed", false, 1);
382 SeedRandomGenerators(RandomNumberSeedInput);
383
384 NucleateEvery = FileInterface::ReadParameterI(inp, moduleLocation, "NucleateEvery", false, 1);
385 NumberOfAttempts = FileInterface::ReadParameterI(inp, moduleLocation, "NumberOfAttempts", false, 100);
386
387 for(size_t n = 0; n < Nphases; n++)
388 for(size_t m = 0; m < Nphases; m++)
389 {
390 stringstream converter;
391 converter << n << "_" << m;
392 string counter = converter.str();
393
394 Parameters(n, m).Allowed = FileInterface::ReadParameterB(inp, moduleLocation, string("Allowed_") + counter, false, "NO");
395
396 if(Parameters(n, m).Allowed)
397 {
398 string location_input = FileInterface::ReadParameterK(inp, moduleLocation, string("Location_") + counter, false, "BULK");
399
400 bool valid_location_input = false;
401
402 // Setting nucleation sites location
403 if(location_input == "BULK")
404 {
405 Parameters(n, m).LocationMode = NucleiLocationModes::Bulk;
406 valid_location_input = true;
407 }
408 if(location_input == "BULKANDGRAINBOUNDARIES")
409 {
410 Parameters(n, m).LocationMode = NucleiLocationModes::BulkAndGrainBoundaries;

414 {
415 Parameters(n, m).LocationMode = NucleiLocationModes::BulkAndInterfaces;

420 Parameters(n, m).LocationMode = NucleiLocationModes::GrainBoundaries;
421 valid_location_input = true;
422 }
423 if(location_input == "PHASEBOUNDARIES")
424 {
425 Parameters(n, m).LocationMode = NucleiLocationModes::PhaseBoundaries;
426 valid_location_input = true;
427 }
428 if(location_input == "INTERFACE" or location_input == "INTERFACES")

430 Parameters(n, m).LocationMode = NucleiLocationModes::Interfaces;

432 }

434 {
435 Parameters(n, m).LocationMode = NucleiLocationModes::Junctions;

437 }
438 if(location_input == "XBOTTOM")
439 {
440 Parameters(n, m).LocationMode = NucleiLocationModes::XBottom;

582 }

584 {

587 }
588 if(orientations_input == "REFERENCE")
589 {
590 Parameters(n, m).OrientationMode = NucleiOrientationModes::Reference;
591 orientations_set = true;
592 }
593 if(!orientations_set)
594 {
595 //Wrong seed orientation input
596 string message = "Wrong or no input for the seeds orientation!";
597 ConsoleOutput::WriteExit(message, thisclassname, "ReadInput()");
598 OP_Exit(EXIT_FAILURE);
599 }
600
601 if(Nvariants[n] > 1)
602 {
603 Parameters(n, m).Nvariants = FileInterface::ReadParameterI(inp, moduleLocation, string("Variants_") + counter, false, 1);
604
605 string variants_mode = FileInterface::ReadParameterK(inp,moduleLocation, string("VariantsMode_")+counter, false, "RANDOM");
606 bool variants_mode_set = false;
607 if(variants_mode == "RANDOM")
608 {
609 Parameters(n, m).VariantsMode = NucleiVariantsModes::Random;
610 Parameters(n, m).VariantSelector.param(uniform_int_distribution<int>::param_type(0, Nvariants[n] - 1));
611 variants_mode_set = true;
612 }
613 if(variants_mode == "LOWESTENERGY")
614 {
615 Parameters(n, m).VariantsMode = NucleiVariantsModes::LowestEnergy;
616 variants_mode_set = true;
617 }
618 if(!variants_mode_set)
619 {
620 //Wrong variants mode input
621 string message = "Wrong or no input for the variants mode!";
622 ConsoleOutput::WriteExit(message, thisclassname, "ReadInput()");
623 OP_Exit(EXIT_FAILURE);
624 }
625 }
626 }
627 else
628 {
629 ConsoleOutput::WriteExit("Nucleation mode for phase pair " + counter +
630 " could not be read", thisclassname, "ReadInput()");
631 OP_Exit(EXIT_NUCMODE_ERROR);
632 }
633 }
634 }
635 }

641void Nucleation::Remesh(int newNx, int newNy, int newNz,
642 const BoundaryConditions& BC)
643{
644 Grid.SetDimensions(newNx, newNy, newNz);
645
646 for(size_t n = 0; n < Nphases; n++)
647 for(size_t m = 0; m < Nphases; m++)
648 if(Parameters(n, m).Allowed)
649 {
650 Parameters(n, m).PositionDistributionX.param(uniform_int_distribution<int>::param_type(0, Grid.TotalNx - 1));
651 Parameters(n, m).PositionDistributionY.param(uniform_int_distribution<int>::param_type(0, Grid.TotalNy - 1));
652 Parameters(n, m).PositionDistributionZ.param(uniform_int_distribution<int>::param_type(0, Grid.TotalNz - 1));
653
654 if(Parameters(n, m).Generated)
655 for(auto it = Parameters(n, m).GeneratedNuclei.begin();
656 it != Parameters(n, m).GeneratedNuclei.end(); )
657 {
658 if(!Grid.PositionInTotalBounds(it->position))
659 {
660 it = Parameters(n, m).GeneratedNuclei.erase(it);
661 }
662 else
663 {
664 it++;
665 }
666 }
667 }
668
669 ConsoleOutput::WriteStandard(thisclassname, "Remeshed");
670}

672void Nucleation::SeedRandomGenerators(int RandomNumberSeedInput)
673{
674 const size_t RandomNumberSeed = (RandomNumberSeedInput < 0)
675 ? static_cast<size_t>(std::chrono::system_clock::now().time_since_epoch().count())
676 : static_cast<size_t>(RandomNumberSeedInput);
