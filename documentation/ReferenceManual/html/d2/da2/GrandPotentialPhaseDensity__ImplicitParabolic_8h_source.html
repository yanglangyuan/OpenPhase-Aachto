43 double PhasePressure (double Temperature, const Tensor<double,1>& ChemicalPotential) const override
44 {
45 double locPressure = 0.0;
46 for(size_t comp = 0; comp < Ncomp; ++comp)
47 {
48 // Use shorthands to improve readability
49 const double& eps = EPS[comp];
50 const double& zeta = Zeta[comp];
51 const double& mu = ChemicalPotential({comp});
52 const double& c0 = C0[comp];
53 const double c = PhaseConcentration(Temperature, mu, comp);
54 locPressure += eps*(c - c0)*(c - c0)/2 + zeta/c + c*mu;
55 }
56 return locPressure;
57 }
58 double PhaseConcentration (double Temperature, double ChemicalPotential, size_t comp) const override
59 {
60 assert(comp < Ncomp);
61 assert(precision < 1);
62 assert(precision > 0);
63
64 // Use shorthands to improve readability
65 const double& eps = EPS[comp];
66 const double& zeta = Zeta[comp];
67 const double& mu = ChemicalPotential;
68 const double& c0 = C0[comp];
69 double c = c0;//DBL_EPSILON;
70
71 auto f = [&eps, &c0, &mu, &zeta](double c) { return eps*(c - c0) - mu - zeta/c/c; };
72 auto df = [&eps, &zeta] (double c) { return eps + 2*zeta/c/c/c; };
73 try
74 {
75 RootFindingAlgorithms::PositivNewton(f, df, c, precision, MaxIterations);
76 }
77 catch (std::runtime_error& ecep)
78 {
79 ConsoleOutput::WriteWarning(ecep.what(),thisclassname,"PhaseConcentration");
80 }
81 return c;
82 }
83 double PhaseSusceptibility (double Temperature, double ChemicalPotential, size_t comp) const override
84 {
85 assert(comp < Ncomp);
86 assert(precision < 1);
87 assert(precision > 0);
88 const double& eps = EPS[comp];
89 const double& zeta = Zeta[comp];
90 const double c = PhaseConcentration(Temperature, ChemicalPotential, comp);
91 return c*c*c/(eps*c*c*c + 2*zeta);
92 }
