48 GrandPotentialPhaseDensity& operator() (size_t PhaseIdx) const
49 {
50 assert(PhaseIdx < Nphases);
51 return *storage[PhaseIdx];
52 };
53 double operator() (int i, int j, int k, const PhaseField& Phase) const
54 {
55 double omega = 0.0;
56 for(auto alpha = Phase.Fields(i,j,k).cbegin(); alpha != Phase.Fields(i,j,k).cend(); ++alpha)
57 {
58 size_t PhaseIdx = Phase.FieldsProperties[alpha->index].Phase;
59 omega += alpha->value*(*storage[PhaseIdx])(i,j,k);
60 }
61 return omega;
62 };
63
64 //NOTE: The set-methods set references to external storage classes.
65 // A set-method needs to be called explicitly only once but will it be
66 // implicitly invoked by the compiler on each function call when the
67 // external external memory is accessed during that function call!
68 void Set(const Storage3D<double,0>& Temperature, const Storage3D<double,1>& ChemicalPotential)
69 {
70 if(DoGravity)
71 {
72 for(auto omega : storage)
73 {
74 omega->Set(Temperature,ChemicalPotential,GravityDirection);
75 }
76 }
77 else
78 {
79 for(auto omega : storage)
80 {
81 omega->Set(Temperature,ChemicalPotential);
82 }
83 }
84
85 };
