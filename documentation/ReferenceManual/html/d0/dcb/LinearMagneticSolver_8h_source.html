100 inline dVector3 H0 (const int i, const int j, const int k) const
101 {
102 const double Hx = H0x + (dH0x_dx*i + dH0x_dy*j + dH0x_dz*k)*Grid.dx;
103 const double Hy = H0y + (dH0y_dx*i + dH0y_dy*j + dH0y_dz*k)*Grid.dx;
104 const double Hz = H0z + (dH0z_dx*i + dH0z_dy*j + dH0z_dz*k)*Grid.dx;
105 return dVector3({Hx,Hy,Hz});
106 }

109 inline dVector3 H (const int i, const int j, const int k) const
110 {
111 const double dchi_dx = (Grid.dNx) ? (phi(i+1,j,k)-phi(i-1,j,k))/2.0/Grid.dx : 0.0;
112 const double dchi_dy = (Grid.dNy) ? (phi(i,j+1,k)-phi(i,j-1,k))/2.0/Grid.dx : 0.0;
113 const double dchi_dz = (Grid.dNz) ? (phi(i,j,k+1)-phi(i,j,k-1))/2.0/Grid.dx : 0.0;
114
115 const double Hx = H0x + (dH0x_dx*i + dH0x_dy*j + dH0x_dz*k)*Grid.dx - dchi_dx;
116 const double Hy = H0y + (dH0y_dx*i + dH0y_dy*j + dH0y_dz*k)*Grid.dx - dchi_dy;
117 const double Hz = H0z + (dH0z_dx*i + dH0z_dy*j + dH0z_dz*k)*Grid.dx - dchi_dz;
118 return dVector3({Hx,Hy,Hz});
119 }

134 inline dVector3 ForceDensity (const int i, const int j, const int k) const
135 {
136 dMatrix3x3 gradB;
137 gradB(0,0) = (Grid.dNx) ? (B(i+1,j,k)[0]-B(i-1,j,k)[0])/(2.0*Grid.dx) : 0.0;
138 gradB(0,1) = (Grid.dNy) ? (B(i,j+1,k)[0]-B(i,j-1,k)[0])/(2.0*Grid.dx) : 0.0;
139 gradB(0,2) = (Grid.dNz) ? (B(i,j,k+1)[0]-B(i,j,k-1)[0])/(2.0*Grid.dx) : 0.0;
140 gradB(1,0) = (Grid.dNx) ? (B(i+1,j,k)[1]-B(i-1,j,k)[1])/(2.0*Grid.dx) : 0.0;
141 gradB(1,1) = (Grid.dNy) ? (B(i,j+1,k)[1]-B(i,j-1,k)[1])/(2.0*Grid.dx) : 0.0;
142 gradB(1,2) = (Grid.dNz) ? (B(i,j,k+1)[1]-B(i,j,k-1)[1])/(2.0*Grid.dx) : 0.0;
143 gradB(2,0) = (Grid.dNx) ? (B(i+1,j,k)[2]-B(i-1,j,k)[2])/(2.0*Grid.dx) : 0.0;
144 gradB(2,1) = (Grid.dNy) ? (B(i,j+1,k)[2]-B(i,j-1,k)[2])/(2.0*Grid.dx) : 0.0;
145 gradB(2,2) = (Grid.dNz) ? (B(i,j,k+1)[2]-B(i,j,k-1)[2])/(2.0*Grid.dx) : 0.0;
146 return gradB*M(i,j,k);
147 }
