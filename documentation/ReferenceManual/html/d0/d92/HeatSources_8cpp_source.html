71void HeatSources::ReadInput(std::stringstream& inp)
72{
73 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
74
75 bool endofsources = false;
76 size_t idx = 0;
77 while(!endofsources)
78 {
79 stringstream converter;
80 converter << idx;
81 string counter = converter.str();
82 if(FileInterface::FindParameter(inp, moduleLocation, string("Source_") + counter) != -1)
83 {
84 ConsoleOutput::WriteLine("=");
85 bool SourceEnabled = FileInterface::ReadParameterB(inp, moduleLocation, string("Source_") + counter, true, false);
86
87 if(SourceEnabled)
88 {
89 HeatSourceStructure locHeatSource;
90 locHeatSource.Active = false;
91 locHeatSource.Value = FileInterface::ReadParameterD(inp, moduleLocation, string("Value_") + counter, true, 0.0);
92
93 bool SourceTypeSet = false;
94 string SourceType = FileInterface::ReadParameterK(inp, moduleLocation, string("Type_") + counter, true, "NONE");
95 if(SourceType == "BCX0" ) {locHeatSource.Type = HeatSourceTypes::BCX0; SourceTypeSet = true;}
96 if(SourceType == "BCXN" ) {locHeatSource.Type = HeatSourceTypes::BCXN; SourceTypeSet = true;}
97 if(SourceType == "BCY0" ) {locHeatSource.Type = HeatSourceTypes::BCY0; SourceTypeSet = true;}
98 if(SourceType == "BCYN" ) {locHeatSource.Type = HeatSourceTypes::BCYN; SourceTypeSet = true;}
99 if(SourceType == "BCZ0" ) {locHeatSource.Type = HeatSourceTypes::BCZ0; SourceTypeSet = true;}
100 if(SourceType == "BCZN" ) {locHeatSource.Type = HeatSourceTypes::BCZN; SourceTypeSet = true;}
101 if(SourceType == "PHASE" ) {locHeatSource.Type = HeatSourceTypes::Phase; SourceTypeSet = true;}
102 if(SourceType == "ELLIPSOIDAL") {locHeatSource.Type = HeatSourceTypes::Ellipsoidal; SourceTypeSet = true;}
103
104 if(not SourceTypeSet)
105 {
106 string message = "No valid type is provided for the heat source " + counter + " --> " + SourceType;
107 ConsoleOutput::WriteExit(message,thisclassname,"ReadInput()");
108 OP_Exit(EXIT_FAILURE);
109 }
110
111 switch(locHeatSource.Type)
112 {
113 case HeatSourceTypes::Phase:
114 {
115 locHeatSource.PhaseIndex = FileInterface::ReadParameterI(inp, moduleLocation, string("PhaseIndex_") + counter, true, 0);
116 break;
117 }
118 case HeatSourceTypes::Ellipsoidal:
119 {
120 locHeatSource.Position = FileInterface::ReadParameterV3(inp, moduleLocation, string("Position_") + counter, true, {0.0,0.0,0.0});
121 locHeatSource.Size = FileInterface::ReadParameterV3(inp, moduleLocation, string("Radii_" ) + counter, true, {0.0,0.0,0.0});
122 locHeatSource.Velocity = FileInterface::ReadParameterV3(inp, moduleLocation, string("Velocity_") + counter, false, {0.0,0.0,0.0});
123
124 string EnergyDistribution = FileInterface::ReadParameterK(inp, moduleLocation, string("EnergyDistribution_") + counter, false, "UNIFORM");
125 bool EnergyDistributionSet = false;
126 if(EnergyDistribution == "UNIFORM" ) {locHeatSource.Distribution = HeatSourceEnergyDistributions::Uniform; EnergyDistributionSet = true;}
127 if(EnergyDistribution == "GAUSSIAN") {locHeatSource.Distribution = HeatSourceEnergyDistributions::Gaussian; EnergyDistributionSet = true;}
128
129 if(not EnergyDistributionSet)
130 {
131 string message = "No valid energy distribution is provided for the heat source " + counter + " --> " + EnergyDistribution;
132 ConsoleOutput::WriteExit(message,thisclassname,"ReadInput()");
133 OP_Exit(EXIT_FAILURE);
134 }
135 break;
136 }
137 default:
138 {
139 break;
140 }
141 }
142
143 bool SourceONtriggerSet = false;
144 string SourceONtrigger = FileInterface::ReadParameterK(inp, moduleLocation, string("TriggerON_") + counter, true, "USER");
145 if(SourceONtrigger == "USER" ) {locHeatSource.TriggerON = EventTriggers::User; SourceONtriggerSet = true;}
146 if(SourceONtrigger == "TMAX" ) {locHeatSource.TriggerON = EventTriggers::Tmax; SourceONtriggerSet = true;}
147 if(SourceONtrigger == "TMIN" ) {locHeatSource.TriggerON = EventTriggers::Tmin; SourceONtriggerSet = true;}
148 if(SourceONtrigger == "TIME" ) {locHeatSource.TriggerON = EventTriggers::Time; SourceONtriggerSet = true;}
149 if(SourceONtrigger == "TIMESTEP" ) {locHeatSource.TriggerON = EventTriggers::TimeStep; SourceONtriggerSet = true;}
150 if(SourceONtrigger == "PHASEFRACTIONMAX") {locHeatSource.TriggerON = EventTriggers::PhaseFractionMax; SourceONtriggerSet = true;}
151 if(SourceONtrigger == "PHASEFRACTIONMIN") {locHeatSource.TriggerON = EventTriggers::PhaseFractionMin; SourceONtriggerSet = true;}
152
153 if(not SourceONtriggerSet)
154 {
155 string message = "No valid ON trigger is provided for the heat source " + counter + " --> " + SourceONtrigger;
156 ConsoleOutput::WriteExit(message,thisclassname,"ReadInput()");
157 OP_Exit(EXIT_FAILURE);
158 }
159
160 if(locHeatSource.TriggerON != EventTriggers::User)
161 {
162 locHeatSource.ONtriggerValue = FileInterface::ReadParameterD(inp, moduleLocation, string("TriggerONvalue_") + counter, true, 0);
163 }
164 if(locHeatSource.TriggerON == EventTriggers::PhaseFractionMax or locHeatSource.TriggerON == EventTriggers::PhaseFractionMin)
165 {
166 locHeatSource.PhaseIndexON = FileInterface::ReadParameterI(inp, moduleLocation, string("PhaseIndexON_") + counter, true, 0);
167 }
168
169 bool SourceOFFtriggerSet = false;
170 string SourceOFFtrigger = FileInterface::ReadParameterK(inp, moduleLocation, string("TriggerOFF_") + counter, false, SourceONtrigger);
171 if(SourceOFFtrigger == "USER" ) {locHeatSource.TriggerOFF = EventTriggers::User; SourceOFFtriggerSet = true;}
172 if(SourceOFFtrigger == "TMAX" ) {locHeatSource.TriggerOFF = EventTriggers::Tmax; SourceOFFtriggerSet = true;}
173 if(SourceOFFtrigger == "TMIN" ) {locHeatSource.TriggerOFF = EventTriggers::Tmin; SourceOFFtriggerSet = true;}
174 if(SourceOFFtrigger == "TIME" ) {locHeatSource.TriggerOFF = EventTriggers::Time; SourceOFFtriggerSet = true;}
175 if(SourceOFFtrigger == "TIMESTEP" ) {locHeatSource.TriggerOFF = EventTriggers::TimeStep; SourceOFFtriggerSet = true;}
176 if(SourceOFFtrigger == "PHASEFRACTIONMAX") {locHeatSource.TriggerOFF = EventTriggers::PhaseFractionMax; SourceOFFtriggerSet = true;}
177 if(SourceOFFtrigger == "PHASEFRACTIONMIN") {locHeatSource.TriggerOFF = EventTriggers::PhaseFractionMin; SourceOFFtriggerSet = true;}
178
179 if(not SourceOFFtriggerSet)
180 {
181 string message = "No valid OFF trigger is provided for the heat source " + counter + " --> " + SourceOFFtrigger;
182 ConsoleOutput::WriteExit(message,thisclassname,"ReadInput()");
183 OP_Exit(EXIT_FAILURE);
184 }
185
186 if(locHeatSource.TriggerOFF != EventTriggers::User)
187 {
188 locHeatSource.OFFtriggerValue = FileInterface::ReadParameterD(inp, moduleLocation, string("TriggerOFFvalue_") + counter, true, 0);
189 }
190
191 if(locHeatSource.TriggerOFF == EventTriggers::PhaseFractionMax or locHeatSource.TriggerOFF == EventTriggers::PhaseFractionMin)
192 {
193 locHeatSource.PhaseIndexOFF = FileInterface::ReadParameterI(inp, moduleLocation, string("PhaseIndexOFF_") + counter, true, 0);
194 }
195
196 locHeatSource.Repeat = FileInterface::ReadParameterI(inp, moduleLocation, string("Repeat_") + counter, false, 1);
197 if(locHeatSource.Repeat <= 0)
198 {
199 string message = "Zero or negative \"Repeat\" parameter for heat source " + counter + " --> " + to_string(locHeatSource.Repeat);
200 ConsoleOutput::WriteExit(message,thisclassname,"ReadInput()");
201 OP_Exit(EXIT_FAILURE);
202 }
203
204 switch(locHeatSource.Type)
205 {
206 case HeatSourceTypes::Phase:
207 {
208 locHeatSource.Mode = HeatSourceModes::ThreeDimensional;
209 break;
210 }
211 case HeatSourceTypes::Ellipsoidal:
212 {
213 if(locHeatSource.Size[0] == 0 or
214 locHeatSource.Size[1] == 0 or
215 locHeatSource.Size[2] == 0)
216 {
217 locHeatSource.Mode = HeatSourceModes::TwoDimensional;
218 }
219 else
220 {
221 locHeatSource.Mode = HeatSourceModes::ThreeDimensional;
222 }
223 if(locHeatSource.Size[0] == 0)
224 {
225 locHeatSource.Size[0] = 0.5;
226 }
227 if(locHeatSource.Size[1] == 0)
228 {
229 locHeatSource.Size[1] = 0.5;
230 }
231 if(locHeatSource.Size[2] == 0)
232 {
233 locHeatSource.Size[2] = 0.5;
234 }
235 break;
236 }
237 default:
238 {
239 locHeatSource.Mode = HeatSourceModes::TwoDimensional;
240 }
241 }
242
243 Sources.push_back(locHeatSource);
244 }
245 idx++;
246 }
247 else
248 {
249 endofsources = true;
250 }
251 }
252
253 ConsoleOutput::WriteLine();
254 ConsoleOutput::WriteBlankLine();
255}

263bool HeatSources::Read(const std::string FileName)
264{
265 std::ifstream inp(FileName.c_str(), std::ios::in | std::ios::binary);
266 if (!inp)
267 {
268 ConsoleOutput::WriteWarning("File \"" + FileName + "\" could not be opened", thisclassname, "Read()");
269 return false;
270 }
271
272 size_t size = 0;
273 inp.read(reinterpret_cast<char*>(&size), sizeof(size_t));
274
275 if(Sources.size() != size)
276 {
277 Sources.resize(size);
278 }
279
280 for(size_t n = 0; n != size; n++) Sources[n].Read(inp);
281
282 inp.close();
283 ConsoleOutput::WriteStandard(thisclassname, "Binary input loaded");
284 return true;
285}

318void HeatSources::Apply(PhaseField& Phase, Temperature& Tx, HeatDiffusion& HD)
319{
320 for(auto source = Sources.begin(); source != Sources.end(); source++)
321 if(source->Active)
322 {
323 double source_value = source->Value;
324 if(source->Mode == HeatSourceModes::TwoDimensional)
325 {
326 source_value /= HD.Grid.dx;
327 }
328
329 switch(source->Type)
330 {
331 case HeatSourceTypes::BCX0:
332 {
333 if(Tx.ExtensionX0.isActive())
334 {
335 Tx.ExtensionX0.Qdot = source_value;
336 }
337 else if(Tx.Grid.OffsetX == 0)
338 {
339 int x = 0;
340#ifdef _OPENMP
341#pragma omp parallel for collapse(2) schedule(OMP_SCHEDULING_TYPE,OMP_CHUNKSIZE)
342#endif
343 for(int y = 0; y < HD.Grid.Ny; y++)
344 for(int z = 0; z < HD.Grid.Nz; z++)
345 {
346 HD.Qdot(x,y,z) += source_value;
347 }
348 }
349 break;
350 }
351 case HeatSourceTypes::BCXN:
352 {
353 if(Tx.ExtensionXN.isActive())
354 {
355 Tx.ExtensionXN.Qdot = source_value;
356 }
357 else if(Tx.Grid.OffsetX + Tx.Grid.Nx == Tx.Grid.TotalNx)
358 {
359 int x = HD.Grid.Nx-1;
360#ifdef _OPENMP
361#pragma omp parallel for collapse(2) schedule(OMP_SCHEDULING_TYPE,OMP_CHUNKSIZE)
362#endif
363 for(int y = 0; y < HD.Grid.Ny; y++)
364 for(int z = 0; z < HD.Grid.Nz; z++)
365 {
366 HD.Qdot(x,y,z) += source_value;
367 }
368 }
369 break;
370 }
371 case HeatSourceTypes::BCY0:
372 {
373 if(Tx.ExtensionY0.isActive())
374 {
375 Tx.ExtensionY0.Qdot = source_value;
376 }
377 else if(Tx.Grid.OffsetY == 0)
378 {
379 int y = 0;
380#ifdef _OPENMP
381#pragma omp parallel for collapse(2) schedule(OMP_SCHEDULING_TYPE,OMP_CHUNKSIZE)
382#endif
383 for(int x = 0; x < HD.Grid.Nx; x++)
384 for(int z = 0; z < HD.Grid.Nz; z++)
385 {
386 HD.Qdot(x,y,z) += source_value;
387 }
388 }
389 break;
390 }
391 case HeatSourceTypes::BCYN:
392 {
393 if(Tx.ExtensionYN.isActive())
394 {
395 Tx.ExtensionYN.Qdot = source_value;
396 }
397 else if(Tx.Grid.OffsetY + Tx.Grid.Ny == Tx.Grid.TotalNy)
398 {
399 int y = HD.Grid.Ny-1;
400#ifdef _OPENMP
401#pragma omp parallel for collapse(2) schedule(OMP_SCHEDULING_TYPE,OMP_CHUNKSIZE)
402#endif
403 for(int x = 0; x < HD.Grid.Nx; x++)
404 for(int z = 0; z < HD.Grid.Nz; z++)
405 {
406 HD.Qdot(x,y,z) += source_value;
407 }
408 }
409 break;
410 }
411 case HeatSourceTypes::BCZ0:
412 {
413 if(Tx.ExtensionZ0.isActive())
414 {
415 Tx.ExtensionZ0.Qdot = source_value;
416 }
417 else if(Tx.Grid.OffsetZ == 0)
418 {
419 int z = 0;
420#ifdef _OPENMP
421#pragma omp parallel for collapse(2) schedule(OMP_SCHEDULING_TYPE,OMP_CHUNKSIZE)
422#endif
423 for(int x = 0; x < HD.Grid.Nx; x++)
424 for(int y = 0; y < HD.Grid.Ny; y++)
425 {
426 HD.Qdot(x,y,z) += source_value;
427 }
428 }
429 break;
430 }
431 case HeatSourceTypes::BCZN:
432 {
433 if(Tx.ExtensionZN.isActive())
434 {
435 Tx.ExtensionZN.Qdot = source_value;
436 }
437 else if(Tx.Grid.OffsetZ + Tx.Grid.Nz == Tx.Grid.TotalNz)
438 {
439 int z = HD.Grid.Nz-1;
440#ifdef _OPENMP
441#pragma omp parallel for collapse(2) schedule(OMP_SCHEDULING_TYPE,OMP_CHUNKSIZE)
442#endif
443 for(int x = 0; x < HD.Grid.Nx; x++)
444 for(int y = 0; y < HD.Grid.Ny; y++)
445 {
446 HD.Qdot(x,y,z) += source_value;
447 }
448 }
449 break;
450 }
451 case HeatSourceTypes::Phase:
452 {
453 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
454 {
455 if(Phase.Fractions(i,j,k,{source->PhaseIndex}) > DBL_EPSILON)
456 {
457 HD.Qdot(i,j,k) += source_value * Phase.Fractions(i,j,k,{source->PhaseIndex});
458 }
459 }
460 OMP_PARALLEL_STORAGE_LOOP_END
461 break;
462 }
463 case HeatSourceTypes::Ellipsoidal:
464 {
465 double radiusX2 = pow(source->Size[0],2);
466 double radiusY2 = pow(source->Size[1],2);
467 double radiusZ2 = pow(source->Size[2],2);
468
469 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
470 {
471 double radX2 = pow((i+Phase.Grid.OffsetX - source->Position[0]), 2);
472 double radY2 = pow((j+Phase.Grid.OffsetY - source->Position[1]), 2);
473 double radZ2 = pow((k+Phase.Grid.OffsetZ - source->Position[2]), 2);
474
475 if ((radX2/radiusX2 + radY2/radiusY2 + radZ2/radiusZ2) <= 1.0)
476 {
477 switch(source->Distribution)
478 {
479 case HeatSourceEnergyDistributions::Uniform:
480 {
481 HD.Qdot(i,j,k) += source_value;
482 break;
483 }
484 case HeatSourceEnergyDistributions::Gaussian:
485 {
486 HD.Qdot(i,j,k) += source_value * exp(-0.5*9.0*(radX2/radiusX2 + radY2/radiusY2 + radZ2/radiusZ2));
487 break;
488 }
489 }
490 }
491 }
492 OMP_PARALLEL_STORAGE_LOOP_END
493 break;
494 }
495
496 default:
497 {
498 break;
499 }
500 }
501 }

504void HeatSources::Activate(PhaseField& Phase, Temperature& Tx, RunTimeControl& RTC)
505{
506 int counter = 0;
507 for(auto source = Sources.begin(); source != Sources.end(); source++)
508 {
509 if(source->Active)
510 {
511 switch(source->TriggerOFF)
512 {
513 case EventTriggers::Tmin:
514 {
515 if(Tx.Tmin < source->OFFtriggerValue)
516 {
517 source->Active = false;
518 }
519 break;
520 }
521 case EventTriggers::Tmax:
522 {
523 if(Tx.Tmax > source->OFFtriggerValue)
524 {
525 source->Active = false;
526 }
527 break;
528 }
529
530 case EventTriggers::TimeStep:
531 {
532 if(RTC.TimeStep == source->OFFtriggerValue)
533 {
534 source->Active = false;
535 }
536 break;
537 }
538 case EventTriggers::Time:
539 {
540 if(RTC.SimulationTime > source->OFFtriggerValue - RTC.dt and
541 RTC.SimulationTime < source->OFFtriggerValue + RTC.dt)
542 {
543 source->Active = false;
544 }
545 break;
546 }
547 case EventTriggers::PhaseFractionMax:
548 {
549 if(Phase.FractionsTotal[source->PhaseIndexOFF] > source->OFFtriggerValue)
550 {
551 source->Active = false;
552 }
553 break;
554 }
555 case EventTriggers::PhaseFractionMin:
556 {
557 if(Phase.FractionsTotal[source->PhaseIndexOFF] < source->OFFtriggerValue)
558 {
559 source->Active = false;
560 }
561 break;
562 }
563 case EventTriggers::User:
564 default:
565 {
566 break;
567 }
568 }
569 if(!source->Active)
570 {
571 std::stringstream message;
572 message << " Activate Heat Sources: heat source " << counter << " is deactivated.";
573 ConsoleOutput::WriteStandard(thisclassname, message.str());
574 }
575 }
576 else if (source->Repeat != 0)
577 {
578 switch(source->TriggerON)
579 {
580 case EventTriggers::Tmin:
581 {
582 if(Tx.Tmin < source->ONtriggerValue)
583 {
584 source->Active = true;
585 source->Repeat -= 1;
586 }
587 break;
588 }
589 case EventTriggers::Tmax:
590 {
591 if(Tx.Tmax > source->ONtriggerValue)
592 {
593 source->Active = true;
594 source->Repeat -= 1;
595 }
596 break;
597 }
598
599 case EventTriggers::TimeStep:
600 {
601 if(RTC.TimeStep == source->ONtriggerValue)
602 {
603 source->Active = true;
604 source->Repeat -= 1;
605 }
606 break;
607 }
608 case EventTriggers::Time:
609 {
610 if(RTC.SimulationTime > source->ONtriggerValue - RTC.dt and
611 RTC.SimulationTime < source->ONtriggerValue + RTC.dt)
612 {
613 source->Active = true;
614 source->Repeat -= 1;
615 }
616 break;
617 }
618 case EventTriggers::PhaseFractionMax:
619 {
620 if(Phase.FractionsTotal[source->PhaseIndexON] > source->ONtriggerValue)
621 {
622 source->Active = true;
623 source->Repeat -= 1;
624 }
625 break;
626 }
627 case EventTriggers::PhaseFractionMin:
628 {
629 if(Phase.FractionsTotal[source->PhaseIndexON] < source->ONtriggerValue)
630 {
631 source->Active = true;
632 source->Repeat -= 1;
633 }
634 break;
635 }
636 case EventTriggers::User:
637 default:
638 {
639 break;
640 }
641 }
642 if(source->Active)
643 {
644 std::stringstream message;
645 message << " Activate Heat Sources: heat source " << counter << " is activated.";
646 ConsoleOutput::WriteStandard(thisclassname, message.str());
647 }
648 }
649 // Move heat source
650 if(source->Active and source->Type == HeatSourceTypes::Ellipsoidal)
651 {
652 source->Position += source->Velocity*RTC.dt/Tx.Grid.dx;
653 }
654
655 counter++;
656 }
657}
