38//Auxiliary function to rotate grain interface normal back to the reference configuration
39dVector3 InterfaceOrientation(const PhaseField& Phase, NodePF::citerator it)
40{
41 dVector3 normal = it->gradient*(-1.0);
42 normal.normalize();
43 normal = Phase.FieldsProperties[it->index].Orientation.RotationMatrix.transposed()*normal;
44 return normal;
45}

481void InterfaceProperties::Coarsen(const PhaseField& Phase)
482{
483 double norm = 1.0/pow(2.0,Grid.Active());
484 long int fx = 1 + Grid.dNx;
485 long int fy = 1 + Grid.dNy;
486 long int fz = 1 + Grid.dNz;
487 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
488 {
489 if(Phase.Fields(i,j,k).wide_interface())
490 {
491 Properties(i,j,k).clear();
492
493 for(int di = -Grid.dNx; di <= Grid.dNx; di+=2)
494 for(int dj = -Grid.dNy; dj <= Grid.dNy; dj+=2)
495 for(int dk = -Grid.dNz; dk <= Grid.dNz; dk+=2)
496 {
497 Properties(i,j,k).add_energies_and_mobilities(PropertiesDR(fx*i+(di+1)/2,fy*j+(dj+1)/2,fz*k+(dk+1)/2));
498 }
499 Properties(i,j,k) *= norm;
500 }
501 }
502 OMP_PARALLEL_STORAGE_LOOP_END
503}

523double InterfaceProperties::ReportMaximumTimeStep(void)
524{
525 double maxEnergy = 0.0;
526 double maxMobility = 0.0;
527 for(size_t n = 0; n < Nphases; n++)
528 for(size_t m = 0; m < Nphases; m++)
529 {
530 maxEnergy = std::max(maxEnergy, maxEnergies(n,m));
531 maxMobility = std::max(maxMobility, maxMobilities(n,m));
532 }
533 double max_dt = DBL_MAX;
534 if(maxEnergy > DBL_EPSILON and maxMobility > 0.0)
535 {
536 switch(Grid.Active())
537 {
538 case 1:
539 {
540 max_dt = 0.50*Grid.dx*Grid.dx/(maxMobility*maxEnergy*RegularizationFactor);
541 break;
542 }
543 case 2:
544 {
545 max_dt = 0.25*Grid.dx*Grid.dx/(maxMobility*maxEnergy*RegularizationFactor);
546 break;
547 }
548 case 3:
549 {
550 max_dt = 0.16*Grid.dx*Grid.dx/(maxMobility*maxEnergy*RegularizationFactor);
551 break;
552 }
553 }
554 }
555 return max_dt;

603void InterfaceProperties::SetSR(const PhaseField& Phase)
604{
605 Matrix<double> locMaxEnergies(Nphases,Nphases);
606 Matrix<double> locMaxMobilities(Nphases,Nphases);
607
608 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Properties, 0, reduction(MatrixDMAX:locMaxEnergies) reduction(MatrixDMAX:locMaxMobilities))
609 {
610 Properties(i,j,k).clear();
611
612 if(Phase.Fields(i,j,k).wide_interface())
613 {
614 for(auto alpha = Phase.Fields(i,j,k).cbegin();
615 alpha != Phase.Fields(i,j,k).cend(); ++alpha)
616 for(auto beta = alpha + 1;
617 beta != Phase.Fields(i,j,k).cend(); ++beta)
618 if(alpha != beta)
619 {
620 int pIndexA = Phase.FieldsProperties[alpha->index].Phase;
621 int pIndexB = Phase.FieldsProperties[ beta->index].Phase;
622
623 // Setting interface energy/stiffness
624 double locEnergy = 0.0;
625 if(InterfaceEnergy(pIndexA, pIndexB).Model == InterfaceEnergyModels::Iso)
626 {
627 locEnergy = InterfaceEnergy(pIndexA, pIndexB).Energy;
628 }
629 else if(InterfaceEnergy(pIndexA, pIndexB).Model == InterfaceEnergyModels::Disorientation)
630 {
631 Quaternion OrientationA = Phase.FieldsProperties[alpha->index].Orientation;
632 Quaternion OrientationB = Phase.FieldsProperties[ beta->index].Orientation;
633 locEnergy = InterfaceEnergy(pIndexA, pIndexB).CalculateDisorientation(OrientationA, OrientationB);
634 }
635 else if(InterfaceEnergy(pIndexA, pIndexB).Model != InterfaceEnergyModels::Ext)
636 {
637 double weightA = 0.0;
638 double weightB = 0.0;
639
640 if(Phase.FieldsProperties[alpha->index].State == AggregateStates::Solid)
641 {
642 weightA = fabs(alpha->value*(1.0 - alpha->value));
643 if(weightA > DBL_EPSILON)
644 {
645 dVector3 Normal_alpha = InterfaceOrientation(Phase,alpha);
646 locEnergy += InterfaceEnergy(pIndexA, pIndexB).Calculate(Normal_alpha)*weightA;
647 }
648 else
649 {
650 weightA = 0.0;
651 }
652 }
653
654 if(Phase.FieldsProperties[beta->index].State == AggregateStates::Solid)
655 {
656 weightB = fabs(beta->value*(1.0 - beta->value));
657
658 if(weightB > DBL_EPSILON)
659 {
660 dVector3 Normal_beta = InterfaceOrientation(Phase,beta);
661 locEnergy += InterfaceEnergy(pIndexA, pIndexB).Calculate(Normal_beta)*weightB;
662 }
663 else
664 {
665 weightB = 0.0;
666 }
667 }
668
669 if (weightA + weightB > DBL_EPSILON)
670 {
671 locEnergy /= weightA + weightB;
672 }
673 else
674 {
675 locEnergy = InterfaceEnergy(pIndexA, pIndexB).MaxEnergy;
676 }
677 }
678
679 // Setting interface mobility
680 double locMobility = 0.0;
681 if(InterfaceMobility(pIndexA, pIndexB).Model == InterfaceMobilityModels::Iso)
682 {
683 locMobility = InterfaceMobility(pIndexA, pIndexB).MaxMobility;
684 }
685 else if (InterfaceMobility(pIndexA, pIndexB).Model != InterfaceMobilityModels::Ext)
686 {
687 int number_of_solids = 0;
688 if(Phase.FieldsProperties[alpha->index].State == AggregateStates::Solid)
689 if(alpha->value != 0.0 and alpha->value != 1.0)
690 {
691 dVector3 Normal_alpha = InterfaceOrientation(Phase,alpha);
692 locMobility += InterfaceMobility(pIndexA, pIndexB).Calculate(Normal_alpha);
693 number_of_solids ++;
694 }
695
696 if(Phase.FieldsProperties[beta->index].State == AggregateStates::Solid)
697 if(beta->value != 0.0 and beta->value != 1.0)
698 {
699 dVector3 Normal_beta = InterfaceOrientation(Phase,beta);
700 locMobility += InterfaceMobility(pIndexA, pIndexB).Calculate(Normal_beta);
701 number_of_solids ++;
702 }
703
704 if(number_of_solids == 0)
705 {
706 locMobility = InterfaceMobility(pIndexA, pIndexB).Mobility;
707 }
708 }
709
710 if(RespectParentBoundaries(pIndexA, pIndexB))
711 {
712 if(pIndexA == pIndexB and
713 Phase.FieldsProperties[alpha->index].Parent != Phase.FieldsProperties[ beta->index].Parent)
714 {
715 locMobility = 0.0;
716 }
717
718 if(pIndexA != pIndexB and
719 Phase.FieldsProperties[alpha->index].Parent != beta->index and
720 Phase.FieldsProperties[ beta->index].Parent != alpha->index)
721 {
722 locMobility = 0.0;
723 }
724 }
725
726 // Modifying mobility for nuclei (if enabled)
727 if(NucleiMobilityFactor(pIndexA, pIndexB) != 1.0 and
728 (Phase.FieldsProperties[alpha->index].IsNucleus() or
729 Phase.FieldsProperties[ beta->index].IsNucleus()))
730 {
731 double volume_ratio = min(Phase.FieldsProperties[alpha->index].VolumeRatio,
732 Phase.FieldsProperties[ beta->index].VolumeRatio);
733
734 locMobility *= NucleiMobilityFactor(pIndexA, pIndexB)*(1.0 - volume_ratio) + 1.0*volume_ratio;
735 }
736
737 // Modifying mobility if growth constraints are violated
738 if(Phase.FieldsProperties[alpha->index].GrowthConstraintsViolation != GrowthConstraintsViolations::None and
739 Phase.FieldsProperties[ beta->index].GrowthConstraintsViolation != GrowthConstraintsViolations::None)
740 {
741 locMobility *= Phase.PairwiseGrowthFactors.get_sym1(alpha->index,beta->index);
742 }
743
744 // Putting interface energy and mobility values into the properties storage
745 if(locEnergy != 0.0 or locMobility != 0.0)
746 {
747 Properties(i,j,k).set_energy_and_mobility(alpha->index, beta->index, locEnergy, locMobility);
748 }
749 locMaxEnergies(pIndexA,pIndexB) = max(locMaxEnergies(pIndexA,pIndexB),locEnergy);
750 locMaxMobilities(pIndexA,pIndexB) = max(locMaxMobilities(pIndexA,pIndexB),locMobility);
751 }
752 }
753 }
754 OMP_PARALLEL_STORAGE_LOOP_END
755 maxEnergies = locMaxEnergies;
756 maxMobilities = locMaxMobilities;
757
758#ifdef MPI_PARALLEL
759 OP_MPI_Allreduce(OP_MPI_IN_PLACE, maxEnergies.data(), Nphases*Nphases, OP_MPI_DOUBLE, OP_MPI_MAX, OP_MPI_COMM_WORLD);
760 OP_MPI_Allreduce(OP_MPI_IN_PLACE, maxMobilities.data(), Nphases*Nphases, OP_MPI_DOUBLE, OP_MPI_MAX, OP_MPI_COMM_WORLD);
761#endif

766 size_t pIndexA = Phase.FieldsProperties[alpha->index].Phase;
767 size_t pIndexB = Phase.FieldsProperties[ beta->index].Phase;

785 }
786

788 if(beta->value != 0.0 and beta->value != 1.0)

812 for(auto alpha = Phase.FieldsDR(i,j,k).cbegin();

815 beta != Phase.FieldsDR(i,j,k).cend(); ++beta)

817 {
818 int pIndexA = Phase.FieldsProperties[alpha->index].Phase;
819 int pIndexB = Phase.FieldsProperties[ beta->index].Phase;

832 }
833 else if(InterfaceEnergy(pIndexA, pIndexB).Model != InterfaceEnergyModels::Ext)

836 double weightB = 0.0;

858 dVector3 Normal_beta = InterfaceOrientation(Phase,beta);
859 locEnergy += InterfaceEnergy(pIndexA, pIndexB).Calculate(Normal_beta)*weightB;
860 }
861 else
862 {
863 weightB = 0.0;
864 }
865 }
866
867 if (weightA + weightB > DBL_EPSILON)
868 {
869 locEnergy /= weightA + weightB;
870 }
871 else
872 {
873 locEnergy = InterfaceEnergy(pIndexA, pIndexB).MaxEnergy;
874 }
875 }
876
877 // Setting interface mobility
878 double locMobility = 0.0;
879 if(InterfaceMobility(pIndexA, pIndexB).Model == InterfaceMobilityModels::Iso)
880 {
881 locMobility = InterfaceMobility(pIndexA, pIndexB).MaxMobility;
882 }
883 else if (InterfaceMobility(pIndexA, pIndexB).Model != InterfaceMobilityModels::Ext)

888 {
889 dVector3 Normal_alpha = InterfaceOrientation(Phase,alpha);
890 locMobility += InterfaceMobility(pIndexA, pIndexB).Calculate(Normal_alpha);
891 number_of_solids ++;
892 }
893
894 if(Phase.FieldsProperties[beta->index].State == AggregateStates::Solid)
895 if(beta->value != 0.0 and beta->value != 1.0)
896 {
897 dVector3 Normal_beta = InterfaceOrientation(Phase,beta);
898 locMobility += InterfaceMobility(pIndexA, pIndexB).Calculate(Normal_beta);
899 number_of_solids ++;
900 }
901
902 if(number_of_solids == 0)
903 {
904 locMobility = InterfaceMobility(pIndexA, pIndexB).MaxMobility;
905 }
906 }
907
908 if(RespectParentBoundaries(pIndexA, pIndexB))
909 {
910 if(pIndexA == pIndexB and
911 Phase.FieldsProperties[alpha->index].Parent != Phase.FieldsProperties[ beta->index].Parent)
912 {
913 locMobility = 0.0;
914 }
915
916 if(pIndexA != pIndexB and
917 Phase.FieldsProperties[alpha->index].Parent != beta->index and
918 Phase.FieldsProperties[ beta->index].Parent != alpha->index)
919 {
920 locMobility = 0.0;
921 }
922 }
923 // Modifying mobility for nuclei (if enabled)
924 if(NucleiMobilityFactor(pIndexA, pIndexB) != 1.0 and
925 (Phase.FieldsProperties[alpha->index].IsNucleus() or
926 Phase.FieldsProperties[ beta->index].IsNucleus()))
927 {
928 double volume_ratio = min(Phase.FieldsProperties[alpha->index].VolumeRatio,
929 Phase.FieldsProperties[ beta->index].VolumeRatio);
930
931 locMobility *= NucleiMobilityFactor(pIndexA, pIndexB)*(1.0 - volume_ratio) + 1.0*volume_ratio;
932 }
933
934 // Modifying mobility if growth constraints are violated
935 if(Phase.FieldsProperties[alpha->index].GrowthConstraintsViolation != GrowthConstraintsViolations::None and
936 Phase.FieldsProperties[ beta->index].GrowthConstraintsViolation != GrowthConstraintsViolations::None)
937 {
938 locMobility *= Phase.PairwiseGrowthFactors.get_sym1(alpha->index,beta->index);
939 }
940
941 // Putting interface energy and mobility values into the properties storage
942 if(locEnergy != 0.0 or locMobility != 0.0)
943 {
944 PropertiesDR(i,j,k).set_energy_and_mobility(alpha->index, beta->index, locEnergy, locMobility);
945 }
946 locMaxEnergies(pIndexA,pIndexB) = max(locMaxEnergies(pIndexA,pIndexB),locEnergy);
947 locMaxMobilities(pIndexA,pIndexB) = max(locMaxMobilities(pIndexA,pIndexB),locMobility);
948 }
949 }
950 }
951 OMP_PARALLEL_STORAGE_LOOP_END

953 maxMobilities = locMaxMobilities;
954
955#ifdef MPI_PARALLEL
956 OP_MPI_Allreduce(OP_MPI_IN_PLACE, maxEnergies.data(), Nphases*Nphases, OP_MPI_DOUBLE, OP_MPI_MAX, OP_MPI_COMM_WORLD);

1151bool InterfaceProperties::FirstOrderExtrapolation(const PhaseField& Phase)
1152{

1155 auto& IP = (Grid.Resolution != Resolutions::Dual) ? Properties : PropertiesDR;
1156
1157 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,PF,0,reduction(&&:Extrapolated))
1158 if (PF(i,j,k).wide_interface())
1159 {
1160 std::set<size_t> locIndex = localPhaseIdx(i,j,k,Phase);
1161
1162 // Extrapolate curvatures for all pairs of phase fields
1163 for (auto alpha = locIndex.begin(); alpha != locIndex.end(); alpha++)
1164 for (auto beta = locIndex.begin(); beta != locIndex.end(); beta++)
1165 if (*beta > *alpha)
1166 {
1167 if (IP(i,j,k).get_energy(*alpha, *beta) == 0.0)
1168 {

1170 int count = 0;
1171 double sumEnergies = 0.0;
1172 double locE1 = 0.0;
1173 double locE2 = 0.0;

1177 if (Grid.dNx ) {locE1 = IP(i-1,j,k) .get_energy(*alpha, *beta); locE2 = IP(i-2,j,k) .get_energy(*alpha, *beta); if (locE1 != 0.0 && locE2 != 0.0) { sumEnergies += 2*locE1 - locE2; count++; }}

1181 if (Grid.dNz ) {locE1 = IP(i,j,k-1) .get_energy(*alpha, *beta); locE2 = IP(i,j,k-2) .get_energy(*alpha, *beta); if (locE1 != 0.0 && locE2 != 0.0) { sumEnergies += 2*locE1 - locE2; count++; }}
1182 if (Grid.dNx && Grid.dNy) {locE1 = IP(i-1,j-1,k).get_energy(*alpha, *beta); locE2 = IP(i-2,j-2,k).get_energy(*alpha, *beta); if (locE1 != 0.0 && locE2 != 0.0) { sumEnergies += 2*locE1 - locE2; count++; }}
1183 if (Grid.dNx && Grid.dNy) {locE1 = IP(i+1,j-1,k).get_energy(*alpha, *beta); locE2 = IP(i+2,j-2,k).get_energy(*alpha, *beta); if (locE1 != 0.0 && locE2 != 0.0) { sumEnergies += 2*locE1 - locE2; count++; }}
1184 if (Grid.dNx && Grid.dNy) {locE1 = IP(i-1,j+1,k).get_energy(*alpha, *beta); locE2 = IP(i-2,j+2,k).get_energy(*alpha, *beta); if (locE1 != 0.0 && locE2 != 0.0) { sumEnergies += 2*locE1 - locE2; count++; }}
