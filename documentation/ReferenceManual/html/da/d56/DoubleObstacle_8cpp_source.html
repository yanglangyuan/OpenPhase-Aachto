66void DoubleObstacle::CalculatePhaseFieldIncrementsSharpSR(PhaseField& Phase,
67 InterfaceProperties& IP)
68{
69 dVector3 n_vector = {1.0, 0.0, 0.0};
70 n_vector.normalize();
71
72 double dx = Phase.Grid.dx;
73 vector<PotentialCorrections> StencilDirections;
74
75 for(auto ds = Phase.LStencil.cbegin(); ds != Phase.LStencil.cend(); ds++)
76 {
77 double d_x = ds->di;
78 double d_y = ds->dj;
79 double d_z = ds->dk;
80
81 PotentialCorrections dirNew;
82
83 dirNew.d_x = d_x;
84 dirNew.d_y = d_y;
85 dirNew.d_z = d_z;
86
87 dirNew.stencil_weight = ds->weight;
88
89 dirNew.scal_prod = (d_x*n_vector[0] + d_y*n_vector[1] + d_z*n_vector[2])*dx;
90
91 dirNew.alpha = cos(Pi*(dirNew.scal_prod)/Phase.Grid.Eta);
92 dirNew.beta = sin(Pi*(dirNew.scal_prod)/Phase.Grid.Eta);
93 dirNew.phi_tl = 0.5*cos(Pi*(1.0 - (dirNew.scal_prod)/Phase.Grid.Eta)) + 0.5;
94 dirNew.phi_tu = 0.5*cos(Pi*(dirNew.scal_prod)/Phase.Grid.Eta) + 0.5;
95
96 StencilDirections.push_back(dirNew);
97 }
98
99 const double Prefactor = Pi*Pi/(Phase.Grid.Eta*Phase.Grid.Eta);
100
101 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
102 {
103 if(Phase.Fields(i,j,k).wide_interface())
104 {
105 const NodePF& locPF = Phase.Fields(i,j,k);
106
107 double norm_1 = 1.0/Phase.LocalN(locPF);
108
109 for(auto alpha = locPF.cbegin();
110 alpha != locPF.cend(); ++alpha)
111 for(auto beta = alpha + 1;
112 beta != locPF.cend(); ++beta)
113 if(alpha != beta)
114 {
115 double dPhi_dt = 0.0;
116
117 double pot_term_a = 0.0;
118 for(auto StDir = StencilDirections.cbegin(); StDir != StencilDirections.cend(); StDir++)
119 {
120 double fik_term = 0;
121 if(StDir->scal_prod < 0 && alpha->value > StDir->phi_tu)
122 {
123 fik_term = 1.0;
124 }
125 else if(StDir->scal_prod > 0 && alpha->value < StDir->phi_tl)
126 {
127 fik_term = 0.0;
128 }
129 else
130 {
131 fik_term = 0.5 + StDir->alpha*(alpha->value - 0.5) - StDir->beta*sqrt(alpha->value*(1.0 - alpha->value));
132 }
133 pot_term_a += -StDir->stencil_weight*(fik_term - alpha->value);
134 }
135
136 double pot_term_b = 0.0;
137 for(auto StDir = StencilDirections.cbegin(); StDir != StencilDirections.cend() ; StDir++)
138 {
139 double fik_term = 0.0;
140 if(StDir->scal_prod < 0 && beta->value > StDir->phi_tu)
141 {
142 fik_term = 1.0;
143 }
144 else if(StDir->scal_prod > 0 && beta->value < StDir->phi_tl)
145 {
146 fik_term = 0.0;
147 }
148 else
149 {
150 fik_term = 0.5 + StDir->alpha*(beta->value - 0.5) - StDir->beta*sqrt(beta->value*(1.0 - beta->value));
151 }
152 pot_term_b += -StDir->stencil_weight*(fik_term - beta->value);
153 }
154
155 double scale = sqrt(Phase.FieldsProperties[alpha->index].VolumeRatio*
156 Phase.FieldsProperties[ beta->index].VolumeRatio);
157
158 dPhi_dt = IP.Properties(i,j,k).get_energy(alpha->index, beta->index)
159 *((alpha->laplacian - beta->laplacian) + 0.5*(1.0 + scale)*(pot_term_a - pot_term_b));
160
161 if(locPF.size() > 2)
162 for(auto gamma = locPF.cbegin();
163 gamma != locPF.cend(); ++gamma)
164 if((gamma != alpha) && (gamma != beta))
165 {
166 double pot_term_c = 0.0;
167 for(auto StDir = StencilDirections.cbegin(); StDir != StencilDirections.cend(); StDir++)
168 {
169 double fik_term = 0.0;
170 if(StDir->scal_prod < 0 && gamma->value > StDir->phi_tu)
171 {
172 fik_term = 1.0;
173 }
174 else if(StDir->scal_prod > 0 && gamma->value < StDir->phi_tl)
175 {
176 fik_term = 0.0;
177 }
178 else
179 {
180 fik_term = 0.5 + StDir->alpha*(gamma->value - 0.5) - StDir->beta*sqrt(gamma->value*(1.0 - gamma->value));
181 }
182 pot_term_c += -StDir->stencil_weight*(fik_term - gamma->value);
183 }
184
185 dPhi_dt += (IP.Properties(i,j,k).get_energy( beta->index, gamma->index) -
186 IP.Properties(i,j,k).get_energy(alpha->index, gamma->index))
187 *(gamma->laplacian + pot_term_c + Prefactor*0.5);
188
189 switch(IP.TripleJunctionModel)
190 {
191 case TripleJunctionModels::Model:
192 {
193 if(IP.TripleJunctionFactor != 0.0 or scale < 1.0)
194 {
195 size_t pIndexA = Phase.FieldsProperties[alpha->index].Phase;
196 size_t pIndexB = Phase.FieldsProperties[ beta->index].Phase;
197 size_t pIndexG = Phase.FieldsProperties[gamma->index].Phase;
198
199 dPhi_dt += max(IP.TripleJunctionFactor, (1.0 - scale))*Prefactor
200 *(IP.InterfaceEnergy(pIndexA, pIndexB).Energy +
201 IP.InterfaceEnergy(pIndexB, pIndexG).Energy +
202 IP.InterfaceEnergy(pIndexA, pIndexG).Energy)
203 *(alpha->value*gamma->value - beta->value*gamma->value);
204 }
205 break;
206 }
207 case TripleJunctionModels::Value:
208 {
209 dPhi_dt += IP.TripleJunctionEnergy*Prefactor
210 *(alpha->value*gamma->value - beta->value*gamma->value);
211 break;
212 }
213 case TripleJunctionModels::None:
214 {
215 break;
216 }
217 }
218 }
219 dPhi_dt *= IP.Properties(i,j,k).get_mobility(alpha->index, beta->index)*norm_1*scale;
220 Phase.FieldsDot(i,j,k).add_asym1(alpha->index, beta->index, dPhi_dt);
221 }
222 }
223 }
224 OMP_PARALLEL_STORAGE_LOOP_END

387void DoubleObstacle::CalculateCurvatureDrivingForce(PhaseField& Phase,
388 InterfaceProperties& IP,
389 DrivingForce& dG)
390{
391 const double Prefactor = Pi*Pi/(Phase.Grid.Eta*Phase.Grid.Eta);
392 const double Prefactor2 = Phase.Grid.Eta/(2.0*Pi);
393
394 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
395 {
396 if (Phase.Fields(i,j,k).interface())
397 {
398 const NodePF& locPF = Phase.Fields(i,j,k);
399
400 for(auto alpha = locPF.cbegin();
401 alpha != locPF.cend(); ++alpha)
402 for(auto beta = alpha + 1;
403 beta != locPF.cend(); ++beta)
404 if(alpha != beta and alpha->value*beta->value > DBL_EPSILON)
405 {
406 double scale = sqrt(Phase.FieldsProperties[alpha->index].VolumeRatio *
407 Phase.FieldsProperties[ beta->index].VolumeRatio);
408
409 double loc_dG = IP.Properties(i,j,k).get_energy(alpha->index,beta->index)
410 *((alpha->laplacian + 0.5*(1.0 + scale)*Prefactor*alpha->value) -
411 ( beta->laplacian + 0.5*(1.0 + scale)*Prefactor* beta->value));
412
413 if(locPF.size() > 2)
414 for(auto gamma = locPF.cbegin();
415 gamma != locPF.cend(); ++gamma)
416 if((gamma != alpha) && (gamma != beta))
417 {
418 loc_dG += (IP.Properties(i,j,k).get_energy( beta->index, gamma->index) -
419 IP.Properties(i,j,k).get_energy(alpha->index, gamma->index))
420 *(gamma->laplacian + 0.5*(1.0 + scale)*Prefactor*gamma->value);
421
422 switch(IP.TripleJunctionModel)
423 {
424 case TripleJunctionModels::Model:
425 {
426 if(IP.TripleJunctionFactor != 0.0 or scale < 1.0)
427 {
428 size_t pIndexA = Phase.FieldsProperties[alpha->index].Phase;
429 size_t pIndexB = Phase.FieldsProperties[ beta->index].Phase;
430 size_t pIndexG = Phase.FieldsProperties[gamma->index].Phase;
431
432 loc_dG += max(IP.TripleJunctionFactor, (1.0 - scale))*Prefactor
433 *(IP.InterfaceEnergy(pIndexA, pIndexB).Energy +
434 IP.InterfaceEnergy(pIndexB, pIndexG).Energy +
435 IP.InterfaceEnergy(pIndexA, pIndexG).Energy)
436 *(alpha->value*gamma->value - beta->value*gamma->value);
437 }
438 break;
439 }
440 case TripleJunctionModels::Value:
441 {
442 loc_dG += IP.TripleJunctionEnergy*Prefactor
443 *(alpha->value*gamma->value - beta->value*gamma->value);
444 break;
445 }
446 case TripleJunctionModels::None:
447 {
448 break;
449 }
450 }
451 }
452 loc_dG *= Prefactor2 * scale / sqrt(alpha->value*beta->value);
453 loc_dG *= IP.CurvatureFactor;
454 dG.Force(i,j,k).add_raw(alpha->index, beta->index, loc_dG);
455 }
456 }
457 }
458 OMP_PARALLEL_STORAGE_LOOP_END
459}

687void DoubleObstacle::CalculatePhaseFieldIncrementsSR(PhaseField& Phase,
688 InterfaceProperties& IP,
689 InterfaceRegularization& Kappa)
690{
691 const double Prefactor = Pi*Pi/(Phase.Grid.Eta*Phase.Grid.Eta);
692 const double Prefactor2 = Phase.Grid.Eta/Pi;
693
694 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
695 {
696 if(Phase.Fields(i,j,k).wide_interface())
697 {
698 const NodePF& locPF = Phase.Fields(i,j,k);
699 double norm_1 = 1.0/Phase.LocalN(locPF);
700
701 for(auto alpha = locPF.cbegin();
702 alpha != locPF.cend(); ++alpha)
703 for(auto beta = alpha + 1;
704 beta != locPF.cend(); ++beta)
705 if(alpha != beta)
706 {
707 double scale = sqrt(Phase.FieldsProperties[alpha->index].VolumeRatio *
708 Phase.FieldsProperties[ beta->index].VolumeRatio);
709
710 double loc_increment = IP.Properties(i,j,k).get_energy(alpha->index,beta->index)
711 *((alpha->laplacian + 0.5*(1.0 + scale)*Prefactor*alpha->value) -
712 ( beta->laplacian + 0.5*(1.0 + scale)*Prefactor* beta->value));
713
714 if(locPF.size() > 2)
715 for(auto gamma = locPF.cbegin();
716 gamma != locPF.cend(); ++gamma)
717 if((gamma != alpha) and (gamma != beta))
718 {
719 loc_increment += (IP.Properties(i,j,k).get_energy( beta->index, gamma->index) -
720 IP.Properties(i,j,k).get_energy(alpha->index, gamma->index))*
721 (gamma->laplacian + 0.5*(1.0 + scale)*Prefactor*gamma->value);
722
723 switch(IP.TripleJunctionModel)
724 {
725 case TripleJunctionModels::Model:
726 {
727 if(IP.TripleJunctionFactor != 0.0 or scale < 1.0)
728 {
729 size_t pIndexA = Phase.FieldsProperties[alpha->index].Phase;
730 size_t pIndexB = Phase.FieldsProperties[ beta->index].Phase;
731 size_t pIndexG = Phase.FieldsProperties[gamma->index].Phase;
732
733 loc_increment += max(IP.TripleJunctionFactor, (1.0 - scale))*Prefactor
734 *(IP.InterfaceEnergy(pIndexA, pIndexB).Energy +
735 IP.InterfaceEnergy(pIndexB, pIndexG).Energy +
736 IP.InterfaceEnergy(pIndexA, pIndexG).Energy)
737 *(alpha->value*gamma->value - beta->value*gamma->value);
738 }
739 break;
740 }
741 case TripleJunctionModels::Value:
742 {
743 loc_increment += IP.TripleJunctionEnergy*Prefactor
744 *(alpha->value*gamma->value - beta->value*gamma->value);
745 break;
746 }
747 case TripleJunctionModels::None:
748 {
749 break;
750 }
751 }
752 }
753 double dPhi_dt = loc_increment * IP.Properties(i,j,k).get_mobility(alpha->index, beta->index) * norm_1 * scale;
754 if(IP.RegularizationFactor != 1.0 or IP.CurvatureFactor != 1.0)
755 {
756 dPhi_dt *= IP.RegularizationFactor;
757
758 if(alpha->value != 0.0 and beta->value != 0.0)
759 {
760 double loc_curvature = loc_increment * Prefactor2 * scale * norm_1 / sqrt(alpha->value*beta->value);
761 loc_curvature *= IP.CurvatureFactor - IP.RegularizationFactor;
762 Kappa.Curvature(i,j,k).add_raw(alpha->index, beta->index, loc_curvature);
763 }
764 }
765 Phase.FieldsDot(i,j,k).add_asym1(alpha->index, beta->index, dPhi_dt);
766 }
767 }
768 }
769 OMP_PARALLEL_STORAGE_LOOP_END
770}

831 }

833 {
834 break;
835 }
836 }
837 }

839 if(IP.RegularizationFactor != 1.0 or IP.CurvatureFactor != 1.0)
840 {
841 dPhi_dt *= IP.RegularizationFactor;

857void DoubleObstacle::StabilizeThinChannels(PhaseField& Phase, InterfaceProperties& IP, double penalty)
858{
859 const double Prefactor = Pi*Pi/(Phase.Grid.Eta*Phase.Grid.Eta);
860 //const double Prefactor2 = Phase.Eta/(2.0*Pi);
861
862 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
863 {
864 if(Phase.Fields(i,j,k).interface_halo())
865 {
866 std::vector<NodeAB<dVector3,dVector3>> allNormals;
867 std::vector<iVector3> directions;
868 // Collect all normals and directions on both sides of the thin channel
869 for(auto it = Phase.LStencil.begin(); it != Phase.LStencil.end(); it++)
870 if(Phase.Fields(i + it->di,j + it->dj,k + it->dk).interface())
871 {
872 allNormals.push_back(Phase.Normals(i + it->di,j + it->dj,k + it->dk));
873 directions.push_back((iVector3){it->di,it->dj,it->dk});
874 }
875
876 std::vector<NodeAB<double,double>> allProducts;
877
878 // Find opposite directions and normals products
879 for(size_t n = 0; n < directions.size() - 1; n++)
880 for(size_t m = n+1; m < directions.size(); m++)
881 if(directions[n] == directions[m]*(-1))
882 {
883 NodeAB<double,double> locProduct;
884 for(auto nm = allNormals[n].begin(); nm != allNormals[n].end(); nm++)
885 if(allNormals[m].present(nm->indexA, nm->indexB))
886 {
887 locProduct.set_asym1(nm->indexA, nm->indexB, nm->value1*allNormals[m].get_asym1(nm->indexA, nm->indexB));
888 }
889 allProducts.push_back(locProduct);
890 }
891
892 // Analyze normal products and detect opposite normal directions
893 for(size_t p = 0; p < allProducts.size(); p++)
894 for(auto it = allProducts[p].begin(); it != allProducts[p].end(); it++)
895 if(it->value1 < 0.0)
896 {
897 Phase.FieldsDot(i,j,k).add_asym1(it->indexA, it->indexB, it->value1*penalty*Prefactor);
898 }
899 }
900 }
901 OMP_PARALLEL_STORAGE_LOOP_END
902}

912 {

914 alpha != Phase.Fields(i,j,k).cend(); ++alpha)

917 if(alpha != beta)

921 }
922 }

1034void DoubleObstacle::CalculateFullAnisotropy(const PhaseField& Phase,
1035 InterfaceProperties& IP,
1036 DrivingForce& dG)
1037{
1038 IP.InterfaceStiffnessTMP.Clear();
1039
1040 const double Prefactor1 = 4.0/Phase.Grid.Eta;
1041 const double Prefactor2 = Phase.Grid.Eta*Phase.Grid.Eta/Pi/Pi;
1042 const double Prefactor3 = - 4.0*Phase.Grid.Eta*Phase.Grid.Eta/Pi/Pi;
1043
1044 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
1045 if(Phase.Fields(i,j,k).interface())
1046 {
1047 for(auto alpha = Phase.Fields(i,j,k).cbegin();
1048 alpha != Phase.Fields(i,j,k).cend(); ++alpha)
1049 for(auto beta = alpha + 1;
1050 beta != Phase.Fields(i,j,k).cend(); ++beta)
1051 if(alpha != beta)
1052 {
1053 const size_t pIndexA = Phase.FieldsProperties[alpha->index].Phase;
1054 const size_t pIndexB = Phase.FieldsProperties[ beta->index].Phase;
1055
1056 if(IP.InterfaceEnergy(pIndexA, pIndexB).Model != InterfaceEnergyModels::Iso)
1057 if(IP.InterfaceEnergy(pIndexA, pIndexB).Type == InterfaceEnergyModelType::Energy)
1058 {
1059 const double I_AB = Prefactor1*(alpha->value*beta->value - Prefactor2*(alpha->gradient*beta->gradient));
1060 const dVector3 value = (IP.dEnergy_dGradientAlpha(Phase, alpha, beta) - IP.dEnergy_dGradientAlpha(Phase, beta, alpha))*(-I_AB);
1061
1062 if (value.abs() > 0.0)
1063 {
1064 IP.InterfaceStiffnessTMP(i,j,k).add_asym1(alpha->index, beta->index, value);
1065 }
1066
1067 for(auto gs = Phase.GStencil.cbegin(); gs != Phase.GStencil.cend(); ++gs)
1068 {
1069 const double sigma = IP.Properties(i+gs->di, j+gs->dj, k+gs->dk).get_energy(alpha->index, beta->index);
1070 const double dsigma_dx = gs->weightX * sigma;
1071 const double dsigma_dy = gs->weightY * sigma;
1072 const double dsigma_dz = gs->weightZ * sigma;
1073 const dVector3 sigma_gradient ={dsigma_dx,dsigma_dy,dsigma_dz};
1074 const double value = Prefactor3*(sigma_gradient*(alpha->gradient - beta->gradient));
1075 dG.Force(i,j,k).add_raw(alpha->index, beta->index, -value);
1076 }
1077 }
1078 }
1079 }
1080 OMP_PARALLEL_STORAGE_LOOP_END
1081
1082 const double DWeights[3] = {-0.5/Phase.Grid.dx, 0.0, 0.5/Phase.Grid.dx};
1083 if (Phase.Grid.dNx)
1084 {
1085 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,IP.InterfaceStiffnessTMP,0,)
1086 {
1087 if(Phase.Fields(i,j,k).interface())
1088 for (int ii = -1; ii <= +1; ii+=2)
1089 for (auto it = IP.InterfaceStiffnessTMP(i+ii,j,k).cbegin();
1090 it < IP.InterfaceStiffnessTMP(i+ii,j,k).cend(); ++it)
1091 {
1092 dG.Force(i,j,k).add_raw(it->indexA, it->indexB,-DWeights[ii+1]*it->value1[0]);
1093 }
1094 }
1095 OMP_PARALLEL_STORAGE_LOOP_END
1096 }
1097
1098 if (Phase.Grid.dNy)
1099 {
1100 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,IP.InterfaceStiffnessTMP,0,)
1101 {
1102 if(Phase.Fields(i,j,k).interface())
1103 for (int ii = -1; ii <= +1; ii+=2)
1104 for (auto it = IP.InterfaceStiffnessTMP(i,j+ii,k).cbegin();
1105 it < IP.InterfaceStiffnessTMP(i,j+ii,k).cend(); ++it)
1106 {
1107 dG.Force(i,j,k).add_raw(it->indexA, it->indexB,-DWeights[ii+1]*it->value1[1]);
1108 }
1109 }
1110 OMP_PARALLEL_STORAGE_LOOP_END
1111 }
1112
1113 if (Phase.Grid.dNz)
1114 {
1115 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,IP.InterfaceStiffnessTMP,0,)
1116 {
1117 if(Phase.Fields(i,j,k).interface())
1118 for (int ii = -1; ii <= +1; ii+=2)
1119 for (auto it = IP.InterfaceStiffnessTMP(i,j,k+ii).cbegin();
1120 it < IP.InterfaceStiffnessTMP(i,j,k+ii).cend(); ++it)
1121 {
1122 dG.Force(i,j,k).add_raw(it->indexA, it->indexB,-DWeights[ii+1]*it->value1[2]);
1123 }
1124 }
1125 OMP_PARALLEL_STORAGE_LOOP_END
1126 }
1127}
