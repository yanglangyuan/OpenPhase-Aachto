55void Noise::Initialize(Settings& locSettings, std::string ObjectNameSuffix)
56{
57 thisclassname = "Noise";
58 thisobjectname = thisclassname + ObjectNameSuffix;
59
60 Grid = locSettings.Grid;
61
62 Nphases = locSettings.Nphases;
63
64 // Allocate three-dimensional Fourier space noise
65 RandomFourier = new complex<double>[Grid.LocalNumberOfCells()]();
66 RandomReal = new complex<double>[Grid.LocalNumberOfCells()]();
67
68 // Allocate raw Noise Storage
69 Raw.Allocate(Grid, 0);
70
71 // Assign storages to fftw pointers
72 fftw_In = reinterpret_cast<fftw_complex*>(RandomFourier);
73 fftw_Out = reinterpret_cast<fftw_complex*>(RandomReal);
74
75 // Create FFT-Plan, which determines how the FFT will be executed
76#ifdef _OPENMP
77 fftw_init_threads();
78 fftw_plan_with_nthreads(omp_get_max_threads());
79#endif
80 FFTBackward = fftw_plan_dft_3d(Grid.Nx, Grid.Ny, Grid.Nz, fftw_In,fftw_Out,FFTW_BACKWARD,FFTW_ESTIMATE);
81
82 // Initialize random number distribution
83 distribution = normal_distribution<double>(0.0,0.5);
84
85 size_t seed = std::chrono::system_clock::now().time_since_epoch().count();
86 generator = std::default_random_engine(seed);
87
88 initialized = true;
89 ConsoleOutput::WriteStandard(thisclassname, "Initialized");
90}

111void Noise::ReadInput(std::stringstream& inp)
112{
113 // Read Parameters
114 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
115
116 gamma = FileInterface::ReadParameterD(inp, moduleLocation, string("dgamma"),false,0);
117 TimeSteps = FileInterface::ReadParameterI(inp, moduleLocation, string("iTime"));
118 CutOffWaveLength = FileInterface::ReadParameterI(inp, moduleLocation, string("dWaveLength"));
119 FixAmpl = FileInterface::ReadParameterI(inp, moduleLocation, string("dFixAmpl"));
120
121 // Check if the input TimeSteps makes sense
122 if (TimeSteps < 1)
123 {
124 std::string message = "A input value TimeSteps < 1 does not make sense! "
125 "Set TimeSteps to 1!";
126 ConsoleOutput::WriteWarning( message, thisclassname, "ReadInput");
127 TimeSteps = 0;
128 }
129
130 // Check if the input CutOffWaveLength makes sense
131 if (CutOffWaveLength < 0)
132 {
133 std::string message = "A input value CutOffWaveLength < 0 does not make sense! "
134 "Set CutOffWaveLength to 0!";
135 ConsoleOutput::WriteWarning( message, thisclassname, "ReadInput");
136 CutOffWaveLength = 0;
137 }
138 ConsoleOutput::WriteLine();
139 ConsoleOutput::WriteBlankLine();
140}

142void Noise::Generate(void)
143{
144 double Lx = 2.0 * Pi / (double(Grid.Nx) * Grid.dx);
145 double Ly = 2.0 * Pi / (double(Grid.Ny) * Grid.dx);
146 double Lz = 2.0 * Pi / (double(Grid.Nz) * Grid.dx);
147
148 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Raw,0,)
149 {
150 dVector3 WaveVector;
151 WaveVector[0] = Lx*(i*(i <= Grid.Nx/2) - (Grid.Nx-i)*(i > Grid.Nx/2));
152 WaveVector[1] = Ly*(j*(j <= Grid.Ny/2) - (Grid.Ny-j)*(j > Grid.Ny/2));
153 WaveVector[2] = Lz*(k*(k <= Grid.Nz/2) - (Grid.Nz-k)*(k > Grid.Nz/2));
154
155 if(WaveVector.abs() < 2.0 * Pi/CutOffWaveLength)
156 {
157 // Generate Gaussian white noise coefficients
158 complex<double> cijk{distribution(generator),distribution(generator)};
159 //cijk /= abs(cijk);
160
161 // Store computed coefficient
162 RandomFourier[k + Grid.Nz*(j + Grid.Ny*i)] = cijk;
163 }
164 else
165 {
166 RandomFourier[k + Grid.Nz*(j + Grid.Ny*i)] = complex<double>(0.0,0.0);
167 }
168 }
169 OMP_PARALLEL_STORAGE_LOOP_END
170 // Do Fourier transform
171 fftw_execute(FFTBackward);
172
173 // Calculate global maximum of RandomReal
174 double Max = 0.0;
175
176 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Raw,0,reduction(max:Max))
177 {
178 Max = max(Max, abs(real(RandomReal[k + Grid.Nz*(j + Grid.Ny*i)])));
179 }
180 OMP_PARALLEL_STORAGE_LOOP_END
181
182 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Raw,0,)
183 {
184 RandomReal[k + Grid.Nz*(j + Grid.Ny*i)] /= Max;
185 RandomReal[k + Grid.Nz*(j + Grid.Ny*i)] -= Raw(i,j,k);
186 RandomReal[k + Grid.Nz*(j + Grid.Ny*i)] /= TimeSteps;
187 }
188 OMP_PARALLEL_STORAGE_LOOP_END
189}

191void Noise::UpdateRaw(int tStep)
192{
193 if (tStep == 0)
194 {
195 // Generate initial noise distribution
196 Generate();
197
198 // Save initial noise distribution and add it to the driving force
199
200 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Raw,0,)
201 {
202 // Set initial noise
203 Raw(i,j,k) = TimeSteps * real(RandomReal[k + Grid.Nz*(j + Grid.Ny*i)]);
204 }
205 OMP_PARALLEL_STORAGE_LOOP_END
206 // Generate new noise distribution in order to able to interpolate
207 // between the old distribution and the new one.
208 Generate();
209 }
210 else
211 {
212 // First check if a new noise Distribution has to be generated
213 if (!(tStep%TimeSteps)) Generate();
214
215 // Interpolate between old and new noise distribution by adding RandomReal
216 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Raw,0,)
217 {
218 // Compute noise for this time step
219 Raw(i,j,k) += real(RandomReal[k + Grid.Nz*(j + Grid.Ny*i)]);
220 }
221 OMP_PARALLEL_STORAGE_LOOP_END
222 }
223
224 // Calculate the global maximum of Raw
225 double Max = 0.0;
226
227 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Raw,0,reduction(max:Max))
228 {
229 Max = max(Max, abs(Raw(i,j,k)));
230 }
231 OMP_PARALLEL_STORAGE_LOOP_END
232
233 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Raw,0,)
234 {
235 Raw(i,j,k) /= Max;
236 }
237 OMP_PARALLEL_STORAGE_LOOP_END
238}

240void Noise::AddToDrivingForce(int tStep, Temperature& Temp, DrivingForce& DF)
241{
242 UpdateRaw(tStep);
243
244 // Interpolate between old and new noise distribution by adding RandomReal
245 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Raw,0,)
246 {
247 // Add noise to driving force
248 for (auto it = DF.Force(i,j,k).cbegin(); it < DF.Force(i,j,k).cend(); ++it)
249 {
250 double Ampl = sqrt( 2 * PhysicalConstants::k_Boltzmann * Temp(i,j,k)/gamma);
251 DF.Force(i,j,k).add_raw(it->indexA, it->indexB, Ampl * Raw(i,j,k));
252 }
253 }
254 OMP_PARALLEL_STORAGE_LOOP_END
255}

257void Noise::AddToDrivingForce(int tStep, DrivingForce& DF)
258{
259 UpdateRaw(tStep);
260
261 // Interpolate between old and new noise distribution by adding RandomReal
262 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Raw,0,)

264 // Add noise to driving force
265 for (auto it = DF.Force(i,j,k).cbegin(); it < DF.Force(i,j,k).cend(); ++it)

267 DF.Force(i,j,k).add_raw(it->indexA, it->indexB, FixAmpl * Raw(i,j,k));

270 OMP_PARALLEL_STORAGE_LOOP_END
