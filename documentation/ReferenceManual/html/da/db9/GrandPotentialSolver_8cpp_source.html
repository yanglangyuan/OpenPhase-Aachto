420void GrandPotentialSolver::SetInitialConcentration(const PhaseField& Phase, const GrandPotentialDensity& omega)
421{
422 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,ChemicalPotential,0,)
423 {
424 for(size_t comp = 0; comp < Ncomp; comp++)
425 {
426 // Calculate initial concentration
427 double FinalConcentration = 0;
428 for(auto alpha = Phase.Fields(i,j,k).cbegin(); alpha != Phase.Fields(i,j,k).cend(); ++alpha)
429 {
430 size_t PhaseIdx = Phase.FieldsProperties[alpha->index].Phase;
431 FinalConcentration += alpha->value*CInitial({PhaseIdx,comp});
432 }
433
434 ChemicalPotential(i,j,k,{comp}) = 0.0;
435
436 // Because an initial chemical potential is needed as input
437 // a chemical potential which maps into the desired initial
438 // concentration is calculated using newtons method
439 double precision = DBL_EPSILON;
440 double f,df, dmu = 0;
441 int n = 0;
442 do
443 {
444 CalculateLocalConcentrations(i,j,k,Phase,omega);
445 f = Concentrations(i,j,k,{comp}) - FinalConcentration;
446 df = Susceptibility(i,j,k,comp,Phase,omega);
447 if (df == 0.0) break;
448 dmu = -f/df/2;
449 ChemicalPotential(i,j,k,{comp})+=dmu;
450 n++;
451 }
452 while (std::abs(dmu) != 0.0 and std::abs(dmu) > std::abs(ChemicalPotential(i,j,k,{comp}))*precision and n < 1000);
453 }
454 CalculateLocalConcentrations (i,j,k,Phase,omega);
455 }
456 OMP_PARALLEL_STORAGE_LOOP_END
457}
458void GrandPotentialSolver::SetInitialPressure(const PhaseField& Phase, const GrandPotentialDensity& omega)
459{
460 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,ChemicalPotential,0,)
461 {
462 // Calculate initial concentration
463 ChemicalPotential(i,j,k,{0}) = InitialChemicalPotential[0];
464
465 // Because an initial chemical potential is needed as input
466 // a chemical potential which maps into the desired initial
467 // concentration is calculated using newtons method
468 double precision = 0.1;
469 auto PressureDelta = [this, &Phase, &omega, i, j, k](double mu)
470 {
471 //TODO mu should not be unused!!!!
472 return (-omega(i,j,k,Phase)) - InitialPressure;
473 };
474 auto dPressureDelta_dChemicalPotential = [ &Phase, &omega, i, j, k]([[maybe_unused]] double mu)
475 {
476 double tmp = 0;
477 for(auto alpha = Phase.Fields(i,j,k).cbegin(); alpha != Phase.Fields(i,j,k).cend(); ++alpha)
478 {
479 size_t PhaseIdx = Phase.FieldsProperties[alpha->index].Phase;
480 tmp -= alpha->value*omega(PhaseIdx).dChemicalPotential(i,j,k,0);
481 }
482 return tmp;
483 };
484
485 try
486 {
487 RootFindingAlgorithms::Newton(PressureDelta, dPressureDelta_dChemicalPotential, ChemicalPotential(i,j,k,{0}), precision, 1000);
488 }
489 catch (std::runtime_error& ecep)
490 {
491 ConsoleOutput::WriteWarning(ecep.what(),thisclassname,"SetInitialPressure");
492 }
493 if (std::abs(-omega(i,j,k,Phase)-InitialPressure) > 1.0 )
494 {
495 std::cerr << i << " " << j << " " << k << " " << (-omega(i,j,k,Phase)) << "\n";
496 OP_Exit(EXIT_FAILURE);
497 }
498 }
499 OMP_PARALLEL_STORAGE_LOOP_END
500 ConsoleOutput::Write("Set Pressure sucessufully!");
501}

581void GrandPotentialSolver::EnforceConservationOfTOC(const PhaseField& Phase, const GrandPotentialDensity& omega)
582{
583 for (std::size_t comp = 0; comp < Ncomp; comp++)
584 {
585 auto residual = [&] (double& delta)
586 {
587 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0)
588 {
589 ChemicalPotential(i,j,k,{comp}) += delta;
590 CalculateLocalConcentrations(i,j,k,comp,Phase,omega);
591 }
592 OMP_PARALLEL_STORAGE_LOOP_END
593
594 double r = (TotalAmountOfComponent(comp) - TOC0[comp])/TOC0[comp];
595
596 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,comp)
597 {
598 ChemicalPotential(i,j,k,{comp}) -= delta;
599 }
600 OMP_PARALLEL_STORAGE_LOOP_END
601
602 return r;
603 };
604
605 double delta0 = 0;
606 double delta1 = 0.1;
607 try
608 {
609 RootFindingAlgorithms::Secant(residual, delta0, delta1, TOCAccuracy, TOCMaxIterations);
610 }
611 catch (std::runtime_error& ecep)
612 {
613 ConsoleOutput::WriteWarning(ecep.what(),thisclassname,"EnforceConservation");
614 }
615
616 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0)
617 {
618 ChemicalPotential(i,j,k,{comp}) += delta0;
619 }
620 OMP_PARALLEL_STORAGE_LOOP_END
621 }

623void GrandPotentialSolver::CaclulateLocalDiffusionFlux(long i, long j, long k, dVector3 locVelocity, const PhaseField& Phase, const Temperature& Temp)
624{

626 {
627 // Calculate chemical potential gradient
628 DiffusionFlux(i,j,k,{comp}) = {0.0,0.0,0.0};
629 dVector3 locForce = {0.0,0.0,0.0};
630 if (Grid.dNx) locForce[0] -= (ChemicalPotential(i+1,j ,k ,{comp}) - ChemicalPotential(i-1,j ,k ,{comp}))/Grid.dx/2.0;
631 if (Grid.dNy) locForce[1] -= (ChemicalPotential(i ,j+1,k ,{comp}) - ChemicalPotential(i ,j-1,k ,{comp}))/Grid.dx/2.0;
632 if (Grid.dNz) locForce[2] -= (ChemicalPotential(i ,j ,k +1,{comp}) - ChemicalPotential(i ,j ,k-1,{comp}))/Grid.dx/2.0;
633
634 DiffusionFlux(i,j,k,{comp}) += locForce*Mobility(i,j,k,comp,Phase,Temp);

696void GrandPotentialSolver::CalculateLocalIncrements(long i, long j, long k, const PhaseField& Phase, const GrandPotentialDensity& omega, double dt)
697{
698 for (size_t comp = 0; comp < Ncomp; comp++)
699 {
700 // Calculate change of concentration due to diffusion
701 double ConcentrationDot = ConcentrationsDot(i,j,k,{comp});
702 if (Grid.dNx) ConcentrationDot -= (DiffusionFlux(i+1,j ,k ,{comp})[0]-DiffusionFlux(i-1,j ,k ,{comp})[0])/Grid.dx/2.0;
703 if (Grid.dNy) ConcentrationDot -= (DiffusionFlux(i ,j+1,k ,{comp})[1]-DiffusionFlux(i ,j-1,k ,{comp})[1])/Grid.dx/2.0;
704 if (Grid.dNz) ConcentrationDot -= (DiffusionFlux(i ,j ,k +1,{comp})[2]-DiffusionFlux(i ,j ,k -1,{comp})[2])/Grid.dx/2.0;
705
706 double locInverseSusceptibility = 1/Susceptibility(i,j,k,comp,Phase,omega);
707 ChemicalPotentialDot(i,j,k,{comp}) += ConcentrationDot*locInverseSusceptibility;
708
709 // Calculate change of concentration due to phase-transformation
710 NodeA locPhaseDot = Phase.Dot(i,j,k, dt);
711 for (auto alpha = Phase.Fields(i,j,k).cbegin();
712 alpha != Phase.Fields(i,j,k).cend(); alpha++)
713 {
714 size_t PhaseIdx = Phase.FieldsProperties[alpha->index].Phase;
715 double locPhaseConcentration = -omega(PhaseIdx).dChemicalPotential(i,j,k,comp);
716 ChemicalPotentialDot(i,j,k,{comp}) -= locPhaseConcentration*locPhaseDot.get_value(alpha->index)*locInverseSusceptibility;
717 }
718 ConcentrationsDot(i,j,k,{comp}) = 0.0;
719 }

721void GrandPotentialSolver::CalculateLocalIncrements1(long i, long j, long k, const PhaseField& Phase, const GrandPotentialDensity& omega, double dt)
722{
723 for (size_t comp = 0; comp < Ncomp; comp++)
724 {
725 // Calculate change of concentration due to diffusion
726 double ConcentrationDot = 0.0;
727 if (Grid.dNx) ConcentrationDot -= (DiffusionFlux(i+1,j ,k ,{comp})[0]-DiffusionFlux(i-1,j ,k ,{comp})[0])/Grid.dx/2.0;
728 if (Grid.dNy) ConcentrationDot -= (DiffusionFlux(i ,j+1,k ,{comp})[1]-DiffusionFlux(i ,j-1,k ,{comp})[1])/Grid.dx/2.0;
729 if (Grid.dNz) ConcentrationDot -= (DiffusionFlux(i ,j ,k +1,{comp})[2]-DiffusionFlux(i ,j ,k -1,{comp})[2])/Grid.dx/2.0;
730
731 double locInverseSusceptibility = 1/Susceptibility(i,j,k,comp,Phase,omega);
732 ChemicalPotentialDot(i,j,k,{comp}) += ConcentrationDot*locInverseSusceptibility;
733
734 // Calculate change of concentration due to phase-transformation due to curvature
735 for (auto alpha = Phase.Fields(i,j,k).cbegin();
736 alpha != Phase.Fields(i,j,k).cend(); alpha++)
737 {
738 size_t PhaseIdx = Phase.FieldsProperties[alpha->index].Phase;
739 double locPhaseConcentration = -omega(PhaseIdx).dChemicalPotential(i,j,k,comp);
740 double locPhaseDot1 = Phase.Dot1(i,j,k, dt).get_value(alpha->index);
741 ChemicalPotentialDot(i,j,k,{comp}) -= locPhaseConcentration*locPhaseDot1*locInverseSusceptibility;
742 }
743 }

745void GrandPotentialSolver::CalculateLocalIncrements2(long i, long j, long k, const PhaseField& Phase, const GrandPotentialDensity& omega, double dt)
746{
747 for (size_t comp = 0; comp < Ncomp; comp++)
748 {
749 double locInverseSusceptibility = 1/Susceptibility(i,j,k,comp,Phase,omega);
750 // Calculate change of concentration due to phase-transformation due to pressure difference
751 for (auto alpha = Phase.Fields(i,j,k).cbegin(); alpha != Phase.Fields(i,j,k).cend(); alpha++)
752 {
753 size_t PhaseIdx = Phase.FieldsProperties[alpha->index].Phase;
754 double locPhaseConcentration = -omega(PhaseIdx).dChemicalPotential(i,j,k,comp);
755 double locPhaseDot2 = Phase.Dot2(i,j,k,dt).get_value(alpha->index);
756 ChemicalPotentialDot2(i,j,k,{comp}) -= locPhaseConcentration*locPhaseDot2*locInverseSusceptibility;
757 }

760void GrandPotentialSolver::CalculateLocalPhaseFieldIncrements (long i, long j, long k, PhaseField& Phase, const GrandPotentialDensity& omega, const InterfaceProperties& IP)
761{
762 for (auto it = Phase.FieldsDot(i,j,k).begin(); it != Phase.FieldsDot(i,j,k).end(); ++it)
763 {
764 Phase.FieldsDot(i,j,k).set_sym2(it->indexA,it->indexB,0.0);
765 }
766
767 double Prefactor = Pi*Pi/(8.0*Phase.Grid.Eta*Phase.LocalN(Phase.Fields(i, j, k)));
768
769 for(auto alpha = Phase.Fields(i, j, k).cbegin();
770 alpha != Phase.Fields(i, j, k).cend(); ++alpha)
771 {
772 size_t AlphaIdx = Phase.FieldsProperties[alpha->index].Phase;
773 double locOmegaAlpha = omega(AlphaIdx)(i,j,k);
774 for(auto beta = alpha + 1;
775 beta != Phase.Fields(i, j, k).cend(); ++beta)
776 {
777 size_t BetaIdx = Phase.FieldsProperties[beta->index].Phase;
778 if (AlphaIdx != BetaIdx)
779 {
780 double locOmegaBeta = omega(BetaIdx)(i,j,k);
781 double locOmegaDelta = locOmegaBeta - locOmegaAlpha ;
782 double loc_dPhi_dt = locOmegaDelta*IP.Properties(i,j,k).get_mobility(alpha->index, beta->index)*Prefactor;
783 Phase.FieldsDot(i,j,k).add_asym2(alpha->index, beta->index, loc_dPhi_dt);
784 }
785 }
786 }

814void GrandPotentialSolver::MergeLocalIncrements2Implicit(long i, long j, long k, PhaseField& Phase, const GrandPotentialDensity& omega, const InterfaceProperties& IP, const double dt)
815{
816 if (Ncomp == 1)
817 {
818 // This method uses the secant method (Newton's method with numeric calculation of the gradient) to solve the semi implicit time-step
819 double ChemicalPotential_Curvature = ChemicalPotential(i,j,k,{0});
820 auto residual = [this,i,j,k,&Phase,&omega,&IP,dt,ChemicalPotential_Curvature](double mu)
821 {
822 double mu0 = ChemicalPotential(i,j,k,{0});
823 ChemicalPotential(i,j,k,{0}) = mu;
824 ChemicalPotentialDot2(i,j,k,{0}) = 0.0;
825
826 CalculateLocalConcentrations (i,j,k,Phase,omega);
827 CalculateLocalPhaseFieldIncrements (i,j,k,Phase,omega,IP);
828 CalculateLocalIncrements2 (i,j,k,Phase,omega,dt);
829
830 double res = ChemicalPotential(i,j,k,{0}) - ChemicalPotential_Curvature - ChemicalPotentialDot2(i,j,k,{0})*dt;
831 ChemicalPotential(i,j,k,{0}) = mu0;
832 return res;
833 };
834
835 // Use chemical potential of the previous time step as starting point for the root fining algorithm
836 ChemicalPotential(i,j,k,{0}) = ChemicalPotentialOld(i,j,k,{0});
837
838 try
839 {
840 double& x0 = ChemicalPotential(i,j,k,{0});
841 double x1 = ChemicalPotential(i,j,k,{0})*1.01;
842 RootFindingAlgorithms::Secant(residual, x0, x1, ChemicalPotentialAccuracy, MaxIterations);
843 //int error = RootFindingAlgorithms::Broyden(residual, ChemicalPotential(i,j,k)({0}), ChemicalPotentialAccuracy, MaxIterations);
844 }
845 catch (std::runtime_error& ecep)
846 {
847 ConsoleOutput::WriteWarning(ecep.what(),thisclassname,"MergeLocalIncrements2Implicit");
848 }
849 }
850 else
851 {
852 // This method uses the Broyden's method to solve the semi implicit
853 // time-step (Newton's method for multiple dimensions where the
854 // inverse Jacobian is calculated numerically only for the first iteration).
855 typedef std::vector<double> vector;
856 vector ChemicalPotential_Curvature(Ncomp);
857 for (std::size_t comp = 0; comp < Ncomp; comp++)
858 {
859 ChemicalPotential_Curvature[comp] = ChemicalPotential(i,j,k,{comp});
860 }
861 auto resiudal = [this,i,j,k,&Phase,&omega,&IP,dt,&ChemicalPotential_Curvature] (vector& x0)
862 {
863 std::vector<double> mu (Ncomp);
864 std::vector<double> f0 (Ncomp);
865 for (std::size_t comp = 0; comp < Ncomp; comp++)
866 {
867 mu[comp] = ChemicalPotential (i,j,k,{comp});
868 ChemicalPotential (i,j,k,{comp}) = x0[comp];
869 ChemicalPotentialDot2 (i,j,k,{comp}) = 0;
870 }
871
872 CalculateLocalConcentrations (i,j,k,Phase,omega);
873 CalculateLocalPhaseFieldIncrements (i,j,k,Phase,omega,IP);
874 CalculateLocalIncrements2 (i,j,k,Phase,omega,dt);
875
876 for (std::size_t comp = 0; comp < Ncomp; comp++)
877 {
878 f0[comp] = ChemicalPotential(i,j,k,{comp}) - ChemicalPotential_Curvature[comp] - ChemicalPotentialDot2(i,j,k,{comp})*dt;
879 ChemicalPotential (i,j,k,{comp}) = mu[comp];
880 }
881 return f0;
882 };
883
884 // Use chemical potential of the previous time step as starting point for the root fining algorithm
885 std::vector<double> mu (Ncomp);
886 for (std::size_t comp = 0; comp < Ncomp; comp++)
887 {
888 mu[comp] = ChemicalPotentialOld(i,j,k,{comp});
889 }
890
891 try
892 {
893 RootFindingAlgorithms::Broyden(resiudal, mu, ChemicalPotentialAccuracy, MaxIterations);
894 }
895 catch (std::runtime_error& ecep)
896 {
897 ConsoleOutput::WriteWarning(ecep.what(),thisclassname,"MergeLocalIncrements2Implicit");
898 }
899
900 for (std::size_t comp = 0; comp < Ncomp; comp++)
901 {
902 ChemicalPotential(i,j,k,{comp}) = mu[comp] ;
903 }
904 }

906double GrandPotentialSolver::Mobility(long i, long j, long k, size_t comp, const PhaseField& Phase, const Temperature& Temp) const
907{
908 assert(Temp.Tx(i,j,k) > 0.0 && "Negative Temperature");
909 double locMobility = 0;
910
911 for (auto alpha = Phase.Fields(i,j,k).cbegin();
912 alpha != Phase.Fields(i,j,k).cend(); alpha++)
913 {
914 size_t PhaseIdx = Phase.FieldsProperties[alpha->index].Phase;
915 locMobility += alpha->value*PhaseMobilities({PhaseIdx,comp})
916 *std::exp(-PhaseActivationEnergies({PhaseIdx,comp})/PhysicalConstants::R/Temp.Tx(i,j,k));
917 }
918
919 if (Use_InterfaceMobilities)
920 for (auto alpha = Phase.Fields(i,j,k).cbegin();
921 alpha != Phase.Fields(i,j,k).cend(); alpha++)
922 for (auto beta = alpha+1;
923 beta != Phase.Fields(i,j,k).cend(); beta++)
924 {
925 size_t PhaseIdx = Phase.FieldsProperties[alpha->index].Phase;
926 {
927 size_t BetaIdx = Phase.FieldsProperties[beta->index].Phase;
928 locMobility += 4*alpha->value*beta->value*InterfaceMobilities({PhaseIdx,BetaIdx,comp});
929 }
930 }
931
932 if (Use_dPhaseMobility_dConcentration)
933 for(size_t compB = 0; compB < Ncomp; compB++)
934 for (auto alpha = Phase.Fields(i,j,k).cbegin();
935 alpha != Phase.Fields(i,j,k).cend(); alpha++)
936 {
937 size_t PhaseIdx = Phase.FieldsProperties[alpha->index].Phase;
938 locMobility += alpha->value*dPhaseMobilities_dConcentration({PhaseIdx,comp,compB});
939 }
940
941 return locMobility;

943double GrandPotentialSolver::MaximumTimeStep(const PhaseField& Phase, const Temperature& Temp, const GrandPotentialDensity& omega)
944{
945 const double factor = (Grid.Active() > 1) ? ((Grid.Active() > 2) ? 6 : 4) : 2; // factor is derived stability criterion (div(grad(mu))
946 double dt_max = std::numeric_limits<double>::max();
947
948 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,ChemicalPotential,0,reduction(min:dt_max))
949 {
950 for(size_t comp = 0; comp < Ncomp; comp++)
951 {
952 double M = std::abs(Mobility(i,j,k,comp,Phase,Temp)/Susceptibility(i,j,k,comp,Phase,omega));
953 double loc_dt_max = Grid.dx*Grid.dx/factor/M;
954 dt_max = std::min(loc_dt_max, dt_max);
955 }
956 }
957 OMP_PARALLEL_STORAGE_LOOP_END
958 return dt_max;

990void GrandPotentialSolver::Solve(PhaseField& Phase, const GrandPotentialDensity& omega, const BoundaryConditions& BC, const InterfaceProperties& IP, const Temperature& Temp, double dt)
991{
992 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,ChemicalPotential,ChemicalPotential.Bcells()-1,)
993 {
994 CalculateLocalPhaseFieldIncrements(i,j,k,Phase,omega,IP);
995 CaclulateLocalDiffusionFlux (i,j,k,{0.0,0.0,0.0},Phase,Temp);
996 }
997 OMP_PARALLEL_STORAGE_LOOP_END
998
999 if (UseImplicitSolver) SolveImplicit(Phase, omega, BC, IP, Temp, dt);
1000 else SolveExplicit(Phase, omega, BC, IP, Temp, dt);
1001
1002 if (ConserveTOC) EnforceConservationOfTOC(Phase,omega);
1003 else
1004 {
1005 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,ChemicalPotential,0,)
1006 {
1007 CalculateLocalConcentrations (i,j,k,Phase,omega);
1008 }
1009 OMP_PARALLEL_STORAGE_LOOP_END
1010 }
1011 SetBoundaryConditions(BC);
