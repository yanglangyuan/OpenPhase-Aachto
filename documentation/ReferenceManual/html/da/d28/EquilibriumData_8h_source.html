51 double eq_composition_extra(size_t comp, double locTx) const
52 {
53 double eq_composition_return = eq_composition({comp});
54 if(eq_slope({comp}) != 0.0)
55 {
56 eq_composition_return += (locTx - eq_temperature)/eq_slope({comp});
57 }
58 return eq_composition_return;
59 }
