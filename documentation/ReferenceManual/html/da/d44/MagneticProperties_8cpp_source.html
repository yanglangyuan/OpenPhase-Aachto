91void MagneticProperties::ReadInput(stringstream& inp)
92{
93 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
94
95 const unsigned int Sublattices =
96 FileInterface::ReadParameterI(inp, moduleLocation, std::string("SUB"));
97
98 Magnetisation.Allocate(Grid, {Sublattices}, 1);
99
100 for(unsigned long int alpha = 0; alpha < Nphases; alpha++)
101 for(unsigned long int i = 0; i < 3; i++)
102 for(unsigned long int j = 0; j < 3; j++)
103 {
104 stringstream converter;
105 converter << alpha << "_" << i+1 << j+1;
106 const std::string name = "PMEA" + converter.str();
107
108 const double value = FileInterface::ReadParameterD(inp, moduleLocation, name, false, false);
109
110 RelativePermeability [alpha](i,j) = value;
111 RelativePermeabilityPhase[alpha](i,j) = value;
112 Permeability [alpha](i,j) = value * PhysicalConstants::mu_0;
113 PermeabilityPhase [alpha](i,j) = value * PhysicalConstants::mu_0;
114 }
115
116 ConsoleOutput::WriteLine("-");
117}

119void MagneticProperties::SetGrainsProperties(const PhaseField& Phase)
120{
121 const unsigned int NPhaseFields = Phase.FieldsProperties.size();
122 if(Permeability.size() != NPhaseFields)
123 {
124 Permeability.Reallocate(NPhaseFields);
125 RelativePermeability.Reallocate(NPhaseFields);
126 }
127
128 for(unsigned int alpha = 0; alpha < NPhaseFields; alpha++)
129 if(Phase.FieldsProperties[alpha].Exist)
130 {
131 const unsigned int pIndex = Phase.FieldsProperties[alpha].Phase;
132 Permeability [alpha] = PermeabilityPhase [pIndex];
133 RelativePermeability[alpha] = RelativePermeabilityPhase[pIndex];
134 }
135}

156void MagneticProperties::WriteMagneticFieldVTK(const int tStep, const Settings& locSettings) const
157{
158 std::vector<VTK::Field_t> ListOfFields;
159 std::string Filename = FileInterface::MakeFileName(locSettings.VTKDir, "MagneticField_", tStep, ".vts");
160 ListOfFields.push_back((VTK::Field_t) {"MagneticField", [this](int i,int j,int k){return MagneticField(i,j,k);}});
161 VTK::Write(Filename, locSettings, ListOfFields);
162}

164void MagneticProperties::WriteMagnetisationVTK(const int tStep, const Settings& locSettings) const
165{
166 std::vector<VTK::Field_t> ListOfFields;
167 std::string Filename = FileInterface::MakeFileName(locSettings.VTKDir, "Magnetisation_", tStep, ".vts");
168 for (unsigned int n = 0; n < Sublattices; n++)
169 {
170 ListOfFields.push_back((VTK::Field_t) {"Magnetisation_" + to_string(n), [this,n](int i,int j,int k){return Magnetisation(i,j,k,{n});}});
171 }
172 VTK::Write(Filename, locSettings, ListOfFields);
173}
