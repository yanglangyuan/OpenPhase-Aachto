37std::string ConsoleOutput::get_time()
38{
39 std::time_t now = std::chrono::system_clock::to_time_t(std::chrono::system_clock::now());
40 std::string s(30, '\0');
41 size_t size = std::strftime(&s[0], s.size(), "%Y-%m-%d %H:%M:%S", std::localtime(&now));
42 s.resize(size);
43 return s;
44}

258void ConsoleOutput::AdvanceProgressIndicator(double start_value, double end_value, double current_value, int& pos)
259{
260#ifdef MPI_PARALLEL
261 if (MPI_RANK == 0)
262#endif
263 if(OutputVerbosity >= VerbosityLevels::Normal)
264 {
265 int current_value_normalized = 80.0*fabs((current_value - start_value)/(end_value - start_value));
266 for(int n = pos; n < current_value_normalized; n++)
267 {
268 pos++;
269 cout << "#";
270 }
271 cout << flush;
272 }
273}
