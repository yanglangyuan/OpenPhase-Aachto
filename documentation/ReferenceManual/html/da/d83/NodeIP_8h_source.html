38struct InterfacePropertiesFieldEntry
39{
40 size_t indexA;
41 size_t indexB;
42 double energy;
43 double stiffness;
44 double mobility;
45
46 InterfacePropertiesFieldEntry() :
47 indexA(0),
48 indexB(0),
49 energy(0.0),
50 stiffness(0.0),
51 mobility(0.0)
52 {
53
54 }
55 InterfacePropertiesFieldEntry(const InterfacePropertiesFieldEntry& entry) :
56 indexA(entry.indexA),
57 indexB(entry.indexB),
58 energy(entry.energy),
59 stiffness(entry.stiffness),
60 mobility(entry.mobility)
61 {
62
63 }
64 InterfacePropertiesFieldEntry& operator=(const InterfacePropertiesFieldEntry& entry)
65 {
66 indexA = entry.indexA;
67 indexB = entry.indexB;
68 energy = entry.energy;
69 stiffness = entry.stiffness;
70 mobility = entry.mobility;
71 return *this;
72 }
73};

193 {

199inline NodeIP NodeIP::operator-(const NodeIP& n) const
200{
201 NodeIP result = n;
202
203 for (auto i = result.begin(); i < result.end(); ++i)
204 {
205 i->energy *= -1.0;
206 i->stiffness *= -1.0;
207 i->mobility *= -1.0;
208 }
209
210 for (auto i = Fields.cbegin(); i < Fields.cend(); ++i)
211 {
212 result.add_energy_stiffness_and_mobility(i->indexA, i->indexB, i->energy, i->stiffness, i->mobility);
213 }
214 return result;
215}

730inline void NodeIP::pack(std::vector<double>& buffer)
731{
732 buffer.push_back(Fields.size());
733 for(auto it = Fields.begin(); it != Fields.end();++it)
734 {
