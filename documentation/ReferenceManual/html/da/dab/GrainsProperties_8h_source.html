453 double GrainRadius(const Grain& grain, int dim = 3) const
454 {
455 switch (dim)
456 {
457 case 1: return grain.Volume;
458 case 2: return std::sqrt(grain.Volume/Pi);
459 case 3: return std::pow(3.0/4.0*grain.Volume/Pi,1.0/3.0);
460 default: return 0.0;

489 double MeanGrainVolume(size_t pIndex) const
490 {
491 size_t number = 0;
492 double volume = 0.0;
493 for (const Grain& grain : GrainsStorage)
494 {
495 //if (grain.Volume != 0.0) \\NOTE: counts invisible small grains
496 if (grain.Volume >= 1.0 && grain.Phase == pIndex)
497 {
498 volume+=grain.Volume;
499 number++;
500 }
501 }
502 if (number > 0) return volume/number;
503 else return 0.0;
504 }

506 {
507 size_t number = 0;
508 double mean = 0;
509 for (const Grain& grain : GrainsStorage)

512 if (grain.Volume > 1.0 && grain.Phase == pIndex)

514 mean += GrainRadius(grain,dim);
515 number++;
516 }
