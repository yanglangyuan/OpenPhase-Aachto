39void MassDensity::Initialize(Settings& locSettings, std::string ObjectNameSuffix)
40{

97void MassDensity::Set(PhaseField& PF, Composition& Cx, Temperature& Tx)
98{
99 size_t Ncomp = Cx.Ncomp;
100 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Total,0,)
101 for(size_t comp = 0; comp < Ncomp; comp++)
102 {
103 Total(i,j,k) = 0.0;
104 }
105 OMP_PARALLEL_STORAGE_LOOP_END
106
107 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Total,0,)
108 if(PF.Fields(i,j,k).wide_interface())
109 {
110 for(size_t comp = 0; comp < Ncomp; comp++)
111 for(size_t alpha = 0; alpha != Nphases; alpha++)
112 {
113 Total(i,j,k) += PF.Fractions(i,j,k,{alpha})*
114 Cx.MoleFractions(i,j,k,{alpha, comp});
115 }
116 }
117 else
118 {
119 size_t alpha = PF.FieldsProperties[PF.Fields(i,j,k).front().index].Phase;
120 for(size_t comp = 0; comp < Ncomp; comp++)
121 {
122 Total(i,j,k) = Cx.MoleFractions(i,j,k,{alpha, comp});
123 }
124 }
125 OMP_PARALLEL_STORAGE_LOOP_END
126}

128void MassDensity::SetInitial(PhaseField& PF)
129{
130 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Total,0,)
131 {
132 Total(i,j,k) = 0.0;
133 Tensor<double,1> locFractions = PF.Fractions(i,j,k);
134 for(size_t alpha = 0; alpha != Nphases; alpha++)
135 {
136 Phase(i,j,k,{alpha}) = Initial({alpha});
137 Total(i,j,k) += locFractions({alpha})*Initial({alpha});
138 }
139 }
140 OMP_PARALLEL_STORAGE_LOOP_END
141}

154void MassDensity::MoveFrame(const int dx, const int dy, const int dz, const BoundaryConditions& BC)
155{
156 int xBeg = (dx >= 0) + (dx < 0)*(Grid.Nx) - 1;
157 int xEnd = (dx >= 0)*(Grid.Nx) + (dx < 0) - 1;
158 int xInc = 1 - 2*(dx < 0);
159
160 int yBeg = (dy >= 0) + (dy < 0)*(Grid.Ny) - 1;
161 int yEnd = (dy >= 0)*(Grid.Ny) + (dy < 0) - 1;
162 int yInc = 1 - 2*(dy < 0);
163
164 int zBeg = (dz >= 0) + (dz < 0)*(Grid.Nz) - 1;
165 int zEnd = (dz >= 0)*(Grid.Nz) + (dz < 0) - 1;
166 int zInc = 1 - 2*(dz < 0);
167
168 SetBoundaryConditions(BC);
169
170 for(int i = xBeg; ((dx >= 0) and (i <= xEnd)) or ((dx < 0) and (i >= xEnd)); i += xInc)
171 for(int j = yBeg; ((dy >= 0) and (j <= yEnd)) or ((dy < 0) and (j >= yEnd)); j += yInc)
172 for(int k = zBeg; ((dz >= 0) and (k <= zEnd)) or ((dz < 0) and (k >= zEnd)); k += zInc)
173 {
174 Phase(i,j,k) = Phase(i + dx, j + dy, k + dz);
175 Total(i, j, k) = Total(i + dx, j + dy, k + dz);
176 }
177
178 SetBoundaryConditions(BC);
179
180 ConsoleOutput::WriteStandard(thisclassname, "Frame moved.");
181}

194bool MassDensity::Write(const Settings& locSettings, const int tStep) const
195{
196#ifdef MPI_PARALLEL
197 string FileName = FileInterface::MakeFileName(locSettings.RawDataDir, thisclassname+"_"+std::to_string(MPI_RANK)+"_", tStep, ".dat");
198#else
199 string FileName = FileInterface::MakeFileName(locSettings.RawDataDir, thisclassname+"_", tStep, ".dat");
200#endif
201
202 ofstream out(FileName.c_str(), ios::out | ios::binary);
203
204 if (!out)
205 {
206 ConsoleOutput::WriteExit("File \"" + FileName + "\" could not be created", thisclassname, "Write()");
207 return false;
208 };
209 STORAGE_LOOP_BEGIN(i,j,k,Phase,Phase.Bcells())
210 out.write(reinterpret_cast<const char*>(Phase(i,j,k).data()), Phase(i,j,k).size()*sizeof(double));
211 STORAGE_LOOP_END
212 STORAGE_LOOP_BEGIN(i,j,k,Total,Total.Bcells())
213 out.write(reinterpret_cast<const char*>(&Total(i,j,k)), sizeof(double));
214 STORAGE_LOOP_END
215 return true;
216}

219{
220#ifdef MPI_PARALLEL
221 string FileName = FileInterface::MakeFileName(locSettings.RawDataDir, thisclassname+"_"+std::to_string(MPI_RANK)+"_", tStep, ".dat");

223 string FileName = FileInterface::MakeFileName(locSettings.RawDataDir, thisclassname+"_", tStep, ".dat");

225
226 fstream inp(FileName.c_str(), ios::in | ios::binary);
227
228 if (!inp)
229 {
230 ConsoleOutput::WriteWarning("File \"" + FileName + "\" could not be opened", thisclassname, "Read");
231 return false;
232 };
233
234 STORAGE_LOOP_BEGIN(i,j,k,Phase,Phase.Bcells())
235 inp.read(reinterpret_cast<char*>(Phase(i,j,k).data()), Phase(i,j,k).size()*sizeof(double));
236 STORAGE_LOOP_END
237 STORAGE_LOOP_BEGIN(i,j,k,Total,Total.Bcells())
238 inp.read(reinterpret_cast<char*>(&Total(i,j,k)), sizeof(double));
239 STORAGE_LOOP_END
240 SetBoundaryConditions(BC);
241 ConsoleOutput::WriteStandard(thisclassname, "Binary Input Read Successfully.");
242 return true;

245void MassDensity::WriteVTK(int tStep, const Settings& locSettings, const int precision)
246{
247 std::vector<VTK::Field_t> ListOfFields;
248 std::string Filename = FileInterface::MakeFileName(locSettings.VTKDir, thisclassname+"_", tStep, ".vts");
249
250 ListOfFields.push_back((VTK::Field_t){"Total", [this](int i,int j,int k){return Total(i,j,k);}});
251 for(size_t n = 0; n < Nphases; n++)
252 {
253 ListOfFields.push_back((VTK::Field_t){"Phase_" + std::to_string(n), [n,this](int i,int j,int k){return Phase(i,j,k,{n});}});
254 }
255 VTK::Write(Filename, locSettings, ListOfFields, precision);
256}

269MassDensity& MassDensity::operator=(const MassDensity& rhs)
270{
271 // protect against self-assignment and copy of unitialized object
272 if (this != &rhs and rhs.thisclassname != "")
273 {
274 thisclassname = rhs.thisclassname;
275
276 Grid = rhs.Grid;
277
278 Nphases = rhs.Nphases;
279
280 if (Phase.IsNotAllocated())
281 {
282 Phase.Allocate(Grid, {Nphases}, rhs.Phase.Bcells());
283 Total.Allocate(Grid, rhs.Total.Bcells());
284 Initial.Allocate({Nphases});
285 }
286 else if (not Phase.IsSize(Grid.Nx, Grid.Ny, Grid.Nz))
287 {
288 Phase.Reallocate(Grid.Nx, Grid.Ny, Grid.Nz);
289 Total.Reallocate(Grid.Nx, Grid.Ny, Grid.Nz);
290 Initial.Reallocate({Nphases});
291 }
292
293 Phase = rhs.Phase;
294 Total = rhs.Total;
295
296 for (size_t n = 0; n < Nphases; n++)
297 {
298 Initial({n}) = rhs.Initial({n});
299 }
300 }
301 return *this;
302}
