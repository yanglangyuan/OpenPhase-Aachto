43 std::string MetaData::to_iso8601(std::time_t t) {
44 std::tm tm = *std::gmtime(&t); // use gmtime for UTC ("Z"), localtime for local offset
45 std::ostringstream oss;
46 oss << std::put_time(&tm, "%Y-%m-%dT%H:%M:%SZ");
47 return oss.str();
48 }

50std::string MetaData::make_relative_path(const std::string& path_to_file,
51 const std::string& path_to_working_directory)
52{
53 try {
54 fs::path file_path(path_to_file);
55 fs::path base_path(path_to_working_directory);
56
57 // Ensure both paths are absolute for correct relative calculation
58 file_path = fs::absolute(file_path);
59 base_path = fs::absolute(base_path);
60
61 fs::path relative = fs::relative(file_path, base_path);
62 return "./"+relative.generic_string(); // Use forward slashes
63 } catch (const fs::filesystem_error& e) {
64 std::cerr << "Filesystem error: " << e.what() << std::endl;
65 return path_to_file; // fallback to original
66 }
67}

69 MetaData::MetaData()
70 {
71 time_t timestamp;
72 time(&timestamp);
73 Program.Title = "OPExe";
74 Program.GitHashCore = GIT_COMMIT_SHA;
75 StartTime = to_iso8601(timestamp);
76 #ifdef _WIN32
77 char exepath[MAX_PATH];
78 std::string s_exepath(exepath);
79 HMODULE hModule = GetModuleHandle(NULL);
80 if (hModule != NULL)
81 {
82 // Use GetModuleFileName() with module handle to get the path
83 GetModuleFileName(hModule, exepath, (sizeof(exepath)));
84 s_exepath = exepath;
85 }
86 Program.Title = s_exepath;
87 #else
88 std::string result;
89 result.resize(PATH_MAX);
90 ssize_t count = readlink("/proc/self/exe", result.data(), PATH_MAX);
91 result.resize(count);
92 Program.Title = result;
93 #endif
94 }

177 void MetaData::ReadAuthorInformation(std::stringstream& inp)
178 {
179 std::string module = " ";
180 int moduleLocation = FileInterface::FindModuleLocation(inp, "AuthorInformation");
181 if (moduleLocation != 0)
182 {
183 int i = 0;
184 do {
185 module = "Author_" + std::to_string(i);
186 std::string tmpMd = FileInterface::ReadParameterS(inp,moduleLocation,module,false,"NOAUTHOR");
187 if (tmpMd != "NOAUTHOR")
188 {
189 AuthorInformation NewAuthor;
190 NewAuthor.Name = tmpMd;
191 module = "Affiliation_" + std::to_string(i);
192 tmpMd = FileInterface::ReadParameterS(inp,moduleLocation,module,false,"");
193 NewAuthor.Affiliation = tmpMd;
194 module = "ORCID_" + std::to_string(i);
195 tmpMd = FileInterface::ReadParameterS(inp,moduleLocation,module,false,"");
196 NewAuthor.ORCID = tmpMd;
197 Authors.push_back(NewAuthor);
198 ++i;
199 }
200 else
201 {
202 break;
203 }
204 }
205 while(true);
206 }
