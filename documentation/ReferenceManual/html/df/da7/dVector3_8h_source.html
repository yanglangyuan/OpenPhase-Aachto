336 dVector3& rotate(const dMatrix3x3& RotationMatrix)
337 {
338 dVector3 tmp;
339 for(int i = 0; i < 3; ++i)
340 for(int j = 0; j < 3; ++j)
341 {
342 tmp[i] += RotationMatrix(i,j) * storage[j];
343 }
344 storage = tmp.storage;
345 return *this;
346 }
