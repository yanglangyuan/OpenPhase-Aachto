71void MechanicalLoads::ReadInput(std::stringstream& inp)
72{
73 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
74
75 bool endofloads = false;
76 size_t idx = 0;
77 while(!endofloads)
78 {
79 stringstream converter;
80 converter << idx;
81 string counter = converter.str();
82 if(FileInterface::FindParameter(inp, moduleLocation, string("Load_") + counter) != -1)
83 {
84 ConsoleOutput::WriteLine("=");
85
86 bool LoadEnabled = FileInterface::ReadParameterB(inp, moduleLocation, string("Load_") + counter, true, false);
87
88 if(LoadEnabled)
89 {
90 MechanicalLoadStructure locLoad;
91 locLoad.Active = false;
92
93 bool LoadONtriggerSet = false;
94 string LoadONtrigger = FileInterface::ReadParameterK(inp, moduleLocation, string("TriggerON_") + counter, true, "USER");
95 if(LoadONtrigger == "USER") {locLoad.TriggerON = EventTriggers::User; LoadONtriggerSet = true;}
96 if(LoadONtrigger == "TIME") {locLoad.TriggerON = EventTriggers::Time; LoadONtriggerSet = true;}
97 if(LoadONtrigger == "TIMESTEP") {locLoad.TriggerON = EventTriggers::TimeStep; LoadONtriggerSet = true;}
98 if(LoadONtrigger == "STRESS") {locLoad.TriggerON = EventTriggers::Stress; LoadONtriggerSet = true;}
99 if(LoadONtrigger == "STRAIN") {locLoad.TriggerON = EventTriggers::Strain; LoadONtriggerSet = true;}
100
101 if(not LoadONtriggerSet)
102 {
103 string message = "No valid ON trigger is provided for the load " + counter + " --> " + LoadONtrigger;
104 ConsoleOutput::WriteExit(message,thisclassname,"ReadInput()");
105 OP_Exit(EXIT_FAILURE);
106 }
107
108 if(locLoad.TriggerON != EventTriggers::User)
109 {
110 locLoad.ONtriggerValue = FileInterface::ReadParameterD(inp, moduleLocation, string("TriggerONvalue_") + counter, true, 0);
111 }
112 if(locLoad.TriggerON == EventTriggers::Strain or locLoad.TriggerON == EventTriggers::Stress)
113 {
114 string OnComp = FileInterface::ReadParameterK(inp, moduleLocation, string("ONindex_") + counter, true, "XX");
115 if (OnComp == "XX") locLoad.ONindex = 0;
116 if (OnComp == "YY") locLoad.ONindex = 1;
117 if (OnComp == "ZZ") locLoad.ONindex = 2;
118 if (OnComp == "YZ" or OnComp == "ZY") locLoad.ONindex = 3;
119 if (OnComp == "XZ" or OnComp == "ZX") locLoad.ONindex = 4;
120 if (OnComp == "XY" or OnComp == "YX") locLoad.ONindex = 5;
121 }
122
123 bool LoadOFFtriggerSet = false;
124 string LoadOFFtrigger = FileInterface::ReadParameterK(inp, moduleLocation, string("TriggerOFF_") + counter, false, LoadONtrigger);
125 if(LoadOFFtrigger == "USER") {locLoad.TriggerOFF = EventTriggers::User; LoadOFFtriggerSet = true;}
126 if(LoadOFFtrigger == "TIME") {locLoad.TriggerOFF = EventTriggers::Time; LoadOFFtriggerSet = true;}
127 if(LoadOFFtrigger == "TIMESTEP") {locLoad.TriggerOFF = EventTriggers::TimeStep; LoadOFFtriggerSet = true;}
128 if(LoadOFFtrigger == "STRESS") {locLoad.TriggerOFF = EventTriggers::Stress; LoadOFFtriggerSet = true;}
129 if(LoadOFFtrigger == "STRAIN") {locLoad.TriggerOFF = EventTriggers::Strain; LoadOFFtriggerSet = true;}
130
131 if(not LoadOFFtriggerSet)
132 {
133 string message = "No valid OFF trigger is provided for the load " + counter + " --> " + LoadOFFtrigger;
134 ConsoleOutput::WriteExit(message,thisclassname,"ReadInput()");
135 OP_Exit(EXIT_FAILURE);
136 }
137
138 if(locLoad.TriggerOFF != EventTriggers::User)
139 {
140 locLoad.OFFtriggerValue = FileInterface::ReadParameterD(inp, moduleLocation, string("TriggerOFFvalue_") + counter, true, 0);
141 }
142
143 if(locLoad.TriggerOFF == EventTriggers::Strain or locLoad.TriggerOFF == EventTriggers::Stress)
144 {
145 string OffComp = FileInterface::ReadParameterK(inp, moduleLocation, string("OFFindex_") + counter, true, "XX");
146
147 if (OffComp == "XX") locLoad.OFFindex = 0;
148 if (OffComp == "YY") locLoad.OFFindex = 1;
149 if (OffComp == "ZZ") locLoad.OFFindex = 2;
150 if (OffComp == "YZ" or OffComp == "ZY") locLoad.OFFindex = 3;
151 if (OffComp == "XZ" or OffComp == "ZX") locLoad.OFFindex = 4;
152 if (OffComp == "XY" or OffComp == "YX") locLoad.OFFindex = 5;
153 }
154
155 locLoad.Repeat = FileInterface::ReadParameterI(inp, moduleLocation, string("Repeat_") + counter, false, 1);
156 if(locLoad.Repeat <= 0)
157 {
158 string message = "Zero or negative \"Repeat\" parameter for load " + counter + " --> " + to_string(locLoad.Repeat);
159 ConsoleOutput::WriteExit(message,thisclassname,"ReadInput()");
160 OP_Exit(EXIT_FAILURE);
161 }
162
164
165 std::vector<std::string> BCnames {"BCX", "BCY","BCZ","BCYZ","BCXZ","BCXY"};
166 std::vector<std::string> BCvalueNames {"BCValueX", "BCValueY","BCValueZ","BCValueYZ","BCValueXZ","BCValueXY"};
167
168 for(int n = 0; n < 6; n++)
169 {
170 string tmp2 = FileInterface::ReadParameterK(inp, moduleLocation, BCnames[n] + "_" + counter, false);
171
172 if (tmp2 == "STRAIN")
173 {
174 locLoad.AppliedStrain[n] = FileInterface::ReadParameterD(inp, moduleLocation, BCvalueNames[n] + "_" + counter);
175 locLoad.AppliedStrainMask[n] = 1.0;
176 }
177 if (tmp2 == "STRAINRATE")
178 {
179 locLoad.AppliedStrainRate[n] = FileInterface::ReadParameterD(inp, moduleLocation, BCvalueNames[n] + "_" + counter);
180 locLoad.AppliedStrainRateMask[n] = 1.0;
181 }
182 if (tmp2 == "STRESS")
183 {
184 locLoad.AppliedStressMask[n] = 1.0;
185 locLoad.AppliedStress[n] = FileInterface::ReadParameterD(inp, moduleLocation, BCvalueNames[n] + "_" + counter);
186 }
187 }
188
189 Loads.push_back(locLoad);
190 }
191 idx++;
192 }
193 else
194 {
195 endofloads = true;
196 }
197 }
198
199 ConsoleOutput::WriteLine();
200 ConsoleOutput::WriteBlankLine();
201}

209bool MechanicalLoads::Read(const std::string FileName)
210{
211 std::ifstream inp(FileName.c_str(), std::ios::in | std::ios::binary);
212 if (!inp)
213 {
214 ConsoleOutput::WriteWarning("File \"" + FileName + "\" could not be opened", thisclassname, "Read()");
215 return false;
216 }
217
218 size_t size = 0;
219 inp.read(reinterpret_cast<char*>(&size), sizeof(size_t));
220
221 if(Loads.size() != size)
222 {
223 Loads.resize(size);
224 }
225
226 for(size_t n = 0; n != size; n++) Loads[n].Read(inp);
227
228 inp.close();
229 ConsoleOutput::WriteStandard(thisclassname, "Binary input loaded");
230 return true;
231}

264void MechanicalLoads::Apply(PhaseField& Phase, ElasticProperties& EP, RunTimeControl& RTC)
265{
266 for(auto load = Loads.begin(); load != Loads.end(); load++)
267 if(load->Active)
268 {
269 if(!load->Set)
270 {
271 for(int n = 0; n < 6; n++)
272 {
273 if(load->AppliedStrainMask[n])
274 {
275 EP.AppliedStrainMask[n] = load->AppliedStrainMask[n];
276 EP.AppliedStrain[n] = load->AppliedStrain[n];
277 }
278 if(load->AppliedStrainRateMask[n])
279 {
280 EP.AppliedStrainRateMask[n] = load->AppliedStrainRateMask[n];
281 EP.AppliedStrainRate[n] = load->AppliedStrainRate[n];
282 }
283 if(load->AppliedStressMask[n])
284 {
285 EP.AppliedStressMask[n] = load->AppliedStressMask[n];
286 EP.AppliedStress[n] = load->AppliedStress[n];
287 }
288 }
289 load->Set = true;
290 }
291 }
292 else
293 {
294 if(load->Set)
295 {
296 for(int n = 0; n < 6; n++)
297 {
298 if(load->AppliedStrainMask[n])
299 {
300 EP.AppliedStrainMask[n] = 0.0;
301 EP.AppliedStrain[n] = 0.0;
302 }
303 if(load->AppliedStrainRateMask[n])
304 {
305 EP.AppliedStrainRateMask[n] = 0.0;
306 EP.AppliedStrainRate[n] = 0.0;
307 }
308 if(load->AppliedStressMask[n])
309 {
310 EP.AppliedStressMask[n] = 0.0;
311 EP.AppliedStress[n] = 0.0;
312 }
313 }
314 load->Set = false;
315 }
316 }

319void MechanicalLoads::Activate(PhaseField& Phase, ElasticProperties& EP, RunTimeControl& RTC)
320{
321 int counter = 0;
322 for(auto load = Loads.begin(); load != Loads.end(); load++)
323 {
324 if(load->Active)
325 {
326 switch(load->TriggerOFF)
327 {
328 case EventTriggers::Time:
329 {
330 if(RTC.SimulationTime > load->OFFtriggerValue - RTC.dt and
331 RTC.SimulationTime < load->OFFtriggerValue + RTC.dt)
332 {
333 load->Active = false;
334 }
335 break;
336 }
337 case EventTriggers::TimeStep:
338 {
339 if(RTC.TimeStep == load->OFFtriggerValue)
340 {
341 load->Active = false;
342 }
343 break;
344 }
345 case EventTriggers::Stress:
346 {
347 if(( signbit(load->OFFtriggerValue) and EP.AverageStress[load->OFFindex] < load->OFFtriggerValue) or
348 (!signbit(load->OFFtriggerValue) and EP.AverageStress[load->OFFindex] > load->OFFtriggerValue))
349 {
350 load->Active = false;
351 }
352 break;
353 }
354 case EventTriggers::Strain:
355 {
356 if(( signbit(load->OFFtriggerValue) and EP.AverageStrain[load->OFFindex] < load->OFFtriggerValue) or
357 (!signbit(load->OFFtriggerValue) and EP.AverageStrain[load->OFFindex] > load->OFFtriggerValue))
358 {
359 load->Active = false;
360 }
361 break;
362 }
363 case EventTriggers::User:
364 default:
365 {
366 break;
367 }
368 }
369 if(!load->Active)
370 {
371 std::stringstream message;
372 message << "load " << counter << " is deactivated.";
373 ConsoleOutput::WriteStandard(thisclassname, message.str());
374 }
375 }
376 else if (load->Repeat != 0)
377 {
378 switch(load->TriggerON)
379 {
380 case EventTriggers::Time:
381 {
382 if(RTC.SimulationTime > load->ONtriggerValue - RTC.dt and
383 RTC.SimulationTime < load->ONtriggerValue + RTC.dt)
384 {
385 load->Active = true;
386 load->Repeat -= 1;
387 }
388 break;
389 }
390 case EventTriggers::TimeStep:
391 {
392 if(RTC.TimeStep == load->ONtriggerValue)
393 {
394 load->Active = true;
395 load->Repeat -= 1;
396 }
397 break;
398 }
399
400 case EventTriggers::Stress:
401 {
402 if(( signbit(load->ONtriggerValue) and EP.AverageStress[load->ONindex] < load->ONtriggerValue) or
403 (!signbit(load->ONtriggerValue) and EP.AverageStress[load->ONindex] > load->ONtriggerValue))
404 {
405 load->Active = true;
406 load->Repeat -= 1;
407 }
408 break;
409 }
410 case EventTriggers::Strain:
411 {
412 if(( signbit(load->ONtriggerValue) and EP.AverageStrain[load->ONindex] < load->ONtriggerValue) or
413 (!signbit(load->ONtriggerValue) and EP.AverageStrain[load->ONindex] > load->ONtriggerValue))
414 {
415 load->Active = true;
416 load->Repeat -= 1;
417 }
418 break;
419 }
420 case EventTriggers::User:
421 default:
422 {
423 break;
424 }
425 }
426 if(load->Active)
427 {
428 std::stringstream message;
429 message << "load " << counter << " is activated.";
430 ConsoleOutput::WriteStandard(thisclassname, message.str());
431 }
432 }
433 counter++;
434 }
435}
