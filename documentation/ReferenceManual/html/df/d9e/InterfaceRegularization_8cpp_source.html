45void InterfaceRegularization::Initialize(Settings& locSettings, std::string ObjectNameSuffix)
46{
47 thisclassname = "InterfaceRegularization";
48 thisobjectname = thisclassname + ObjectNameSuffix;
49
50 Grid = locSettings.Grid;
51
52 Nphases = locSettings.Nphases;
53
54 Averaging = true;
55
56 PhiThreshold = DBL_EPSILON;
57
58 Range = (Grid.iWidth + 1)/2;
59
60 switch(Grid.Resolution)
61 {
62 case Resolutions::Single:
63 {
64 Curvature.Allocate(Grid, Range);
65 break;
66 }
67 case Resolutions::Dual:
68 {
69 GridParameters DoubleDimensions = Grid.DoubleResolution();
70 Curvature.Allocate(DoubleDimensions, Range);
71 break;
72 }
73 }
74
75 locSettings.AddForRemeshing(*this);
76
77 initialized = true;
78 ConsoleOutput::WriteStandard(thisclassname, "Initialized");
79}

101void InterfaceRegularization::ReadInput(stringstream& inp)
102{
103 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
104
105 Averaging = FileInterface::ReadParameterB(inp, moduleLocation, string("Average"), false, true);
106 Range = FileInterface::ReadParameterI(inp, moduleLocation, string("Range"), false, Range);
107 PhiThreshold = FileInterface::ReadParameterD(inp, moduleLocation, string("Threshold"), false, PhiThreshold);
108
109 string tmp = FileInterface::ReadParameterK(inp, moduleLocation, string("ClearingMode"), false, "AUTOMATIC");
110
111 if(tmp == string("AUTOMATIC")) ClearingMode = ClearingModes::Automatic;
112 if(tmp == string("MANUAL")) ClearingMode = ClearingModes::Manual;
113
114 ConsoleOutput::WriteLine();
115 ConsoleOutput::WriteBlankLine();
116}

134void InterfaceRegularization::Average(const PhaseField& Phase, const BoundaryConditions& BC)
135{
136 if(Averaging)
137 {
138 if(Grid.Resolution == Resolutions::Single)
139 {
140 SetWeightsSR(Phase);
141 SetBoundaryConditions(BC);
142 CollectAverageSR(Phase);
143 SetBoundaryConditions(BC);
144 DistributeAverageSR(Phase);
145 }
146 else
147 {
148 SetWeightsDR(Phase);
149 SetBoundaryConditions(BC);
150 CollectAverageDR(Phase);
151 SetBoundaryConditions(BC);
152 DistributeAverageDR(Phase);
153 }
154 }
155}

157void InterfaceRegularization::SetWeightsSR(const PhaseField& Phase)
158{
159 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Curvature,0,)
160 {
161 if (Phase.Fields(i,j,k).interface())
162 for(auto it = Curvature(i,j,k).begin();
163 it != Curvature(i,j,k).end(); ++it)
164 {
165 double PhiAlpha = Phase.Fields(i,j,k).get_value(it->indexA);
166 double PhiBeta = Phase.Fields(i,j,k).get_value(it->indexB);
167
168 it->weight = sqrt(PhiAlpha*PhiBeta);
169 }
170 }
171 OMP_PARALLEL_STORAGE_LOOP_END
172}

174void InterfaceRegularization::CollectAverageSR(const PhaseField& Phase)
175{
176 const int Xrange = min(Range, Grid.Nx-1)*Grid.dNx;
177 const int Yrange = min(Range, Grid.Ny-1)*Grid.dNy;
178 const int Zrange = min(Range, Grid.Nz-1)*Grid.dNz;
179
180 const double weight_threshold = sqrt(PhiThreshold*(1.0 - PhiThreshold));
181
182 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Curvature,0,)
183 {
184 if (Phase.Fields(i,j,k).interface())
185 for(auto it = Curvature(i,j,k).begin();
186 it != Curvature(i,j,k).end(); ++it)
187 {
188 if (it->weight > weight_threshold)
189 {
190 double value = 0.0;
191
192 double SumWeights = 0.0;
193
194 for(int ii = -Xrange; ii <= Xrange; ii++)
195 for(int jj = -Yrange; jj <= Yrange; jj++)
196 for(int kk = -Zrange; kk <= Zrange; kk++)
197 {
198 double dist = sqrt(ii*ii + jj*jj + kk*kk);
199
200 if((Range - dist) > 0.0 and Phase.Fields(i+ii,j+jj,k+kk).interface())
201 {
202 double weight = Curvature(i+ii, j+jj, k+kk).get_weight(it->indexA, it->indexB)*(Range - dist);
203
204 if (weight > DBL_EPSILON)
205 {
206 SumWeights += weight;
207 value += weight*Curvature(i+ii, j+jj, k+kk).get_raw(it->indexA, it->indexB);
208 }
209 }
210 }
211 if(SumWeights > DBL_EPSILON)
212 {
213 it->tmp = value/SumWeights;
214 }
215 }
216 else if(Phase.FieldsProperties[it->indexA].Stage == GrainStages::Seed or
217 Phase.FieldsProperties[it->indexB].Stage == GrainStages::Seed)
218 {
219 it->tmp = it->raw;
220 }
221 }
222 }
223 OMP_PARALLEL_STORAGE_LOOP_END
224}

425void InterfaceRegularization::MergePhaseFieldIncrementsSR(PhaseField& Phase,
426 InterfaceProperties& IP)
427{
428 const double Prefactor = Pi/Phase.Grid.Eta;
429
430 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i, j, k, Curvature, 0, )
431 {
432 if (Phase.Fields(i,j,k).interface())
433 {
434 for(auto it = Curvature(i,j,k).begin();
435 it != Curvature(i,j,k).end(); ++it)
436 {
437 double locCurvature = 0.0;
438 if(Averaging)
439 {
440 locCurvature = it->average;
441 }
442 else
443 {
444 locCurvature = it->raw;
445 }
446
447 double norm = it->weight;
448
449 double dPsi_dt = locCurvature*IP.Properties(i,j,k).get_mobility(it->indexA,it->indexB)*norm*Prefactor;
450
451 Phase.FieldsDot(i,j,k).add_asym1(it->indexA,it->indexB,dPsi_dt);
452 }
453 }
454 }
455 OMP_PARALLEL_STORAGE_LOOP_END
456}

514void InterfaceRegularization::WriteVTK(const Settings& locSettings,
515 const int tStep,
516 const size_t indexA,
517 const size_t indexB,
518 const int precision) const
519{
520 stringstream converter;
521 converter << indexA << "," << indexB;
522 string phases = converter.str();
523
524 std::vector<VTK::Field_t> ListOfFields;
525 ListOfFields.push_back((VTK::Field_t) {"Curvature_avg(" + phases + ")", [indexA,indexB,this](int i,int j,int k){return Curvature(i,j,k).get_average(indexA, indexB);}});
526 ListOfFields.push_back((VTK::Field_t) {"Curvature_raw(" + phases + ")", [indexA,indexB,this](int i,int j,int k){return Curvature(i,j,k).get_raw(indexA, indexB);}});
527 ListOfFields.push_back((VTK::Field_t) {"Curvature_tmp(" + phases + ")", [indexA,indexB,this](int i,int j,int k){return Curvature(i,j,k).get_tmp(indexA, indexB);}});
528 std::string Filename = FileInterface::MakeFileName(locSettings.VTKDir, "InterfaceRegularization_", tStep, ".vts");
529
530 if(Grid.Resolution == Resolutions::Single)
531 {
532 VTK::Write(Filename, locSettings, ListOfFields, precision);
533 }
534 else
535 {
536 VTK::Write(Filename, locSettings, ListOfFields, precision, 2);
537 }
538}

575NodeDF InterfaceRegularization::at(const double x, const double y, const double z) const
576{
577#ifdef DEBUG
578 if(x > Curvature.sizeX() + Curvature.BcellsX() - 1 or
579 y > Curvature.sizeY() + Curvature.BcellsY() - 1 or
580 z > Curvature.sizeZ() + Curvature.BcellsZ() - 1 or
581 x < -Curvature.BcellsX() or
582 y < -Curvature.BcellsY() or
583 z < -Curvature.BcellsZ())
584 {
585 std::stringstream message;
586 message << "ERROR: InterfaceCurvature::Curvature_at()\n"
587 << "Access beyond storage range -> ("
588 << x << "," << y << "," << z << ")" << " is outside of storage bounds ["
589 << -Curvature.BcellsX() << ", " << -Curvature.BcellsY() << ", " << -Curvature.BcellsZ() << "] and ("
590 << Curvature.sizeX() + Curvature.BcellsX() << ", "
591 << Curvature.sizeY() + Curvature.BcellsY() << ", "
592 << Curvature.sizeZ() + Curvature.BcellsZ() << ")\n"
593 << "Terminating!!!\n";
594 throw std::logic_error(message.str());
595 }
596#endif
597
598 long int x0 = floor(x)*Grid.dNx;
599 long int y0 = floor(y)*Grid.dNy;
600 long int z0 = floor(z)*Grid.dNz;
601 double dx = fabs(x - x0)*Grid.dNx;
602 double dy = fabs(y - y0)*Grid.dNy;
603 double dz = fabs(z - z0)*Grid.dNz;
604
605 NodeDF locCurvature;
606 double weight = 0.0;
607 if(Curvature(x0, y0, z0).size())
608 {
609 locCurvature += (Curvature(x0, y0, z0)*((1.0 - dx)*(1.0 - dy)*(1.0 - dz)));
610 weight += ((1.0 - dx)*(1.0 - dy)*(1.0 - dz));
611 }
612 if(Curvature(x0+Grid.dNx, y0, z0).size())
613 {
614 locCurvature += (Curvature(x0+Grid.dNx, y0, z0)*(dx*(1.0 - dy)*(1.0 - dz)));
615 weight += (dx*(1.0 - dy)*(1.0 - dz));
616 }
617 if(Curvature(x0, y0+Grid.dNy, z0).size())
618 {
619 locCurvature += (Curvature(x0, y0+Grid.dNy, z0)*((1.0 - dx)*dy*(1.0 - dz)));
620 weight += ((1.0 - dx)*dy*(1.0 - dz));
621 }
622 if(Curvature(x0, y0, z0+Grid.dNz).size())
623 {
624 locCurvature += (Curvature(x0, y0, z0+Grid.dNz)*((1.0 - dx)*(1.0 - dy)*dz));
625 weight += ((1.0 - dx)*(1.0 - dy)*dz);
626 }
627 if(Curvature(x0+Grid.dNx, y0+Grid.dNy, z0).size())
628 {
629 locCurvature += (Curvature(x0+Grid.dNx, y0+Grid.dNy, z0)*(dx*dy*(1.0 - dz)));
630 weight += (dx*dy*(1.0 - dz));
631 }
632 if(Curvature(x0+Grid.dNx, y0, z0+Grid.dNz).size())
633 {
634 locCurvature += (Curvature(x0+Grid.dNx, y0, z0+Grid.dNz)*(dx*(1.0 - dy)*dz));
635 weight += (dx*(1.0 - dy)*dz);
636 }
637 if(Curvature(x0, y0+Grid.dNy, z0+Grid.dNz).size())
638 {
639 locCurvature += (Curvature(x0, y0+Grid.dNy, z0+Grid.dNz)*((1.0 - dx)*dy*dz));
640 weight += ((1.0 - dx)*dy*dz);
641 }
642 if(Curvature(x0+Grid.dNx, y0+Grid.dNy, z0+Grid.dNz).size())
643 {
644 locCurvature += (Curvature(x0+Grid.dNx, y0+Grid.dNy, z0+Grid.dNz)*(dx*dy*dz));
645 weight += (dx*dy*dz);
646 }
647 if(weight >= DBL_EPSILON)
648 {
649 locCurvature *= 1.0/weight;
650 }
651 return locCurvature;
652}
