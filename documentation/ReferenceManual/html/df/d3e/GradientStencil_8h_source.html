120class GradientStencil
121{
122 public:
123 struct GradientStencilEntry
124 {
125 int di;
126 int dj;
127 int dk;
128 double weightX;
129 double weightY;
130 double weightZ;
131 };
132 void Set(const double UserStencil[3][3][3], double dx,
133 int dNx = 1, int dNy = 1, int dNz = 1)
134 {
135 if(not StencilElements.empty())
136 {
137 StencilElements.clear();
138 }
139
140 for(int x = -dNx; x <= dNx; ++x)
141 for(int y = -dNy; y <= dNy; ++y)
142 for(int z = -dNz; z <= dNz; ++z)
143 if (UserStencil[x+1][y+1][z+1] != 0.0)
144 {
145 GradientStencilEntry locElement;
146 locElement.di = x;
147 locElement.dj = y;
148 locElement.dk = z;
149
150 locElement.weightX = x*UserStencil[x+1][y+1][z+1]/dx;
151 locElement.weightY = y*UserStencil[x+1][y+1][z+1]/dx;
152 locElement.weightZ = z*UserStencil[x+1][y+1][z+1]/dx;
153
154 StencilElements.push_back(locElement);
155 }
156 }
157 void Set(const double UserStencil[3][3][3], GridParameters& Dimensions)
158 {
159 Set(UserStencil, Dimensions.dx, Dimensions.dNx, Dimensions.dNy, Dimensions.dNz);
160 }
161 size_t size() const {return StencilElements.size();};
162 typedef std::vector<GradientStencilEntry>::iterator iterator;
163 typedef std::vector<GradientStencilEntry>::const_iterator citerator;
164 iterator begin() {return StencilElements.begin();};
165 iterator end() {return StencilElements.end();};
166 citerator cbegin() const {return StencilElements.cbegin();};
167 citerator cend() const {return StencilElements.cend();};
168 iterator erase(iterator it) {it = StencilElements.erase(it); return it;}
169 protected:
170 private:
171 std::vector< GradientStencilEntry > StencilElements;
172};
173}
