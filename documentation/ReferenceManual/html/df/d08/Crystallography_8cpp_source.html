34void Crystallography::ReadInput(string InputFileName)
35{
36 ConsoleOutput::WriteLineInsert("Crystal Symmetry input");
37 ConsoleOutput::WriteStandard("Source", InputFileName);
38
39 fstream inpF(InputFileName.c_str(), ios::in | ios_base::binary);
40
41 if (!inpF)
42 {
43 ConsoleOutput::WriteExit("File \"" + InputFileName + "\" could not be opened", thisclassname, "ReadInput()");
44 OP_Exit(EXIT_FAILURE);
45 };
46 stringstream inp;
47 inp << inpF.rdbuf();
48 inpF.close();
49
50 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
51
52 string Symmetry = FileInterface::ReadParameterK(inp, moduleLocation, "CrystalSymmetry", false, "CUBIC");
53
54 // Symmetry matrices for crystal lattice. Taken from Kocks, Tome and Wenk
55 // 'Texture and Anisotropy', Cambridge University Press 1998
56 if (Symmetry == "CUBIC")
57 {
58 // Load cubic parameters (class 432)
59 CrystalSym = LatticeSystems::Cubic;
60
61 nsym = 24;
62 Theta_min = 0 * (Pi / 180);
63 Theta_max = 45 * (Pi / 180);
64 Phi_min = 0 * (Pi);
65 Phi_max = acos(sqrt(1.0 / (2.0 + (pow(tan(Theta_max),2.0)))));
66
67 CrystalSymmetries.Allocate(nsym);
68
69 CrystalSymmetries[0].set(1, 0, 0, 0, 1, 0, 0, 0, 1);
70 CrystalSymmetries[1].set(0, 0, 1, 1, 0, 0, 0, 1, 0);
71 CrystalSymmetries[2].set(0, 1, 0, 0, 0, 1, 1, 0, 0);
72 CrystalSymmetries[3].set(0,-1, 0, 0, 0, 1,-1, 0, 0);
73 CrystalSymmetries[4].set(0,-1, 0, 0, 0,-1, 1, 0, 0);
74 CrystalSymmetries[5].set(0, 1, 0, 0, 0,-1,-1, 0, 0);
75
76 CrystalSymmetries[6].set( 0, 0,-1, 1, 0, 0, 0,-1, 0);
77 CrystalSymmetries[7].set( 0, 0,-1,-1, 0, 0, 0, 1, 0);
78 CrystalSymmetries[8].set( 0, 0, 1,-1, 0, 0, 0,-1, 0);
79 CrystalSymmetries[9].set(-1, 0, 0, 0, 1, 0, 0, 0,-1);
80 CrystalSymmetries[10].set(-1, 0, 0, 0,-1, 0, 0, 0, 1);
81 CrystalSymmetries[11].set( 1, 0, 0, 0,-1, 0, 0, 0,-1);
82
83 CrystalSymmetries[12].set( 0, 0,-1, 0,-1, 0,-1, 0, 0);
84 CrystalSymmetries[13].set( 0, 0, 1, 0,-1, 0, 1, 0, 0);
85 CrystalSymmetries[14].set( 0, 0, 1, 0, 1, 0,-1, 0, 0);
86 CrystalSymmetries[15].set( 0, 0,-1, 0, 1, 0, 1, 0, 0);
87 CrystalSymmetries[16].set(-1, 0, 0, 0, 0,-1, 0,-1, 0);
88 CrystalSymmetries[17].set( 1, 0, 0, 0, 0,-1, 0, 1, 0);
89
90 CrystalSymmetries[18].set( 1, 0, 0, 0, 0, 1, 0,-1, 0);
91 CrystalSymmetries[19].set(-1, 0, 0, 0, 0, 1, 0, 1, 0);
92 CrystalSymmetries[20].set( 0,-1, 0,-1, 0, 0, 0, 0,-1);
93 CrystalSymmetries[21].set( 0, 1, 0,-1, 0, 0, 0, 0,-1);
94 CrystalSymmetries[22].set( 0, 1, 0, 1, 0, 0, 0, 0,-1);
95 CrystalSymmetries[23].set( 0,-1, 0, 1, 0, 0, 0, 0, 1);
96
97 ConsoleOutput::WriteSimple("CrystalSymmetry::Cubic");
98 }
99 else if (Symmetry == "HEXAGONAL")
100 {
101 // Load hexagonal parameters (class 622)
102 CrystalSym = LatticeSystems::Hexagonal;
103
104 nsym = 12;
105 Theta_min = 0 * (Pi / 180);
106 Theta_max = 30 * (Pi / 180);
107 Phi_min = 0 * (Pi / 180);
108 Phi_max = Pi / 2;
109
110 CrystalSymmetries.Allocate(nsym);
111
112 CrystalSymmetries[0].set( 1, 0, 0, 0, 1, 0, 0, 0, 1);
113 CrystalSymmetries[1].set(-0.5, a, 0, -a, -0.5, 0, 0, 0, 1);
114 CrystalSymmetries[2].set(-0.5,-a, 0, a, -0.5, 0, 0, 0, 1);
115 CrystalSymmetries[3].set( 0.5, a, 0, -a, 0.5, 0, 0, 0, 1);
116 CrystalSymmetries[4].set(-1, 0, 0, 0,-1, 0, 0, 0, 1);
117 CrystalSymmetries[5].set( 0.5,-a, 0, a, 0.5, 0, 0, 0, 1);
118
119 CrystalSymmetries[6].set(-0.5, -a, 0, -a, 0.5, 0, 0, 0, -1);
120 CrystalSymmetries[7].set( 1, 0, 0, 0, -1, 0, 0, 0, -1);
121 CrystalSymmetries[8].set(-0.5, a, 0, a, 0.5, 0, 0, 0, -1);
122 CrystalSymmetries[9].set( 0.5, a, 0, a,-0.5, 0, 0, 0, -1);
123 CrystalSymmetries[10].set(-1, 0, 0, 0, 1, 0, 0, 0, -1);
124 CrystalSymmetries[11].set(0.5, -a, 0, -a,-0.5, 0, 0, 0, -1);
125
126 ConsoleOutput::WriteSimple("CrystalSymmetry::Hexagonal");
127 }
128 else if (Symmetry == "TETRAGONAL")
129 {
130 // Load tetragonal parameters (class 422)
131 CrystalSym = LatticeSystems::Tetragonal;
132
133 nsym = 8;
134 Theta_min = 0 * (Pi / 180);
135 Theta_max = 45 * (Pi / 180);
136 Phi_min = 0 * (Pi / 180);
137 Phi_max = Pi / 2;
138
139 CrystalSymmetries.Allocate(nsym);
140
141 CrystalSymmetries[0].set( 1, 0, 0, 0, 1, 0, 0, 0, 1);
142 CrystalSymmetries[1].set(-1, 0, 0, 0, 1, 0, 0, 0,-1);
143 CrystalSymmetries[2].set( 1, 0, 0, 0,-1, 0, 0, 0,-1);
144 CrystalSymmetries[3].set(-1, 0, 0, 0,-1, 0, 0, 0, 1);
145 CrystalSymmetries[4].set( 0, 1, 0,-1, 0, 0, 0, 0, 1);
146 CrystalSymmetries[5].set( 0,-1, 0, 1, 0, 0, 0, 0, 1);
147 CrystalSymmetries[6].set( 0, 1, 0, 1, 0, 0, 0, 0,-1);
148 CrystalSymmetries[7].set( 0,-1, 0,-1, 0, 0, 0, 0,-1);
149
150 ConsoleOutput::WriteSimple("CrystalSymmetry::Tetragonal");
151 }
152 else if (Symmetry == "TRIGONAL" or Symmetry == "RHOMBOHEDRAL")
153 {
154 // Load trigonal parameters (class 32)
155 CrystalSym = LatticeSystems::Trigonal;
156
157 nsym = 6;
158 Theta_min = 0 * (Pi / 180);
159 Theta_max = 60 * (Pi / 180);
160 Phi_min = 0 * (Pi / 180);
161 Phi_max = Pi / 2;
162
163 CrystalSymmetries.Allocate(nsym);
164
165 CrystalSymmetries[0].set( 1, 0, 0, 0, 1, 0, 0, 0, 1);
166 CrystalSymmetries[1].set(-0.5, a, 0,-a, -0.5, 0, 0, 0, 1);
167 CrystalSymmetries[2].set(-0.5,-a, 0, a, -0.5, 0, 0, 0, 1);
168 CrystalSymmetries[3].set( 0.5, a, 0, a, -0.5, 0, 0, 0,-1);
169 CrystalSymmetries[4].set(-1, 0, 0, 0, 1, 0, 0, 0, 1);
170 CrystalSymmetries[5].set( 0.5,-a, 0,-a, -0.5, 0, 0, 0,-1);
171
172 ConsoleOutput::WriteSimple("CrystalSymmetry::Trigonal");
173 }
174 else if (Symmetry == "ORTHORHOMBIC")
175 {
176 // Load orthorhombic parameters (class 22)
177 CrystalSym = LatticeSystems::Orthorhombic;
178
179 nsym = 4;
180 Theta_min = 0 * (Pi / 180);
181 Theta_max = 90 * (Pi / 180);
182 Phi_min = 0 * (Pi / 180);
183 Phi_max = Pi / 2;
184
185 CrystalSymmetries.Allocate(nsym);
186
187 CrystalSymmetries[0].set( 1, 0, 0, 0, 1, 0, 0, 0, 1);
188 CrystalSymmetries[1].set(-1, 0, 0, 0, 1, 0, 0, 0,-1);
189 CrystalSymmetries[2].set( 1, 0, 0, 0,-1, 0, 0, 0, 1);
190 CrystalSymmetries[3].set(-1, 0, 0, 0,-1, 0, 0, 0, 1);
191
192 ConsoleOutput::WriteSimple("CrystalSymmetry::Orthorhombic");
193 }
194 else if (Symmetry == "MONOCLINIC")
195 {
196 // Load monoclinic parameters (class 2)
197 CrystalSym = LatticeSystems::Monoclinic;
198
199 nsym = 2;
200 Theta_min = 0 * (Pi / 180);
201 Theta_max = 180 * (Pi / 180);
202 Phi_min = 0 * (Pi / 180);
203 Phi_max = Pi / 2;
204
205 CrystalSymmetries.Allocate(nsym);
206
207 CrystalSymmetries[0].set( 1, 0, 0, 0, 1, 0, 0, 0, 1);
208 CrystalSymmetries[1].set(-1, 0, 0, 0, 1, 0, 0, 0,-1);
209
210 ConsoleOutput::WriteSimple("CrystalSymmetry::Monoclinic");
211 }
212 else if (Symmetry == "TRICLINIC")
213 {
214 // Load triclinic parameters (class 1)
215 CrystalSym = LatticeSystems::Triclinic;
216
217 nsym = 1;
218 Theta_min = 0 * (Pi / 180);
219 Theta_max = 360 * (Pi / 180);
220 Phi_min = 0 * (Pi / 180);
221 Phi_max = Pi / 2;
222
223 CrystalSymmetries.Allocate(nsym);
224
225 CrystalSymmetries[0].set(1, 0, 0, 0, 1, 0, 0, 0, 1);
226
227 ConsoleOutput::WriteSimple("CrystalSymmetry::Triclinic");
228 }
229 else
230 {
231 nsym = 0;
232 ConsoleOutput::WriteWarning("No or wrong CrystalSymmetry specified!\nThe default \"Cubic\" model is used!", thisclassname, "ReadInput()");
233 }
234
235 ConsoleOutput::WriteLine();
236 ConsoleOutput::WriteBlankLine();
237}
