33void GrandPotentialPhaseDensity_CompressibleFluid::Initialize(Settings& locSettings)
34{
35 GrandPotentialPhaseDensity::Initialize(locSettings);
36
37 if (Ncomp != 1)
38 {
39 ConsoleOutput::WriteExit("Model is only suitable for single component simulation!", thisclassname, "Initialize");
40 OP_Exit(EXIT_FAILURE);
41 }
42
43 BulkModulus .assign(Ncomp, 0.0);
44 ReferenceChemicalPotential.assign(Ncomp, 0.0);
45 ReferenceMolarVolume .assign(Ncomp, 0.0);
46 ReferencePressure .assign(Ncomp, 0.0);
47 MolarMass .assign(Ncomp, 0.0);
48}
49void GrandPotentialPhaseDensity_CompressibleFluid::ReadInput(std::stringstream& InputFile, int moduleLocation)
50{
51 ConsoleOutput::Write("");
52 ConsoleOutput::Write("Compressible non-gas grand potential density properties of "+PhaseNames[PhaseIdx]);
53 std::string converter = "_"+std::to_string(PhaseIdx)+"_"+ElementNames[0];
54 BulkModulus [0] = FileInterface::ReadParameterD(InputFile, moduleLocation, "K" +converter);
55 ReferenceChemicalPotential [0] = FileInterface::ReadParameterD(InputFile, moduleLocation, "MU0"+converter);
56 ReferencePressure [0] = FileInterface::ReadParameterD(InputFile, moduleLocation, "P0" +converter);
57 ReferenceMolarVolume [0] = FileInterface::ReadParameterD(InputFile, moduleLocation, "VM0"+converter);
58 MolarMass [0] = FileInterface::ReadParameterD(InputFile, moduleLocation, "M_" +ElementNames[0]);
59 GravityAcceleration = FileInterface::ReadParameterD(InputFile, moduleLocation, "g");
60 ConsoleOutput::Write("");
61}
