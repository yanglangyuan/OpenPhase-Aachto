169 void SetNoCenter(const double UserStencil[3][3][3], double dx,
170 int dNx = 1, int dNy = 1, int dNz = 1)
171 {
172 if(not StencilElements.empty())
173 {
174 StencilElements.clear();
175 }
176
177 for(int x = -dNx; x <= dNx; ++x)
178 for(int y = -dNy; y <= dNy; ++y)
179 for(int z = -dNz; z <= dNz; ++z)
180 if(x != 0 or y != 0 or z!= 0)
181 if (UserStencil[x+1][y+1][z+1] != 0.0)
182 {
183 LaplacianStencilEntry locElement;
184 locElement.di = x;
185 locElement.dj = y;
186 locElement.dk = z;
187
188 locElement.weight = UserStencil[x+1][y+1][z+1]/(dx*dx);
189 StencilElements.push_back(locElement);
190 }
191 weight_zero = UserStencil[1][1][1];
192 }
193 void SetNoCenter(const double UserStencil[3][3][3], GridParameters& Dimensions)
194 {
195 SetNoCenter(UserStencil,Dimensions.dx, Dimensions.dNx, Dimensions.dNy, Dimensions.dNz);
196 }
197 void Set(const double UserStencil[3][3][3], double dx,
198 int dNx = 1, int dNy = 1, int dNz = 1)
199 {
200 if(not StencilElements.empty())
201 {
202 StencilElements.clear();
203 }
204
205 for(int x = -dNx; x <= dNx; ++x)
206 for(int y = -dNy; y <= dNy; ++y)
207 for(int z = -dNz; z <= dNz; ++z)
208 if (UserStencil[x+1][y+1][z+1] != 0.0)
209 {
210 LaplacianStencilEntry locElement;
211 locElement.di = x;
212 locElement.dj = y;
213 locElement.dk = z;
214
215 locElement.weight = UserStencil[x+1][y+1][z+1]/(dx*dx);
216 StencilElements.push_back(locElement);
217 }
218 weight_zero = UserStencil[1][1][1];
219 }
220 void Set(const double UserStencil[3][3][3], GridParameters& Dimensions)
221 {
222 Set(UserStencil,Dimensions.dx, Dimensions.dNx, Dimensions.dNy, Dimensions.dNz);
223 }
