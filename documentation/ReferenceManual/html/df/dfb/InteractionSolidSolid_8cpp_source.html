35{

81void InteractionSolidSolid::ReadInput(std::stringstream& inp)
82{
83 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
84
85 std::string tmp = FileInterface::ReadParameterK(inp, moduleLocation, "Model");
86
87 if (tmp == "STANDARD") Model = SolidSolidInteractionModel::Standard;
88 else if (tmp == "WANG") Model = SolidSolidInteractionModel::Wang;
89 else
90 {
91 ConsoleOutput::WriteExit("Nonexistent solid solid interaction model, "+tmp+", selected", thisclassname, "ReadInput()");
92 OP_Exit(EXIT_FAILURE);
93 }
94
95 switch (Model)
96 {
97 case SolidSolidInteractionModel::Wang:
98 strength_wang = FileInterface::ReadParameterD(inp, moduleLocation, "SWang");
99 break;
100
101 default:
102 order = FileInterface::ReadParameterI(inp, moduleLocation, "Order", false, 4);
103 cutoff = FileInterface::ReadParameterI(inp, moduleLocation, "Cutoff", false, 3);
104 strength = FileInterface::ReadParameterD(inp, moduleLocation, "Strength");
105
106 for (size_t n = 0; n < Nphases; ++n)
107 if (PhaseAggregateStates[n] == AggregateStates::Solid)
108 for (size_t m = n; m < Nphases; ++m)
109 if (PhaseAggregateStates[m] == AggregateStates::Solid)
110 {
111 std::stringstream TEMPCstr;
112 TEMPCstr << "Elastic[" << n << "][" << m << "]";
113 elastic[n][m] = FileInterface::ReadParameterD(inp, moduleLocation, TEMPCstr.str());
114 }
115 }
116}

118void InteractionSolidSolid::CalculateLocal(
119 const int i, const int j, const int k,
120 PhaseField& Phase,
121 const BoundaryConditions& BC,
122 const double dt) const
123{
124 const double dV = Grid.CellVolume(true);
125 //const NodeAB<dVector3,dVector3> locNormal = Phase.Normals(i,j,k);
126
127 for (auto alpha = Phase.Fields(i,j,k).cbegin();
128 alpha != Phase.Fields(i,j,k).cend(); ++alpha)
129 if (applicable(Phase.FieldsProperties[alpha->index]))
130 {
131 const size_t pidxA = Phase.FieldsProperties[alpha->index].Phase;
132 const dVector3& pos_cm_alpha = Phase.FieldsProperties[alpha->index].Rcm;
133
134 for (int ii = -cutoff*Grid.dNx; ii <= cutoff*Grid.dNx; ++ii)
135 for (int jj = -cutoff*Grid.dNy; jj <= cutoff*Grid.dNy; ++jj)
136 for (int kk = -cutoff*Grid.dNz; kk <= cutoff*Grid.dNz; ++kk)
137 if (ii*ii+jj*jj+kk*kk > 0)
138 if (ii*ii+jj*jj+kk*kk < cutoff*cutoff)
139 {
140 // Calculate position of collision event
141 const dVector3 loc_pos_c = {i+0.5*ii,j+0.5*jj,k+0.5*kk};
142 const dVector3 pos_c = Tools::Position(loc_pos_c, Phase.Grid.OffsetX, Phase.Grid.OffsetY, Phase.Grid.OffsetZ);
143
144 for(auto beta = Phase.Fields(i+ii,j+jj,k+kk).cbegin();
145 beta != Phase.Fields(i+ii,j+jj,k+kk).cend(); ++beta)
146 if (applicable(Phase.FieldsProperties[beta->index]))
147 if(alpha->index != beta->index)
148 {
149 const double r = sqrt(ii*ii+jj*jj+kk*kk); // Distance between (i,j,k) and (i+ii,j+jj,k+kk).
150 const size_t pidxB = Phase.FieldsProperties[beta->index].Phase;
151 const dVector3& pos_cm_beta = Phase.FieldsProperties[beta->index].Rcm;
152
153 // Calculate collision velocity v_c projected on connection between (i,j,k) and (i+ii,j+jj,k+kk).
154 const dVector3 r_c_alpha = Tools::Distance(pos_c, pos_cm_alpha, Phase.Grid.TotalNx, Phase.Grid.TotalNy, Phase.Grid.TotalNz, BC)*Grid.dx;
155 const dVector3 r_c_beta = Tools::Distance(pos_c, pos_cm_beta, Phase.Grid.TotalNx, Phase.Grid.TotalNy, Phase.Grid.TotalNz, BC)*Grid.dx;
156 const dVector3 v_alpha = Phase.FieldsProperties[alpha->index].Vcm + Phase.FieldsProperties[alpha->index].aVel.cross(r_c_alpha);
157 const dVector3 v_beta = Phase.FieldsProperties[beta ->index].Vcm + Phase.FieldsProperties[beta ->index].aVel.cross(r_c_beta );
158 const dVector3 normal = {ii/r,jj/r,kk/r};
159 const double v_c = normal*(v_alpha-v_beta);
160
161 // Calculate force density
162 const double Vol1 = 4.0/3.0*Pi*alpha->value*alpha->value*alpha->value;
163 const double Vol2 = 4.0/3.0*Pi*beta ->value*beta ->value*beta ->value;
164 const double v0 = Grid.dx/dt; // Reference velocity to get physical dimensions right
165 double dEnergyDensity_dr = 0.5*strength*Vol1*Vol2*order*std::pow((r-cutoff)/cutoff,order-1)/cutoff;
166 dEnergyDensity_dr = elastic[pidxA][pidxB]*dEnergyDensity_dr + (1.0-elastic[pidxA][pidxB])*v_c/v0*dEnergyDensity_dr;
167 const dVector3 force_density = {dEnergyDensity_dr*ii/r,dEnergyDensity_dr*jj/r,dEnergyDensity_dr*kk/r};
168
169 // Apply force density
170 const dVector3 pos_alpha = Tools::Position(dVector3({double(i), double(j), double(k)}), Phase.Grid.OffsetX, Phase.Grid.OffsetY, Phase.Grid.OffsetZ);
171 const dVector3 pos_beta = Tools::Position(dVector3({double(i+ii), double(j+jj), double(k+kk)}), Phase.Grid.OffsetX, Phase.Grid.OffsetY, Phase.Grid.OffsetZ);
172 const dVector3 r_alpha = Tools::Distance(pos_alpha, pos_cm_alpha, Phase.Grid.TotalNx, Phase.Grid.TotalNy, Phase.Grid.TotalNz, BC)*Grid.dx;
173 const dVector3 r_beta = Tools::Distance(pos_beta, pos_cm_beta, Phase.Grid.TotalNx, Phase.Grid.TotalNy, Phase.Grid.TotalNz, BC)*Grid.dx;
174 #ifdef _OPENMP
175 #pragma omp critical
176 #endif
177 {
178 Phase.FieldsProperties[alpha->index].Force += force_density*dV;
179 Phase.FieldsProperties[beta ->index].Force -= force_density*dV;
180 Phase.FieldsProperties[alpha->index].Torque += r_alpha.cross(force_density)*dV;
181 Phase.FieldsProperties[beta ->index].Torque -= r_beta .cross(force_density)*dV;
182 }
183 }
184 }
185 }

188void InteractionSolidSolid::CalculateLocalWang(
189 const int i, const int j, const int k,
190 PhaseField& Phase,
191 const BoundaryConditions& BC,
192 const std::function<double(int,int,int)>& MassDensity) const
193{
194 const double dV = Grid.CellVolume(true);
195 //The calculation of the force density is based on, Wang, Y. U. (2006).
196 //Acta materialia, 54(4), 953-961.
197
198 const dVector3 pos = Tools::Position(dVector3({double(i), double(j), double(k)}), Phase.Grid.OffsetX, Phase.Grid.OffsetY, Phase.Grid.OffsetZ);
199
200 double locMassDensity = MassDensity(i,j,k); // actual local mass
201 double locMassDensity_Uncompressed = 0.0; // local equilibrium mass density without pressure
202 for (auto alpha = Phase.Fields(i,j,k).cbegin(); alpha != Phase.Fields(i,j,k).cend(); ++alpha)
203 {
204 locMassDensity_Uncompressed += Phase.FieldsProperties[alpha->index].Density*alpha->value;
205 }
206
207 for (auto alpha = Phase.Fields(i,j,k).cbegin(); alpha != Phase.Fields(i,j,k).cend(); ++alpha)
208 if (applicable(Phase.FieldsProperties[alpha->index]))
209 {
210 const dVector3 gradient_alpha = Phase.Fields(i,j,k).get_gradient(alpha->index);
211 const dVector3 pos_cm_alpha = Phase.FieldsProperties[alpha->index].Rcm;
212 const dVector3 r_alpha = Tools::Distance(pos, pos_cm_alpha, Phase.Grid.TotalNx, Phase.Grid.TotalNy, Phase.Grid.TotalNz, BC)*Grid.dx;
213 for (auto beta = alpha + 1; beta != Phase.Fields(i,j,k).cend(); ++beta)
214 if (applicable(Phase.FieldsProperties[beta->index]))
215 {
216 const dVector3 gradient_beta = Phase.Fields(i,j,k).get_gradient(beta->index);
217 const dVector3 ForceDensity = (gradient_alpha-gradient_beta)*(locMassDensity-locMassDensity_Uncompressed)*strength_wang;
218 const dVector3 pos_cm_beta = Phase.FieldsProperties[beta->index].Rcm;
219 const dVector3 r_beta = Tools::Distance(pos, pos_cm_beta, Phase.Grid.TotalNx, Phase.Grid.TotalNy, Phase.Grid.TotalNz, BC)*Grid.dx;
220 #ifdef _OPENMP
221 #pragma omp critical
222 #endif
223 {
224 Phase.FieldsProperties[alpha->index].Force += ForceDensity*dV;
225 Phase.FieldsProperties[beta ->index].Force -= ForceDensity*dV;
226 Phase.FieldsProperties[alpha->index].Torque += r_alpha.cross(ForceDensity)*dV;
227 Phase.FieldsProperties[beta ->index].Torque -= r_beta .cross(ForceDensity)*dV;
228 }
229 }
230 }

233void InteractionSolidSolid::Calculate(PhaseField& Phase,
234 const BoundaryConditions& BC,
235 const std::function<double(int,int,int)>& MassDensity,
236 const double dt) const
237{
238 switch (Model)
239 {
240 case SolidSolidInteractionModel::Wang:
241 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
242 {
243 CalculateLocalWang(i,j,k,Phase,BC,MassDensity);
244 }
245 OMP_PARALLEL_STORAGE_LOOP_END
246 break;
247
248 default:
249 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
250 {
251 CalculateLocal(i,j,k,Phase,BC,dt);
252 }
253 OMP_PARALLEL_STORAGE_LOOP_END
254 }
255}
