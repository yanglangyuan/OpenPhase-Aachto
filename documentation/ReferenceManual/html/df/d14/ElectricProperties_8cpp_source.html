43void ElectricProperties::Initialize(Settings& locSettings, std::string ObjectNameSuffix)
44{
45 thisclassname = "ElectricProperties";
46 thisobjectname = thisclassname + ObjectNameSuffix;
47
48 Grid = locSettings.Grid;
49 Nphases = locSettings.Nphases;
50 Ncomp = locSettings.Ncomp;

65void ElectricProperties::ReadInput(const std::string InputFileName)
66{
67 ConsoleOutput::WriteLineInsert("Electric properties input");
68 ConsoleOutput::WriteStandard("Source", InputFileName);
69
70 std::fstream inpF(InputFileName.c_str(), std::ios::in | std::ios_base::binary);
71
72 if (!inpF)
73 {
74 ConsoleOutput::WriteExit("File \"" + InputFileName + "\" could not be opened", thisclassname, "ReadInput()");
75 OP_Exit(EXIT_FAILURE);
76 };
77
78 std::stringstream inp;
79 inp << inpF.rdbuf();
80
81 ReadInput(inp);
82
83 inpF.close();
84}

86void ElectricProperties::ReadInput(std::stringstream& inp)
87{
88 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
89 for(size_t comp = 0; comp < Ncomp; comp++)
90 {
91 const std::string name = "MolarCharge_" + comp;
92 MolarCharge[comp] = FileInterface::ReadParameterD(inp, moduleLocation, name, false, 0.0);
93 }

96void ElectricProperties::CalculateChargeDensity(const Composition& Cx)
97{
98 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,ChargeDensity,ChargeDensity.Bcells(),)
99 {
100 ChargeDensity(i,j,k) = 0.0;
101 for(size_t comp = 0; comp < Ncomp; comp++)
102 {
103 const double MolarComponentDensity = Cx.MoleFractionsTotal(i,j,k,{comp})*Cx.PT.GetData(Cx.Component[comp].Name).MolarVolume;
104 ChargeDensity(i,j,k) += MolarCharge[comp] * MolarComponentDensity;
105 }
106 }
107 OMP_PARALLEL_STORAGE_LOOP_END
108}

129void ElectricProperties::WriteVTK(const Settings& locSettings, const int tStep, const int precision) const
130{
131 std::vector<VTK::Field_t> ListOfFields;
132 ListOfFields.push_back((VTK::Field_t) {"Electric Charge Density [A*s/m^3]", [this](int i,int j,int k){return ChargeDensity(i,j,k);}});
133 ListOfFields.push_back((VTK::Field_t) {"Electric Potential [V]", [this](int i,int j,int k){return ElectricPotential(i,j,k);}});
134 ListOfFields.push_back((VTK::Field_t) {"Wave Vector [1/m]", [this](int i,int j,int k){return WaveVector(i,j,k);}});
135
136 std::string Filename = FileInterface::MakeFileName(locSettings.VTKDir, "ElectricProperties", tStep, ".vts");
137
138 VTK::Write(Filename, locSettings, ListOfFields);
139}
