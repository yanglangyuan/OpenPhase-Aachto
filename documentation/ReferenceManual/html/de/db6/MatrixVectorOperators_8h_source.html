298extern inline void Eigensystem(dMatrix3x3 M, dMatrix3x3& Eigenvectors,
299 dMatrix3x3& Eigenvalues,
300 double accuracy = FLT_EPSILON)
301{
302 // M MUST BE _SYMMETRIC_
303 assert((std::fabs(M(0,1) - M(1,0)) <= std::numeric_limits<double>::epsilon() and
304 std::fabs(M(0,2) - M(2,0)) <= std::numeric_limits<double>::epsilon() and
305 std::fabs(M(1,2) - M(2,1)) <= std::numeric_limits<double>::epsilon()) and
306 "Matrix is not symmetric in MatrixVectorOperators::Eigensystem()");
307
308 // Set elements which are already below specified accuracy to zero:
309 for(int x = 0; x < 3; x++)
310 for(int y = 0; y < 3; y++)
311 {
312 if(fabs(M(x,y)) < accuracy)
313 {
314 M(x,y) = 0.0;
315 }
316 }
317
318 Eigenvectors.set_to_unity();
319
320 do
321 {
322 // Maximum off-diagonal element value:
323 double maxValue = 0.0;
324 // Givens rotation angle:
325 double theta = 0.0;
326 // Indices of the maximum off-diagonal element:
327 int indI = -1;
328 int indJ = -1;
329
330 // Find maximum off-diagonal element:
331 for(int i = 0; i < 3; i++)
332 for(int j = 0; j < 3; j++)
333 if((i != j))
334 {
335 if(fabs(M(i,j)) > fabs(maxValue))
336 {
337 maxValue = M(i,j);
338 indI = i;
339 indJ = j;
340 }
341 }
342
343 // Perform iteration if there are non-zero off-diagonal elements
344 if(indI != -1 and indJ != -1)
345 {
346 // Set rotation angle for Givens rotation:
347 if(M(indI,indI) == M(indJ,indJ))
348 {
349 if(M(indI,indJ) > 0) theta = Pi/4.0;
350 else theta = -Pi/4.0;
351 }
352 else
353 {
354 theta = 0.5 * atan((2 * maxValue) / (M(indI,indI) - M(indJ,indJ)));
355 }
356
357 // Set Givens rotation matrix:
358 dMatrix3x3 S;
359 S.set_to_unity();
360 S(indI,indI) = cos(theta);
361 S(indJ,indJ) = cos(theta);
362 S(indI,indJ) = -sin(theta);
363 S(indJ,indI) = sin(theta);
364
365 // Compute update of the eigenvalue matrix:
366 M = S.transposed() * M * S;
367
368 // Compute update of the eigenvector matrix:
369 Eigenvectors = Eigenvectors * S;
370
371 // Set elements which have reached specified accuracy to zero:
372 for(int x = 0; x < 3; x++)
373 for(int y = 0; y < 3; y++)
374 {
375 if(fabs(M(x,y)) < accuracy)
376 {
377 M(x,y) = 0.0;
378 }
379 }
380 }
381 }
382 while (M(0,1) != 0.0 or M(0,2) != 0.0 or M(1,2) != 0.0 or
383 M(1,0) != 0.0 or M(2,0) != 0.0 or M(2,1) != 0.0);
384
385 // Assign eigenvalues:
386 Eigenvalues = M;
387}

389extern inline dMatrix3x3 sqrtM3x3(const dMatrix3x3& M,
390 double accuracy = FLT_EPSILON)
391{
392 dMatrix3x3 Eigenvectors;
393 dMatrix3x3 Eigenvalues;
394
395 Eigensystem(M,Eigenvectors,Eigenvalues,accuracy);
396
397 for(int i = 0; i < 3; i++)
398 {
399 Eigenvalues(i,i) = sqrt(Eigenvalues(i,i));
400 }
401 // sqrt(M) = Eigenvectors_column . sqrt(Eigenvalues) . Eigenvectors_row
402 return Eigenvectors * Eigenvalues * Eigenvectors.transposed();
403}

466extern inline dMatrix3x3 AlignBaseAxes(dMatrix3x3 M)
467{
468 /*
469 * This function uses Givens rotation algorithm to zero (1,0), (2,0) and
470 * (2,1) elements of M.
471 */
472 double c = 0.0;
473 double s = 0.0;
474 double den = 0.0;
475 dMatrix3x3 G;
476
477 auto denom = [](double x, double y) {return sqrt(x*x + y*y);};
478
479 // Zeroing element (1,0)
480 if(M(1,0) > DBL_EPSILON or M(1,0) < -DBL_EPSILON)
481 {
482 den = denom (M(1,0),M(0,0));
483 if(den > DBL_EPSILON)
484 {
485 c = M(0,0) / den;
486 s = M(1,0) / den;
487 }
488 G(0,0) = c; G(0,1) = s; G(0,2) = 0;
489 G(1,0) = -s; G(1,1) = c; G(1,2) = 0;
490 G(2,0) = 0; G(2,1) = 0; G(2,2) = 1;
491 M = G*M;
492 }
493
494 // Zeroing element (2,0)
495 if(M(2,0) > DBL_EPSILON or M(2,0) < -DBL_EPSILON)
496 {
497 den = denom (M(2,0),M(0,0));
498 if(den > DBL_EPSILON)
499 {
500 c = M(0,0) / den;
501 s = M(2,0) / den;
502 }
503 G(0,0) = c; G(0,1) = 0; G(0,2) = s;
504 G(1,0) = 0; G(1,1) = 1; G(1,2) = 0;
505 G(2,0) = -s; G(2,1) = 0; G(2,2) = c;
506 M = G*M;
507 }
508
509 // Zeroing element (2,1)
510 if(M(2,1) > DBL_EPSILON or M(2,1) < -DBL_EPSILON)
511 {
512 den = denom (M(2,1),M(1,1));
513 if(den > DBL_EPSILON)
514 {
515 c = M(1,1) / den;
516 s = M(2,1) / den;
517 }
518 G(0,0) = 1; G(0,1) = 0; G(0,2) = 0;
519 G(1,0) = 0; G(1,1) = c; G(1,2) = s;
520 G(2,0) = 0; G(2,1) = -s; G(2,2) = c;
521 M = G*M;
522 }
523
524 return M;
525}
