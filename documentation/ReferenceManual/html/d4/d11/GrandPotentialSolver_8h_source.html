170 template<typename field_t>
171 double CalculateVolumeIntegral(field_t field, double dx) const
172 {
173 double local_value = 0.0;
174 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,ChemicalPotential,0,reduction(+:local_value))
175 {
176 local_value += field(i,j,k);
177 }
178 OMP_PARALLEL_STORAGE_LOOP_END
179 local_value*=dx*dx*dx;
180 #ifdef MPI_PARALLEL
181 double global_value = 0.0;
182 OP_MPI_Allreduce(&local_value, &global_value, 1, OP_MPI_DOUBLE, OP_MPI_SUM, OP_MPI_COMM_WORLD);
183 return global_value;
184 #else
185 return local_value;
186 #endif
187 }
