32void Quaternion::set(const dMatrix3x3& RotMatrix)
33{
34 assert(std::fabs((RotMatrix.transposed()*RotMatrix - dMatrix3x3::UnitTensor()).determinant()) <= std::numeric_limits<float>::epsilon() && "matrix is not orthogonal");
35
36 // Found at http://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/
37 RotationMatrix = RotMatrix;
38
39 double trace = RotMatrix.trace();
40
41 if(1.0 + trace > 0)
42 {
43 double ss = sqrt(trace + 1.0)*2.0;
44 set(0.25*ss,
45 (RotMatrix(2,1) - RotMatrix(1,2))/ss,
46 (RotMatrix(0,2) - RotMatrix(2,0))/ss,
47 (RotMatrix(1,0) - RotMatrix(0,1))/ss);
48 }
49 else
50 {
51 if (RotMatrix(0,0) > RotMatrix(1,1) && RotMatrix(0,0) > RotMatrix(2,2))
52 {
53 double ss = 2.0 * sqrt( 1.0 + RotMatrix(0,0) - RotMatrix(1,1) - RotMatrix(2,2));
54 set((RotMatrix(2,1) - RotMatrix(1,2))/ss,
55 0.25 * ss,
56 (RotMatrix(0,1) + RotMatrix(1,0))/ss,
57 (RotMatrix(0,2) + RotMatrix(2,0))/ss);
58 }
59 else if (RotMatrix(1,1) > RotMatrix(2,2))
60 {
61 double ss = 2.0 * sqrt(1.0 + RotMatrix(1,1) - RotMatrix(0,0) - RotMatrix(2,2));
62 set((RotMatrix(0,2) - RotMatrix(2,0))/ss,
63 (RotMatrix(0,1) + RotMatrix(1,0))/ss,
64 0.25 * ss,
65 (RotMatrix(1,2) + RotMatrix(2,1))/ss);
66 }
67 else
68 {
69 double ss = 2.0 * sqrt(1.0 + RotMatrix(2,2) - RotMatrix(0,0) - RotMatrix(1,1));
70 set((RotMatrix(1,0) - RotMatrix(0,1))/ss,
71 (RotMatrix(0,2) + RotMatrix(2,0))/ss,
72 (RotMatrix(1,2) + RotMatrix(2,1))/ss,
73 0.25 * ss);
74 }
75 }
76}

78void Quaternion::setRotationMatrix()
79{
80 if(length() < DBL_EPSILON)
81 {
82 RotationMatrix.set_to_unity();
83 }
84 else
85 {
86 Quaternion Qcopy = normalized();
87
88 RotationMatrix(0,0) = 1.0 - 2.0*(Qcopy.v[1]*Qcopy.v[1] + Qcopy.v[2]*Qcopy.v[2]);
89 RotationMatrix(0,1) = 2.0*(Qcopy.v[0]*Qcopy.v[1] - Qcopy.s*Qcopy.v[2]);
90 RotationMatrix(0,2) = 2.0*(Qcopy.v[0]*Qcopy.v[2] + Qcopy.s*Qcopy.v[1]);
91 RotationMatrix(1,0) = 2.0*(Qcopy.v[0]*Qcopy.v[1] + Qcopy.s*Qcopy.v[2]);
92 RotationMatrix(1,1) = 1.0 - 2.0*(Qcopy.v[0]*Qcopy.v[0] + Qcopy.v[2]*Qcopy.v[2]);
93 RotationMatrix(1,2) = 2.0*(Qcopy.v[1]*Qcopy.v[2] - Qcopy.s*Qcopy.v[0]);
94 RotationMatrix(2,0) = 2.0*(Qcopy.v[0]*Qcopy.v[2] - Qcopy.s*Qcopy.v[1]);
95 RotationMatrix(2,1) = 2.0*(Qcopy.v[1]*Qcopy.v[2] + Qcopy.s*Qcopy.v[0]);
96 RotationMatrix(2,2) = 1.0 - 2.0*(Qcopy.v[0]*Qcopy.v[0] + Qcopy.v[1]*Qcopy.v[1]);
97 }
98}

100dMatrix3x3 Quaternion::getRotationMatrix(const bool Active)
101{
102 dMatrix3x3 RotationMatrix;
103
104 Quaternion Qcopy = normalized();
105
106 double q = Qcopy.s * Qcopy.s - (Qcopy.v[0] * Qcopy.v[0] + Qcopy.v[1] * Qcopy.v[1] + Qcopy.v[2] * Qcopy.v[2]);
107
108 if(Active)
109 {
110 RotationMatrix(0,0) = q + 2.0 * Qcopy.v[0] * Qcopy.v[0];
111 RotationMatrix(0,1) = 2.0 * (Qcopy.v[0] * Qcopy.v[1] - Qcopy.s * Qcopy.v[2]);
112 RotationMatrix(0,2) = 2.0 * (Qcopy.v[0] * Qcopy.v[2] + Qcopy.s * Qcopy.v[1]);
113 RotationMatrix(1,0) = 2.0 * (Qcopy.v[1] * Qcopy.v[0] + Qcopy.s * Qcopy.v[2]);
114 RotationMatrix(1,1) = q + 2.0 *Qcopy.v[1] * Qcopy.v[1];
115 RotationMatrix(1,2) = 2.0 * (Qcopy.v[1] * Qcopy.v[2] - Qcopy.s * Qcopy.v[0]);
116 RotationMatrix(2,0) = 2.0 * (Qcopy.v[2] * Qcopy.v[0] - Qcopy.s * Qcopy.v[1]);
117 RotationMatrix(2,1) = 2.0 * (Qcopy.v[2] * Qcopy.v[1] + Qcopy.s * Qcopy.v[0]);
118 RotationMatrix(2,2) = q + 2.0 *Qcopy.v[2] * Qcopy.v[2];
119 }
120 else
121 {
122 RotationMatrix(0,0) = q + 2.0 * Qcopy.v[0] * Qcopy.v[0];
123 RotationMatrix(0,1) = 2.0 * (Qcopy.v[0] * Qcopy.v[1] + Qcopy.s * Qcopy.v[2]);
124 RotationMatrix(0,2) = 2.0 * (Qcopy.v[0] * Qcopy.v[2] - Qcopy.s * Qcopy.v[1]);
125 RotationMatrix(1,0) = 2.0 * (Qcopy.v[1] * Qcopy.v[0] - Qcopy.s * Qcopy.v[2]);
126 RotationMatrix(1,1) = q + 2.0 *Qcopy.v[1] * Qcopy.v[1];
127 RotationMatrix(1,2) = 2.0 * (Qcopy.v[1] * Qcopy.v[2] + Qcopy.s * Qcopy.v[0]);
128 RotationMatrix(2,0) = 2.0 * (Qcopy.v[2] * Qcopy.v[0] + Qcopy.s * Qcopy.v[1]);
129 RotationMatrix(2,1) = 2.0 * (Qcopy.v[2] * Qcopy.v[1] - Qcopy.s * Qcopy.v[0]);
130 RotationMatrix(2,2) = q + 2.0 *Qcopy.v[2] * Qcopy.v[2];
131 }
132 return RotationMatrix;
133}

135Quaternion Quaternion::lerp(const Quaternion& rhSQ1, const Quaternion& rhSQ2, const double t)
136{
137 // linear interpolation between two quaternions, where 0.0 < t < 1.0
138 Quaternion result = (rhSQ1*(1.0 - t) + rhSQ2*t).normalized();
139 return result;
140}

142Quaternion Quaternion::slerp(const Quaternion& Qa, const Quaternion& Qb, const double t)
143{
144 // linear interpolation between two quaternions, where 0.0 < t < 1.0
145 // taken from http://www.euclideanspace.com/maths/algebra/realNormedAlgebra/quaternions/slerp/index.htm
146 Quaternion result;
147
148 double cosHalfTheta = Qa.s * Qb.s
149 + Qa.v[0] * Qb.v[0]
150 + Qa.v[1] * Qb.v[1]
151 + Qa.v[2] * Qb.v[2];
152 double sign = 1.0;
153 if (cosHalfTheta < 0)
154 {
155 sign = -1.0;
156 }
157 if(fabs(cosHalfTheta) >= 1.0) // both rotations equal
158 {
159 result = Qa;
160 return result;
161 }
162 else
163 {
164 double halfTheta = acos(sign*cosHalfTheta);
165 double sinHalfTheta = sqrt(1.0 - cosHalfTheta*cosHalfTheta);
166 if(fabs(sinHalfTheta) < 2.0*DBL_EPSILON) // the rotations are opposite of each other
167 {
168 result.s = (Qa.s *0.5 + sign*Qb.s *0.5);
169 result.v[0] = (Qa.v[0]*0.5 + sign*Qb.v[0]*0.5);
170 result.v[1] = (Qa.v[1]*0.5 + sign*Qb.v[1]*0.5);
171 result.v[2] = (Qa.v[2]*0.5 + sign*Qb.v[2]*0.5);
172 }
173 else // general case
174 {
175 double ratioA = sin((1 - t) * halfTheta) / sinHalfTheta;
176 double ratioB = sin(t * halfTheta) / sinHalfTheta;
177
178 result.s = (Qa.s * ratioA + sign*Qb.s * ratioB);
179 result.v[0] = (Qa.v[0] * ratioA + sign*Qb.v[0] * ratioB);
180 result.v[1] = (Qa.v[1] * ratioA + sign*Qb.v[1] * ratioB);
181 result.v[2] = (Qa.v[2] * ratioA + sign*Qb.v[2] * ratioB);
182 }
183 result.setRotationMatrix();
184 }
185 return result;
186}
