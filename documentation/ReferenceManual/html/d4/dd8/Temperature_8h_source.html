69class Temperature1Dextension
70{
71 public:
72 void Initialize(size_t size, iVector3 direction)
73 {
74 if(size <= 0) // Checks for sufficient extension size
75 {
76 std::stringstream message;
77 message << "1D temperature extension size = " << size << " is incorrect!\n"
78 << "It should be in the range [1, inf) for correct operation\n";
79 ConsoleOutput::WriteExit(message.str(), "Temperature1Dextension", "Initialize()");
80 OP_Exit(EXIT_FAILURE);
81 }
82
83 Data.Allocate(size, 1);
84 DataTMP.Allocate(size, 1);
85 Direction = direction;
86 Qdot = 0.0;
87 }
88 Temperature1Dextension& operator=(const Temperature1Dextension& RHS)
89 {
90 Qdot = RHS.Qdot;
91 Data = RHS.Data;
92 DataTMP = RHS.DataTMP;
93 Direction = RHS.Direction;
94
95 return *this;
96 }
97 void setBC(BoundaryConditionTypes extBC)
98 {
99 switch(extBC)
100 {
101 case BoundaryConditionTypes::Periodic:
102 {
103 std::string message = std::string("If 1D temperature extension is active, periodic boundary conditions\n")
104 + std::string("along the extension dimension are not permitted.\n")
105 + std::string("Permitted boundary conditions: NoFlux, Free and Fixed.");
106 ConsoleOutput::WriteWarning(message, "Temperature1Dextension", "setBC()");
107 break;
108 }
109 case BoundaryConditionTypes::NoFlux:
110 case BoundaryConditionTypes::Mirror:
111 {
112 Data(size()) = Data(size()-1);
113 break;
114 }
115 case BoundaryConditionTypes::Free:
116 {
117 Data(size()) = Data(size()-1)*2.0 - Data(size()-2);
118 break;
119 }
120 default:
121 case BoundaryConditionTypes::Fixed:
122 {
123 break;
124 }
125 }
126 }
127 void setFixedBC(double value)
128 {
129 Data(size()) = value;
130 }
131 void moveFrame(const int dx, const BoundaryConditionTypes extBC)
132 {
133 if(dx > 0)
134 for(long int x = 0; x < (long int) Data.size(); x++)
135 {
136 Data(x) = Data(x+dx);
137 }
138 if(dx < 0)
139 for(long int x = (long int) Data.size() - 1; x >= 0; x--)
140 {
141 Data(x) = Data(x+dx);
142 }
143 setBC(extBC);
144 }
145 bool isActive() const
146 {
147 return Data.size() > 0;
148 }
149 size_t size() const
150 {
151 return Data.size();
152 }
153 void store_temporary(void)
154 {
155 DataTMP = Data;
156 }
157 void read(std::ifstream& out)
158 {
159 out.read(reinterpret_cast<char*>(Data.data()),Data.total_size()*sizeof(double));
160 }
161 void write(std::ofstream& out) const
162 {
163 out.write(reinterpret_cast<const char*>(Data.data()),Data.total_size()*sizeof(double));
164 }
165
166 double Qdot;
167 Storage1D<double,0> Data;
168 Storage1D<double,0> DataTMP;
169 iVector3 Direction;
170
171 protected:
172 private:
173
174};
