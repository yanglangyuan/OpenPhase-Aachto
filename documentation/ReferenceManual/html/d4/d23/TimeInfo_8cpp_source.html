83void TimeInfo::PrintWallClockSummary(void)
84{
85 if(counter)
86 {
87 ConsoleOutput::WriteLine("=");
88 ConsoleOutput::WriteSimple(TimerName);
89 ConsoleOutput::WriteLine("-");
90 double TotalTime = 0;
91 for (auto const& [message,time] : TimeMap) TotalTime += time;
92 double TotalConsumedTime = TotalTime/double(counter);
93 std::vector<std::pair<std::string, double>> tmp(TimeMap.begin(), TimeMap.end());
94 std::sort(tmp.begin(), tmp.end(), [](const std::pair<std::string, double>& a, const std::pair<std::string, double>& b) { return a.second > b.second; });
95 for (auto const& [message,time] : tmp)
96 {
97 if (message != "#")
98 {
99 std::stringstream showtime;
100
101 double SectionConsumedTime = time/double(counter);
102 showtime << std::fixed << std::setprecision(2) << std::scientific << SectionConsumedTime << " "
103 << std::fixed << std::setprecision(2) << ((SectionConsumedTime/TotalConsumedTime)*100.0) << " %";
104
105 ConsoleOutput::WriteStandard(message, showtime.str());
106 }
107 }
108 counter = 0;
109 for (auto & [message,time] : TimeMap) time = 0;
110 ConsoleOutput::WriteLine("-");
111 ConsoleOutput::WriteStandard("Total time per time step [s]", TotalConsumedTime);
112 ConsoleOutput::WriteLine("=");
113 }
114}
