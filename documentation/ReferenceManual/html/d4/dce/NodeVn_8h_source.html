110template<size_t N>
111inline void NodeVn<N>::set_comp(const size_t idx, const size_t ss, const double value)
112{
113 for (iterator i = begin(); i < end(); ++i)
114 if (i->index == idx)
115 {
116 i->value[ss] = value;
117 return;
118 }
119
120 VectorVnEntry<N> NewEntry;
121 NewEntry.index = idx;
122 for(size_t ii = 0; ii < N; ii++) NewEntry.value[ii] = 0.0;
123 NewEntry.value[ss] = value;
124
125 VectorVnFields.push_back(NewEntry);

194template<size_t N>
195inline void NodeVn<N>::change_index(const size_t n, const size_t m)
196{
197 // Set values and index of m to index n
198 // Warning: Values are averaged arithmetically!
199 if (n != m)
200 {
201 double returndV[N];
202 for (size_t ss = 0; ss < N; ss++) returndV[ss] = 0.0;
203
204 bool found = false;
205 for (auto i = cbegin(); i < cend(); ++i)
206 {
207 if (i->index == m)
208 {
209 for (size_t ss = 0; ss < N; ss++)
210 {
211 returndV[ss] = (i->value)[ss];
212 }
213 found = true;
214 break;
215 }
216 }
217 if (found == true)
218 {
219 for (auto i = begin(); i < end(); ++i)
220 if (i->index == m)
221 {
222 for (size_t ss = 0; ss < N; ss++)
223 {
224 (i->value)[ss] = 0.5*((i->value)[ss]+returndV[ss]);
225 }
226 i->index = n;
227 break;
228 }
229 }
230 }
231}
