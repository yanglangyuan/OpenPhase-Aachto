123template<class T>
124inline bool NodeA<T>::present(const size_t index) const
125{
126 for(auto i = Fields.cbegin(); i != Fields.cend(); ++i)
127 {
128 if(i->index == index) return true;
129 }
130 return false;
131}
132template<class T>
133inline void NodeA<T>::set_value(const size_t n, const T& value)
134{
135 for(auto i = Fields.begin(); i < Fields.end(); ++i)
136 if(i->index == n)
