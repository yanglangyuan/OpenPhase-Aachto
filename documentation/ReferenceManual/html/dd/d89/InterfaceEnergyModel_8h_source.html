196 void ReadInputFaceted(std::stringstream& inp, int moduleLocation, std::string counter)
197 {
198 bool FacetFamilies = true;
199 Nfacets = FileInterface::ReadParameterI(inp, moduleLocation, std::string("NfacetFamilies") + counter, false, 0);
200 if(Nfacets == 0)
201 {
202 Nfacets = FileInterface::ReadParameterI(inp, moduleLocation, std::string("Nfacets") + counter);
203
204 if(Nfacets == 0)
205 {
206 ConsoleOutput::WriteExit("Zero Nfacets specified!\nThe input should contain non-zero number of facets!", "InterfaceEnergyModel", "ReadInputFaceted()");
207 OP_Exit(EXIT_FAILURE);
208 }
209 FacetFamilies = false;
210 }
211
212 FacetVectors.resize(Nfacets);
213 FacetEnergy.resize(Nfacets,0.0);
214 FacetEpsilon.resize(Nfacets,0.0);
215
216 for(size_t n = 0 ; n < Nfacets; n++)
217 {
218 std::stringstream converter;
219 converter << "_" << n;
220 std::string counter1 = converter.str();
221
222 iVector3 locFacet = FileInterface::ReadParameterV3I(inp, moduleLocation, std::string("Facet") + counter + counter1, true);
223
224 if(FacetFamilies)
225 {
226 FacetVectors[n] = Tools::findIndexPermutations(locFacet);
227 }
228 else
229 {
230 FacetVectors[n].push_back(locFacet);
231 }
232
233 FacetEnergy[n] = FileInterface::ReadParameterD(inp, moduleLocation, std::string("SigmaF") + counter + counter1);
234 FacetEpsilon[n] = FileInterface::ReadParameterD(inp, moduleLocation, std::string("EpsilonEF") + counter + counter1);
235 }
236
237 for(size_t n = 0; n < Nfacets; n++)
238 {
239 Energy += FacetEnergy[n]/Nfacets;
240
241 if(FacetEnergy[n] > MaxEnergy)
242 {
243 MaxEnergy = FacetEnergy[n];
244 }
245 }
246 Model = InterfaceEnergyModels::Faceted;
247 };

308 double CalculateDisorientation(Quaternion& OrientationA, Quaternion& OrientationB)
309 {
310 double constant = 0.85; // Shear Modulus*Burgers vector/4*Pi*(1-v)
311 double misorientation = Tools::getDisorientationCubic(OrientationA, OrientationB);
312 double locEnergy = 0.0;
313
314 if (misorientation <= 0.349066)
315 {
316 locEnergy = misorientation*constant*(0.8 - log(misorientation));
317 }
318 else
319 {
320 locEnergy = 0.349066*constant*(0.8 - log(0.349066));
321 }
322 return locEnergy;
323 }

325 double Calculate(dVector3& locNormal)
326 {
327 double locEnergy = 0.0;
328 switch(Model)
329 {
330 case InterfaceEnergyModels::Ext:
331 {
332 break;
333 }
334 case InterfaceEnergyModels::Iso:
335 {
336 locEnergy = MaxEnergy;
337 break;
338 }
339 case InterfaceEnergyModels::Cubic: //
340 {
341 locEnergy = Energy*(1.0 + Epsilon1 * (1.5 - 2.5*(pow(locNormal[0], 4) +
342 pow(locNormal[1], 4) +
343 pow(locNormal[2], 4))));
344 break;
345 }
346 case InterfaceEnergyModels::CubicFull:
347 {
348 const double nx4 = std::pow(locNormal[0],4);
349 const double ny4 = std::pow(locNormal[1],4);
350 const double nz4 = std::pow(locNormal[2],4);
351 locEnergy = Sigma0*(1.0 + Epsilon1*(nx4+ny4+nz4));
352 break;
353 }
354 case InterfaceEnergyModels::HexBoettger: // Bottger, B.; Eiken, J.; Steinbach, I. Phase field simulation of equiaxed solidification in technical alloys. Acta Mater. 2006, 54, 26972704. [http://doi.org/10.1016/j.actamat.2006.02.008]
355 {
356 locEnergy = Energy*(1.0 - Epsilon1 *
357 ( std::pow(locNormal[0], 6) -
358 std::pow(locNormal[1], 6) -
359 15.0 * std::pow(locNormal[0], 4) * locNormal[1]*locNormal[1] +
360 15.0 * std::pow(locNormal[1], 4) * locNormal[0]*locNormal[0] +
361 (5.0 * std::pow(locNormal[2], 4) -
362 5.0 * std::pow(locNormal[2], 2) +
363 std::pow(locNormal[2], 6))));
364 break;
365 }
366 case InterfaceEnergyModels::HexSun: // Sun, D.; Mendelev, M.; Becker, C.; Kudin, K.; Haxhimali, T.; Asta, M.; Hoyt, J.; Karma, A.; Srolovitz, D. Crystal-melt interfacial free energies in hcp metals: A molecular dynamics study of Mg. Phys. Rev. B 2006, 73, 024116. [http://doi.org/10.1103/PhysRevB.73.024116]
367 {
368 locEnergy = Energy*(1.0
369 + Epsilon1*sqrt(5.0/16.0/Pi)*(3.0*locNormal[2]*locNormal[2] - 1.0)
370 + Epsilon2*3.0/16.0/sqrt(Pi)*(35.0*pow(locNormal[2],4) -
371 30.0*locNormal[2]*locNormal[2] + 3.0)
372 + Epsilon3*sqrt(13.0/Pi)/32.0*(231.0*pow(locNormal[2],6) -
373 315.0*pow(locNormal[2],4) +
374 105.0*locNormal[2]*locNormal[2] - 5.0)
375 + Epsilon4*sqrt(6006.0/Pi)/64.0*(pow(locNormal[0],6) -
376 15.0*pow(locNormal[0],4)*locNormal[1]*locNormal[1] +
377 15.0*locNormal[0]*locNormal[0]*pow(locNormal[1],4) -
378 pow(locNormal[1],6)));
379 break;
380 }
381 case InterfaceEnergyModels::HexYang: // Yang, M.; Xiong, S.; Guo, Z. Characterisation of the 3-D dendrite morphology of magnesium alloys using synchrotron X-ray tomography and 3-D phase-field modelling. Acta Mater. 2015, 92, 817. [http://doi.org/10.1016/j.actamat.2015.03.044]
382 {
383 locEnergy = Energy*(1.0
384 + Epsilon1 * pow((3.0*locNormal[2]*locNormal[2] - 1.0), 2)
385 + Epsilon2 * pow((locNormal[0]*locNormal[0]*locNormal[0] - 3.0*locNormal[0]*locNormal[1]*locNormal[1]),2)
386 * pow((9.0*locNormal[2]*locNormal[2] - 1.0 + Epsilon3),2));
387 break;
388 }
389 case InterfaceEnergyModels::FacetedEXP: // Inspired by M. Salvalaglio et al. Cryst. Growth Des. 2015, 15, 27872794 DOI: 10.1021/acs.cgd.5b00165
390 {
391 locEnergy = Energy;
392 for(size_t n = 0; n < Nfacets; n++)
393 for(size_t j = 0; j < FacetVectors[n].size() ; j++)
394 {
395 dVector3 locFacet = FacetVectors[n][j].normalized();
396 double cos_inc = locNormal * locFacet;
397 if(cos_inc > 0.0)
398 {
399 // part of the energy function
400 locEnergy -= Energy*FacetEpsilon[n]*pow(cos_inc,FacetPower[n]);
401 // second order derivative
402 locEnergy += Energy*FacetEpsilon[n]*FacetPower[n]*pow(cos_inc,FacetPower[n]);
404 //locEnergy -= FacetEnergy[n]*FacetEpsilon[n]*FacetPower[n]*(FacetPower[n]-1)*pow(cos_inc,FacetPower[n]-2)*(1.0 - cos_inc*cos_inc);
405 }
406 }
407 if(locEnergy < 0.0)
408 {
409 ConsoleOutput::WriteWarning("Negative interface stiffness encountered for InterfaceEnergyModels::FacetedEXP!", "InterfaceEnergyModels","Calculate()");
410 }
411
412 break;
413 }
414 case InterfaceEnergyModels::Faceted: // H. F. M. A. Salama, J. Kundin, O. Shchyglo, V. Mohles, K. Marquardt, I. Steinbach. Role of inclination dependence of grain boundary energy on the microstructure evolution during grain growth, Acta Materialia, 188, 641-651, (2020) [http://dx.doi.org/10.1016/j.actamat.2020.02.043]
415 {
416 double CosInc2 = 0.0;
417 double EnergyA = 0.0;
418 double EpsilonA = 0.0;
419
420 for(size_t n = 0; n < Nfacets; n++)
421 for(size_t j = 0; j < FacetVectors[n].size() ; j++)
422 {
423 dVector3 locFacet = FacetVectors[n][j].normalized();
424 double cos_inc = locNormal * locFacet;
425 double cos_inc2 = cos_inc*cos_inc;
426
427 if(cos_inc2 > CosInc2)
428 {
429 CosInc2 = cos_inc2;
430 EnergyA = FacetEnergy[n];
431 EpsilonA = FacetEpsilon[n];
432 }
433 }
434 double SinInc2 = 1.0 - CosInc2;
435 locEnergy = EnergyA * EpsilonA * EpsilonA * pow(sqrt(SinInc2 + EpsilonA*EpsilonA*CosInc2),-3.0);
436 break;
437 }
438 case InterfaceEnergyModels::FacetedFull:
439 {
440 double Inclination = Pi;
441 double EnergyA = 0.0;
442 double EpsilonA = 0.0;
443
444 // Calculating inclination angle between facet normal and interface normal
445 for(size_t n = 0; n < Nfacets; n++)
446 for(size_t j = 0; j < FacetVectors[n].size() ; j++)
447 {
448 const dVector3 locFacet = FacetVectors[n][j].normalized();
449 const double dotVec = std::acos(locNormal * locFacet);
450
451 if(std::fabs(dotVec) <= std::fabs(Inclination))
452 {
453 Inclination = dotVec;
454 EnergyA = FacetEnergy[n];
455 EpsilonA = FacetEpsilon[n];
456 }
457 }
458
459 const double sin_inc = std::sin(Inclination);
460 const double cos_inc = std::cos(Inclination);
461 locEnergy = EnergyA * std::sqrt(sin_inc*sin_inc + EpsilonA*EpsilonA*cos_inc*cos_inc);
462
463 break;
464 }
465 case InterfaceEnergyModels::Disorientation:
466 {
467 // The disorientation model is implemented in its own method
468 break;
469 }
470 }
471 assert(locEnergy > 0.0 && "Negative interface energy reduce anisotropy");
472 return locEnergy;
473 };

475 dVector3 Derivative(const dVector3& locNormal) const
476 {
477 switch(Model)
478 {
479 case InterfaceEnergyModels::CubicFull:
480 {
481 const double nx3 = std::pow(locNormal[0],3);
482 const double ny3 = std::pow(locNormal[1],3);
483 const double nz3 = std::pow(locNormal[2],3);
484 return {Sigma0*Epsilon1*4*nx3,Sigma0*Epsilon1*4*ny3,Sigma0*Epsilon1*4*nz3};
485 }
486 case InterfaceEnergyModels::FacetedFull:
487 {
488 double Inclination = Pi;
489 double EnergyA = 0.0;
490 double EpsilonA = 0.0;
491 dVector3 Facet = {0.0,0.0,0.0};
492
493 // Calculating inclination angle between facet normal and interface normal
494 for(size_t n = 0; n < Nfacets; n++)
495 for(size_t j = 0; j < FacetVectors[n].size() ; j++)
496 {
497 const dVector3 locFacet = FacetVectors[n][j].normalized();
498 const double dotVec = std::acos(locNormal * locFacet);
499
500 if(std::fabs(dotVec) <= std::fabs(Inclination))
501 {
502 Inclination = dotVec;
503 EnergyA = FacetEnergy[n];
504 EpsilonA = FacetEpsilon[n];
505 Facet = locFacet;
506 }
507 }
508
509 const double sin_inc = std::sin(Inclination);
510 const double cos_inc = std::cos(Inclination);
511 const double dEnergy_dInclination = EnergyA * (1.0-EpsilonA*EpsilonA)*sin_inc*cos_inc/std::sqrt(sin_inc*sin_inc + EpsilonA*EpsilonA*cos_inc*cos_inc);
512 const dVector3 dInclination_dn = Facet*(-1.0)/std::sqrt(1-(locNormal*Facet)*(locNormal*Facet));
513
514 return dInclination_dn*dEnergy_dInclination;
515 }
516 default:
517 return {0.0,0.0,0.0};
518 }
519 };
