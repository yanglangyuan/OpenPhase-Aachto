46inline double psi(double rho, double rho0)
47{
48 assert(rho > 0.0);
49 return 1.0 - exp(-rho/rho0);

52inline double Phi(
53 const double lbDensity,
54 const double ReducedPressure,
55 const double GasParameter)
56{
57 const double potential = GasParameter*ReducedPressure - lbDensity/3.0;
58 assert(potential <= 0.0);
59 return std::sqrt(-potential);
60}

62 void CalculateDistancePeriodic(dVector3 A, dVector3 B,
63 dVector3 &dist, double Nx, double Ny, double Nz)
64{
65 dist[0] = A[0]-B[0];
66 if (std::abs(dist[0]) > std::abs(dist[0]+Nx)) dist[0] += Nx;
67 else if (std::abs(dist[0]) > std::abs(dist[0]-Nx)) dist[0] -= Nx;
68 dist[1] = A[1]-B[1];
69 if (std::abs(dist[1]) > std::abs(dist[1]+Ny)) dist[1] += Ny;
70 else if (std::abs(dist[1]) > std::abs(dist[1]-Ny)) dist[1] -= Ny;
71 dist[2] = A[2]-B[2];
72 if (std::abs(dist[2]) > std::abs(dist[2]+Nz)) dist[2] += Nz;
73 else if (std::abs(dist[2]) > std::abs(dist[2]-Nz)) dist[2] -= Nz;
74};

93void FlowSolverLBM::Initialize(Settings& locSettings, std::stringstream& inp, double in_dt)
94{
95 thisclassname = "FlowSolverLBM";
96
97 Grid = locSettings.Grid;
98
99 dt = in_dt;
100 Nphases = locSettings.Nphases;
101 Ncomp = (locSettings.Ncomp > 0) ? locSettings.Ncomp - 1 : 0;
102
103 Pth = 1.0e05;
104 PthOld = 1.0e05;
105 Pth0 = 1.0e05;
106 Poutlet = 1e-6;
107
108 cs2 = lbcs2*Grid.dx*Grid.dx/dt/dt;
109
110 ObstaclesChanged = false;
111
112 Bcells = Grid.Bcells;
113 ElementNames = locSettings.ElementNames;
114 ReadInput(inp); // NOTE N_Fluid_Comp needs to be read
115
116 Obstacle.Allocate (Grid, Bcells);
117 ObstacleAppeared.Allocate (Grid, Bcells);
118 ObstacleChangedDensity.Allocate (Grid, Bcells);
119 ObstacleVanished.Allocate (Grid, Bcells);
120 DensityWetting.Allocate (Grid, {N_Fluid_Comp}, Bcells);
121 ForceDensity.Allocate (Grid, {N_Fluid_Comp}, Bcells);
122 MomentumDensity.Allocate (Grid, {N_Fluid_Comp}, Bcells);
123 lbPopulations.Allocate (Grid, {N_Fluid_Comp}, Bcells);
124 lbPopulationsTMP.Allocate (Grid, {N_Fluid_Comp}, Bcells);
125 nut.Allocate (Grid, {N_Fluid_Comp}, Bcells);
126 HydroPressure.Allocate (Grid, {N_Fluid_Comp}, Bcells);
127 DivVel.Allocate (Grid, {N_Fluid_Comp}, Bcells);
128 GradRho.Allocate (Grid, {N_Fluid_Comp}, Bcells);
129
130 switch(Grid.Active())
131 {
132 case 1:
133 {
134 for(int ii = 0; ii < 3; ii++)
135 for(int jj = 0; jj < 3; jj++)
136 for(int kk = 0; kk < 3; kk++)
137 {
138 lbWeights[ii][jj][kk] = LBStencil1D[ii][jj][kk];
139 }
140 break;
141 }
142 case 2:
143 {
144 for(int ii = 0; ii < 3; ii++)
145 for(int jj = 0; jj < 3; jj++)
146 for(int kk = 0; kk < 3; kk++)
147 {
148 lbWeights[ii][jj][kk] = LBStencil2D[ii][jj][kk];
149 }
150 break;
151 }
152 case 3:
153 {
154 for(int ii = 0; ii < 3; ii++)
155 for(int jj = 0; jj < 3; jj++)
156 for(int kk = 0; kk < 3; kk++)
157 {
158 lbWeights[ii][jj][kk] = LBStencil3D[ii][jj][kk];
159 }
160 break;
161 }
162 default:
163 {
164
165 }
166 }
167
168 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,DensityWetting,DensityWetting.Bcells(),)
169 {
170 Obstacle (i,j,k) = false;
171 ObstacleAppeared (i,j,k) = false;
172 ObstacleChangedDensity (i,j,k) = false;
173 ObstacleVanished (i,j,k) = false;
174
175 for (size_t n = 0; n < N_Fluid_Comp; ++n)
176 {
177 DensityWetting (i,j,k,{n}) = 0.0;
178 HydroPressure (i,j,k,{n}) = 0.0;
179 DivVel (i,j,k,{n}) = 0.0;
180 GradRho (i,j,k,{n}).set_to_zero();
181 ForceDensity (i,j,k,{n}).set_to_zero();
182 MomentumDensity (i,j,k,{n}).set_to_zero();
183 lbPopulations (i,j,k,{n}).set_to_zero();
184 lbPopulationsTMP (i,j,k,{n}).set_to_zero();
185 }
186 }
187 OMP_PARALLEL_STORAGE_LOOP_END
188
189 InitialTotalMass = -1.0;
190
191 locSettings.AddForAdvection(*this);
192 locSettings.AddForRemeshing(*this);
193 locSettings.AddForReading(*this);
194
195 initialized = true;
196 ConsoleOutput::Write(thisclassname, "Initialized");
197}

219void FlowSolverLBM::ReadInput(std::stringstream& inp)
220{

222
223 N_Fluid_Comp = FileInterface::ReadParameterD(inp, moduleLocation, std::string("N_FLUID_COMP"), false, 1);

225 for (size_t n = 0; n < N_Fluid_Comp; ++n)
226 {
227 std::stringstream NUstr;
228 NUstr << "NU[" << n << "]";

233 Do_Benzi = FileInterface::ReadParameterB(inp, moduleLocation, std::string("BENZI"), false, false);

239 Do_ThermalComp = FileInterface::ReadParameterB(inp, moduleLocation, std::string("THERMALCOMP"), false, false);
240 Do_DI = FileInterface::ReadParameterB(inp, moduleLocation, std::string("Diffuse_Interface"), false, false);
241 FluidRedistributionRange = FileInterface::ReadParameterI(inp, moduleLocation, std::string("FluidRedistributionRange"), false, 1);
242 ParaKuper = FileInterface::ReadParameterD(inp, moduleLocation, std::string("ParaKuper"), false, -0.0152);
243 h_star = FileInterface::ReadParameterD(inp, moduleLocation, std::string("H_STAR"), Do_Drag, 0.0);
244 Do_TwoPhase = (Do_Benzi or Do_Kupershtokh) ? true : false;
245 Do_GuoForcing = FileInterface::ReadParameterB(inp, moduleLocation, std::string("GUO_FORCING"), false, Do_TwoPhase);
246 Do_EDForcing = FileInterface::ReadParameterB(inp, moduleLocation, std::string("ED_FORCING"), false, !Do_TwoPhase);
247 GradRho_Upwind = FileInterface::ReadParameterB(inp, moduleLocation, std::string("GradRho_Upwind"), false, false);
248 GradRho_Central = FileInterface::ReadParameterB(inp, moduleLocation, std::string("GradRho_Central"), false, false);
249 GradRho_VanLeer = FileInterface::ReadParameterB(inp, moduleLocation, std::string("GradRho_VanLeer"), false, true);
250
251 if (Bcells < FluidRedistributionRange)
252 {
253 std::stringstream message;
254 message << Bcells << " boundary cells are too few, at least "
255 << FluidRedistributionRange << " boundary cells are needed!";
256
257 ConsoleOutput::WriteExit(message.str(), thisclassname, "ReadInput");
258 std::exit(EXIT_FAILURE);
259 }
260
261 // Determine density discretization
262 // NOTE that dRho is the equilibrium fluid density [Kg/m^2] if no liquid-vapor phase separation is considered!
263 dRho = FileInterface::ReadParameterD(inp, moduleLocation, std::string("dRho"));
264 dM = dRho*Grid.CellVolume(true);
265 dP = dM/Grid.dx/dt/dt;
266 dm = dM/Grid.dx/Grid.dx/dt;
267 df = dM/Grid.dx/Grid.dx/dt/dt;
268 dnu = Grid.dx*Grid.dx/dt;
269 dv = Grid.dx/dt;
270
271 if (Ncomp > 0)
272 {
273 dDensity_dc.resize(Ncomp);
274 for (size_t n = 0; n < Ncomp; ++n)
275 {
276 std::stringstream dDensity_dc_str;
277 dDensity_dc_str << "DDENSITY_DC_" << ElementNames[n];
278 dDensity_dc[n] = FileInterface::ReadParameterD(inp, moduleLocation, dDensity_dc_str.str(), Do_Gravity,0.0);
279 }
280 }
281
282 // GA Gravitational acceleration
283 for (size_t i = 0; i < 3; i++)
284 {
285 std::stringstream GAstr;
286 GAstr << "G0_" << i;
287 GA[i] = FileInterface::ReadParameterD(inp, moduleLocation, GAstr.str(), Do_Gravity, 0.0);
288 }
289
290 // Read Benzi interaction parameter
291 Gb.resize(N_Fluid_Comp);
292 for (size_t n = 0; n < N_Fluid_Comp; ++n)
293 {
294 Gb[n].resize(N_Fluid_Comp);
295 for (size_t m = 0; m < N_Fluid_Comp; ++m)
296 {
297 std::stringstream Gbstr;
298 Gbstr << "GB_" << n << "_" << m;
299 Gb[n][m] = FileInterface::ReadParameterD(inp, moduleLocation, Gbstr.str(), Do_Benzi, (n==m) ? 1 : 0);
300
301 }
302 }
303
304 // Read Kupershtokh interaction parameter
305 lbGK.resize(N_Fluid_Comp);
306 for (size_t n = 0; n < N_Fluid_Comp; ++n)
307 {
308 lbGK[n].resize(N_Fluid_Comp);
309 for (size_t m = 0; m < N_Fluid_Comp; ++m)
310 {
311 std::stringstream GKstr;
312 GKstr << "lbGK_" << n << "_" << m;
313 lbGK[n][m] = FileInterface::ReadParameterD(inp, moduleLocation, GKstr.str(), Do_Kupershtokh, (n==m) ? 1 : 0);
314 }
315 }
316
317 rho_0.resize(N_Fluid_Comp);
318 for (size_t n = 0; n < N_Fluid_Comp; ++n)
319 {
320 std::stringstream RHO0str;
321 RHO0str << "RHO0_" << n;
322 rho_0[n] = FileInterface::ReadParameterD(inp, moduleLocation, RHO0str.str(), false, dRho);
323 }
324
325 CriticalDensity.resize(N_Fluid_Comp);
326 for (size_t n = 0; n < N_Fluid_Comp; ++n)
327 {
328 std::stringstream RHOCstr;
329 RHOCstr << "RHOC_" << n;
330 CriticalDensity[n] = FileInterface::ReadParameterD(inp, moduleLocation, RHOCstr.str(), Do_Kupershtokh, dRho);
331 }
332
333 lbTemperature.resize(N_Fluid_Comp);
334 for (size_t n = 0; n < N_Fluid_Comp; ++n)
335 {
336 std::stringstream TEMPstr;
337 TEMPstr << "TEMP_" << n;
338 lbTemperature[n] = FileInterface::ReadParameterD(inp, moduleLocation, TEMPstr.str(), Do_Kupershtokh, 0.0);
339 }
340
341 lbCriticalTemperature.resize(N_Fluid_Comp);
342 for (size_t n = 0; n < N_Fluid_Comp; ++n)
343 {
344 std::stringstream TEMPCstr;
345 TEMPCstr << "TEMPC_" << n;
346 lbCriticalTemperature[n] = FileInterface::ReadParameterD(inp, moduleLocation, TEMPCstr.str(), Do_Kupershtokh, 0.0);
347 }
348
349 GasParameter.resize(N_Fluid_Comp);
350 lbCriticalPressure.resize(N_Fluid_Comp);
351 for (size_t n = 0; n < N_Fluid_Comp; ++n)
352 {
353 std::stringstream TEMPCstr;
354 TEMPCstr << "CRITP_" << n;
355 lbCriticalPressure[n] = FileInterface::ReadParameterD(inp, moduleLocation, TEMPCstr.str(), Do_Kupershtokh, 0.0);
356 GasParameter[n] = lbCriticalPressure[n]/CriticalDensity[n]*dRho*(dt*dt)/(Grid.dx*Grid.dx);
357 }
358
359 LiquidDensity .resize(N_Fluid_Comp);
360 VaporDensity .resize(N_Fluid_Comp);
361 SurfaceTension.resize(N_Fluid_Comp);
362 InterfaceWidth.resize(N_Fluid_Comp);
363 if (Do_Benzi)
364 for (size_t n = 0; n < N_Fluid_Comp; ++n)
365 {
366
367 const BenziGas::EquilibriumValues_t Equilibrium =
368 BenziGas::EquilibriumValues(Gb[n][n]/dRho);
369
370 LiquidDensity [n] = Equilibrium.lbLiquidDensity*dRho;
371 VaporDensity [n] = Equilibrium.lbVaporDensity*dRho;
372 SurfaceTension [n] = Equilibrium.lbSurfaceTension*dM/dt/dt;
373 InterfaceWidth [n] = 2.0; //TODO calculate
374
375 ConsoleOutput::Write("Vapor density ["+std::to_string(n)+"]", VaporDensity [n]);
376 ConsoleOutput::Write("Liquid density ["+std::to_string(n)+"]", LiquidDensity [n]);
377 ConsoleOutput::Write("Surface tension ["+std::to_string(n)+"]", SurfaceTension [n]);
378 ConsoleOutput::Write("Interface width ["+std::to_string(n)+"]", InterfaceWidth [n]);
379 }
380 else if (Do_Kupershtokh)
381 for (size_t n = 0; n < N_Fluid_Comp; ++n)
382 {
383 const VanDerWaalsGas::EquilibriumValues_t Equilibrium =
384 VanDerWaalsGas::EquilibriumValues(lbTemperature[n], lbCriticalTemperature[n]);
385
386 LiquidDensity [n] = Equilibrium.LiquidDensity*dRho;
387 VaporDensity [n] = Equilibrium.VaporDensity*dRho;
388 SurfaceTension [n] = 0.0*dM/dt/dt; //TODO
389 InterfaceWidth [n] = 0.402819*std::pow(GasParameter[n],-0.470509);
390
391 ConsoleOutput::Write("Vapor density ["+std::to_string(n)+"]", VaporDensity [n]);
392 ConsoleOutput::Write("Liquid density ["+std::to_string(n)+"]", LiquidDensity [n]);
393 ConsoleOutput::Write("Surface tension ["+std::to_string(n)+"]", SurfaceTension [n]);
394 ConsoleOutput::Write("Interface width ["+std::to_string(n)+"]", InterfaceWidth [n]);
395 }
396 else
397 for (size_t n = 0; n < N_Fluid_Comp; ++n)
398 {
399 LiquidDensity [n] = dRho;
400 VaporDensity [n] = dRho;
401 InterfaceWidth [n] = 1.0;
402 }
403
404 U0X = FileInterface::ReadParameterD(inp, moduleLocation, std::string("U0X"), false, 0.0);
405 U0Y = FileInterface::ReadParameterD(inp, moduleLocation, std::string("U0Y"), false, 0.0);
406 U0Z = FileInterface::ReadParameterD(inp, moduleLocation, std::string("U0Z"), false, 0.0);
407 Pth0 = FileInterface::ReadParameterD(inp, moduleLocation, std::string("Pth0"), false, 1.0e5);
408 Poutlet = FileInterface::ReadParameterD(inp, moduleLocation, std::string("Poutlet"), false, 1.0e5);
409 Pth = Pth0;
410 PthOld = Pth0;
411 Poutlet = Poutlet - 1.0e5 + 1.0e-4;
412
413 Wetting.resize(N_Fluid_Comp);
414 if (Do_TwoPhase)
415 for (size_t n = 0; n < N_Fluid_Comp; ++n)
416 {
417 Wetting[n].resize(Nphases);
418 for (size_t m = 0; m < Nphases; ++m)
419 {
420 std::stringstream TEMPCstr;
421 TEMPCstr << "Wetting_" << n << "_" << m;
422 Wetting[n][m] = FileInterface::ReadParameterD(inp, moduleLocation, TEMPCstr.str());
423 }
424 }
425
426 FluidMass.resize(N_Fluid_Comp, 0.0);
427
428 // NOTE: the minimum and maximum kinematic viscosity values are based
429 // on experience and may be adjusted.
430 const double nu_min = 0.05*Grid.dx*Grid.dx/dt;
431 const double nu_max = 1.00*Grid.dx*Grid.dx/dt;
432 const double nu_opt = 0.5/3.0*Grid.dx*Grid.dx/dt; // optimal value !
433
434 // Set relaxation parameter tau
435 lbtau.resize(N_Fluid_Comp);
436 for (size_t n = 0; n < N_Fluid_Comp; ++n)
437 {
438 if ((nu[n] < nu_min) or (nu[n] > nu_max))
439 {
440 std::stringstream message;
441 message << "Bad kinematic viscosity"
442 << "nu_lb = "<< std::scientific << nu[n]
443 << "\n\t (Min nu_lb = "<< std::scientific << nu_min
444 << "; Opt nu_lb = "<< std::scientific << nu_opt
445 << "; Max nu_lb = "<< std::scientific << nu_max
446 <<")!\n\t Simulation may be unstable! Adjust dx or dt.";
447 ConsoleOutput::WriteWarning(message.str(), thisclassname, "Initialize");
448 }
449
450 // Set relaxation parameter tau
451 lbtau [n] = 3*nu[n]/Grid.dx/Grid.dx*dt + 0.5;
452 ConsoleOutput::WriteStandard("lbtau["+std::to_string(n)+"]", lbtau[n]);
453 }
454
455 ConsoleOutput::WriteLine();
456 ConsoleOutput::WriteBlankLine();
457}

494bool FlowSolverLBM::Read(const Settings& locSettings, const BoundaryConditions& BC, const int tStep)
495{
496#ifdef MPI_PARALLEL
497 std::string FileName = FileInterface::MakeFileName(locSettings.InputRawDataDir, thisclassname+'_'+std::to_string(MPI_RANK)+"_", tStep, ".dat");
498#else
499 std::string FileName = FileInterface::MakeFileName(locSettings.InputRawDataDir, thisclassname+'_', tStep, ".dat");
500#endif
501
502 std::ifstream inp(FileName.c_str(), std::ios::in | std::ios::binary);
503
504 if (!inp)
505 {
506 std::stringstream message;
507 message << "File \"" << FileName << "\" could not be opened";
508 ConsoleOutput::WriteWarning(message.str(), thisclassname, "Read()");
509 return false;
510 };
511
512 int locNx = Grid.Nx;
513 int locNy = Grid.Ny;
514 int locNz = Grid.Nz;
515 inp.read(reinterpret_cast<char*>(&locNx), sizeof(int));
516 inp.read(reinterpret_cast<char*>(&locNy), sizeof(int));
517 inp.read(reinterpret_cast<char*>(&locNz), sizeof(int));
518
519 if(locNx != Grid.Nx or locNy != Grid.Ny or locNz != Grid.Nz)
520 {
521 std::stringstream message;
522 message << "Inconsistent system dimensions!\n"
523 << "Input data dimensions: ("
524 << locNx << ", "
525 << locNy << ", "
526 << locNz << ") grid points.\n"
527 << "Required data dimensions: ("
528 << Grid.Nx << ", "
529 << Grid.Ny << ", "
530 << Grid.Nz << ") grid points.\n";
531#ifdef DEBUG
532 throw std::runtime_error(message.str());
533#else
534 return false;
535#endif
536 }
537
538 size_t locN_Fluid_Comp = N_Fluid_Comp;
539 inp.read(reinterpret_cast<char*>(&N_Fluid_Comp), sizeof(size_t));
540
541 if(locN_Fluid_Comp != N_Fluid_Comp)
542 {
543 std::stringstream message;
544 message << "Inconsistent number of fluid components\n"
545 << "Input number of fluid components: "
546 << locN_Fluid_Comp << "\n"
547 << "Required number of fluid components: "
548 << N_Fluid_Comp << "\n";
549#ifdef DEBUG
550 throw std::runtime_error(message.str());
551#else
552 return false;
553#endif
554 }
555
556 STORAGE_LOOP_BEGIN(i,j,k,lbPopulations,0)
557 {
558 for(size_t n = 0; n < N_Fluid_Comp; ++n)
559 for(int ii = -Grid.dNx; ii <= Grid.dNx; ++ii)
560 for(int jj = -Grid.dNy; jj <= Grid.dNy; ++jj)
561 for(int kk = -Grid.dNz; kk <= Grid.dNz; ++kk)
562 {

710void FlowSolverLBM::CalculateForceTwoPhase(const int i, const int j,
711 const int k, PhaseField& Phase)
712{
713 for(int ii = -Grid.dNx; ii <= Grid.dNx; ++ii)
714 for(int jj = -Grid.dNy; jj <= Grid.dNy; ++jj)
715 for(int kk = -Grid.dNz; kk <= Grid.dNz; ++kk)
716 {
717 const dVector3 vel {ii*Grid.dx/dt,jj*Grid.dx/dt,kk*Grid.dx/dt};
718
719 for (size_t n = 0; n < N_Fluid_Comp; ++n)
720 for (size_t m = 0; m < N_Fluid_Comp; ++m)
721 {
722 double tmp = 0.0;
723 if (Do_Benzi)
724 {
725 // Benzi et. al. (2006) - Mesoscopic modeling of a two-phase flow
726 // in the presence of boundaries
727 tmp = - Gb[n][m]/dRho * psi(DensityWetting(i,j,k,{n}), rho_0[n]) * lbWeights[ii+1][jj+1][kk+1] * psi(DensityWetting(i+ii,j+jj,k+kk,{m}), rho_0[m]);
728 }
729 else if (Do_Kupershtokh)
730 {
731 // Calculate phase sepration force according to:
732 // Kupershtokh, Medvedev and Karpov, 2009, On equations of state in a lattice Boltzmann method
733 // Kupershtokh, Medvedev and Gribanov, 2017, Thermal lattice Boltzmann method for multiphase flows
734
735 // Evaluate equation of state (Van der Waals)
736 const double p_000 = VanDerWaalsGas::ReducedPressure(DensityWetting(i ,j ,k ,{n})/dRho, lbTemperature[n], CriticalDensity[n]/dRho, lbCriticalTemperature[n]);
737 const double p_ijk = VanDerWaalsGas::ReducedPressure(DensityWetting(i+ii,j+jj,k+kk,{m})/dRho, lbTemperature[m], CriticalDensity[m]/dRho, lbCriticalTemperature[m]);
738
739 // Calculate interaction potential
740 const double Phi_000 = Phi(DensityWetting(i, j, k ,{n})/dRho, p_000, GasParameter[n]);
741 const double Phi_ijk = Phi(DensityWetting(i+ii,j+jj,k+kk,{m})/dRho, p_ijk, GasParameter[m]);
742
743 // Interpolate between both way
744 tmp = 6.0*lbWeights[ii+1][jj+1][kk+1]*lbGK[n][m]*(ParaKuper*(Phi_ijk*Phi_ijk) + (1.0-2.0*ParaKuper)*(Phi_000*Phi_ijk));
745 }
746
747 const dVector3 BenziForceDensity = {tmp*ii*df, tmp*jj*df, tmp*kk*df};
748
749 ForceDensity(i,j,k,{n}) += BenziForceDensity;
750
751 if (!Obstacle(i,j,k) && Obstacle(i+ii,j+jj,k+kk))
752 for(auto& it : Phase.Fields(i+ii,j+jj,k+kk))
753 {
754 Grain& grain = Phase.FieldsProperties[it.index];
755 if(grain.State == AggregateStates::Solid)
756 {
757 const dVector3 pos = {double(i+ii), double(j+jj), double(k+kk)};
758 dVector3 distanceCM;
759 CalculateDistancePeriodic(pos, grain.Rcm, distanceCM, Grid.Nx, Grid.Ny, Grid.Nz);
760 const dVector3 R = distanceCM * Grid.dx;
761
762 #ifdef _OPENMP
763 #pragma omp critical
764 #endif
765 {
766 grain.Force -= BenziForceDensity * Grid.CellVolume(true);
767 grain.Torque -= R.cross(BenziForceDensity) * Grid.CellVolume(true);
768 }
769 }
770 }
771 }
772 }
773}

790void FlowSolverLBM::AcountForAndLimitPhaseTransformation(PhaseField& Phase, int tStep)
791{
792 // This method calculates the fluid density change according to phase-
793 // transformations like solidification. Further it limits the
794 // phase-transformation if not enough liquid is present.
795 // It only needs to be calculated if the fluid density is assumed to be
796 // inhomogeneous.
797
798 if (tStep == 0)
799 {
800 ConsoleOutput::WriteWarning("Simulation of a two phase fluid and Solidification and Melting is still experimental", thisclassname, "AcountForAndLimitPhaseTransformation");
801 }
802
803 if (Do_TwoPhase)
804 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
805 if(Phase.Fields(i,j,k).wide_interface())
806 {
807 double locDensity = DensityWetting(i,j,k,{0});
808 double locIncomingDensityChange = CalculateLocalIncomingSolidDensityChange(Phase,i,j,k);
809 double locDensityChange = locIncomingDensityChange;
810
811 // Limit incoming density change
812 const double locDensityMinium = 2*VaporDensity[0];
813 if (locIncomingDensityChange > 0.0) // TODO also consider melting
814 {
815 double Limiter = 1.0;
816 if (locDensity < locDensityMinium)
817 {
818 locDensityChange = 0.0;
819 Limiter = 0.0;
820 }
821 else if (locDensity - locIncomingDensityChange < locDensityMinium)
822 {
823 locDensityChange = locDensityMinium-locDensity;
824 Limiter = locDensityChange/locIncomingDensityChange;
825 }
826
827 // Limit phase-transformation to ensure mass conservation
828 for(auto& psi : Phase.FieldsDot(i,j,k))
829 {
830 const Grain GrainA = Phase.FieldsProperties[psi.indexA];
831 const Grain GrainB = Phase.FieldsProperties[psi.indexB];
832 if (psi.value1 != 0.0)
833 if ((GrainA.is_solid() and GrainB.is_fluid()) or
834 (GrainB.is_solid() and GrainA.is_fluid()))
835 {
836 psi.value1 *= Limiter;
837 }
838 }
839 }
840
841 // Apply density change
842 DensityWetting(i,j,k,{0}) -= locDensityChange;
843 lbPopulations (i,j,k,{0}) = EquilibriumDistribution(DensityWetting(i,j,k,{0})/dRho, lbWeights, MomentumDensity(i,j,k,{0})/dm);
844 }
845 OMP_PARALLEL_STORAGE_LOOP_END
846}

949void FlowSolverLBM::CalculateDensityGradient(PhaseField& Phase, Velocities& Vel)
950{
951 double eps=1e-10;
952 std::vector<int> dir(3);
953 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,DensityWetting,0,)
954 {
955 if(!Obstacle(i,j,k))
956 for (int direction=0; direction<3; ++direction)
957 {
958 dir[0] = (direction == 0)?1*Grid.dNx:0;
959 dir[1] = (direction == 1)?1*Grid.dNy:0;
960 dir[2] = (direction == 2)?1*Grid.dNz:0;
961 int ip=i+dir[0];
962 int jp=j+dir[1];
963 int kp=k+dir[2];
964 int im=i-dir[0];
965 int jm=j-dir[1];
966 int km=k-dir[2];
967
968 for (size_t n = 0; n < N_Fluid_Comp; ++n)
969 {
970 double v = 0.0;
971 for (auto it = Phase.Fields(i,j,k).cbegin(); it != Phase.Fields(i,j,k).cend(); it++)
972 if (Phase.FieldsProperties[it->index].State != AggregateStates::Solid)
973 {
974 size_t PhaseIdx = Phase.FieldsProperties[it->index].Phase;
975 v = Vel.Phase(i,j,k,{PhaseIdx})[direction] ;
976 }
977 double Rho = DensityWetting(i,j,k,{n});
978 double Rhop = DensityWetting(ip,jp,kp,{n});
979 double Rhom = DensityWetting(im,jm,km,{n});
980
981 if(GradRho_Upwind)
982 {
983 if(v>=0)
984 {
985 GradRho(i,j,k,{n})[direction] = (Rho-Rhom)/Grid.dx;
986 }
987 else
988 {
989 GradRho(i,j,k,{n})[direction] = (Rhop-Rho)/Grid.dx;
990 }
991 }
992 else if(GradRho_VanLeer)
993 {
994 double Rhohp = (Rho+Rhop)/2.0;
995 double Rhohm = (Rho+Rhom)/2.0;
996 if(v>=0)
997 {
998 int imm=i-2.0*dir[0];
999 int jmm=j-2.0*dir[1];
1000 int kmm=k-2.0*dir[2];
1001
1002 double Rhomm = DensityWetting(imm,jmm,kmm,{n});
1003 double r = (Rho-Rhom)/(Rhop-Rho+eps);
1004 double phi=(r+fabs(r))/(1.0+fabs(r));
1005 Rhohp=Rho+0.5*phi*(Rhop-Rho);
1006
1007 double rm = (Rhom-Rhomm)/(Rho-Rhom+eps);
1008 double phim=(rm+fabs(rm))/(1.0+fabs(rm));
1009 Rhohm=Rhom+0.5*phim*(Rho-Rhom);
1010 GradRho(i,j,k,{n})[direction] = (Rhohp-Rhohm)/Grid.dx;
1011 }
1012 else
1013 {
1014 int ipp=i+2.0*dir[0];
1015 int jpp=j+2.0*dir[1];
1016 int kpp=k+2.0*dir[2];
1017 double Rhopp = DensityWetting(ipp,jpp,kpp,{n});
1018
1019 double r = (Rhopp-Rhop)/(Rhop-Rho+eps);
1020 double phi=(r+fabs(r))/(1.0+fabs(r));
1021 Rhohp=Rhop-0.5*phi*(Rhop-Rho);
1022
1023 double rm = (Rhop-Rho)/(Rho-Rhom+eps);
1024 double phim=(rm+fabs(rm))/(1.0+fabs(rm));
1025 Rhohm=Rho-0.5*phim*(Rho-Rhom);
1026 GradRho(i,j,k,{n})[direction] = (Rhohp-Rhohm)/Grid.dx;
1027 }
1028 }
1029 else if(GradRho_Central)
1030 {
1031 GradRho(i,j,k,{n})[direction] = (Rhop-Rhom)/Grid.dx/2.0;
1032 }
1033 }
1034 }
1035 }
1036 OMP_PARALLEL_STORAGE_LOOP_END
1037}
1038void FlowSolverLBM::CalculateHydrodynamicPressureAndMomentum(Velocities& Vel)
1039{
1040 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,DensityWetting,0,)
1041 if(!Obstacle(i,j,k))
1042 for (size_t n = 0; n < N_Fluid_Comp; ++n)
1043 {
1044 HydroPressure(i,j,k,{n}) = 0.0;
1045 MomentumDensity(i,j,k,{n}).set_to_zero();
1046
1047 for (int ii = -Grid.dNx; ii <= Grid.dNx; ++ii)
1048 for (int jj = -Grid.dNy; jj <= Grid.dNy; ++jj)
1049 for (int kk = -Grid.dNz; kk <= Grid.dNz; ++kk)
1050 {
1051 double rr = lbPopulations(i,j,k,{n})(ii,jj,kk);
1052 HydroPressure (i,j,k,{n}) += rr*dP;
1053 MomentumDensity(i,j,k,{n})[0] += rr*ii;
1054 MomentumDensity(i,j,k,{n})[1] += rr*jj;
1055 MomentumDensity(i,j,k,{n})[2] += rr*kk;
1056 }
1057
1058 MomentumDensity(i,j,k,{n}) = MomentumDensity(i,j,k,{n})/cs2*pow(Grid.dx,3)/pow(dt,3) + ForceDensity(i,j,k,{n})*dt/2.0;
1059 dVector3 vel = MomentumDensity(i,j,k,{n}) /DensityWetting(i,j,k,{n});
1060 HydroPressure (i,j,k,{n}) += cs2*dt/2.0 * ( (vel[0]*GradRho(i,j,k,{n})[0] + vel[1]*GradRho(i,j,k,{n})[1] + vel[2]*GradRho(i,j,k,{n})[2]) + DensityWetting(i,j,k,{n}) * DivVel(i,j,k,{n}));
1061 }
1062 OMP_PARALLEL_STORAGE_LOOP_END
1063}

1065void FlowSolverLBM::CollisionTC( Velocities& Vel)
1066{
1067 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,DensityWetting,0,)
1068 if (!Obstacle(i,j,k))
1069 for (size_t n = 0; n < N_Fluid_Comp; ++n)

1507void FlowSolverLBM::CalculateForceDrag(const int i,const int j,const int k,
1508 PhaseField& Phase, const Velocities& Vel)

1515 double locSolidFraction = SolidFraction(i,j,k,Phase);

1518 {
1519 Grain& grain = Phase.FieldsProperties[it.index];
1520 if(grain.State == AggregateStates::Solid)

1524 (Phase.Grid.iWidth * Phase.Grid.iWidth);

1532 #pragma omp critical
1533 #endif

1535 grain.Force += DragForceDensity * dx3;
1536 grain.Torque += locR.cross(DragForceDensity) * dx3;
1537 }
1538 }

1543double FlowSolverLBM::BounceBack(const int i, const int j, const int k,
1544 const int ii, const int jj, const int kk, const size_t n,
1545 PhaseField& Phase, const BoundaryConditions& BC, double& lbDensityChange)
1546{
1547 double dx3 = Grid.CellVolume(true);
1548
1549 double NewPopulation = lbPopulations(i,j,k,{n})(-ii,-jj,-kk);
1550 if (Do_BounceBack)
1551 {
1552 for(auto it : Phase.Fields(i-ii,j-jj,k-kk))
1553 {
1554 Grain& grain = Phase.FieldsProperties[it.index];
1555 if(grain.State == AggregateStates::Solid)
1556 {
1557 dMatrix3x3 W;
1558 W(0,0) = 0.0;
1559 W(1,1) = 0.0;
1560 W(2,2) = 0.0;
1561 W(0,1) = -grain.aVel[2];
1562 W(0,2) = grain.aVel[1];
1563 W(1,2) = -grain.aVel[0];
1564 W(1,0) = grain.aVel[2];
1565 W(2,0) = -grain.aVel[1];
1566 W(2,1) = grain.aVel[0];
1567
1568 const dVector3 pos = {double(i-0.5*ii), double(j-0.5*jj), double(k-0.5*kk)};
1569 dVector3 distanceCM;
1570 CalculateDistancePeriodic(pos, grain.Rcm, distanceCM, Grid.Nx, Grid.Ny, Grid.Nz);
1571 const dVector3 locR = distanceCM * Grid.dx;
1572 const dVector3 Vel = grain.Vcm + W * locR;
1573
1574 const double tmp = lbWeights[ii+1][jj+1][kk+1] *
1575 DensityWetting(i,j,k,{n})/dRho * (Vel[0]*ii + Vel[1]*jj + Vel[2]*kk) * dt/Grid.dx;
1576
1577 const double lbBBDensity =
1578 2.0 * (lbPopulations(i,j,k,{n})(-ii,-jj,-kk) + 3.0 * tmp);
1579
1580 NewPopulation += it.value * 6.0 * tmp;
1581
1582 // NOTE: Bounce Back Momentum Density is now in physical units
1583 dVector3 lbBounceBackForceDensity;
1584 lbBounceBackForceDensity[0] = - ii * lbBBDensity * it.value;
1585 lbBounceBackForceDensity[1] = - jj * lbBBDensity * it.value;
1586 lbBounceBackForceDensity[2] = - kk * lbBBDensity * it.value;
1587
1588 const dVector3 BounceBackForceDensity = lbBounceBackForceDensity * df;
1589
1590 #ifdef _OPENMP
1591 #pragma omp critical
1592 #endif
1593 {
1594 grain.Force += BounceBackForceDensity * dx3;
1595 grain.Torque += locR.cross(BounceBackForceDensity) * dx3;
1596 }
1597 }
1598 }
1599 }
1600 else if (Do_BounceBackElastic)
1601 {
1602 // Author: raphael.schiedung@rub.de
1603 const double rho1 = lbPopulations(i,j,k,{n})(-ii,-jj,-kk);
1604 const double ci = std::sqrt(ii*ii+jj*jj+kk*kk);
1605 const double ici = 1.0/ci;
1606 const double p1 = rho1*ci;
1607 const dVector3 nn = {-ii*ici, -jj*ici, -kk*ici};
1608 const double Elastic = 1.00; // 0.0 models fully inelastic impact
1609
1610 NewPopulation = 0.0;
1611 for(auto it : Phase.Fields(i-ii,j-jj,k-kk))
1612 {
1613 Grain& grain = Phase.FieldsProperties[it.index];
1614 if(grain.State == AggregateStates::Solid)
1615 {
1616 dMatrix3x3 W;
1617 W(0,0) = 0.0;
1618 W(1,1) = 0.0;
1619 W(2,2) = 0.0;
1620 W(0,1) = -grain.aVel[2];
1621 W(0,2) = grain.aVel[1];
1622 W(1,2) = -grain.aVel[0];
1623 W(1,0) = grain.aVel[2];
1624 W(2,0) = -grain.aVel[1];
1625 W(2,1) = grain.aVel[0];
1626
1627 const dVector3 pos = {double(i-0.5*ii), double(j-0.5*jj), double(k-0.5*kk)};
1628 dVector3 distanceCM;
1629 CalculateDistancePeriodic(pos, grain.Rcm,distanceCM, Grid.Nx, Grid.Ny, Grid.Nz);
1630 const dVector3 locR = distanceCM * Grid.dx;
1631 const dVector3 Vel = grain.Vcm + W * locR;
1632
1633 const double v2 = Vel*nn*dt/Grid.dx;
1634 const double rho2 = grain.Density/dRho;
1635 const double p2 = rho2*v2;
1636 const double p1p = (p1*(rho1-rho2*Elastic) + p2*(rho1+rho1*Elastic))/(rho1+rho2);
1637 const double p2p = (p1*(rho2+rho2*Elastic) + p2*(rho2-rho1*Elastic))/(rho1+rho2);
1638
1639 if (p1p < 0)
1640 {
1641 NewPopulation -= it.value*p1p*ici;
1642 }
1643 else
1644 {
1645 // The solid bounce back from the fluid (locally).
1646 // Check grain Densities !
1647 std::cerr << "Warning fluid momentum is too high!"
1648 << "Check parameter (eg. Densities).\n";
1649 }
1650
1651 dVector3 BounceBackForceDensity = {0,0,0};
1652 BounceBackForceDensity += nn*it.value*(p2p-p2)*df;
1653 #ifdef _OPENMP
1654 #pragma omp critical
1655 #endif
1656 {
1657 grain.Force += BounceBackForceDensity * dx3;
1658 grain.Torque += locR.cross(BounceBackForceDensity) * dx3;
1659 }
1660 }
1661 }
1662 }
1663 lbDensityChange += NewPopulation - lbPopulations(i,j,k,{n})(-ii,-jj,-kk);
1664 return NewPopulation;
1665}

1886void FlowSolverLBM::Collision()
1887{
1888 // Apply body force
1889 if (Do_GuoForcing) // Guo, Zheng, and, Shi, Phy. Rev. E (2002)
1890 {
1891 // Apply body force with force projection method
1892 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,DensityWetting,0,)
1893 if (!Obstacle(i,j,k))
1894 for (size_t n = 0; n < N_Fluid_Comp; ++n)
1895 {
1896 const dVector3 EqMomentum = MomentumDensity(i,j,k,{n}) + ForceDensity(i,j,k,{n})*dt/2;
1897 lbPopulations(i,j,k,{n}) += (EquilibriumDistribution(DensityWetting(i,j,k,{n})/dRho, lbWeights, EqMomentum/dm) - lbPopulations(i,j,k,{n}))/lbtau[n];
1898 }
1899 OMP_PARALLEL_STORAGE_LOOP_END
1900
1901 // Apply body force with force projection method
1902 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,DensityWetting,0,)
1903 if (!Obstacle(i,j,k))
1904 for (size_t n = 0; n < N_Fluid_Comp; ++n)
1905 {
1906 assert(DensityWetting(i,j,k,{n})/dRho > DBL_EPSILON);
1907 assert(DensityWetting(i,j,k,{n}) < DBL_MAX);
1908
1909 dVector3 lbVel;
1910 if (Do_ThermalComp)
1911 {
1912 lbVel = (MomentumDensity(i,j,k,{n}) * 3.0 + ForceDensity(i,j,k,{n})*dt/2)/DensityWetting(i,j,k,{n})*dt/Grid.dx;
1913 }
1914 else
1915 {
1916 lbVel = (MomentumDensity(i,j,k,{n}) + ForceDensity(i,j,k,{n})*dt/2)/DensityWetting(i,j,k,{n})*dt/Grid.dx;
1917 }
1918
1919 D3Q27 lblocForcing;
1920 lblocForcing.set_to_zero();
1921 for(int ii = -Grid.dNx; ii <= Grid.dNx; ++ii)
1922 for(int jj = -Grid.dNy; jj <= Grid.dNy; ++jj)
1923 for(int kk = -Grid.dNz; kk <= Grid.dNz; ++kk)
1924 {
1925 lblocForcing(ii,jj,kk) =
1926 (1.0-0.5/lbtau[n]) * lbWeights[ii+1][jj+1][kk+1]*
1927 (3.0*((ii-lbVel[0])*ForceDensity(i,j,k,{n})[0]/df +
1928 (jj-lbVel[1])*ForceDensity(i,j,k,{n})[1]/df +
1929 (kk-lbVel[2])*ForceDensity(i,j,k,{n})[2]/df) +
1930 9.0*(lbVel[0]*ii + lbVel[1]*jj + lbVel[2]*kk)*
1931 (ii*ForceDensity(i,j,k,{n})[0]/df +
1932 jj*ForceDensity(i,j,k,{n})[1]/df +
1933 kk*ForceDensity(i,j,k,{n})[2]/df));
1934 }
1935
1936 lbPopulations(i,j,k,{n}) += lblocForcing;
1937 }
1938 OMP_PARALLEL_STORAGE_LOOP_END
1939 }
1940 else if (Do_EDForcing)
1941 {
1942 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,DensityWetting,0,)
1943 if (!Obstacle(i,j,k))
1944 for (size_t n = 0; n < N_Fluid_Comp; ++n)
1945 {
1946 const double q = 1.0 - 1.0/lbtau[n];
1947 lbPopulations(i,j,k,{n}) = (lbPopulations(i,j,k,{n}) -
1948 EquilibriumDistribution(DensityWetting(i,j,k,{n})/dRho, lbWeights, MomentumDensity(i,j,k,{n})/dm))*q +
1949 EquilibriumDistribution(DensityWetting(i,j,k,{n})/dRho, lbWeights, (MomentumDensity(i,j,k,{n})/dm + ForceDensity(i,j,k,{n})/df*dt));
1950 }
1951 OMP_PARALLEL_STORAGE_LOOP_END
1952 }

2501 double CF = (ii-lbvel[0])*lbFb[0]+(jj-lbvel[1])*lbFb[1]+(kk-lbvel[2])*lbFb[2];
2502 double lbw = lbWeights[ii+1][jj+1][kk+1];
2503 double lbcu = ii*lbvel[0] + jj*lbvel[1] + kk*lbvel[2];

2508 }
