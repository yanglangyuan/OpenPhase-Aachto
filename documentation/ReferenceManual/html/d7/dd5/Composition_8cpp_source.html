380Tensor<double, 1> Composition::WeightFractionsTotal(int x, int y, int z) const
381{
385 Tensor<double, 1> result({Ncomp});
386
387 double div = 0.0;
388 for(size_t comp = 0; comp < Ncomp; comp++)
389 {
390 double locWeight = MoleFractionsTotal(x,y,z,{comp})*PT.GetData(Component[comp].Name).AtomicWeight;
391 div += locWeight;
392 result({comp}) = locWeight;
393 }
394 for(size_t comp = 0; comp < Ncomp; comp++)
395 {
396 result({comp}) /= div;
397 }
398 return result;

445void Composition::CalculateMoleFractionsAverage(PhaseField& Phase)
446{
447 Tensor<double, 3> locMoleFractionsInterfaceAverage({Nphases,Nphases,Ncomp});
448 Tensor<double, 2> locMoleFractionsAverage({Nphases,Ncomp});
449
450 Tensor<double, 2> NpointsAB({Nphases,Nphases});
451 Tensor<double, 1> NpointsA({Nphases});
452
453 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,MoleFractions,0,reduction(TensorD2Sum: locMoleFractionsAverage) reduction(TensorD3Sum: locMoleFractionsInterfaceAverage) reduction(TensorD1Sum: NpointsA) reduction(TensorD2Sum: NpointsAB))
454 {
455 for(size_t alpha = 0; alpha < Nphases; alpha++)
456 if(Phase.Fractions(i,j,k,{alpha}) != 0.0)
457 {
458 NpointsA({alpha}) += 1;
459 for(size_t comp = 0; comp < Ncomp; comp++)
460 {
461 locMoleFractionsAverage({alpha, comp}) += MoleFractions(i,j,k,{alpha, comp});
462 }
463 }
464
465 if(Phase.Fields(i,j,k).interface())
466 {
467 for(size_t alpha = 0; alpha < Nphases; alpha++)
468 if(Phase.Fractions(i,j,k,{alpha}) != 0.0)
469 for(size_t beta = 0; beta < Nphases; beta++)
470 if(Phase.Fractions(i,j,k,{beta}) != 0.0)
471 {
472 NpointsAB({alpha, beta}) += 1;
473 for(size_t comp = 0; comp < Ncomp; comp++)
474 {
475 locMoleFractionsInterfaceAverage({alpha, beta, comp}) +=
476 Phase.Fractions(i,j,k,{alpha})*MoleFractions(i,j,k,{alpha, comp}) +
477 Phase.Fractions(i,j,k,{ beta})*MoleFractions(i,j,k,{ beta, comp});
478 }
479 }
480 }
481 }
482 OMP_PARALLEL_STORAGE_LOOP_END
483
484 MoleFractionsAverage = locMoleFractionsAverage;
485 MoleFractionsInterfaceAverage = locMoleFractionsInterfaceAverage;
486
487#ifdef MPI_PARALLEL
488 Tensor<double, 1> tmpNpointsA = NpointsA;
489 Tensor<double, 2> tmpNpointsAB = NpointsAB;
490 OP_MPI_Allreduce(tmpNpointsA.data(), NpointsA.data(), Nphases, OP_MPI_DOUBLE, OP_MPI_SUM, OP_MPI_COMM_WORLD);
491 OP_MPI_Allreduce(tmpNpointsAB.data(), NpointsAB.data(), Nphases*Nphases, OP_MPI_DOUBLE, OP_MPI_SUM, OP_MPI_COMM_WORLD);
492
493 OP_MPI_Allreduce(locMoleFractionsAverage.data(), MoleFractionsAverage.data(), Nphases*Ncomp, OP_MPI_DOUBLE, OP_MPI_SUM, OP_MPI_COMM_WORLD);
494 OP_MPI_Allreduce(locMoleFractionsInterfaceAverage.data(), MoleFractionsInterfaceAverage.data(), Nphases*Nphases*Ncomp, OP_MPI_DOUBLE, OP_MPI_SUM, OP_MPI_COMM_WORLD);
495#endif
496
497 for(size_t alpha = 0; alpha < Nphases; alpha++)
498 {
499 if(NpointsA({alpha}) != 0)
500 {
501 for(size_t comp = 0; comp < Ncomp; comp++)
502 {
503 MoleFractionsAverage({alpha, comp}) /= NpointsA({alpha});
504 }
505 }
506 else
507 {
508 for(size_t comp = 0; comp < Ncomp; comp++)
509 {
510 MoleFractionsAverage({alpha, comp}) = MoleFractionsTotalAverage({comp});
511 }
512 }
513 }
514
515 for(size_t alpha = 0; alpha < Nphases; alpha++)
516 for(size_t beta = 0; beta < Nphases; beta++)
517 {
518 if(NpointsAB({alpha, beta}) != 0)
519 {
520 for(size_t comp = 0; comp < Ncomp; comp++)
521 {
522 MoleFractionsInterfaceAverage({alpha, beta, comp}) /= NpointsAB({alpha, beta});
523 }
524 }
525 else
526 {
527 for(size_t comp = 0; comp < Ncomp; comp++)
528 {
529 MoleFractionsInterfaceAverage({alpha, beta, comp}) = 0.5*(MoleFractionsAverage({alpha, comp}) + MoleFractionsAverage({beta, comp}));
530 }
531 }
532 }
533
534 for(size_t alpha = 0; alpha < Nphases; alpha++)
535 for(size_t beta = 0; beta < Nphases; beta++)
536 {
537 if(NpointsAB({alpha, beta}) != 0)
538 {
539 Interface({alpha, beta}) = true;
540 }
541 else
542 {
543 Interface({alpha, beta}) = false;
544 }
545 }

568void Composition::MoveFrame(const int dx, const int dy, const int dz,
569 const BoundaryConditions& BC)
570{
571 int xBeg = (dx >= 0) + (dx < 0)*(Grid.Nx) - 1;
572 int xEnd = (dx >= 0)*(Grid.Nx) + (dx < 0) - 1;
573 int xInc = 1 - 2*(dx < 0);
574
575 int yBeg = (dy >= 0) + (dy < 0)*(Grid.Ny) - 1;
576 int yEnd = (dy >= 0)*(Grid.Ny) + (dy < 0) - 1;
577 int yInc = 1 - 2*(dy < 0);
578
579 int zBeg = (dz >= 0) + (dz < 0)*(Grid.Nz) - 1;
580 int zEnd = (dz >= 0)*(Grid.Nz) + (dz < 0) - 1;
581 int zInc = 1 - 2*(dz < 0);
582
583 for(int i = xBeg; ((dx >= 0) and (i <= xEnd)) or ((dx < 0) and (i >= xEnd)); i += xInc)
584 for(int j = yBeg; ((dy >= 0) and (j <= yEnd)) or ((dy < 0) and (j >= yEnd)); j += yInc)
585 for(int k = zBeg; ((dz >= 0) and (k <= zEnd)) or ((dz < 0) and (k >= zEnd)); k += zInc)
586 {
587 MoleFractions(i,j,k) = MoleFractions(i + dx*Grid.dNx, j + dy*Grid.dNy, k + dz*Grid.dNz);
588 MoleFractionsTotal(i,j,k) = MoleFractionsTotal(i + dx*Grid.dNx, j + dy*Grid.dNy, k + dz*Grid.dNz);
589 }
590 SetBoundaryConditions(BC);
591
592 if(ExtensionX0.isActive()) ExtensionX0.moveFrame(-dx*Grid.dNx, BC.BC0X);
593 if(ExtensionXN.isActive()) ExtensionXN.moveFrame( dx*Grid.dNx, BC.BCNX);
594 if(ExtensionY0.isActive()) ExtensionY0.moveFrame(-dy*Grid.dNy, BC.BC0Y);
595 if(ExtensionYN.isActive()) ExtensionYN.moveFrame( dy*Grid.dNy, BC.BCNY);
596 if(ExtensionZ0.isActive()) ExtensionZ0.moveFrame(-dz*Grid.dNz, BC.BC0Z);
597 if(ExtensionZN.isActive()) ExtensionZN.moveFrame( dz*Grid.dNz, BC.BCNZ);
598
599 ConsoleOutput::WriteStandard(thisclassname, "Frame moved");
600}

620 for(int k = zBeg; ((dz >= 0) and (k <= zEnd)) or ((dz < 0) and (k >= zEnd)); k += zInc)
621 if((i-x)*dx+(j-y)*dy+(k-z)*dz >= 0)
622 {
623 MoleFractions(i,j,k) = MoleFractions(i + dx*Grid.dNx, j + dy*Grid.dNy, k + dz*Grid.dNz);
624 MoleFractionsTotal(i,j,k) = MoleFractionsTotal(i + dx*Grid.dNx, j + dy*Grid.dNy, k + dz*Grid.dNz);
625 }
626 xBeg = (dx >= 0)*(Grid.Nx) + (dx < 0) - 1;
627 xEnd = (dx >= 0) + (dx < 0)*(Grid.Nx) - 1;
628 xInc = 2*(dx < 0) - 1;
629
630 yBeg = (dy >= 0)*(Grid.Ny) + (dy < 0) - 1;
631 yEnd = (dy >= 0) + (dy < 0)*(Grid.Ny) - 1;
632 yInc = 2*(dy < 0) - 1;
633
634 zBeg = (dz >= 0)*(Grid.Nz) + (dz < 0) - 1;
635 zEnd = (dz >= 0) + (dz < 0)*(Grid.Nz) - 1;
636 zInc = 2*(dz < 0) - 1;
637
638 for(int i = xBeg; ((dx >= 0) and (i >= xEnd)) or ((dx < 0) and (i <= xEnd)); i += xInc)
639 for(int j = yBeg; ((dy >= 0) and (j >= yEnd)) or ((dy < 0) and (j <= yEnd)); j += yInc)
640 for(int k = zBeg; ((dz >= 0) and (k >= zEnd)) or ((dz < 0) and (k <= zEnd)); k += zInc)
641 if((i-x)*dx+(j-y)*dy+(k-z)*dz < 0)
642 {
643 MoleFractions(i,j,k) = MoleFractions(i - dx*Grid.dNx, j - dy*Grid.dNy, k - dz*Grid.dNz);
644 MoleFractionsTotal(i,j,k) = MoleFractionsTotal(i - dx*Grid.dNx, j - dy*Grid.dNy, k - dz*Grid.dNz);
645 }
646 SetBoundaryConditions(BC);
647
648 if(ExtensionX0.isActive()) ExtensionX0.moveFrame( dx*Grid.dNx, BC.BC0X);
649 if(ExtensionXN.isActive()) ExtensionXN.moveFrame( dx*Grid.dNx, BC.BCNX);
650 if(ExtensionY0.isActive()) ExtensionY0.moveFrame( dy*Grid.dNy, BC.BC0Y);
651 if(ExtensionYN.isActive()) ExtensionYN.moveFrame( dy*Grid.dNy, BC.BCNY);
652 if(ExtensionZ0.isActive()) ExtensionZ0.moveFrame( dz*Grid.dNz, BC.BC0Z);
653 if(ExtensionZN.isActive()) ExtensionZN.moveFrame( dz*Grid.dNz, BC.BCNZ);
654
655 ConsoleOutput::WriteStandard(thisclassname, "Plane consumed");
656}
