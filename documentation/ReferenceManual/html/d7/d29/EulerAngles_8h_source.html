223 void set(dMatrix3x3 RotMatrix, EulerConvention locConvention)
224 {
225 switch(locConvention)
226 {
227 case XYZ:
228 {
229 Q[1] = asin(std::max(-1.0,std::min(1.0,RotMatrix(0,2))));
230 if (fabs( RotMatrix(0,2) ) < 0.9999999 )
231 {
232 Q[0] = atan2( - RotMatrix(1,2), RotMatrix(2,2) );
233 Q[2] = atan2( - RotMatrix(0,1), RotMatrix(0,0) );
234 }
235 else
236 {
237 Q[0] = atan2( RotMatrix(2,1), RotMatrix(1,1) );
238 Q[2] = 0;
239 }
240 break;
241 }
242 case YXZ:
243 {
244
245 Q[0] = asin(-std::max(-1.0,std::min(RotMatrix(1,2),1.0)));
246
247 if (fabs( RotMatrix(1,2) ) < 0.9999999 )
248 {
249 Q[1] = atan2( RotMatrix(0,2), RotMatrix(2,2) );
250 Q[2] = atan2( RotMatrix(1,0), RotMatrix(1,1) );
251 }
252 else
253 {
254 Q[1] = atan2( RotMatrix(2,0), RotMatrix(0,0) );
255 Q[2] = 0;
256 }
257 break;
258 }
259 case ZXY:
260 {
261 Q[0] = asin(std::max(-1.0,std::min(RotMatrix(2,1),1.0)));
262
263 if (fabs( RotMatrix(2,0) ) < 0.9999999 )
264 {
265 Q[1] = atan2( RotMatrix(2,0), RotMatrix(2,2) );
266 Q[2] = atan2( RotMatrix(0,1), RotMatrix(1,1) );
267 }
268 else
269 {
270 Q[1] = 0;
271 Q[2] = atan2( RotMatrix(1,0), RotMatrix(0,0) );
272 }
273 break;
274 }
275 case ZYX:
276 {
277 Q[1] = asin(-std::max(-1.0,std::min(RotMatrix(2,0),1.0)));
278
279 if (fabs( RotMatrix(2,0) ) < 0.9999999 )
280 {
281
282 Q[0] = atan2( RotMatrix(2,1), RotMatrix(2,2) );
283 Q[2] = atan2( RotMatrix(1,0), RotMatrix(0,0) );
284 }
285 else
286 {
287 Q[0] = 0;
288 Q[2] = atan2( RotMatrix(0,1), RotMatrix(1,1) );
289 }
290 break;
291 }
292 case YZX:
293 {
294 Q[2] = asin(std::max(-1.0,std::min(RotMatrix(1,0),1.0)));
295
296 if (fabs( RotMatrix(1,0) ) < 0.9999999 )
297 {
298 Q[0] = atan2( RotMatrix(1,2), RotMatrix(1,1) );
299 Q[1] = atan2( RotMatrix(2,0), RotMatrix(0,0) );
300 }
301 else
302 {
303 Q[0] = 0;
304 Q[1] = atan2( RotMatrix(0,2), RotMatrix(2,2) );
305 }
306 break;
307 }
308 case XZY:
309 {
310 Q[2] = asin(-std::max(-1.0,std::min(RotMatrix(0,1),1.0)));
311
312 if (fabs( RotMatrix(0,1) ) < 0.9999999 )
313 {
314 Q[0] = atan2( RotMatrix(2,1), RotMatrix(1,1) );
315 Q[1] = atan2( RotMatrix(0,2), RotMatrix(0,0) );
316 }
317 else
318 {
319 Q[0] = atan2( RotMatrix(1,2), RotMatrix(2,2) );
320 Q[1] = 0;
321 }
322 break;
323 }
324 default:
325 {
326 std::cerr << " Wrong Euler Convention selected"
327 << " Check->EulerAngles::setFromRotationMatrix(dMatrix3x3)\n";
328 OP_Exit(EXIT_FAILURE);
329 }
330 }
331
332 Convention = locConvention;
333
334 SinQ[0] = sin(Q[0]);
335 SinQ[1] = sin(Q[1]);
336 SinQ[2] = sin(Q[2]);
337
338 CosQ[0] = cos(Q[0]);
339 CosQ[1] = cos(Q[1]);
340 CosQ[2] = cos(Q[2]);
341
342 IsSet = true;
343 }

347 void set(dVector3 Axis, const double Angle)
348 {
349 double s = sin(Angle);
350 double c = cos(Angle);
351 double t = 1.0 - c;
352 double norm = Axis.length();
353 if(norm == 0)
354 {
355 std::cerr << "Axis length must be greater than zero"
356 << "EulerAngles::set(axis,angle)"
357 << "Terminating!\n";
358 OP_Exit(EXIT_FAILURE);
359 }
360 else
361 {
362 Axis.normalize();
363 }
364 // north pole singularity detected
365 if ((Axis[0]*Axis[1]*t + Axis[2]*s) > 0.998)
366 {
367 Q[0] = 2.0*atan2(Axis[0]*sin(Angle/2),cos(Angle/2));
368 Q[1] = Pi/2.0;
369 Q[2] = 0;
370 }
371 // south pole singularity detected
372 if ((Axis[0]*Axis[1]*t + Axis[2]*s) < -0.998)
373 {
374 Q[0] = -2.0*atan2(Axis[0]*sin(Angle/2),cos(Angle/2));
375 Q[1] = -Pi/2.0;
376 Q[2] = 0;
377 }
378 else
379 {
380 Q[0] = atan2(Axis[1] * s - Axis[0] * Axis[2] * t , 1.0 - (Axis[1]*Axis[1] + Axis[2]*Axis[2]) * t);
381 Q[1] = asin (Axis[0] * Axis[1] * t + Axis[2] * s);
382 Q[2] = atan2(Axis[0] * s - Axis[1] * Axis[2] * t , 1.0 - (Axis[0]*Axis[0] + Axis[2]*Axis[2]) * t);
383 }
384
385 SinQ[0] = sin(Q[0]);
386 SinQ[1] = sin(Q[1]);
387 SinQ[2] = sin(Q[2]);
388
389 CosQ[0] = cos(Q[0]);
390 CosQ[1] = cos(Q[1]);
391 CosQ[2] = cos(Q[2]);
392
393 IsSet = true;
394 }
