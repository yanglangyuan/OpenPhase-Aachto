54void MovingFrame::ReadInput(const std::string InputFileName)
55{
56 ConsoleOutput::WriteLineInsert("MovingFrame input");
57 ConsoleOutput::WriteStandard("Source", InputFileName);
58
59 fstream inp(InputFileName.c_str(), ios::in | ios_base::binary);
60
61 if (!inp)
62 {
63 ConsoleOutput::WriteExit("File \"" + InputFileName + "\" could not be opened", thisclassname, "ReadInput()");
64 OP_Exit(EXIT_FAILURE);
65 };
66
67 std::stringstream data;
68 data << inp.rdbuf();
69 ReadInput(data);
70 inp.close();
71
72 ConsoleOutput::WriteLine();
73}
