187 template <typename T>
188 static std::string StandardMatrix(const T value,
189 const int precision = StandardPrecision,
190 const floatfield notation = StandardNotation,
191 const size_t ColumnWidth = StandardColumnWidth/5)
192 {
193 std::stringstream ss;
194 SetFloationgPointOutputFormat(ss, precision, notation);
195 for (size_t i = 0; i < value.sizeX(); i++)
196 {
197 ss << "\n|| ";
198 for (size_t j = 0; j < value.sizeY(); j++)
199 {
200 ss << sgn(value(i,j)) << std::setw(ColumnWidth) << value(i,j) << " ";
201 }
202 ss << "||";
203 }
204 ss << "\n";
205 return ss.str();
206 };

249 template <typename T>
250 static std::string GetStandard(const std::string& Left, const T Right,
251 [[maybe_unused]] const int precision = StandardPrecision,
252 [[maybe_unused]] const floatfield notation = StandardNotation,
253 const size_t ColumnWidth = StandardColumnWidth)
254 {
255 std::string out = StandardLHS(Left,ColumnWidth);
256 if constexpr (is_matrix<T>::value)
257 {
258 out.append(StandardMatrix(Right,precision,notation,ColumnWidth/5));
259 }
260 else if constexpr (is_vector<T>::value)
261 {
262 out.append(StandardVector(Right,precision,notation,ColumnWidth/5));
263 }
264 else if constexpr (std::is_floating_point<T>::value)
265 {
266 out.append(StandardFloat(Right,precision,notation));
267 }
268 else if constexpr (std::is_signed<T>::value)
269 {
270 out.append(StandardIntegral(Right,precision,notation));
271 }
272 else
273 {
274 std::stringstream ss;
275 ss << " " << Right << "\n";
276 out.append(ss.str());
277 }
278 return out;
279 }
