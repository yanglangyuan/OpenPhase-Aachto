197/*
198 A dynamic array of Plain Old Data. Doesn't support constructors, etc.
199 Has a small initial memory pool, so that low or no usage will not
200 cause a call to new/delete
201*/
202template <class T, size_t INITIAL_SIZE>
203class DynArray
204{
205public:
206 DynArray() :
207 _mem( _pool ),
208 _allocated( INITIAL_SIZE ),
209 _size( 0 )
210 {
211 }
212
213 ~DynArray() {
214 if ( _mem != _pool ) {
215 delete [] _mem;
216 }
217 }
218
219 void Clear() {
220 _size = 0;
221 }
222
223 void Push( T t ) {
224 TIXMLASSERT( _size < INT_MAX );
225 EnsureCapacity( _size+1 );
226 _mem[_size] = t;
227 ++_size;
228 }
229
230 T* PushArr( size_t count ) {
231 TIXMLASSERT( _size <= SIZE_MAX - count );
232 EnsureCapacity( _size+count );
233 T* ret = &_mem[_size];
234 _size += count;
235 return ret;
236 }
237
238 T Pop() {
239 TIXMLASSERT( _size > 0 );
240 --_size;
241 return _mem[_size];
242 }
243
244 void PopArr( size_t count ) {
245 TIXMLASSERT( _size >= count );
246 _size -= count;
247 }
248
249 bool Empty() const {
250 return _size == 0;
251 }
252
253 T& operator[](size_t i) {
254 TIXMLASSERT( i < _size );
255 return _mem[i];
256 }
257
258 const T& operator[](size_t i) const {
259 TIXMLASSERT( i < _size );
260 return _mem[i];
261 }
262
263 const T& PeekTop() const {
264 TIXMLASSERT( _size > 0 );
265 return _mem[ _size - 1];
266 }
267
268 size_t Size() const {
269 return _size;
270 }
271
272 size_t Capacity() const {
273 TIXMLASSERT( _allocated >= INITIAL_SIZE );
274 return _allocated;
275 }
276
277 void SwapRemove(size_t i) {
278 TIXMLASSERT(i < _size);
279 TIXMLASSERT(_size > 0);
280 _mem[i] = _mem[_size - 1];
281 --_size;
282 }
283
284 const T* Mem() const {
285 TIXMLASSERT( _mem );
286 return _mem;
287 }
288
289 T* Mem() {
290 TIXMLASSERT( _mem );
291 return _mem;
292 }
293
294private:
295 DynArray( const DynArray& ); // not supported
296 void operator=( const DynArray& ); // not supported
297
298 void EnsureCapacity( size_t cap ) {
299 TIXMLASSERT( cap > 0 );
300 if ( cap > _allocated ) {
301 TIXMLASSERT( cap <= SIZE_MAX / 2 / sizeof(T));
302 const size_t newAllocated = cap * 2;
303 T* newMem = new T[newAllocated];
304 TIXMLASSERT( newAllocated >= _size );
305 memcpy( newMem, _mem, sizeof(T) * _size ); // warning: not using constructors, only works for PODs
306 if ( _mem != _pool ) {
307 delete [] _mem;
308 }
309 _mem = newMem;
310 _allocated = newAllocated;
311 }
312 }
313
314 T* _mem;
315 T _pool[INITIAL_SIZE];
316 size_t _allocated; // objects allocated
317 size_t _size; // number objects in use
318};

339 Template child class to create pools of the correct type.
340*/
341template< size_t ITEM_SIZE >
342class MemPoolT : public MemPool
343{
344public:
345 MemPoolT() : _blockPtrs(), _root(0), _currentAllocs(0), _nAllocs(0), _maxAllocs(0), _nUntracked(0) {}
346 ~MemPoolT() {
347 MemPoolT< ITEM_SIZE >::Clear();
348 }
349
350 void Clear() {
351 // Delete the blocks.
352 while( !_blockPtrs.Empty()) {
353 Block* lastBlock = _blockPtrs.Pop();
354 delete lastBlock;
355 }
356 _root = 0;
357 _currentAllocs = 0;
358 _nAllocs = 0;
359 _maxAllocs = 0;
360 _nUntracked = 0;
361 }
362
363 virtual size_t ItemSize() const override {
364 return ITEM_SIZE;
365 }
366 size_t CurrentAllocs() const {
367 return _currentAllocs;
368 }
369
370 virtual void* Alloc() override{
371 if ( !_root ) {
372 // Need a new block.
373 Block* block = new Block;
374 _blockPtrs.Push( block );
375
376 Item* blockItems = block->items;
377 for( size_t i = 0; i < ITEMS_PER_BLOCK - 1; ++i ) {
378 blockItems[i].next = &(blockItems[i + 1]);
379 }
380 blockItems[ITEMS_PER_BLOCK - 1].next = 0;
381 _root = blockItems;
382 }
383 Item* const result = _root;
384 TIXMLASSERT( result != 0 );
385 _root = _root->next;
386
387 ++_currentAllocs;
388 if ( _currentAllocs > _maxAllocs ) {
389 _maxAllocs = _currentAllocs;
390 }
391 ++_nAllocs;
392 ++_nUntracked;
393 return result;
394 }
395
396 virtual void Free( void* mem ) override {
397 if ( !mem ) {
398 return;
399 }
400 --_currentAllocs;
401 Item* item = static_cast<Item*>( mem );
402#ifdef TINYXML2_DEBUG
403 memset( item, 0xfe, sizeof( *item ) );
404#endif
405 item->next = _root;
406 _root = item;
407 }
408 void Trace( const char* name ) {
409 printf( "Mempool %s watermark=%d [%dk] current=%d size=%d nAlloc=%d blocks=%d\n",
410 name, _maxAllocs, _maxAllocs * ITEM_SIZE / 1024, _currentAllocs,
411 ITEM_SIZE, _nAllocs, _blockPtrs.Size() );
412 }
413
414 void SetTracked() override {
415 --_nUntracked;
416 }
417
418 size_t Untracked() const {
419 return _nUntracked;
420 }
421
422 // This number is perf sensitive. 4k seems like a good tradeoff on my machine.
423 // The test file is large, 170k.
424 // Release: VS2010 gcc(no opt)
425 // 1k: 4000
426 // 2k: 4000
427 // 4k: 3900 21000
428 // 16k: 5200
429 // 32k: 4300
430 // 64k: 4000 21000
431 // Declared public because some compilers do not accept to use ITEMS_PER_BLOCK
432 // in private part if ITEMS_PER_BLOCK is private
433 enum { ITEMS_PER_BLOCK = (4 * 1024) / ITEM_SIZE };
434
435private:
436 MemPoolT( const MemPoolT& ); // not supported
437 void operator=( const MemPoolT& ); // not supported
438
439 union Item {
440 Item* next;
441 char itemData[static_cast<size_t>(ITEM_SIZE)];
442 };
443 struct Block {
444 Item items[ITEMS_PER_BLOCK];
445 };
446 DynArray< Block*, 10 > _blockPtrs;
447 Item* _root;
448
449 size_t _currentAllocs;
450 size_t _nAllocs;
451 size_t _maxAllocs;
452 size_t _nUntracked;
453};
