140SparseMatrixCSR SparseMatrixCSR::operator*(const SparseMatrixCSR& rMatrix) const
141{
142 SparseMatrixCSR tmp(storage.size(),reserve);
143 #pragma omp parallel for
144 for (size_t i = 0; i < storage.size(); ++i)
145 for (size_t j = 0; j < storage.size(); ++j)
146 {
147 for (auto k = storage[i].cbegin(); k != storage[i].cend(); ++k)
148 {
149 if (k->value*rMatrix(k->index,j) != 0.)
150 {
151 tmp(i,j) += k->value*rMatrix(k->index,j);
152 }
153 }
154 }
155 return tmp;
156}

303size_t SparseMatrixCSR::MaxRow(const size_t i, const size_t j)
304{
305 assert(i < storage.size() && "MaxRow::operator(): Access beyond matrix size");
306
307 size_t max_i = i;
308 double max_value = 0.0;
309
310 #pragma omp parallel
311 {
312 size_t loc_max_i = max_i;
313 double loc_max_value = max_value;
314
315 #pragma omp for
316 for (size_t k = i; k < storage.size(); k++)
317 for (auto it : storage[k])
318 if (it.index == j)
319 {
320 if (std::abs(it.value) > std::abs(loc_max_value))
321 {
322 loc_max_i = k;
323 loc_max_value = it.value;
324 }
325 break;
326 }
327
328 #pragma omp critical
329 {
330 if (std::abs(loc_max_value) > std::abs(max_value))
331 {
332 max_i = loc_max_i;
333 max_value = loc_max_value;
334 }
335 }
336 }
337 return max_i;
338}

383bool operator==(const SparseMatrixCSR& lhs, const SparseMatrixCSC& rhs)
384{
385 bool eq = true;
386 for (size_t i = 0; i < lhs.storage.size(); i++)
387 for (auto j = lhs.storage[i].cbegin(); j != lhs.storage[i].cend(); j++)
388 {
389 eq = eq and (j->value == rhs(i,j->index));
390 }
391
392 for (size_t j = 0; j < rhs.storage.size(); j++)
393 for (auto i = rhs.storage[j].cbegin(); i != rhs.storage[j].cend(); i++)
394 {
395 eq = eq and (lhs(i->index,j) == i->value);
396 }
397
398 return eq;
