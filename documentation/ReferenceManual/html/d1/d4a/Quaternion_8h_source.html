114inline Quaternion::Quaternion(std::array<double,4> vecinit)
115{
116 //assert(vecinit.size() == 4 && "Initialization list size is not equal to storage range");
117
118 int ii = 0;
119 s = *vecinit.begin();
120 for (auto it = vecinit.begin()+1; it != vecinit.end(); it++)
121 {
122 v[ii] = *it;
123 ii += 1;
124 }
125 setRotationMatrix();
126}

176inline void Quaternion::set(dVector3 Axis, const double Angle)
177{
178 Axis.normalize();
179 // https://www.euclideanspace.com/maths/geometry/rotations/conversions/angleToQuaternion/index.htm
180 s = cos(Angle/2);
181 v[0] = Axis[0] * sin(Angle/2);
182 v[1] = Axis[1] * sin(Angle/2);
183 v[2] = Axis[2] * sin(Angle/2);
184 normalize();
185 setRotationMatrix();
186}

257inline Quaternion Quaternion::operator*(const Quaternion rhS) const
258{
259 Quaternion result;
260 result.v[0] = v[0]*rhS.s + v[1]*rhS.v[2] - v[2]*rhS.v[1] + s*rhS.v[0];
261 result.v[1] = -v[0]*rhS.v[2] + v[1]*rhS.s + v[2]*rhS.v[0] + s*rhS.v[1];
262 result.v[2] = v[0]*rhS.v[1] - v[1]*rhS.v[0] + v[2]*rhS.s + s*rhS.v[2];
263 result.s = -v[0]*rhS.v[0] - v[1]*rhS.v[1] - v[2]*rhS.v[2] + s*rhS.s;
264 //result.setRotationMatrix(); // RotationMatrix will be set in assignment operator
265 return result;
266}

268inline Quaternion& Quaternion::operator*=(const Quaternion rhS)
269{
270 s = s*rhS.s - v*rhS.v;
271 v = v.cross(rhS.v) + rhS.v*s + v*rhS.s;
272 setRotationMatrix();
273 return *this;
274}

327inline Quaternion& Quaternion::invert()
328{
329 // if |q|=1 then q inverse = q conjugate
330 v = v*(-1.0);
331 *this /= (s*s + v[0]*v[0] + v[1]*v[1] + v[2]*v[2]);
332 return *this;
333}
