44void LinearMagneticSolver::Initialize(Settings& locSettings, std::string ObjectNameSuffix)
45{
46 thisclassname = "LinearMagneticSolver";
47 thisobjectname = thisclassname + ObjectNameSuffix;
48
49 Grid = locSettings.Grid;
50
51 Nphases = locSettings.Nphases;
52
53 int Bcells = std::max(Grid.Bcells, 2);

88void LinearMagneticSolver::ReadInput(std::stringstream& inp)
89{
90 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
91
92 H0x = FileInterface::ReadParameterD(inp, moduleLocation,"H0X", false, 0.0);
93 H0y = FileInterface::ReadParameterD(inp, moduleLocation,"H0Y", false, 0.0);
94 H0z = FileInterface::ReadParameterD(inp, moduleLocation,"H0Z", false, 0.0);
95 dH0x_dx = FileInterface::ReadParameterD(inp, moduleLocation,"H0XX", false, 0.0);
96 dH0y_dx = FileInterface::ReadParameterD(inp, moduleLocation,"H0YX", false, 0.0);
97 dH0z_dx = FileInterface::ReadParameterD(inp, moduleLocation,"H0ZX", false, 0.0);
98 dH0x_dy = FileInterface::ReadParameterD(inp, moduleLocation,"H0XY", false, 0.0);
99 dH0y_dy = FileInterface::ReadParameterD(inp, moduleLocation,"H0YY", false, 0.0);
100 dH0z_dy = FileInterface::ReadParameterD(inp, moduleLocation,"H0ZY", false, 0.0);
101 dH0x_dz = FileInterface::ReadParameterD(inp, moduleLocation,"H0XZ", false, 0.0);
102 dH0y_dz = FileInterface::ReadParameterD(inp, moduleLocation,"H0YZ", false, 0.0);
103 dH0z_dz = FileInterface::ReadParameterD(inp, moduleLocation,"H0ZZ", false, 0.0);
104 //MaxResidual = UserInterface::ReadParameterD(inp, moduleLocation,"MaxRes", false, 1.0e-2);
105
106 PhaseChi.resize(Nphases);
107 for (size_t i = 0; i < Nphases; ++i)
108 {
109 std::stringstream converter;
110 converter << i;
111 std::string counter = converter.str();
112 PhaseChi[i] = FileInterface::ReadParameterD(inp, moduleLocation, std::string("chi_") + counter);
113 }
114
115 ConsoleOutput::WriteLine();
116 ConsoleOutput::WriteBlankLine();
117}

119void LinearMagneticSolver::SetEffectiveSusceptibility(
120 const PhaseField& Phase, const BoundaryConditions& BC)
121{
122 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,chi,0)
123 {
124 chi(i,j,k) = 0.0;
125 for (auto it = Phase.Fields(i,j,k).cbegin(); it != Phase.Fields(i,j,k).cend(); ++it)
126 {
127 chi(i,j,k) += PhaseChi[Phase.FieldsProperties[it->index].Phase]*it->value;
128 }
129 }
130 OMP_PARALLEL_STORAGE_LOOP_END
131
132 BC.SetX(chi);
133 BC.SetY(chi);
134 BC.SetZ(chi);
135}

137void LinearMagneticSolver::Solve(const BoundaryConditions& BC, double MaxResidual)
138{
139#ifdef MPI_PARALLEL
140 ConsoleOutput::WriteExit("MPI Parallelism not implemented !", "LinearMagneticSolver", "Solve");
141 OP_Exit(EXIT_FAILURE);
142#endif
143
144 A.clear();
145 for (int i = 0; i < Grid.Nx; ++i)
146 for (int j = 0; j < Grid.Ny; ++j)
147 for (int k = 0; k < Grid.Nz; ++k)
148 {
149 const double dchi_dx = (Grid.dNx) ? (chi(i+1,j,k)-chi(i-1,j,k))/2.0/Grid.dx : 0;
150 const double dchi_dy = (Grid.dNy) ? (chi(i,j+1,k)-chi(i,j-1,k))/2.0/Grid.dx : 0;
151 const double dchi_dz = (Grid.dNz) ? (chi(i,j,k+1)-chi(i,j,k-1))/2.0/Grid.dx : 0;
152
153 const double mu_rel = (chi(i,j,k)+1);
154
155 const double dphi0_dx = (Grid.dNx) ? -H0x - (dH0x_dx*i + dH0x_dy*j + dH0x_dz*k)*Grid.dx : 0;
156 const double dphi0_dy = (Grid.dNy) ? -H0y - (dH0y_dx*i + dH0y_dy*j + dH0y_dz*k)*Grid.dx : 0;
157 const double dphi0_dz = (Grid.dNz) ? -H0z - (dH0z_dx*i + dH0z_dy*j + dH0z_dz*k)*Grid.dx : 0;
158 const double Laplacephi0 = -dH0x_dx - dH0y_dy - dH0z_dz;
159
160 x[idx(i,j,k)] = 0.0;
161 b[idx(i,j,k)] = -dchi_dx*dphi0_dx - dchi_dy*dphi0_dy - dchi_dz*dphi0_dz - mu_rel*Laplacephi0;
162 A(idx(i,j,k),idx(i,j,k)) = -6.0*mu_rel/Grid.dx/Grid.dx;
163
164 const double Axp = dchi_dx/2.0/Grid.dx + mu_rel/Grid.dx/Grid.dx;
165 const double Axm = -dchi_dx/2.0/Grid.dx + mu_rel/Grid.dx/Grid.dx;
166 const double Ayp = dchi_dy/2.0/Grid.dx + mu_rel/Grid.dx/Grid.dx;
167 const double Aym = -dchi_dy/2.0/Grid.dx + mu_rel/Grid.dx/Grid.dx;
168 const double Azp = dchi_dz/2.0/Grid.dx + mu_rel/Grid.dx/Grid.dx;
169 const double Azm = -dchi_dz/2.0/Grid.dx + mu_rel/Grid.dx/Grid.dx;
170
171 if (i+1 < Grid.Nx) A(idx(i,j,k),idx(i+1,j,k)) = Axp;
172 else switch (BC.BCNX)
173 {
174 case BoundaryConditionTypes::NoFlux: { A(idx(i,j,k),idx(i-1,j,k)) += Axp; break; }
175 case BoundaryConditionTypes::Periodic: { A(idx(i,j,k),idx( 0,j,k)) += Axp; break; }
176 //case Free: { A(idx(i,j,k),idx(Nx-1,j,k)) += 2*Axp;
177 // A(idx(i,j,k),idx(Nx-2,j,k)) -= Axp; break; }
178 case BoundaryConditionTypes::Fixed: break; // Fixes potential to zero which results in an arbitrary but constant magnetic field along the boundary
179 default: { throw std::invalid_argument("Boundary Condition not implemented!"); }
180 }
181 if (i >= 1 ) A(idx(i,j,k),idx(i-1,j,k)) = Axm;
182 else switch (BC.BC0X)
183 {
184 case BoundaryConditionTypes::NoFlux: { A(idx(i,j,k),idx( i+1,j,k)) += Axm; break; }
185 case BoundaryConditionTypes::Periodic: { A(idx(i,j,k),idx(Grid.Nx-1,j,k)) += Axm; break; }
186 //case Free: { A(idx(i,j,k),idx( 0,j,k)) += 2*Axm;
187 // A(idx(i,j,k),idx( 1,j,k)) -= Axm; break; }
188 case BoundaryConditionTypes::Fixed: break; // Fixes potential to zero which results in an arbitrary but constant magnetic field along the boundary
189 default: { throw std::invalid_argument("Boundary Condition not implemented!"); }
190 }
191 if (j+1 < Grid.Ny) A(idx(i,j,k),idx(i,j+1,k)) = Ayp;
192 else switch (BC.BCNY)
193 {
194 case BoundaryConditionTypes::NoFlux: { A(idx(i,j,k),idx(i, j-1,k)) += Ayp; break; }
195 case BoundaryConditionTypes::Periodic: { A(idx(i,j,k),idx(i, 0,k)) += Ayp; break; }
196 case BoundaryConditionTypes::Fixed: break; // Fixes potential to zero which results in an arbitrary but constant magnetic field along the boundary
197 default: { throw std::invalid_argument("Boundary Condition not implemented!"); }
198 }
199 if (j >= 1 ) A(idx(i,j,k),idx(i,j-1,k)) = Aym;
200 else switch (BC.BC0Y)
201 {
202 case BoundaryConditionTypes::NoFlux: { A(idx(i,j,k),idx(i, j+1,k)) += Aym; break; }
203 case BoundaryConditionTypes::Periodic: { A(idx(i,j,k),idx(i, Grid.Ny-1,k)) += Aym; break; }
204 case BoundaryConditionTypes::Fixed: break; // Fixes potential to zero which results in an arbitrary but constant magnetic field along the boundary
205
206 default: { throw std::invalid_argument("Boundary Condition not implemented!"); }
207 }
208 if (k+1 < Grid.Nz) A(idx(i,j,k),idx(i,j,k+1)) = Azp;
209 else switch (BC.BCNZ)
210 {
211 case BoundaryConditionTypes::NoFlux: { A(idx(i,j,k),idx(i,j, k-1)) += Azp; break; }
212 case BoundaryConditionTypes::Periodic: { A(idx(i,j,k),idx(i,j, 0)) += Azp; break; }
213 case BoundaryConditionTypes::Fixed: break; // Fixes potential to zero which results in an arbitrary but constant magnetic field along the boundary
214 default: { throw std::invalid_argument("Boundary Condition not implemented!"); }
215 }
216 if (k >= 1 ) A(idx(i,j,k),idx(i,j,k-1)) = Azm;
217 else switch (BC.BC0Z)
218 {
219 case BoundaryConditionTypes::NoFlux: { A(idx(i,j,k),idx(i,j, k+1)) += Azm; break; }
220 case BoundaryConditionTypes::Periodic: { A(idx(i,j,k),idx(i,j, Grid.Nz-1)) += Azm; break; }
221 case BoundaryConditionTypes::Fixed: break; // Fixes potential to zero which results in an arbitrary but constant magnetic field along the boundary
222 default: { throw std::invalid_argument("Boundary Condition not implemented!"); }
223 }
224 }
225
226 //SystemOfLinearEquationsSolvers::Pivote(A,b);
227 SystemOfLinearEquationsSolvers::FastPivote(A,b);
228 SystemOfLinearEquationsSolvers::PreconditionJacobi(A,b);
229 SystemOfLinearEquationsSolvers::BiconjugateGradient(A,x,b,MaxResidual);
230
231 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,phi,0)
232 {
233 phi(i,j,k) = x[idx(i,j,k)];
234 }
235 OMP_PARALLEL_STORAGE_LOOP_END
236
237 BC.SetX(phi);
238 BC.SetY(phi);
239 BC.SetZ(phi);
240}

242void LinearMagneticSolver::CalcForceDensity(ElasticProperties& EP,
243 const BoundaryConditions& BC) const
244{
245 if(EP.ConsiderExternalForces)
246 {
247 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,EP.ForceDensity,0)
248 {
249 EP.ForceDensity(i,j,k) = ForceDensity(i,j,k);
250 }
251 OMP_PARALLEL_STORAGE_LOOP_END
252 BC.SetX(EP.ForceDensity);
253 BC.SetY(EP.ForceDensity);
254 BC.SetZ(EP.ForceDensity);
255 }
256 else
257 {
258 ConsoleOutput::WriteExit("ForceDensity Storage not allocated!", "LinearMagneticSolver", "CalcForceDensity");
259 OP_Exit(EXIT_FAILURE);
260 }

263void LinearMagneticSolver::WriteVTK(const int tStep, const Settings& locSettings)
264{
265 std::vector<VTK::Field_t> ListOfFields;
266 ListOfFields.push_back((VTK::Field_t){"H" , [this](int i,int j,int k){return H (i, j, k);}});
267 ListOfFields.push_back((VTK::Field_t){"B" , [this](int i,int j,int k){return B (i, j, k);}});
268 ListOfFields.push_back((VTK::Field_t){"M" , [this](int i,int j,int k){return M (i, j, k);}});
269 ListOfFields.push_back((VTK::Field_t){"mu" , [this](int i,int j,int k){return mu (i, j, k);}});
270 ListOfFields.push_back((VTK::Field_t){"phi" , [this](int i,int j,int k){return phi (i, j, k);}});
271 ListOfFields.push_back((VTK::Field_t){"chi" , [this](int i,int j,int k){return chi (i, j, k);}});
272 ListOfFields.push_back((VTK::Field_t){"ForceDensity", [this](int i,int j,int k){return ForceDensity(i, j, k);}});
273
274 std::string Filename = FileInterface::MakeFileName(locSettings.VTKDir, thisclassname, tStep, ".vts");
275
276 VTK::Write(Filename, locSettings, ListOfFields);
277}
