155double ThermodynamicPhase::getSubstitutionalSites()
156{
157 double result = 0.0;
158 for(size_t sub = 0; sub < Nsubs; sub++)
159 {
160 bool substitutional = true;
161 for(size_t cons = 0; cons < Sublattice[sub].Ncons; cons++)
162 if(Sublattice[sub].Constituent[cons].Index == -1)
163 substitutional = false;
164
165 if(substitutional)
166 result += Sublattice[sub].Site;
167 }
168
169 return result;
170}

211double ThermodynamicPhase::Nmoles(const Tensor<double,1> Cx)
212{
217 size_t counter = 0;
218 double tempNmoles = 0.0;
219 for(size_t n = 0; n < Nsubs; n++)
220 {
221 double tempSite = Sublattice[n].Site;
222 double temp = 0.0;
223 for(size_t i = 0; i < Sublattice[n].Ncons; i++)
224 {
225 if(Sublattice[n].Constituent[i].Index >= 0)
226 {
227 temp += Cx({counter});
228 }
229 counter++;
230 }
231 tempNmoles += temp*tempSite;
232 }
233 return tempNmoles;
234}

343bool ThermodynamicPhase::isEndmember(int CIdxI)
344{
349 bool Iispresent = false;
350 bool Iisendmember = false;
351
352 for(size_t sub = 0; sub < Nsubs; sub++)
353 if(Sublattice[sub].isElementPresent(CIdxI))
354 Iispresent = true;
355
356 if(Iispresent)
357 {
358 Iisendmember = true;
359 for(size_t sub = 0; sub < Nsubs; sub++)
360 if((not Sublattice[sub].isElementPresent(CIdxI))
361 and(not Sublattice[sub].hasVacancies)
362 /*and(Sublattice[sub].Ncons > 1)*/)
363 {// no I nor Va on sublattice -> I not an endmember (if Ncons > 1)
364 Iisendmember = false;
365 }
366 }

385bool ThermodynamicPhase::isStoichiometric(int CIdxI)
386{
390 bool aloneinsubl = true;
391 bool nointerstitials = true;
392
393 bool isStoich = false;
394
395 for(size_t sub = 0; sub < Nsubs; sub++)
396 {
397 if((Sublattice[sub].isElementPresent(CIdxI)) and Sublattice[sub].Ncons > 1)
398 {
399 aloneinsubl = false;
400 }
401
402 if(Sublattice[sub].hasVacancies and Sublattice[sub].Ncons > 1)
403 {
404 nointerstitials = false;
405 }
406 }
407
408 if(aloneinsubl and nointerstitials)
409 isStoich = true;
410
411 return isStoich;
412}
