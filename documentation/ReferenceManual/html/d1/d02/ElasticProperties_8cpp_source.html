257 string tmp2 = FileInterface::ReadParameterK(inp, moduleLocation, BCnames[n], 0);

259 if (tmp2 == "FREEBOUNDARIES")

261 AppliedStressMask[n] = 1.0;

272 AppliedStrainRateMask[n] = 1.0;
273 }
274 if (tmp2 == "APPLIEDSTRESS")
275 {
276 AppliedStressMask[n] = 1.0;
277 AppliedStress[n] = FileInterface::ReadParameterD(inp, moduleLocation, BCvalueNames[n]);
278 }
279 }
280
281 string tmp3 = FileInterface::ReadParameterK(inp, moduleLocation, "Restrict", 0);
282
283 if (tmp3 == "ASPECTRATIO")
284 {
285 KeepAspectRatio = true;
286 }
287
288 if (tmp3 == "SHEAR")
289 {
290 PreventShear = true;
291 }

293 for(size_t alpha = 0; alpha != Nphases; alpha++)
294 {
295 PhaseElasticConstants[alpha].set_to_zero();
296 }
297
298 string tmp4 = FileInterface::ReadParameterK(inp, moduleLocation, "ElasticConstantMode", false, "ELASTICMODULI");
299
300 if (tmp4 != "ELASTICMODULI" and tmp4 != "TENSOR")
301 {
302 std::cerr << "WARNING: ElasticConstantMode should be ELASTICMODULI or TENSOR, trying ELASTICMODULI." << std::endl;
303 tmp4 = "ELASTICMODULI";
304 }
305
306 for(size_t pIndex = 0; pIndex < Nphases; pIndex++)
307 {
308 // Read elastic moduli if provided
309 std::stringstream KConv;
310 std::stringstream EConv;
311 std::stringstream LConv;
312 std::stringstream GConv;
313 std::stringstream NConv;
314 std::stringstream MConv;
315
316 KConv << "K_" << pIndex;
317 EConv << "E_" << pIndex;
318 LConv << "L_" << pIndex;
319 GConv << "G_" << pIndex;
320 NConv << "Nu_"<< pIndex;
321 MConv << "M_" << pIndex;
322
323 double K = FileInterface::ReadParameterD(inp, moduleLocation, KConv.str(), false, 0);
324 double E = FileInterface::ReadParameterD(inp, moduleLocation, EConv.str(), false, 0);
325 double lambda = FileInterface::ReadParameterD(inp, moduleLocation, LConv.str(), false, 0);
326 double G = FileInterface::ReadParameterD(inp, moduleLocation, GConv.str(), false, 0);
327 double nu = FileInterface::ReadParameterD(inp, moduleLocation, NConv.str(), false, 0);
328 double M = FileInterface::ReadParameterD(inp, moduleLocation, MConv.str(), false, 0);
329
330 if (K != 0.0 and E != 0.0) {lambda = 3.0*K*(3.0*K-E)/(9.0*K-E); G = 3.0*K*E/(9.0*K-E);}
331 else if (K != 0.0 and lambda != 0.0) { G = 3*(K-lambda)/2.0;}
332 else if (K != 0.0 and G != 0.0) {lambda = K-2.0*G/3.0;}
333 else if (K != 0.0 and nu != 0.0) {lambda = 3.0*K*nu/(1+nu); G = 3*K*(1-2.0*nu)/(2.0*(1.0+nu));}
334 else if (K != 0.0 and M != 0.0) {lambda = (3.0*K-M)/2; G = 3.0*(M-K)/4.0;}
335 else if (E != 0.0 and lambda != 0.0) { G = (E-3.0*lambda+std::sqrt(E*E+9.0*lambda*lambda+2.0*E*lambda))/4.0;}
336 else if (E != 0.0 and G != 0.0) {lambda = G*(E-2.0*G)/(3.0*G-E);}
337 else if (E != 0.0 and nu != 0.0) {lambda = E*nu/((1.0+nu)*(1.0-2.0*nu)); G = E/(2.0*(1.0+nu));}
338 else if (E != 0.0 and M != 0.0) {ConsoleOutput::WriteExit("Choice of elastic moduli is not unique", thisclassname, "ReadInput"); std::exit(EXIT_FAILURE);}
339 else if (lambda != 0.0 and nu != 0.0) { G = lambda*(1.0-2.0*nu)/(2.0*nu);}
340 else if (lambda != 0.0 and M != 0.0) { G = (M-lambda)/2.0;}
341 else if (G != 0.0 and nu != 0.0) {lambda = 2.0*G*nu/(1.0-2.0*nu);}
342 else if (G != 0.0 and M != 0.0) {lambda = M-2.0*G;}
343 else if (nu != 0.0 and M != 0.0) {lambda = M*nu/(1.0-nu); G = M*(1.0-2.0*nu)/(2.0*(1.0-nu));}
344
345 if (std::fabs(lambda) > DBL_EPSILON and std::fabs(G) > DBL_EPSILON and tmp4 == "ELASTICMODULI")
346 {
347 PhaseElasticConstants[pIndex](0,0) = 2.0*G + lambda;
348 PhaseElasticConstants[pIndex](1,1) = 2.0*G + lambda;
349 PhaseElasticConstants[pIndex](2,2) = 2.0*G + lambda;
350
351 PhaseElasticConstants[pIndex](0,1) = lambda;
352 PhaseElasticConstants[pIndex](0,2) = lambda;
353 PhaseElasticConstants[pIndex](1,2) = lambda;
354
355 PhaseElasticConstants[pIndex](1,0) = lambda;
356 PhaseElasticConstants[pIndex](2,0) = lambda;
357 PhaseElasticConstants[pIndex](2,1) = lambda;
358
359 PhaseElasticConstants[pIndex](3,3) = G;
360 PhaseElasticConstants[pIndex](4,4) = G;
361 PhaseElasticConstants[pIndex](5,5) = G;
362
363 }
364 else
365 {
366 stringstream converter;
367 converter << "C" << "_" << pIndex;
368 PhaseElasticConstants[pIndex] = FileInterface::ReadParameterM6x6(inp, moduleLocation, converter.str(),false,dMatrix6x6::ZeroTensor());
369 if(PhaseElasticConstants[pIndex] == dMatrix6x6::ZeroTensor())
370 {
371 for(int ii = 1; ii <= 6; ii++)
372 for(int jj = ii; jj <= 6; jj++)
373 {
374 stringstream Cij;
375 Cij << "C" << ii << jj << "_" << pIndex;
376
377 PhaseElasticConstants[pIndex](ii-1,jj-1) = FileInterface::ReadParameterD(inp, moduleLocation, Cij.str(),false,0);
378 if(ii != jj)
379 {
380 PhaseElasticConstants[pIndex](jj-1,ii-1) = PhaseElasticConstants[pIndex](ii-1,jj-1);
381 }
382 }
383 }
384 }
385 PoissonRatio[pIndex] = PhaseElasticConstants[pIndex](0,1) / (PhaseElasticConstants[pIndex](0,1) + PhaseElasticConstants[pIndex](0,0));
386
387 // Check if stiffness constants are read.
388 if (PhaseElasticConstants[pIndex].norm() <= DBL_EPSILON)
389 {
390 std::string message = "Elastic constants for phase " + to_string(pIndex) + " not set.";
391 ConsoleOutput::WriteExit(message, thisclassname, "ReadInput()");
392 exit(3);
393 }
394 }
395
396 for(size_t pIndex = 0; pIndex < Nphases; pIndex++)
397 {
398 stringstream converter;
399 converter << "U_" << pIndex;
400 PhaseTransformationStretches[pIndex] = FileInterface::ReadParameterM3x3(inp, moduleLocation, converter.str(),true,dMatrix3x3::UnitTensor());
401 }
402 // Considering external forces
403 ConsiderExternalForces = FileInterface::ReadParameterB(inp, moduleLocation, "ConsiderExternalForces", false, false);
404 if(ConsiderExternalForces)
405 {
406 ForceDensity.Allocate(Grid, 1);
407 }
408
409 // Reading chemo-mechanical coupling parameters if considered

412 {
413 int counter = 0;
414 for(size_t comp = 0; comp < Ncomp; comp++)
415 for(size_t alpha = 0; alpha != Nphases; alpha++)
416 {

425 for (size_t comp = 0; comp < Ncomp; comp++)

544void ElasticProperties::Remesh(int newNx, int newNy, int newNz, const BoundaryConditions& BC)
545{
546 SetBoundaryConditions(BC);
548 dMatrix3x3 stretch;
549 stretch.set_to_zero();
550 stretch(0,0) = double(newNx)/double(Grid.Nx);
551 stretch(1,1) = double(newNy)/double(Grid.Ny);
552 stretch(2,2) = double(newNz)/double(Grid.Nz);
553
554 RemeshedStrain[0] += stretch(0,0) - 1.0;
555 RemeshedStrain[1] += stretch(1,1) - 1.0;
556 RemeshedStrain[2] += stretch(2,2) - 1.0;
557
558 StrainToRemesh[0] -= stretch(0,0) - 1.0;
559 StrainToRemesh[1] -= stretch(1,1) - 1.0;
560 StrainToRemesh[2] -= stretch(2,2) - 1.0;
561
562 Grid.SetDimensions(newNx, newNy, newNz);
563
564 Stresses.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
565 StressIncrements.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
566
567 DeformationGradientsTotal.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
568 DeformationGradientsEigen.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
569
570 DeformationGradientsPlastic.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
571 VelocityGradientsPlastic.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
572
573 Displacements.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
574 EffectiveElasticConstants.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
575 ElasticConstants.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
576 TransformationStretches.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
577
578 if(VelocityGradientsTotal.IsAllocated()) VelocityGradientsTotal.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
579 if(LocalRotations.IsAllocated()) LocalRotations.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
580
581 DeformationJumps.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
582 NonLocalDeformationJumpTerm.Remesh(Grid.Nx, Grid.Ny, Grid.Nz);
583
584 SetBoundaryConditions(BC);
585
586 //WriteRemeshingData(tStep, sim_time);
587 ConsoleOutput::Write(thisclassname, "Remeshed");
588}

590void ElasticProperties::SetGrainsProperties(const PhaseField& Phase)
591{
592 size_t size = Phase.FieldsProperties.size();
593 if(GrainTransformationStretches.size() != size)
594 {
595 GrainTransformationStretches.Reallocate(size);
596 GrainElasticConstants.Reallocate(size);
597 if(Ncomp)
598 {
599 GrainLambda.Reallocate({size, Ncomp});
600 GrainKappa.Reallocate({size, Ncomp});
601 }
602 GrainAlpha.Reallocate(size);
603 GrainGamma.Reallocate(size);
604 }
605 for(size_t alpha = 0; alpha != size; alpha++)
606 if(Phase.FieldsProperties[alpha].Exist)
607 {
608 size_t pIndex = Phase.FieldsProperties[alpha].Phase;
609 size_t vIndex = Phase.FieldsProperties[alpha].Variant;
610
611 GrainTransformationStretches[alpha] = PhaseTransformationStretches[pIndex];
612 GrainElasticConstants[alpha] = PhaseElasticConstants[pIndex];
613 GrainAlpha[alpha] = PhaseAlpha[pIndex];
614 GrainGamma[alpha] = PhaseGamma[pIndex];
615
616 if(Variants.set)
617 {
618 GrainTransformationStretches[alpha].rotate(Variants(pIndex, vIndex));
619 GrainElasticConstants[alpha].rotate(Variants(pIndex, vIndex));
620 GrainAlpha[alpha].rotate(Variants(pIndex, vIndex));
621 GrainGamma[alpha].rotate(Variants(pIndex, vIndex));
622 }
623
624 GrainTransformationStretches[alpha].rotate(Phase.FieldsProperties[alpha].Orientation.RotationMatrix);

915void ElasticProperties::AddInterfaceStressContribution(const PhaseField& Phase, const InterfaceProperties& IP)
916{
917 //Raphael Schiedung, Ingo Steinbach, and Fathollah Varnik.
918 //"Multi-phase-field method for surface tension induced elasticity."
919 //Physical Review B 97.3 (2018): 035410.
920 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i, j, k, DeformationGradientsEigen,0,)
921 if (Phase.Fields(i,j,k).interface())
922 {
923 const double pre1 = 4.0/Phase.Grid.Eta;
924 const double pre2 = Phase.Grid.Eta * Phase.Grid.Eta/Pi;
925
926 vStress locInterfaceStress;
927 // Calculate phase gradients and normals
928 const NodeAB<dVector3,dVector3>& locNormals = Phase.Normals(i,j,k);
929 const NodeA<dVector3>& locGradients = Phase.Fields(i,j,k).get_gradients();
930
931 for (auto it = locNormals.cbegin(); it < locNormals.cend(); ++it)
932 {
933 // Calculate phase field part of the interface stress
934 const double PhiA = abs(Phase.Fields(i,j,k).get_value(it->indexA));
935 const double PhiB = abs(Phase.Fields(i,j,k).get_value(it->indexB));
936
937 if (PhiA * PhiB < 0.02) continue;
938
939 const double interpol = pre1 * (pre2 *
940 (locGradients.get_value(it->indexA) *
941 locGradients.get_value(it->indexB)) + PhiA * PhiB);
942
943 // Calculate local projection matrix
944 dMatrix3x3 Projection;
945 Projection.set_to_unity();
946 Projection(0,0) -= it->value1[0]*it->value1[0];
947 Projection(1,0) -= it->value1[1]*it->value1[0];
948 Projection(2,0) -= it->value1[2]*it->value1[0];
949 Projection(0,1) -= it->value1[0]*it->value1[1];
950 Projection(1,1) -= it->value1[1]*it->value1[1];
951 Projection(2,1) -= it->value1[2]*it->value1[1];
952 Projection(0,2) -= it->value1[0]*it->value1[2];
953 Projection(1,2) -= it->value1[1]*it->value1[2];
954 Projection(2,2) -= it->value1[2]*it->value1[2];
955
956 // Calculate local interface stress
957 locInterfaceStress -= VoigtStress(Projection) * interpol *
958 IP.Properties(i,j,k).get_energy(it->indexA,it->indexB);
959 }
960
961 //TODO account for derivatives Interface energy with respect to the strain
962 vStrain locEigenStrainInterface =
963 EffectiveElasticConstants(i,j,k).inverted() * locInterfaceStress *(-1.);
964
965 if(StrainModel == StrainModels::Small)
966 {
967 DeformationGradientsEigen(i,j,k) += locEigenStrainInterface.tensor();
968 }
969// else
970// {
971// vStrain locEigenStrain = StrainGreenLagrange(DeformationGradientsEigen(i,j,k));
972// locEigenStrain += locEigenStrainInterface;
973//
974// DeformationGradientsEigen(i,j,k) =
975// sqrtM3x3(locEigenStrain.tensor()*2.0 + dMatrix3x3::UnitTensor());
976// }
977 }
978 OMP_PARALLEL_STORAGE_LOOP_END
979}

1201void ElasticProperties::CalculateDrivingForce(const PhaseField& Phase, DrivingForce& dGab)
1202{
1203 if(ElasticityModel == ElasticityModels::Rank1)
1204 {
1205 CalculateDeformationJumps(Phase);
1206 }
1207 if(ElasticityModel == ElasticityModels::Rank1NL)
1208 {
1209 CalculateDeformationJumps(Phase);
1210 CalculateNonlocalJumpContribution(Phase);
1211 }
1212
1213 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i, j, k, DeformationGradientsTotal,0,)
1214 if(Phase.Fields(i,j,k).interface())
1215 {
1216 const vStrain ElasticStrains = EffectiveElasticConstants(i,j,k).inverted()*Stresses(i, j, k);
1217 NodeAB<dVector3,dVector3> locNormals = Phase.Normals(i,j,k);
1218
1219 for(auto alpha = Phase.Fields(i, j, k).cbegin();
1220 alpha != Phase.Fields(i, j, k).cend() - 1; ++alpha)
1221 for(auto beta = alpha + 1;
1222 beta != Phase.Fields(i, j, k).cend(); ++beta)
1223 {
1224 dMatrix3x3 locStretchesAlpha = TransformationStretches(i,j,k).get_value(alpha->index);
1225 dMatrix3x3 locStretchesBeta = TransformationStretches(i,j,k).get_value( beta->index);
1226
1227 vStrain locEigenStrainDifference =
1228 EigenStrainDifference(locStretchesAlpha, locStretchesBeta, i,j,k);
1229
1230 double dG_AB = 0.0;
1231 switch(ElasticityModel)
1232 {
1233 case ElasticityModels::Khachaturyan:
1234 {
1235 dG_AB += (ElasticStrains*
1236 ((ElasticConstants(i,j,k).get_value( beta->index) -
1237 ElasticConstants(i,j,k).get_value(alpha->index))*
1238 ElasticStrains))*0.5;
1239
1240 dG_AB -= Stresses(i, j, k)*locEigenStrainDifference;
1241 break;
1242 }
1243 case ElasticityModels::Steinbach:
1244 case ElasticityModels::Reuss:
1245 {
1246 dG_AB += (Stresses(i, j, k)*
1247 ((ElasticConstants(i,j,k).get_value(alpha->index).inverted() -
1248 ElasticConstants(i,j,k).get_value( beta->index).inverted())*
1249 Stresses(i, j, k)))*0.5;
1250
1251 dG_AB -= Stresses(i, j, k)*locEigenStrainDifference;
1252 break;
1253 }
1254 case ElasticityModels::Voigt: // Defined in small strain setting
1255 {
1256 vStrain locStrain = StrainSmall(DeformationGradientsTotal(i,j,k))
1257 - StrainSmall(DeformationGradientsPlastic(i,j,k));
1258
1259 vStrain locEigenStrainAlpha = StrainSmall(TransformationStretches(i,j,k).get_value(alpha->index));
1260 vStrain locEigenStrainBeta = StrainSmall(TransformationStretches(i,j,k).get_value( beta->index));
1261
1262 dG_AB += ((locStrain - locEigenStrainBeta)*
1263 (ElasticConstants(i,j,k).get_value(beta->index)*
1264 (locStrain - locEigenStrainBeta)) -
1265
1266 (locStrain - locEigenStrainAlpha)*
1267 (ElasticConstants(i,j,k).get_value(alpha->index)*
1268 (locStrain - locEigenStrainAlpha)))*0.5;
1269 break;
1270 }
1271 case ElasticityModels::Rank1: // Defined in small strain setting
1272 case ElasticityModels::Rank1NL: // Defined in small strain setting
1273 {
1274 if(alpha->value > DBL_EPSILON and beta->value > DBL_EPSILON)
1275 {
1276 vStrain locStrain = StrainSmall(DeformationGradientsTotal(i,j,k))
1277 - StrainSmall(DeformationGradientsPlastic(i,j,k));
1278
1279 double scale = 1.0;
1280 if(Phase.FieldsProperties[alpha->index].Stage != GrainStages::Stable)
1281 {
1282 scale *= Phase.FieldsProperties[alpha->index].VolumeRatio;
1283 }
1284 if(Phase.FieldsProperties[beta->index].Stage != GrainStages::Stable)
1285 {
1286 scale *= Phase.FieldsProperties[beta->index].VolumeRatio;
1287 }
1288
1289 dMatrix3x3 locFjumpAlpha = DeformationJumps(i,j,k).get_sym2(alpha->index, beta->index);
1290 dMatrix3x3 locFjumpBeta = DeformationJumps(i,j,k).get_sym2( beta->index, alpha->index);
1291
1292// for(auto gamma = beta + 1;
1293// gamma != Phase.Fields(i, j, k).cend(); ++gamma)
1294// if(gamma->value > DBL_EPSILON)
1295// {
1296// locFjumpAlpha += DeformationJumps(i,j,k).get_asym2(alpha->index, gamma->index);
1297// locFjumpBeta += DeformationJumps(i,j,k).get_asym2( beta->index, gamma->index);
1298// }
1299
1300 double sign = 1.0;
1301 if(alpha->index > beta->index)
1302 {
1303 sign = -1.0;
1304 }
1305
1306 vStrain locStrainJumpA = StrainSmall(locFjumpAlpha + dMatrix3x3::UnitTensor())*scale*sign;
1307 vStrain locStrainJumpB = StrainSmall(locFjumpBeta + dMatrix3x3::UnitTensor())*scale*(-sign);
1308
1309 vStrain StrainAlpha = locStrain - locStrainJumpA* beta->value/(alpha->value + beta->value);// TODO: check effect of normalization by (alpha->value + beta->value)
1310 vStrain StrainBeta = locStrain - locStrainJumpB*alpha->value/(alpha->value + beta->value);// TODO: check effect of normalization by (alpha->value + beta->value)
1311
1312 vStrain locEigenStrainAlpha = StrainSmall(locStretchesAlpha);
1313 vStrain locEigenStrainBeta = StrainSmall(locStretchesBeta);
1314
1315 dG_AB += ((StrainBeta - locEigenStrainBeta)*
1316 (ElasticConstants(i,j,k).get_value(beta->index)*
1317 (StrainBeta - locEigenStrainBeta)) -
1318
1319 (StrainAlpha - locEigenStrainAlpha)*
1320 (ElasticConstants(i,j,k).get_value(alpha->index)*
1321 (StrainAlpha - locEigenStrainAlpha)))*0.5*scale;
1322
1323 dG_AB -= Stresses(i,j,k)*(locStrainJumpA - locStrainJumpB)*0.5*scale;
1324
1325 if(ElasticityModel == ElasticityModels::Rank1NL)
1326 {
1327 if(Grid.dNx)
1328 {
1329 dG_AB -= (NonLocalDeformationJumpTerm(i+1,j,k).get_asym1(alpha->index, beta->index)[0] - NonLocalDeformationJumpTerm(i-1,j,k).get_asym1(alpha->index, beta->index)[0])*scale/(2.0*Grid.dx);
1330 }
1331 if(Grid.dNy)
1332 {
1333 dG_AB -= (NonLocalDeformationJumpTerm(i,j+1,k).get_asym1(alpha->index, beta->index)[1] - NonLocalDeformationJumpTerm(i,j-1,k).get_asym1(alpha->index, beta->index)[1])*scale/(2.0*Grid.dx);
1334 }
1335 if(Grid.dNz)
1336 {
1337 dG_AB -= (NonLocalDeformationJumpTerm(i,j,k+1).get_asym1(alpha->index, beta->index)[2] - NonLocalDeformationJumpTerm(i,j,k-1).get_asym1(alpha->index, beta->index)[2])*scale/(2.0*Grid.dx);
1338 }
1339 }
1340
1341 //Use Khachaturyan's driving force for small nuclei
1342 if((1.0 - scale) > FLT_EPSILON)
1343 {
1344 dG_AB += (ElasticStrains*
1345 ((ElasticConstants(i,j,k).get_value( beta->index) -
1346 ElasticConstants(i,j,k).get_value(alpha->index))*
1347 ElasticStrains))*0.5*(1.0 - scale);
1348
1349 dG_AB -= Stresses(i, j, k)*locEigenStrainDifference*(1.0 - scale);
1350 }
1351 }
1352 break;
1353 }
1354 }
1355 dGab.Force(i,j,k).add_raw(alpha->index, beta->index, dG_AB);
1356 }
1357 }
1358 OMP_PARALLEL_STORAGE_LOOP_END
1359}

1430void ElasticProperties::CalculateNonlocalJumpContribution(const PhaseField& Phase)
1431{
1432 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,DeformationJumps,0,)
1433 {
1434 NonLocalDeformationJumpTerm(i,j,k).clear();
1435
1436 if(Phase.Fields(i,j,k).interface())
1437 {
1438 vStrain locStrain = StrainSmall(DeformationGradientsTotal(i,j,k))
1439 - StrainSmall(DeformationGradientsPlastic(i,j,k));
1440 NodeAB<dVector3,dVector3> locNormals = Phase.Normals(i,j,k);
1441
1442 for(auto alpha = Phase.Fields(i, j, k).cbegin();
1443 alpha != Phase.Fields(i, j, k).cend() - 1; ++alpha)
1444 for(auto beta = alpha + 1;
1445 beta != Phase.Fields(i, j, k).cend(); ++beta)
1446 if(alpha->value > DBL_EPSILON and beta->value > DBL_EPSILON)
1447 {
1448 dVector3 locNormalAB = locNormals.get_asym1(alpha->index, beta->index);
1449
1450 dVector3 defJumpV = DeformationJumps(i,j,k).get_asym1(alpha->index, beta->index);
1451 dMatrix3x3 defJumpT = DeformationJumps(i,j,k).get_asym2(alpha->index, beta->index);
1452 dMatrix3x3 strainJumpAB = (defJumpT.transposed() + defJumpT)*0.5;
1453
1454 vStrain dStrainA = (locStrain - StrainSmall(TransformationStretches(i,j,k).get_value(alpha->index)) - VoigtStrain(strainJumpAB)*beta->value);
1455 vStrain dStrainB = (locStrain - StrainSmall(TransformationStretches(i,j,k).get_value( beta->index)) + VoigtStrain(strainJumpAB)*alpha->value);
1456
1457 vStress stressAlpha = ElasticConstants(i,j,k).get_value(alpha->index) * dStrainA;
1458 vStress stressBeta = ElasticConstants(i,j,k).get_value( beta->index) * dStrainB;
1459
1460 dMatrix3x3 locUnity = dMatrix3x3::UnitTensor() - locNormalAB.dyadic(locNormalAB);
1461
1462 dVector3 nonLocalTerm = (((stressAlpha - stressBeta).tensor() * locUnity) * defJumpV) * (alpha->value*beta->value)/alpha->gradient.length();
1463
1464 NonLocalDeformationJumpTerm(i,j,k).set_asym1(alpha->index, beta->index, nonLocalTerm);
1465 }
1466 }
1467 }
1468 OMP_PARALLEL_STORAGE_LOOP_END

1950{

1952 {
1953 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,VelocityGradientsTotal,0,)
1954 {
1955 // Update rotation via Euler-Rodrigues formula
1956 dMatrix3x3 W = (VelocityGradientsTotal(i,j,k) - VelocityGradientsTotal(i,j,k).transposed())*0.5;

1958 dMatrix3x3 RotMat;

1961 double wnorm = 0.0;

1972 }

2113 {

2116 case ElasticityModels::Voigt:

2120 energy = 0.5*(locElasticStrains*(EffectiveElasticConstants(i,j,k)*locElasticStrains));

2125 case ElasticityModels::Rank1NL:
2126 {
2127 if(Phase.Fields(i,j,k).interface())
2128 {
2129 vStrain locStrain = TotalStrains(i,j,k);
2130
2131 for(auto alpha = Phase.Fields(i, j, k).cbegin();
2132 alpha != Phase.Fields(i, j, k).cend() - 1; ++alpha)
2133 {

2136 for(auto beta = alpha + 1;
2137 beta != Phase.Fields(i, j, k).cend(); ++beta)
2138 if(alpha->value > DBL_EPSILON and beta->value > DBL_EPSILON)
2139 {
2140 double scale = 1.0;
2141 if(Phase.FieldsProperties[alpha->index].Stage != GrainStages::Stable)
2142 {
2143 scale = Phase.FieldsProperties[alpha->index].VolumeRatio;
2144 }

2146 {
2147 scale = Phase.FieldsProperties[beta->index].VolumeRatio;
2148 }
2149
2150 dMatrix3x3 locFjumpAlpha = DeformationJumps(i,j,k).get_asym2(alpha->index, beta->index);
2151
2152 dMatrix3x3 locStrainJumpA = (locFjumpAlpha + locFjumpAlpha.transposed())*0.5;

2155 if(alpha->index > beta->index)
2156 {
2157 sign = -1.0;

2160 }
2161 vStrain locEigenStrainAlpha = StrainSmall(GrainTransformationStretches[alpha->index]);

2166 }

2174 }
