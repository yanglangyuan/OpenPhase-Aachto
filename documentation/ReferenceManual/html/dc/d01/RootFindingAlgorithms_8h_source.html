49 template<typename UNARY, typename T>
50 static void Bisection(UNARY func, T& x0, T xa, T xb, T accuracy, unsigned long iterations)
51 {
52 T fb = func(xb);
53 T fa = func(xa);
54 if ((xa > xb) or (fa > 0 and fb > 0) or (fa < 0 and fb < 0))
55 {
56 std::stringstream message;
57 message << "Unsuitable initial conditions!\n"
58 << "xa = " << xa << "\n"
59 << "xb = " << xb << "\n"
60 << "fa = " << fa << "\n"
61 << "fb = " << fb;
62 throw std::runtime_error(message.str());
63 }
64
65 T f0 = func(x0);
66 unsigned long itr = 0;
67 while (std::abs(f0) > accuracy && std::abs((xb - xa)/xa) > std::numeric_limits<double>::epsilon())
68 {
69 if ((fa > 0 and f0 > 0) or (fa < 0 and f0 < 0))
70 {
71 xa = x0;
72 fa = f0;
73 }
74 else
75 {
76 xb = x0;
77 fb = f0;
78 }
79
80 itr++; if (itr >= iterations) throw std::runtime_error(MaxIterations);
81 x0 = (xb + xa)/2.0;
82 f0 = func(x0);
83 }
84 }

86 template<typename UNARY1, typename UNARY2, typename T>
87 static void Newton(UNARY1 func, UNARY2 dfunc, T& x0, T accuracy, unsigned long iterations)
88 {
89 T f0 = func(x0);
90 unsigned long itr = 0;
91 while (std::abs(f0) > accuracy)
92 {
93 T df0 = dfunc(x0); if (df0 == 0) throw std::runtime_error(ZeroGradient);
94 //std::cout << itr << " " << x0 << " " << f0 << " " << df0 << "\n";
95 T p0 = - f0/df0;
96 x0 += p0;
97 itr++; if (itr >= iterations) throw std::runtime_error(MaxIterations);
98 f0 = func(x0);
99 }
100 }

102 template<typename UNARY1, typename UNARY2, typename T>
103 static void PositivNewton(UNARY1 func, UNARY2 dfunc, T& x0, T accuracy, unsigned long iterations)
104 {
105 T f0 = func(x0);
106 unsigned long itr = 0;
107 while (std::abs(f0) > accuracy)
108 {
109 T df0 = dfunc(x0); if (df0 == 0.0) throw std::runtime_error(ZeroGradient);
110 T p0 = - f0/df0;
111 while (x0 + p0 <= 0.0 ) p0/=2.0;
112 x0 += p0;
113 itr++; if (itr >= iterations and x0 > 0.0) throw std::runtime_error(MaxIterations);
114 f0 = func(x0);
115 }
116 }

118 template<typename UNARY, typename T>
119 static void Secant(UNARY func, T& x0, T& x1, T accuracy, unsigned long iterations)
120 {
121 T f0 = func(x0);
122 T f1 = func(x1);
123 if (std::abs(f0) > accuracy)
124 {
125 unsigned long itr = 0;
126 do
127 {
128 T dx = (x0 - x1);
129 T df = (f0 - f1); if (df == 0) throw std::runtime_error(ZeroGradient);
130 T p0 = - f0*dx/df;
131 x1 = x0;
132 x0 += p0;
133 f1 = f0;
134 f0 = func(x0);
135 itr++; if (itr >= iterations) throw std::runtime_error(MaxIterations);
136 }
137 while(std::abs(f0) > accuracy);
138 }

141 template<typename UNARY, typename T>
142 static void Broyden(UNARY func, T& x0, T accuracy, unsigned long iterations)
143 {
144 auto gradient = [func] (T x0, T dx)
145 {
146 T x_p1 = x0 + dx;
147 T x_m1 = x0 - dx;
148 T f_m1 = func(x_m1);
149 T f_p1 = func(x_p1);
150 return (f_p1 - f_m1)/dx/2;
151 };
152
153 T x1 = x0;
154 T f0 = func(x0);
155 T f1 = f0;
156
157 if (std::abs(f0) > accuracy)
158 {
159 unsigned long itr = 1;
160 T eps = 1.0;
161 T del = 0.5;
162 T dx = f0/1000;
163 T J0 = gradient(x0,dx);
164 if (J0 == 0) throw std::runtime_error(ZeroGradient);
165 T Jinv0 = 1.0/J0;
166 T p0 = - Jinv0*f0;
167 x0 += eps*p0;
168 f0 = func(x0);
169 itr++;
170
171 do
172 {
173 dx = del*(x0 - x1);
174 J0 = gradient(x0,dx);
175 if (J0 == 0) throw std::runtime_error(ZeroGradient);
176 Jinv0 = 1.0/J0;
177 eps = 1.0;
178 del = 0.5;
179 x1 = x0;
180 f1 = f0;
181 while(true)
182 {
183 T p0 = - Jinv0*f0;
184 x0 += eps*p0;
185 f0 = func(x0);
186 if (f0*f0 < f1*f1 or itr < 3) break;
187 T eps_new = 0.75*eps;
188 del = eps_new-eps;
189 eps = eps_new;
190 x0 = x1;
191 if (eps < accuracy) throw std::runtime_error(NotConverging);
192 }
193 itr++; if (itr>=iterations) throw std::runtime_error(MaxIterations);
194 }
195 while (std::abs(f0) > accuracy);
196 }
197 }

205 auto calculate_jacobian = [&func,&size,&f_p1] (matrix& J, const vector& x0, const vector& f0, const vector& dx)
206 {
207 f_p1 = f0;
208 for (std::size_t m = 0; m < size; m++)
209 {
210 vector x_p1 = x0;
211 x_p1[m] += dx[m];
212 vector f_p1 = func(x_p1);
213 for (std::size_t n = 0; n < size; n++)
214 {
215 J[n][m] = (f_p1[n] - f0[n])/dx[m];
216 }
217 }
218 };

220 auto calculate_inverted = [size](matrix& Jinv, const matrix& J)
221 {
222 if (size == 1)
223 {
224 if (J[0][0] == 0) throw std::runtime_error(ZeroGradient);
225 Jinv[0][0] = 1.0/J[0][0];
226 }
227 else if (size == 2)
228 {
229 T det = J[0][0]*J[1][1]-J[1][0]*J[1][0];
230 if (det == 0) throw std::runtime_error(ZeroGradient);
231 Jinv[0][0] = J[1][1]/det;
232 Jinv[0][1] = -J[0][1]/det;
233 Jinv[1][0] = -J[1][0]/det;
234 Jinv[1][1] = J[0][0]/det;
235 }
236 else if (size == 3)
237 {
238 const T& a = J[0][0];
239 const T& b = J[0][1];
240 const T& c = J[0][2];
241 const T& d = J[1][0];
242 const T& e = J[1][1];
243 const T& f = J[1][2];
244 const T& g = J[2][0];
245 const T& h = J[2][1];
246 const T& i = J[2][2];
247 T A = (e*i-f*h);
248 T B = -(d*i-f*g);
249 T C = (d*h-e*g);
250 T D = -(b*i-c*h);
251 T E = (a*i-c*g);
252 T F = -(a*h-b*g);
253 T G = (b*f-c*e);
254 T H = -(a*f-c*d);
255 T I = (a*e-b*d);
256 T det = a*A+b*B+c*C;
257 if (det == 0) throw std::runtime_error(ZeroGradient);
258 Jinv[0][0] = A/det;
259 Jinv[0][1] = B/det;
260 Jinv[0][2] = C/det;
261 Jinv[1][0] = D/det;
262 Jinv[1][1] = E/det;
263 Jinv[1][2] = F/det;
264 Jinv[2][0] = G/det;
265 Jinv[2][1] = H/det;
266 Jinv[2][2] = I/det;
267 }
268 else throw std::invalid_argument("Matrix inversion for more that 3 components is not yet implemented");
269 };
