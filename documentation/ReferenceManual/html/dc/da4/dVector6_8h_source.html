257 double norm(void) const
258 {
259 /*
260 * multiplication by 0.5 of the off diagonal elements before taking
261 * square and doubling the result due to double appearance of the off
262 * diagonal elements => 0.5^2 * 2 = 0.5
263 */
264 double tmp = storage[0] * storage[0]
265 + storage[1] * storage[1]
266 + storage[2] * storage[2]
267 + storage[3] * storage[3] * 0.5
268 + storage[4] * storage[4] * 0.5
269 + storage[5] * storage[5] * 0.5;
270 return sqrt(tmp);
271 }
