45void DrivingForce::Initialize(Settings& locSettings, std::string ObjectNameSuffix)
46{
47 thisclassname = "DrivingForce";
48 thisobjectname = thisclassname + ObjectNameSuffix;
49
50 Grid = locSettings.Grid;
51
52 Nphases = locSettings.Nphases;
53
54 Limiting = true;
55 Limit.Allocate(Nphases,Nphases);
56 Averaging = false;
57
58 WeightsMode = AveragingWeightsModes::PhaseFields;
59
60 switch(Grid.Resolution)
61 {
62 case Resolutions::Single:
63 {
64 Range = Grid.iWidth;
65
66 if(Grid.iWidth < 5.0)
67 {
68 PhiThreshold = Grid.iWidth/15.0;
69 }
70 else
71 {
72 PhiThreshold = 1.0/3.0;
73 }
74 break;
75 }
76 case Resolutions::Dual:
77 {
78 Range = (Grid.iWidth)/2;
79
80 if(Grid.iWidth < 5.0)
81 {
82 PhiThreshold = Grid.iWidth/30.0;
83 }
84 else
85 {
86 PhiThreshold = 1.0/6.0;
87 }
88 break;
89 }
90 }
91
92 Force.Allocate(Grid, Range);
93
94 OvershootCounter = 0;
95 MAXOvershootPOS.Allocate(Nphases,Nphases);
96 MAXOvershootNEG.Allocate(Nphases,Nphases);
97 MAXDrivingForcePOS.Allocate(Nphases,Nphases);
98 MAXDrivingForceNEG.Allocate(Nphases,Nphases);
99
100 locSettings.AddForRemeshing(*this);
101
102 initialized = true;
103 ConsoleOutput::WriteStandard(thisclassname, "Initialized");
104}

126void DrivingForce::ReadInput(stringstream& inp)
127{
128 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
129
130 Averaging = FileInterface::ReadParameterB(inp, moduleLocation, string("Average"), false, true);
131 Unifying = FileInterface::ReadParameterB(inp, moduleLocation, string("bUnify"), false, false);
132 Range = FileInterface::ReadParameterI(inp, moduleLocation, string("Range"), false, Range);
133 PhiThreshold = FileInterface::ReadParameterD(inp, moduleLocation, string("Threshold"), false, PhiThreshold);
134 Limiting = FileInterface::ReadParameterB(inp, moduleLocation, vector<string>{"Limiting","dGcut"}, false, true);
135 string tmp2 = FileInterface::ReadParameterK(inp, moduleLocation, "WeightsMode", false, "PHASEFIELDS");
136
137 if(tmp2 == string("RANGE"))
138 {
139 WeightsMode = AveragingWeightsModes::Range;
140 }
141 else if(tmp2 == string("PHASEFIELDS"))
142 {
143 WeightsMode = AveragingWeightsModes::PhaseFields;
144 }
145 else if(tmp2 == string("COUNTER"))
146 {
147 WeightsMode = AveragingWeightsModes::Counter;
148 }
149 else
150 {
151 ConsoleOutput::WriteWarning("Wrong driving force weights mode specified!\nThe default \"PHASEFIELDS\" mode is used!", thisclassname, "ReadInput()");
152 }
153
154 for(size_t m = 0; m < Nphases; ++m)
155 for(size_t n = m; n < Nphases; ++n)
156 {
157 stringstream converter;
158 converter << m << "_" << n;
159 string counter = converter.str();
160
161 Limit(m,n) = FileInterface::ReadParameterD(inp, moduleLocation, vector<string>{"Limit_" + counter, "CutOff_" + counter}, false, 0.95);
162 if(Limit(m,n) == 0.0)
163 {
164 stringstream message;
165 message << "Limit(" << m << ", " << n << ") = 0.0, please correct the driving force input!";
166 ConsoleOutput::WriteExit(message.str(), thisclassname, "ReadInput()");
167 exit(13);
168 }
169 Limit(n,m) = Limit(m,n);
170 }
171
172 ConsoleOutput::WriteLine();
173 ConsoleOutput::WriteBlankLine();
174}

192NodeDF DrivingForce::CalcUnified(const PhaseField& Phase, const BoundaryConditions& BC)
193{
194 NodeDF Unified;
195 //OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Force,0, reduction(+:UnifiedDrivingForce),reduction(+:Counter))
196 STORAGE_LOOP_BEGIN(i,j,k,Force,0)
197 {
198 if (Phase.Fields(i,j,k).interface())
199 for(const auto& it : Force(i,j,k))
200 {
201 const double PhiAlpha = Phase.Fields(i,j,k).get_value(it.indexA);
202 const double PhiBeta = Phase.Fields(i,j,k).get_value(it.indexB);
203 const double weight = std::sqrt(PhiAlpha*PhiBeta);
204
205 Unified.add_raw(it.indexA, it.indexB, it.raw*weight);
206 Unified.add_average(it.indexA, it.indexB, it.average*weight);
207 Unified.add_weight(it.indexA, it.indexB, weight);
208 }
209 }
210 STORAGE_LOOP_END
211
212 for(auto& it : Unified)
213 {
214 const double norm = Unified.get_weight(it.indexA,it.indexB);
215 if (norm > DBL_EPSILON)
216 {
217 it.raw /= norm;
218 it.average /= norm;
219 }
220 }
221 return Unified;
222}

293void DrivingForce::CollectAverage(const PhaseField& Phase)
294{
295 const int Xrange = min(Range, Grid.Nx-1)*Grid.dNx;
296 const int Yrange = min(Range, Grid.Ny-1)*Grid.dNy;
297 const int Zrange = min(Range, Grid.Nz-1)*Grid.dNz;
298
299 const double weight_threshold = sqrt(PhiThreshold*(1.0 - PhiThreshold));
300
301 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Force,0,)
302 {
303 if (Phase.Fields(i,j,k).interface())
304 for(auto it = Force(i,j,k).begin();
305 it != Force(i,j,k).end(); ++it)
306 {
307 double scale = Phase.FieldsProperties[it->indexA].VolumeRatio *
308 Phase.FieldsProperties[it->indexB].VolumeRatio;
309
310 if (it->weight > weight_threshold*scale)
311 {
312 double value = 0.0;
313 double sum_weights = 0.0;
314 int counter = 0;
315
316 for(int ii = -Xrange; ii <= Xrange; ii++)
317 for(int jj = -Yrange; jj <= Yrange; jj++)
318 for(int kk = -Zrange; kk <= Zrange; kk++)
319 {
320 double weight_dist = Range - sqrt(ii*ii + jj*jj + kk*kk);
321 double weight_phi = Force(i+ii, j+jj, k+kk).get_weight(it->indexA, it->indexB);
322
323 if(weight_dist > 0.0 and weight_phi > 0.0)
324 {
325 double weight = 0.0;
326 switch(WeightsMode)
327 {
328 case AveragingWeightsModes::Range:
329 {
330 weight = weight_dist;
331 sum_weights += weight;
332 break;
333 }
334 case AveragingWeightsModes::PhaseFields:
335 {
336 weight = weight_phi;
337 sum_weights += weight;
338 break;
339 }
340 case AveragingWeightsModes::Counter:
341 {
342 counter++;
343 weight = 1.0;
344 break;
345 }
346 }
347 value += weight*Force(i+ii, j+jj, k+kk).get_raw(it->indexA, it->indexB);
348 }
349 }
350 switch(WeightsMode)
351 {
352 case AveragingWeightsModes::Range:
353 case AveragingWeightsModes::PhaseFields:
354 {
355 if(sum_weights > 0.0)
356 {
357 it->tmp = value/sum_weights;
358 }
359 else
360 {
361 it->tmp = it->raw;
362 }
363 break;
364 }
365 case AveragingWeightsModes::Counter:
366 {
367 if(counter != 0)
368 {
369 it->tmp = value/counter;
370 }
371 else
372 {
373 it->tmp = it->raw;
374 }
375 break;
376 }
377 }
378 }
379 }
380 }
381 OMP_PARALLEL_STORAGE_LOOP_END

387 const int Yrange = min(Range, Grid.Ny-1)*Grid.dNy;

391
392 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Force,0,)
393 {
394 if (Phase.Fields(i,j,k).interface())
395 for(auto it = Force(i,j,k).begin();
396 it != Force(i,j,k).end(); ++it)
397 {
398 if (it->weight > 0.0)
399 {
400 double value = 0.0;
401 double sum_weights = 0.0;
402 int counter = 0;
403

408 for(int jj = -Yrange; jj <= Yrange; jj++)

413 if(weight_dist > 0.0 and weight_phi > weight_threshold*scale)
414 {
415 double weight = 0.0;

497void DrivingForce::MergePhaseFieldIncrementsSR(PhaseField& Phase,
498 InterfaceProperties& IP)
499{
500 double locMaxPsi = 0.0;
501 long int locOvershootCounter = 0;
502 Matrix<double> locMAXOvershootPOS = MAXOvershootPOS;
503 Matrix<double> locMAXOvershootNEG = MAXOvershootNEG;
504 Matrix<double> locMAXDrivingForcePOS = MAXDrivingForcePOS;
505 Matrix<double> locMAXDrivingForceNEG = MAXDrivingForceNEG;
506
507 const double Prefactor = Pi/Phase.Grid.Eta;
508 //const double Prefactor = 2.0*Pi/(Phase.Eta*Phase.LocalN(Phase.Fields(i,j,k)));
509
510 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i, j, k, Phase.Fields, 0, \
511 reduction(+:locOvershootCounter) \
512 reduction(MatrixDMAX:locMAXOvershootPOS, locMAXDrivingForcePOS) \
513 reduction(MatrixDMIN:locMAXOvershootNEG, locMAXDrivingForceNEG) \
514 reduction(max: locMaxPsi))
515 {
516 if (Phase.Fields(i,j,k).interface())
517 {
518 for(auto it = Force(i,j,k).begin();
519 it != Force(i,j,k).end(); ++it)
520 {
521 double locdG = 0.0;
522 if(Averaging)
523 {
524 locdG = it->average;
525 }
526 else
527 {
528 locdG = it->raw;
529 }
530
531 double norm = it->weight;
532
533 size_t indexA = it->indexA;
534 size_t indexB = it->indexB;
535 size_t pIndexA = Phase.FieldsProperties[indexA].Phase;
536 size_t pIndexB = Phase.FieldsProperties[indexB].Phase;
537
538 if(Phase.FieldsProperties[indexA].Stage == GrainStages::Seed or
539 Phase.FieldsProperties[indexB].Stage == GrainStages::Seed)
540 {
541 /* In the case of a freshly nucleated grain, a small value
542 is assigned to the normalization coefficient to allow the
543 initial growth of the nucleus. */
544
545 norm += 1.0e-6;
546 }
547
548 if(Limiting)
549 {
550 /* If the local driving force exceeds the interface stabilizing
551 force, the interface profile can distort leading to simulation
552 artifacts. Therefore the driving force is limited to a safe
553 value controlled by the user specified Limit(pIndexA,pIndexB)
554 parameter.*/
555
556 double absDG = fabs(locdG);
557 double allowedDG = Limit(pIndexA,pIndexB)*Prefactor
558 *IP.InterfaceEnergy(pIndexA,pIndexB).MaxEnergy;
559
560 allowedDG *= IP.RegularizationFactor;
561
562 /* Collecting statistics for later output */
563 if(absDG > 0.4*allowedDG)
564 {
565 locOvershootCounter++;
566 }
567
568 double tmpMAXOvershoot = locdG/allowedDG;
569
570 if(tmpMAXOvershoot > locMAXOvershootPOS(pIndexA,pIndexB))
571 {
572 locMAXOvershootPOS(pIndexA,pIndexB) = tmpMAXOvershoot;
573 }
574 if(tmpMAXOvershoot < locMAXOvershootNEG(pIndexA,pIndexB))
575 {
576 locMAXOvershootNEG(pIndexA,pIndexB) = tmpMAXOvershoot;
577 }
578 locMAXDrivingForcePOS(pIndexA,pIndexB) = max(locdG, locMAXDrivingForcePOS(pIndexA,pIndexB));
579 locMAXDrivingForceNEG(pIndexA,pIndexB) = min(locdG, locMAXDrivingForceNEG(pIndexA,pIndexB));
580
581 /* End collecting statistics for later output */
582 locdG = allowedDG*tanh(locdG/allowedDG);
583 }
584
585 double dPsi_dt = locdG*IP.Properties(i,j,k).get_mobility(indexA,indexB)*norm*Prefactor;
586
587 locMaxPsi = max(fabs(dPsi_dt), locMaxPsi);
588
589 Phase.FieldsDot(i,j,k).add_asym1(indexA,indexB,dPsi_dt);
590 }
591 }
592 }
593 OMP_PARALLEL_STORAGE_LOOP_END
594
595 maxPsi = locMaxPsi;
596
597 OvershootCounter += locOvershootCounter;
598 MAXOvershootPOS = locMAXOvershootPOS;
599 MAXOvershootNEG = locMAXOvershootNEG;
600 MAXDrivingForcePOS = locMAXDrivingForcePOS;
601 MAXDrivingForceNEG = locMAXDrivingForceNEG;
602}

604void DrivingForce::MergePhaseFieldIncrementsDR(PhaseField& Phase,
605 InterfaceProperties& IP)
606{
607 double locMaxPsi = 0.0;
608 long int locOvershootCounter = 0;
609 Matrix<double> locMAXOvershootPOS = MAXOvershootPOS;
610 Matrix<double> locMAXOvershootNEG = MAXOvershootNEG;
611 Matrix<double> locMAXDrivingForcePOS = MAXDrivingForcePOS;
612 Matrix<double> locMAXDrivingForceNEG = MAXDrivingForceNEG;
613
614 const double Prefactor = Pi/Phase.Grid.Eta;
615 //const double Prefactor = 2.0*Pi/(Phase.Eta*Phase.LocalN(Phase.FieldsDR(i,j,k)));
616
617 const double di = Grid.dNx*0.25;
618 const double dj = Grid.dNy*0.25;
619 const double dk = Grid.dNz*0.25;
620
621 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i, j, k, Phase.FieldsDR, 0, \
622 reduction(+:locOvershootCounter) \
623 reduction(MatrixDMAX:locMAXOvershootPOS, locMAXDrivingForcePOS) \
624 reduction(MatrixDMIN:locMAXOvershootNEG, locMAXDrivingForceNEG) \
625 reduction(max: locMaxPsi))
626 {
627 if (Phase.FieldsDR(i,j,k).interface())
628 {
629 NodeDF locForce = Force_at(i/2 - di, j/2 - dj, k/2 - dk);
630
631 for(auto alpha = Phase.FieldsDR(i,j,k).begin();
632 alpha != Phase.FieldsDR(i,j,k).end(); ++alpha)
633 for(auto beta = alpha + 1;
634 beta != Phase.FieldsDR(i,j,k).end(); ++beta)
635 {
636 /* For each pair of grains the driving force is treated
637 individually. */
638
639 double locdG = 0.0;
640 if(Averaging)
641 {
642 locdG = locForce.get_average(alpha->index, beta->index);
643 }
644 else
645 {
646 locdG = locForce.get_raw(alpha->index, beta->index);
647 }
648
649 size_t pIndexA = Phase.FieldsProperties[alpha->index].Phase;
650 size_t pIndexB = Phase.FieldsProperties[ beta->index].Phase;
651
652 double norm = sqrt(alpha->value * beta->value);
653
654 if(Phase.FieldsProperties[alpha->index].Stage == GrainStages::Seed or
655 Phase.FieldsProperties[ beta->index].Stage == GrainStages::Seed)
656 {
657 /* In the case of a freshly nucleated grain, a small value
658 is assigned to the normalization coefficient to allow the
659 initial growth of the nucleus. */
660
661 norm += 1.0e-6;
662 }
663
664 if(Limiting)
665 {
666 /* If the local driving force exceeds the interface stabilizing
667 force, the interface profile can distort leading to simulation
668 artifacts. Therefore the driving force is limited to a safe
669 value controlled by the user specified Limit(pIndexA,pIndexB)
670 parameter.*/
671
672 double absDG = fabs(locdG);
673 double allowedDG = Limit(pIndexA,pIndexB)*Prefactor
674 *IP.InterfaceEnergy(pIndexA,pIndexB).MaxEnergy;
675
676 allowedDG *= IP.RegularizationFactor;
677
678 /* Collecting statistics for later output */
679 if(absDG > 0.4*allowedDG)
680 {
681 locOvershootCounter++;
682 }
683
684 double tmpMAXOvershoot = locdG/allowedDG;
685
686 if(tmpMAXOvershoot > locMAXOvershootPOS(pIndexA,pIndexB))
687 {
688 locMAXOvershootPOS(pIndexA,pIndexB) = tmpMAXOvershoot;
689 }
690 if(tmpMAXOvershoot < locMAXOvershootNEG(pIndexA,pIndexB))
691 {
692 locMAXOvershootNEG(pIndexA,pIndexB) = tmpMAXOvershoot;
693 }
694
695 locMAXDrivingForcePOS(pIndexA,pIndexB) = max(locdG, locMAXDrivingForcePOS(pIndexA,pIndexB));
696 locMAXDrivingForceNEG(pIndexA,pIndexB) = min(locdG, locMAXDrivingForceNEG(pIndexA,pIndexB));
697
698 /* End collecting statistics for later output */
699
700 locdG = allowedDG*tanh(locdG/allowedDG);
701 }
702
703 double dPsi_dt = locdG*IP.PropertiesDR(i,j,k).get_mobility(alpha->index, beta->index)*norm*Prefactor;
704
705 locMaxPsi = max(fabs(dPsi_dt), locMaxPsi);
706
707 Phase.FieldsDotDR(i,j,k).add_asym1(alpha->index, beta->index, dPsi_dt);
708 }
709 }
710 }
711 OMP_PARALLEL_STORAGE_LOOP_END
712
713 maxPsi = locMaxPsi;
714
715 OvershootCounter += locOvershootCounter;
716 MAXOvershootPOS = locMAXOvershootPOS;
717 MAXOvershootNEG = locMAXOvershootNEG;
718 MAXDrivingForcePOS = locMAXDrivingForcePOS;
719 MAXDrivingForceNEG = locMAXDrivingForceNEG;
