139void ElectricalPotential::ReadInput(stringstream& inp)
140{
141 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
142
143 Nelectrodes = FileInterface::ReadParameterI(inp, moduleLocation, "Nelectrodes");
144
145 electrodes.resize(Nelectrodes);
146 for(int i = 0; i < Nelectrodes; i++)
147 {
148 electrodes[i].index = i+1;
149 electrodes[i].charge = 0;
150 electrodes[i].interfaceVolume = 0.0;
151 for(int dir = 0; dir < 3; dir++)
152 {
153 electrodes[i].position[dir] = 0;
154 }
155 stringstream converter;
156 converter << "CntElectr" << i+1;
157 electrodes[i].counterElectrode = FileInterface::ReadParameterI(inp, moduleLocation, converter.str())-1;
158 }
159
160 for (int q = 0 ; q < Ncomp; q++)
161 {
162 stringstream converter;
163 converter << "Z" << q;
164 ElementarCharges.push_back(FileInterface::ReadParameterI(inp, moduleLocation, converter.str()));
165 MolarCharge.push_back(ElementarCharges[q] * PhysicalConstants::Faraday); // [C/mol]
166 }
167
168 ConsoleOutput::WriteLine();
169 ConsoleOutput::WriteBlankLine();
170}

277void ElectricalPotential::QXYZ(void)
278{
279 double DPi_nX = 2.0*Pi/double(Grid.Nx);
280 double DPi_nY = 2.0*Pi/double(Grid.Ny);
281 double DPi_nZ = 2.0*Pi/double(Grid.Nz);
282
284 for(int i = 0; i < Grid.Nx ; i++)
285 for(int j = 0; j < Grid.Ny ; j++)
286 for(int k = 0; k < Grid.Nz2; k++)
287 {
288 int XYZ = k + Grid.Nz2*(j + Grid.Ny*i);
289
290 Q[0][XYZ] = DPi_nX*(i*(i <= Grid.Nx/2) - (Grid.Nx-i)*(i > Grid.Nx/2))/Grid.dx;
291 Q[1][XYZ] = DPi_nY*(j*(j <= Grid.Ny/2) - (Grid.Ny-j)*(j > Grid.Ny/2))/Grid.dx;
292 Q[2][XYZ] = DPi_nZ*(k*(k <= Grid.Nz/2) - (Grid.Nz-k)*(k > Grid.Nz/2))/Grid.dx;
293 }
294}

296void ElectricalPotential::Solve(PhaseField& Phase, Composition& Cx, BoundaryConditions& BC)
297{
298 CalculateElectrodeInterfaceVolumes(Phase);
299 for (int i = 0; i < Grid.Nx; i++)
300 for (int j = 0; j < Grid.Ny; j++)
301 for (int k = 0; k < Grid.Nz; k++)
302 {
303 RHS[k + Grid.Nz*(j + Grid.Ny*i)] = ChargeDensity(i, j, k, Phase, Cx);//Rho(i+1,j+1,k+1); //sin(i*6.28/Nx);
304 }
305
306 fftw_execute(ForwardPlan);
307
308 for(int XYZ = 0; XYZ < ftSize; XYZ++)
309 {
311 double Q_sqr = Q[0][XYZ]*Q[0][XYZ] + Q[1][XYZ]*Q[1][XYZ] + Q[2][XYZ]*Q[2][XYZ];
312 if (Q_sqr == 0)
313 {
314 ftPotential[XYZ] = 0.0;
315 }
316 else
317 {
318 ftPotential[XYZ] = ftRHS[XYZ] / Q_sqr * EpsInv;
319 }
320 }
321
322 fftw_execute(BackwardPlan);
323
324 for (int i = 0; i < Grid.Nx; i++)
325 for (int j = 0; j < Grid.Ny; j++)
326 for (int k = 0; k < Grid.Nz; k++)
327 {
328 Potential(i,j,k) = rlPotential[k + Grid.Nz*(j + Grid.Ny*i)]/Size;
329 }
330
332 BC.SetX(Potential);
333 BC.SetY(Potential);
334 BC.SetZ(Potential);
335 // End Boundary Conditions
336}
