294 inline double determinant(void) const
295 {
296 return (storage[idx(0,0)]*storage[idx(1,1)]*storage[idx(2,2)] +
297 storage[idx(0,1)]*storage[idx(1,2)]*storage[idx(2,0)] +
298 storage[idx(0,2)]*storage[idx(1,0)]*storage[idx(2,1)] -
299 storage[idx(0,2)]*storage[idx(1,1)]*storage[idx(2,0)] -
300 storage[idx(0,1)]*storage[idx(1,0)]*storage[idx(2,2)] -
301 storage[idx(0,0)]*storage[idx(1,2)]*storage[idx(2,1)]);
302 }
303 dMatrix3x3& invert(void)
304 {
305 dMatrix3x3 tmp;
306
307 double detInv = determinant();
308
309 if(detInv != 0.0)
310 {
311 detInv = 1.0/detInv;
312 }
313 else
314 {
315 std::cerr << "dMatrix3x3: Can Not Compute Inverse Matrix.\n"
316 << this->print() << "Matrix is Singular !!!\n";
317 OP_Exit(EXIT_FAILURE);
318 }
319
320 tmp(0,0) = (storage[idx(1,1)]*storage[idx(2,2)] -
321 storage[idx(1,2)]*storage[idx(2,1)])*detInv;
322 tmp(1,0) =-(storage[idx(1,0)]*storage[idx(2,2)] -
323 storage[idx(1,2)]*storage[idx(2,0)])*detInv;
324 tmp(2,0) = (storage[idx(1,0)]*storage[idx(2,1)] -
325 storage[idx(1,1)]*storage[idx(2,0)])*detInv;
326 tmp(0,1) =-(storage[idx(0,1)]*storage[idx(2,2)] -
327 storage[idx(0,2)]*storage[idx(2,1)])*detInv;
328 tmp(1,1) = (storage[idx(0,0)]*storage[idx(2,2)] -
329 storage[idx(0,2)]*storage[idx(2,0)])*detInv;
330 tmp(2,1) =-(storage[idx(0,0)]*storage[idx(2,1)] -
331 storage[idx(0,1)]*storage[idx(2,0)])*detInv;
332 tmp(0,2) = (storage[idx(0,1)]*storage[idx(1,2)] -
333 storage[idx(1,1)]*storage[idx(0,2)])*detInv;
334 tmp(1,2) =-(storage[idx(0,0)]*storage[idx(1,2)] -
335 storage[idx(0,2)]*storage[idx(1,0)])*detInv;
336 tmp(2,2) = (storage[idx(0,0)]*storage[idx(1,1)] -
337 storage[idx(0,1)]*storage[idx(1,0)])*detInv;
338
339 memmove(storage.data(), tmp.data(), 9*sizeof(double));
340 return *this;
341 }
