105bool Orientations::Write(const Settings& locSettings, const int tStep) const
106{
107 string FileName = FileInterface::MakeFileName(locSettings.RawDataDir,"Rotations_", tStep, ".dat");
108
109 fstream out(FileName.c_str(), ios::out | ios::binary);
110
111 if (!out)
112 {
113 ConsoleOutput::WriteExit("File \"" + FileName + "\" could not be created", thisclassname, "WriteRotations");
114 OP_Exit(EXIT_FAILURE);
115 };
116
117 for(long int i = 0; i < Quaternions.sizeX(); ++i)
118 for(long int j = 0; j < Quaternions.sizeY(); ++j)
119 for(long int k = 0; k < Quaternions.sizeZ(); ++k)
120 {
121 for(int n = 0; n < 4; n++)
122 {
123 double tmp = Quaternions(i,j,k)[n];
124 out.write(reinterpret_cast<char*>(&tmp), sizeof(double));
125 }
126 }
127 return true;
128}

130bool Orientations::Read(const Settings& locSettings, const BoundaryConditions& BC, const int tStep)
131{
132 string FileName = FileInterface::MakeFileName(locSettings.RawDataDir,"Rotations_",
133 tStep, ".dat");
134
135 fstream inp(FileName.c_str(), ios::in | ios::binary);
136
137 if (!inp)
138 {
139 ConsoleOutput::WriteExit("File \"" + FileName + "\" could not be created",
140 thisclassname, "ReadRotations");
141 OP_Exit(EXIT_FAILURE);
142 };
143
144 for(long int i = 0; i < Quaternions.sizeX(); ++i)
145 for(long int j = 0; j < Quaternions.sizeY(); ++j)
146 for(long int k = 0; k < Quaternions.sizeZ(); ++k)
147 {
148 double tmp[4];
149
150 for(int n = 0; n < 4; n++)
151 {
152 inp.read(reinterpret_cast<char*>(&tmp[n]), sizeof(double));
153 }
154 Quaternions(i,j,k).set(tmp[0],tmp[1],tmp[2],tmp[3]);
155 }
156 return true;
157}

202void Orientations::WriteGrainEBSDDataQuaternions(const Settings& locSettings,
203 const PhaseField& Phase,
204 const int tStep)
205{
206 stringstream outbuffer;
207 outbuffer << "Index\t"
208 << "Phase\t"
209 << "X\t"
210 << "Y\t"
211 << "Z\t"
212 << "Quat real\t"
213 << "Quat i\t"
214 << "Quat j\t"
215 << "Quat k\t" << endl;
216
217 int index = 1;
218 STORAGE_LOOP_BEGIN(i,j,k,Quaternions,0)
219 {
220 EulerAngles tempAng;
221 dMatrix3x3 tempMatrix;
222 Quaternion tempQuat;
223
224 outbuffer << index << "\t";
225 if (Phase.Fields(i,j,k).interface())
226 {
227 // linear interpolation in interface via quaternions
228 for(auto alpha = Phase.Fields(i, j, k).cbegin();
229 alpha < Phase.Fields(i, j, k).cend(); ++alpha)
230 {
231 tempQuat += (Quaternions(i,j,k) + Phase.FieldsProperties[alpha->index].Orientation)*alpha->value;
232 }
233 tempQuat.normalize();
234 // Set phase index to 0 in interface
235 outbuffer << 0 << "\t";
236 }
237 else
238 {
239 tempQuat = Quaternions(i,j,k) + Phase.FieldsProperties[Phase.Fields(i,j,k).front().index].Orientation;
240 outbuffer << Phase.FieldsProperties[Phase.Fields(i,j,k).front().index].Phase + 1 << "\t";
241 }
242
243 outbuffer << i*Phase.Grid.dx << "\t"
244 << j*Phase.Grid.dx << "\t"
245 << k*Phase.Grid.dx << "\t"
246 << tempQuat[0] << "\t"
247 << tempQuat[1] << "\t"
248 << tempQuat[2] << "\t"
249 << tempQuat[3] << endl;
250 index++;
251 }
252 STORAGE_LOOP_END
253
254 string FileName = FileInterface::MakeFileName(locSettings.RawDataDir, "EBSD_", tStep, ".dat");
255
256 ofstream ebsd_file(FileName.c_str());
257 ebsd_file << outbuffer.rdbuf();
258 ebsd_file.close();
259
260 ConsoleOutput::WriteStandard("EBSD file", FileName);
261}

263void Orientations::WriteRotated100VectorVTK(const Settings& locSettings,
264 const int tStep,
265 const int precision)
266{
267 stringstream outbufer;
268 dVector3 X;
269 X[0] = 1.0;
270 X[1] = 0.0;
271 X[2] = 0.0;
272
273 outbufer << "# vtk DataFile Version 3.0\n";
274 outbufer << "Rotated100Vector" << "\n";
275 outbufer << "ASCII\n";
276 outbufer << "DATASET RECTILINEAR_GRID\n";
277 outbufer << "DIMENSIONS " << Grid.Nx << " " << Grid.Ny << " " << Grid.Nz << "\n";
278 outbufer << "X_COORDINATES " << Grid.Nx << " double\n";
279 for (int i = 0; i < Grid.Nx; i++) outbufer << i << " ";
280 outbufer << "\n";
281 outbufer << "Y_COORDINATES " << Grid.Ny << " double\n";
282 for (int j = 0; j < Grid.Ny; j++) outbufer << j << " ";
283 outbufer << "\n";
284 outbufer << "Z_COORDINATES " << Grid.Nz << " double\n";
285 for (int k = 0; k < Grid.Nz; k++) outbufer << k << " ";
286 outbufer << "\n";
287 outbufer << "POINT_DATA " << Grid.LocalNumberOfCells() << "\n";
288
289 for (int dir = 0; dir < 3; ++dir)
290 {
291 outbufer << "SCALARS X_" << dir << " double 1\n";
292 outbufer << "LOOKUP_TABLE default\n";
293
294 for (int k = 0; k < Grid.Nz; k++)
295 for (int j = 0; j < Grid.Ny; j++)
296 for (int i = 0; i < Grid.Nx; i++)
297 {
298 outbufer << (Quaternions(i,j,k).RotationMatrix*X)[dir] <<"\n";
299 }
300 }
301 outbufer << "VECTORS X double\n";
302
303 for (int k = 0; k < Grid.Nz; k++)
304 for (int j = 0; j < Grid.Ny; j++)
305 for (int i = 0; i < Grid.Nx; i++)
306 {
307 outbufer << (Quaternions(i,j,k).RotationMatrix*X)[0] << " "
308 << (Quaternions(i,j,k).RotationMatrix*X)[1] << " "
309 << (Quaternions(i,j,k).RotationMatrix*X)[2] << "\n";
310 }
311
312 string FileName = FileInterface::MakeFileName(locSettings.VTKDir,"Rotated100Vector_",
313 tStep, ".vtk");
314
315 ofstream vtk_file(FileName.c_str());
316 vtk_file << outbufer.rdbuf();
317 vtk_file.close();
318}

320void Orientations::Advect(AdvectionHR& Adv,
321 const Velocities& Vel,
322 PhaseField& Phi,
323 const BoundaryConditions& BC,
324 const double dt, const double tStep)
325{
326 if(QuaternionsDot.IsNotAllocated())
327 {
328 QuaternionsDot.Allocate(Quaternions);
329 }
330 if(not QuaternionsDot.IsSize(Grid.Nx, Grid.Ny, Grid.Nz))
331 {
332 QuaternionsDot.Reallocate(Grid.Nx, Grid.Ny, Grid.Nz);
333 }
334
335 Adv.AdvectField(Quaternions, QuaternionsDot, Vel, BC, dt);
336 ConsoleOutput::WriteStandard(thisclassname, "Advected");
337}
