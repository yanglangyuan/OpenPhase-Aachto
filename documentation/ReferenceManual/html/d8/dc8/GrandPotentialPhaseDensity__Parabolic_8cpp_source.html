39void GrandPotentialPhaseDensity_Parabolic::ReadInput (std::stringstream& InputFile, int moduleLocation)
40{
41 ConsoleOutput::Write("");
42 ConsoleOutput::Write("Parabolic grad potential density material properties for "+PhaseNames[PhaseIdx]);
43 for(size_t comp = 0; comp < Ncomp; comp++)
44 {
45 std::string converter = ""+std::to_string(PhaseIdx)+"_"+ElementNames[comp];
46 C0[comp] = FileInterface::ReadParameterD(InputFile, moduleLocation, "C0_"+converter);
47 if (C0[comp] < 0) ConsoleOutput::WriteWarning("C0 needs to positive!",thisclassname,"Read Input");
48 }
49 for(size_t comp = 0; comp < Ncomp; comp++)
50 {
51 std::string converter = ""+std::to_string(PhaseIdx)+"_"+ElementNames[comp];
52 EPS[comp] = FileInterface::ReadParameterD(InputFile, moduleLocation, "EPS_"+converter, false, 0.0);
53 if (EPS[comp]==0.0)
54 {
55 double BulkModulus = FileInterface::ReadParameterD(InputFile, moduleLocation, "K0_"+converter);
56 EPS[comp] = BulkModulus/C0[comp]/C0[comp];
57 }
58 if (EPS[comp] <= 0) ConsoleOutput::WriteWarning("EPS needs to positive!",thisclassname,"Read Input");
59 }
60 ConsoleOutput::Write("");
61}
