50 /*
51 * This is a special version of the six component Voigt vector.
52 * The off diagonal elements of the strain tensor are multiplied by two(!)
53 * while transforming the 3x3 strain tensor to a 6-component Voigt vector.
54 * Consequently, the last three Voigt vector components have to be divided
55 * by two if they are transformed back to a 3x3 tensor.
56 * Therefore the transformation functions and some other functions
57 * have to be specified in different way then for the stress-type
58 * Voigt vector. The differences are specified in the functions.
59 */

253 double norm(void) const
254 {
255 /*
256 * Multiplication by 0.5 of the off diagonal elements before taking
257 * square and doubling the result due to double appearance of the off
258 * diagonal elements => 0.5^2 * 2 = 0.5
259 */
260 double tmp = storage[0] * storage[0]
261 + storage[1] * storage[1]
262 + storage[2] * storage[2]
263 + storage[3] * storage[3] * 0.5
264 + storage[4] * storage[4] * 0.5
265 + storage[5] * storage[5] * 0.5;
266 return sqrt(tmp);
267 }
268
269 double double_contract(const vStrain& Bstrain) const // "double-dot product"
270 {
271 double tmp =
272 storage[0] * Bstrain[0] +
273 storage[1] * Bstrain[1] +
274 storage[2] * Bstrain[2] +
275 2.0*(0.5*storage[3]) * (0.5*Bstrain[3]) +
276 2.0*(0.5*storage[4]) * (0.5*Bstrain[4]) +
277 2.0*(0.5*storage[5]) * (0.5*Bstrain[5]);
278 return tmp;
279 }
280 double PEEQ() const
281 {
282 return sqrt((2.0 / 3.0) * double_contract (*this));
283 }

298 vStrain rotated(const dMatrix3x3& RotationMatrix)
299 {
300 dMatrix3x3 locStrainTensor = tensor().rotated(RotationMatrix);
301 vStrain locStrain;
302 locStrain[0] = locStrainTensor(0,0);
303 locStrain[1] = locStrainTensor(1,1);
304 locStrain[2] = locStrainTensor(2,2);
305 locStrain[3] = locStrainTensor(1,2)*2.0;
306 locStrain[4] = locStrainTensor(0,2)*2.0;
307 locStrain[5] = locStrainTensor(0,1)*2.0;
308 return locStrain;
309 }

314

316 storage[1] = locStrainTensor(1,1);

318 storage[3] = locStrainTensor(1,2)*2.0;
319 storage[4] = locStrainTensor(0,2)*2.0;
320 storage[5] = locStrainTensor(0,1)*2.0;

330 double get_tensor(const int i, const int j) const
331 {
332 /*
333 * Multiplication by 0.5 of the off diagonal elements during translation
334 * to a 3x3 tensor
335 */
336 return (storage[(i==j)?(i):(6-(i+j))] * ((i==j)?(1.0):(0.5)));
337 }

339 dMatrix3x3 tensor(void) const
340 {
341 /*
342 * Multiplication by 0.5 of the off diagonal elements during translation
343 * to a 3x3 tensor
344 */
345 dMatrix3x3 tmp;
346 tmp(0,0) = storage[0];
347 tmp(0,1) = storage[5]*0.5;
348 tmp(0,2) = storage[4]*0.5;
349 tmp(1,0) = storage[5]*0.5;
350 tmp(1,1) = storage[1];
351 tmp(1,2) = storage[3]*0.5;
352 tmp(2,0) = storage[4]*0.5;
353 tmp(2,1) = storage[3]*0.5;
354 tmp(2,2) = storage[2];
355 return tmp;
356 }
