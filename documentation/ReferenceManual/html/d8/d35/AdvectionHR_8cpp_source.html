85void AdvectionHR::ReadInput(std::stringstream& inp)
86{
87 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
88 std::string schemeString = FileInterface::ReadParameterK(inp, moduleLocation, "scheme", false, "UPWIND");
89
90 if (schemeString == "MINMOD")
91 {
92 Scheme = AdvectionSchemes::Minmod;
93 }
94 else if (schemeString == "MC" or schemeString == "MONOTONISEDCENTRAL")
95 {
96 Scheme = AdvectionSchemes::MonotonizedCentral;
97 }
98 else if (schemeString == "SUPERBEE")
99 {
100 Scheme = AdvectionSchemes::Superbee;
101 }
102 /*else if (schemeString == "LAXWENDROFF")
103 {
104 Scheme = AdvectionSchemes::LaxWendroff;
105 }*/
106 else if (schemeString == "UPWIND")
107 {
108 Scheme = AdvectionSchemes::Upwind;
109 }
110 else
111 {
112 ConsoleOutput::WriteWarning("Wrong advection scheme specified!\nThe default \"Upwind\" scheme is used!", thisclassname, "ReadInput()");
113 }
114 switch(Scheme)
115 {
116 case AdvectionSchemes::Minmod:
117 {
118 Limiter = &Slope_Limiter_Minmod;
119 break;
120 }
121 case AdvectionSchemes::MonotonizedCentral:
122 {
123 Limiter = &Slope_Limiter_MC;
124 break;
125 }
126 case AdvectionSchemes::Superbee:
127 {
128 Limiter = &Slope_Limiter_Superbee;
129 break;
130 }
131 /*case AdvectionSchemes::LaxWendroff:
132 {
133 Limiter = &Slope_Limiter_LaxWendroff;
134 break;
135 }*/
136 case AdvectionSchemes::Upwind:
137 default:
138 {
139 Limiter = &Slope_Limiter_Upwind;
140 break;
141 }
142 }
143}

145void AdvectionHR::CalculateLocalAdvectionPhaseField(
146 const int i, const int j, const int k,
147 const int ii, const int jj, const int kk,
148 PhaseField& Phase,
149 const BoundaryConditions& BC,
150 const Velocities& Vel,
151 int& CFL,
152 const int direction,
153 const double dt,
154 double (*Limiter)(const double,const double),
155 size_t FluidIdx)
156{
157 if (Phase.Fields(i,j,k).wide_interface())
158 {
159 const auto& q = Phase.Fields(i ,j ,k );
160 const auto& qp = Phase.Fields(i+ ii,j+ jj,k+ kk);
161 const auto& qm = Phase.Fields(i- ii,j- jj,k- kk);
162 const auto& qpp = Phase.Fields(i+2*ii,j+2*jj,k+2*kk);
163 const auto& qmm = Phase.Fields(i-2*ii,j-2*jj,k-2*kk);
164
165 // Determine indices of close by grains
166 std::set<size_t> GrainIdxs;
167 for (auto alpha = q .cbegin(); alpha != q .cend(); alpha++) GrainIdxs.insert(alpha->index);
168 for (auto alpha = qp .cbegin(); alpha != qp .cend(); alpha++) GrainIdxs.insert(alpha->index);
169 for (auto alpha = qm .cbegin(); alpha != qm .cend(); alpha++) GrainIdxs.insert(alpha->index);
170 for (auto alpha = qpp.cbegin(); alpha != qpp.cend(); alpha++) GrainIdxs.insert(alpha->index);
171 for (auto alpha = qmm.cbegin(); alpha != qmm.cend(); alpha++) GrainIdxs.insert(alpha->index);
172
173 bool SolidInterface = false;
174 for (auto idx = GrainIdxs.cbegin(); idx != GrainIdxs.cend(); idx++)
175 {
176 const Grain& grain = Phase.FieldsProperties[*idx];
177 if (grain.State == AggregateStates::Solid)
178 {
179 SolidInterface = true;
180 const double v = InteractionSolidFluid::LocalGrainVelocity(i,j,k,grain,BC, Phase)[direction];
181 const double dphi_dt = AdvectionKernel<false>(CFL,v,v,v,q.get_value(*idx),qp.get_value(*idx),qm.get_value(*idx),qpp.get_value(*idx),qmm.get_value(*idx),dt,Vel.Grid.dx,Limiter);
182 Phase.FieldsAdvectionDot(i,j,k).add_value( *idx, dphi_dt);
183 Phase.FieldsAdvectionDot(i,j,k).add_value(FluidIdx,-dphi_dt);
184 }
185 }
186
187 if (!SolidInterface)
188 {
189 const double v = Vel.Average(i ,j ,k )[direction];
190 const double vp = Vel.Average(i+ii,j+jj,k+kk)[direction];
191 const double vm = Vel.Average(i-ii,j-jj,k-kk)[direction];
192
193 for (auto idx = GrainIdxs.cbegin(); idx != GrainIdxs.cend(); idx++)
194 {
195 const double dphi_dt = AdvectionKernel<true>(CFL,v,vp,vm,q.get_value(*idx),qp.get_value(*idx),qm.get_value(*idx),qpp.get_value(*idx),qmm.get_value(*idx),dt,Vel.Grid.dx,Limiter);
196 Phase.FieldsAdvectionDot(i,j,k).set_value(*idx, dphi_dt);
197 }
198 }
199 }
200}

241void AdvectionHR::CalculateAdvectionPhaseField(
242 PhaseField& Phase,
243 const BoundaryConditions& BC,
244 const Velocities& Vel,
245 const int direction,
246 double dt,
247 double (*Limiter)(const double,const double),
248 size_t FluidIdx,
249 bool& warning)
250{
251 Phase.FieldsAdvectionBackup = Phase.Fields;
252
253 const int ii = (direction == 0) ? 1 : 0;
254 const int jj = (direction == 1) ? 1 : 0;
255 const int kk = (direction == 2) ? 1 : 0;
256
257 unsigned int iterations = 1;
258 int CFL;
259 do
260 {
261 CFL = 0;
262 for (unsigned int it = 0; it < iterations; ++it)
263 {
264 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,reduction(+:CFL))
265 {
266 Phase.FieldsAdvectionDot(i,j,k).clear();
267 CalculateLocalAdvectionPhaseField(i,j,k,ii,jj,kk,Phase,BC,Vel,CFL,direction,dt,Limiter, FluidIdx);
268 }
269 OMP_PARALLEL_STORAGE_LOOP_END
270
271 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,reduction(+:CFL))
272 {
273 for (auto alpha = Phase.FieldsAdvectionDot(i,j,k).begin();
274 alpha != Phase.FieldsAdvectionDot(i,j,k).end(); alpha++)
275 {
276 Phase.Fields(i,j,k).add_value(alpha->index, alpha->value*dt);
277 }
278 }
279 OMP_PARALLEL_STORAGE_LOOP_END
280
281 BC.Set(Phase.Fields,direction);
282 }
283 if (CFL != 0)
284 {
285 Phase.Fields = Phase.FieldsAdvectionBackup;
286
287 dt *= 0.5;
288 warning = true;
289 }
290 iterations *= 2;
291 }
292 while (CFL > 0);
293}

348size_t AdvectionHR::GetFluidIndex(const PhaseField& Phase, long tStep)
349{
350 const auto methodname = "GetFluidIndex()";
351 size_t num_gas_grain = 0;
352 std::set<size_t> gas_phases;
353 for (size_t idx = 0; idx != Phase.FieldsProperties.size(); idx++)
354 {
355 if (Phase.FieldsProperties[idx].State == AggregateStates::Gas)
356 {
357 gas_phases.insert(Phase.FieldsProperties[idx].Phase);
358 num_gas_grain++;
359 }
360 }
361
362 if (gas_phases.size() > 1)
363 {
364 ConsoleOutput::WriteExit("More than one gas phase found! Immiscible gas phases can not be considered!", thisclassname, methodname);
365 OP_Exit(EXIT_FAILURE);
366 }
367 else if (num_gas_grain > 1)
368 {
369 ConsoleOutput::WriteExit("More than one gas grain found! Immiscible gas grain can not be considered!", thisclassname, methodname);
370 OP_Exit(EXIT_FAILURE);
371 }
372 else if (gas_phases.size() == 1 and num_gas_grain == 1)
373 {
374 for (size_t idx = 0; idx != Phase.FieldsProperties.size(); idx++)
375 {
376 if (Phase.FieldsProperties[idx].State == AggregateStates::Gas) return idx;
377 }
378 }
379
380 size_t num_liquid_grain = 0;
381 std::set<size_t> liquid_phases;
382 for (size_t idx = 0; idx != Phase.FieldsProperties.size(); idx++)
383 {
384 if (Phase.FieldsProperties[idx].State == AggregateStates::Liquid)
385 {
386 liquid_phases.insert(Phase.FieldsProperties[idx].Phase);
387 num_liquid_grain++;
388 }
389 }
390
391 if (liquid_phases.size() > 1)
392 {
393 ConsoleOutput::WriteExit("Multiple liquid phases found. Add vapor phase to calculate advection!", thisclassname, methodname);
394 OP_Exit(EXIT_FAILURE);
395 }
396 else if (num_liquid_grain > 1)
397 {
398 ConsoleOutput::WriteExit("More than one liquid grain found! Add vapor phase to calculate advection!", thisclassname, methodname);
399 OP_Exit(EXIT_FAILURE);
400 }
401 else if (liquid_phases.size() == 1 and num_liquid_grain == 1)
402 {
403 if (tStep == 0) ConsoleOutput::WriteWarning("Consider to add a vapor phase to calculate advection! To allow modeling of cavitation with phase-field.", thisclassname, methodname);
404 for (size_t idx = 0; idx != Phase.FieldsProperties.size(); idx++)
405 {
406 if (Phase.FieldsProperties[idx].State == AggregateStates::Liquid) return idx;
407 }
408 }
409
410 //NOTE that there must be a unique Fluid index for the current algorithm
411 // to function. If you want to consider multiple fluid grains consider
412 // to implement a nucleation algorithm which may nucleate gas grain in
413 // the solid-solid interface due to advection.
414 ConsoleOutput::WriteExit("No fluid phase found! Can not calculate phase-field advection.", thisclassname, methodname);
415 OP_Exit(EXIT_FAILURE);
416 return 0;
417}

419void AdvectionHR::AdvectPhaseField(
420 PhaseField& Phase,
421 const Velocities& Vel,
422 const BoundaryConditions& BC,
423 const double dt,
424 const int tStep,
425 const bool finalize)
426{
427 assert(Phase.Fields.Bcells() >= 2 && "Number of Bcells for storage PhaseField::Fields needs to be 2 or higher.");
428
429 size_t FluidIdx = GetFluidIndex(Phase, tStep);
430
431 bool warning = false;
432 if (tStep % 2 == 0)
433 {
434 if (Vel.Grid.dNx) CalculateAdvectionPhaseField(Phase, BC, Vel, 0, dt, Limiter, FluidIdx, warning);
435 if (Vel.Grid.dNy) CalculateAdvectionPhaseField(Phase, BC, Vel, 1, dt, Limiter, FluidIdx, warning);
436 if (Vel.Grid.dNz) CalculateAdvectionPhaseField(Phase, BC, Vel, 2, dt, Limiter, FluidIdx, warning);
437 }
438 else
439 {
440 if (Vel.Grid.dNz) CalculateAdvectionPhaseField(Phase, BC, Vel, 2, dt, Limiter, FluidIdx, warning);
441 if (Vel.Grid.dNy) CalculateAdvectionPhaseField(Phase, BC, Vel, 1, dt, Limiter, FluidIdx, warning);
442 if (Vel.Grid.dNx) CalculateAdvectionPhaseField(Phase, BC, Vel, 0, dt, Limiter, FluidIdx, warning);
443 }
444
445 if (warning)
446 {
447 ConsoleOutput::WriteWarning("Warning reduce time step, cfl > 0.5!", "AdvectionHR", "CalculateAdvection");
448 }
449
450 Phase.Finalize(BC,finalize);
451}
