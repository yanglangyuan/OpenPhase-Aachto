83void UserDrivingForce::ReadInput(std::stringstream& inp)
84{
85 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
86
87 // Reading driving force modes and parameters for each phase pair
88 for (size_t n = 0; n < Nphases; n++)
89 for (size_t m = n; m < Nphases; m++)
90 {
91 stringstream idx;
92 idx << "_" << n << "_" << m;
93 if(n == m)
94 {
95 Mode(n, m) = UserDrivingForceModes::None;
96 }
97 else
98 {
99 string locMode = FileInterface::ReadParameterK(inp, moduleLocation, "UDF_Mode" + idx.str(), false, "NONE");
100
101 if(locMode == "NONE") Mode(n, m) = Mode(m, n) = UserDrivingForceModes::None;
102 if(locMode == "VALUE") Mode(n, m) = Mode(m, n) = UserDrivingForceModes::Value;
103 if(locMode == "FORMULA") Mode(n, m) = Mode(m, n) = UserDrivingForceModes::Formula;
104 }
105
106 switch(Mode(n, m))
107 {
108 case UserDrivingForceModes::Value:
109 {
110 Value(n,m) = FileInterface::ReadParameterD(inp, moduleLocation, "UDF_Value" + idx.str(), true, 0.0);
111 Value(m,n) = -Value(n,m);
112
113 LatentHeat(n,m) = LatentHeat(m,n) = 0.0;
114 Teq(n,m) = Teq(m,n) = 1.0;
115 Slope(n,m) = Slope(m,n) = 0.0;
116
117 break;
118 }
119 case UserDrivingForceModes::Formula:
120 {
121 Value(n,m) = Value(m,n) = 0.0;
122
123 LatentHeat(n,m) = FileInterface::ReadParameterD(inp, moduleLocation, "UDF_LatentHeat" + idx.str(), true, 0.0);
124 LatentHeat(m,n) = -LatentHeat(n,m);
125
126 Teq(n,m) = FileInterface::ReadParameterD(inp, moduleLocation, "UDF_Teq" + idx.str(), true, 1.0);
127 Teq(m,n) = Teq(n,m);
128
129 Slope(n,m) = FileInterface::ReadParameterD(inp, moduleLocation, "UDF_Slope" + idx.str(), false, 0.0);
130 Slope(m,n) = Slope(n,m);
131
132 Component(n,m) = FileInterface::ReadParameterI(inp, moduleLocation, "UDF_Component" + idx.str(), false, 0);
133 Component(m,n) = Component(n,m);
134
135 break;
136 }
137 case UserDrivingForceModes::None:
138 default:
139 {
140 Value(n,m) = Value(m,n) = 0.0;
141 LatentHeat(n,m) = LatentHeat(m,n) = 0.0;
142 Teq(n,m) = Teq(m,n) = 1.0;
143 break;
144 }
145 }
146 }
147
148 ConsoleOutput::WriteLine();
149 ConsoleOutput::WriteBlankLine();
150}

152void UserDrivingForce::SetDrivingForce(PhaseField& Phase,
153 DrivingForce& dGab)
154{
155 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
156 {
157 if(Phase.Fields(i,j,k).interface())
158 {
159 for(auto alpha = Phase.Fields(i,j,k).cbegin();
160 alpha != Phase.Fields(i,j,k).cend()-1; ++alpha)
161 for(auto beta = alpha + 1;
162 beta != Phase.Fields(i,j,k).cend(); ++beta)
163 {
164 int pIndexA = Phase.FieldsProperties[alpha->index].Phase;
165 int pIndexB = Phase.FieldsProperties[ beta->index].Phase;
166
167 if(Mode(pIndexA,pIndexB) == UserDrivingForceModes::Value)
168 {
169 dGab.Force(i,j,k).add_raw(alpha->index, beta->index, Value(pIndexA,pIndexB));
170 }
171 }
172 }
173 }
174 OMP_PARALLEL_STORAGE_LOOP_END

177void UserDrivingForce::SetDrivingForce(PhaseField& Phase,
178 DrivingForce& dGab,
179 Temperature& Tx)
180{
181 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
182 {
183 if(Phase.Fields(i,j,k).interface())
184 {
185 for(auto alpha = Phase.Fields(i,j,k).cbegin();
186 alpha != Phase.Fields(i,j,k).cend()-1; ++alpha)
187 for(auto beta = alpha + 1;
188 beta != Phase.Fields(i,j,k).cend(); ++beta)
189 {
190 size_t pIndexA = Phase.FieldsProperties[alpha->index].Phase;
191 size_t pIndexB = Phase.FieldsProperties[ beta->index].Phase;
192
193 switch(Mode(pIndexA,pIndexB))
194 {
195 case UserDrivingForceModes::None:
196 {
197 break;
198 }
199 case UserDrivingForceModes::Formula:
200 {
201 double dG_AB = LatentHeat(pIndexA,pIndexB)*
202 (Tx(i,j,k) - Teq(pIndexA,pIndexB))/
203 Teq(pIndexA,pIndexB);
204
205 dGab.Force(i,j,k).add_raw(alpha->index, beta->index, dG_AB);
206 break;
207 }
208 case UserDrivingForceModes::Value:
209 {
210 dGab.Force(i,j,k).add_raw(alpha->index, beta->index, Value(pIndexA,pIndexB));
211 break;
212 }
213 }
214 }
215 }
216 }
217 OMP_PARALLEL_STORAGE_LOOP_END

220void UserDrivingForce::SetDrivingForce(PhaseField& Phase,
221 DrivingForce& dGab,
222 Temperature& Tx,
223 Composition& Cx)
224{
225 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0,)
226 {
227 if(Phase.Fields(i,j,k).interface())
228 {
229 for(auto alpha = Phase.Fields(i,j,k).cbegin();
230 alpha != Phase.Fields(i,j,k).cend()-1; ++alpha)
231 for(auto beta = alpha + 1;
232 beta != Phase.Fields(i,j,k).cend(); ++beta)
233 {
234 size_t pIndexA = Phase.FieldsProperties[alpha->index].Phase;
235 size_t pIndexB = Phase.FieldsProperties[ beta->index].Phase;
236
237 switch(Mode(pIndexA,pIndexB))
238 {
239 case UserDrivingForceModes::None:
240 {
241 break;
242 }
243 case UserDrivingForceModes::Formula:
244 {
245 double locTeq = Teq(pIndexA,pIndexB)
246 + Slope(pIndexA,pIndexB)*Cx.MoleFractions(i,j,k,{pIndexA,Component(pIndexA,pIndexB)});
247 double dG_AB = LatentHeat(pIndexA,pIndexB)*
248 (Tx(i,j,k) - locTeq)/locTeq;
249
250 dGab.Force(i,j,k).add_raw(alpha->index, beta->index, dG_AB);
251 break;
252 }
253 case UserDrivingForceModes::Value:
254 {
255 dGab.Force(i,j,k).add_raw(alpha->index, beta->index, Value(pIndexA,pIndexB));
256 break;
257 }
258 }
259 }
260 }
261 }
262 OMP_PARALLEL_STORAGE_LOOP_END
