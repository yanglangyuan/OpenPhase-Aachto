47class Composition1Dextension
48{
49 public:
50 void Initialize(const size_t size, const size_t phase_index,
51 const size_t n_comp, iVector3 direction)
52 {
53 if(size <= 0) // Checks for sufficient extension size
54 {
55 std::stringstream message;
56 message << "1D composition extension size = " << size << " is incorrect!\n"
57 << "It should be in the range [1, inf) for correct operation\n";
58 ConsoleOutput::WriteExit(message.str(), "Composition1Dextension", "Initialize()");
59 OP_Exit(EXIT_FAILURE);
60 }
61
62 PhaseIndex = phase_index;
63 Ncomp = n_comp;
64 Data.Allocate(size, {Ncomp},1);
65 DataDot.Allocate(size, {Ncomp},1);
66 Direction = direction;
67 }
68 Composition1Dextension& operator=(const Composition1Dextension& RHS)
69 {
70 Data = RHS.Data;
71 DataDot = RHS.DataDot;
72 Direction = RHS.Direction;
73 PhaseIndex = RHS.PhaseIndex;
74 Ncomp = RHS.Ncomp;
75
76 return *this;
77 }
78 void setBC(const BoundaryConditionTypes extBC)
79 {
80 switch(extBC)
81 {
82 case BoundaryConditionTypes::Periodic:
83 {
84 std::string message = std::string("If 1D composition extension is active, periodic boundary conditions\n")
85 + std::string("along the extension dimension are not permitted.\n")
86 + std::string("Permitted boundary conditions: NoFlux, Free and Fixed.");
87 ConsoleOutput::WriteWarning(message, "Composition1Dextension", "setBC()");
88 break;
89 }
90 case BoundaryConditionTypes::NoFlux:
91 case BoundaryConditionTypes::Mirror:
92 {
93 Data(size()) = Data(size()-1);
94 break;
95 }
96 case BoundaryConditionTypes::Free:
97 {
98 Data(size()) = Data(size()-1)*2.0 - Data(size()-2);
99 break;
100 }
101 default:
102 case BoundaryConditionTypes::Fixed:
103 {
104 break;
105 }
106 }
107 }
108 void setFixedBC(Tensor<double,1> value)
109 {
110 Data(size()) = value;
111 }
112 void moveFrame(const int dx, const BoundaryConditionTypes extBC)
113 {
114 if(dx > 0)
115 for(long int x = 0; x < (long int) Data.size(); x++)
116 {
117 Data(x) = Data(x+dx);
118 }
119 if(dx < 0)
120 for(long int x = (long int) Data.size() - 1; x >= 0; x--)
121 {
122 Data(x) = Data(x+dx);
123 }
124 setBC(extBC);
125 }
126 bool isActive() const
127 {
128 return Data.size() > 0;
129 }
130 size_t size() const
131 {
132 return Data.size();
133 }
134 void read(std::ifstream& out)
135 {
136 out.read(reinterpret_cast<char*>(Data.data()),Data.total_size()*Ncomp*sizeof(double));
137 }
138 void write(std::ofstream& out)
139 {
140 out.write(reinterpret_cast<char*>(Data.data()),Data.total_size()*Ncomp*sizeof(double));
141 }
142
143 size_t PhaseIndex;
144 size_t Ncomp;
145
146 Storage1D<double,1> Data;
147 Storage1D<double,1> DataDot;
148 iVector3 Direction;
149
150 protected:
151 private:
152
153};
