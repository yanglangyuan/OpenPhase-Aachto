40void Settings::Initialize(std::string ObjectNameSuffix)

58void Settings::ReadInput(const string InputFileName)
59{
60 ConsoleOutput::WriteLineInsert(thisclassname+" input");
61 ConsoleOutput::WriteStandard("Source", InputFileName);
62
63 std::string filetype = FileInterface::getFileExtension(InputFileName);
64 if (filetype == "opi")
65 {
66 fstream inp(InputFileName.c_str(), ios::in | ios_base::binary);
67
68 if (!inp)
69 {
70 ConsoleOutput::WriteExit("File \"" + InputFileName + "\" could not be opened", thisclassname, "ReadInput()");
71 OP_Exit(EXIT_FAILURE);
72 };
73
74 std::stringstream data;
75 data << inp.rdbuf();
76 ReadInput(data);
77
78 inp.close();
79 }
80 else
81 if (filetype == "json")
82 {
83 ReadJSON(InputFileName);
84 }
85 else
86 {
87 std::cerr << "Filetype " << filetype << " not recognized. Filetype must be opi or json." << std::endl;
88 OP_Exit(1);
89 }

308

442void Settings::RemeshAll(int newNx, int newNy, int newNz, int tStep, const BoundaryConditions& BC)
443{
444 if(RemeshingAllowed)
445 {
446 if(!Grid.dNx) newNx = 0;
447 if(!Grid.dNy) newNy = 0;
448 if(!Grid.dNz) newNz = 0;
449
450 for(size_t n = 0; n < ObjectsToRemesh.size(); n++)
451 {
452 ObjectsToRemesh[n]->Remesh(newNx, newNy, newNz, BC);
453 }
454
455 Grid.SetDimensions(newNx, newNy, newNz);
456
457 Grid.Idx = tStep;
458
459 GridHistory.push_back(Grid);
460 }
461}

558GridParameters Settings::GridHistoryParameters(int time_step) const
559{
560 if(time_step < 0)
561 {
562 time_step = 0;
563 }
564
565 size_t history_record_idx = 0;
566 for(size_t n = 0; n < GridHistory.size(); n++)

568 {

570 }
571 else

573 break;
574 }
