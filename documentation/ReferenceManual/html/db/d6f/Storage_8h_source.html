35template <class T>
36class Storage
37{
38 public:
39 Storage()
40 {
41 Array = nullptr;
42 Size_X = 0;
43 }
44 Storage(const Storage<T>& rhs)
45 {
46 Size_X = rhs.size();
47 Array = new T[Size_X] ();
48 for(size_t n = 0; n != Size_X; n++)
49 {
50 Array[n] = rhs[n];
51 }
52 }
53 Storage<T>& operator=(const Storage<T>& rhs)
54 {
55 if(Size_X == rhs.size())
56 {
57 Reallocate(rhs.size());
58 }
59 for(size_t n = 0; n != Size_X; n++)
60 {
61 Array[n] = rhs[n];
62 }
63 return *this;
64 }
65
66 T& operator[](const size_t x)
67 {
68 assert(x < Size_X && "Access beyond storage range");
69 return Array[x];
70 }
71 T const& operator[](const size_t x) const
72 {
73 assert(x < Size_X && "Access beyond storage range");
74 return Array[x];
75 }
76 void Allocate(const size_t nx)
77 {
78 Size_X = nx;
79 Array = new T[Size_X] ();
80 }
81 void Reallocate(const size_t nx)
82 {
83 delete[] Array;
84 Size_X = nx;
85 Array = new T[Size_X] ();
86 }
87 size_t size() const
88 {
89 return Size_X;
90 }
91 bool IsNotAllocated() const
92 {
93 return (Array == nullptr);
94 }
95 bool IsAllocated() const
96 {
97 return !(Array == nullptr);
98 }
99 ~Storage()
100 {
101 delete[] Array;
102 }
103
104 protected:
105 private:
106 T* Array;
107 size_t Size_X;
108};
