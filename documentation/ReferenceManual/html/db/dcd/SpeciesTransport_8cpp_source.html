112void SpeciesTransport::CalculateSpeciesHeatCapacitiesAndEnthalpies(EnergyTransport& ET, FlowSolverLBM& FL)
113{
114 double R = 8.314510; // J/mol.K
115 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Cp_Species,Cp_Species.Bcells(),)
116 {
117 if(!FL.Obstacle(i,j,k))
118 {
119 double T = ET.Tx(i,j,k);
120 for (size_t ic = 0 ; ic < nSpecies; ic++)
121 {
122 if(T < PolyNomCoeffs({ic,0}))
123 {
124 Cp_Species(i,j,k,{ic}) = (PolyNomCoeffs({ic,8}) + PolyNomCoeffs({ic,9})*T + PolyNomCoeffs({ic,10})*pow(T,2.0)
125 + PolyNomCoeffs({ic,11})*pow(T,3.0) + PolyNomCoeffs({ic,12})*pow(T,4.0) )
126 * R / MolecularWeight({ic});
127 h_Species(i,j,k,{ic}) = (PolyNomCoeffs({ic,8}) + PolyNomCoeffs({ic,9})/2.0*T + PolyNomCoeffs({ic,10})/3.0*pow(T,2.0)
128 + PolyNomCoeffs({ic,11})/4.0*pow(T,3.0) + PolyNomCoeffs({ic,12})/5.0*pow(T,4.0) + PolyNomCoeffs({ic,13})/T)
129 * R * T / MolecularWeight({ic});
130 }
131 else
132 {
133 Cp_Species(i,j,k,{ic}) = (PolyNomCoeffs({ic,1}) + PolyNomCoeffs({ic,2})*T + PolyNomCoeffs({ic,3})*pow(T,2.0)
134 + PolyNomCoeffs({ic,4})*pow(T,3.0) + PolyNomCoeffs({ic,5})*pow(T,4.0) )
135 * R / MolecularWeight({ic});
136 h_Species(i,j,k,{ic}) = (PolyNomCoeffs({ic,1}) + PolyNomCoeffs({ic,2})/2.0*T + PolyNomCoeffs({ic,3})/3.0*pow(T,2.0)
137 + PolyNomCoeffs({ic,4})/4.0*pow(T,3.0) + PolyNomCoeffs({ic,5})/5.0*pow(T,4.0) + PolyNomCoeffs({ic,6})/T)
138 * R * T / MolecularWeight({ic});
139 }
140
141 }
142 }
143 }
144 OMP_PARALLEL_STORAGE_LOOP_END

147void SpeciesTransport::SetFreeBCNX(PhaseField& Phase, EnergyTransport& ET)
148{
149 int iEnd = Grid.Nx;
150 double Temp = ET.T0; // Temporary variable for extrapolated temperature
151 double f4[nSpecies]; // Temporary array for extrapolated compositions
152 double f3[nSpecies];
153 double f2[nSpecies];
154 double f1[nSpecies];
155 // Loop over the extrapolated boundary nodes
156 if(Grid.OffsetX+Grid.Nx==Grid.TotalNx)
157 for (int i = iEnd; i < Grid.Nx + MassFractions.Bcells(); ++i)
158 {
159 for (int j = 0; j< Grid.Ny ; ++j)
160 {
161 for (int k = 0; k< Grid.Nz ; ++k)
162 {
163 for (size_t ic = 0; ic < nSpecies; ic++)
164 {
165 // Load values from previous points for extrapolation
166 f4[ic] = MassFractions(iEnd - 1, j, k,{ic});
167 f3[ic] = MassFractions(iEnd - 2, j, k,{ic});
168 f2[ic] = MassFractions(iEnd - 3, j, k,{ic});
169 f1[ic] = MassFractions(iEnd - 4, j, k,{ic});
170 }
171
172 if (BCOrder==0)
173 {
174 Temp = ET.Tx(iEnd - 1, j, k);
175 for (size_t ic = 0; ic < nSpecies; ic++)
176 {
177 MassFractions(i, j, k,{ic}) = f4[ic];
178 }
179 }
180 else if (BCOrder==1)
181 {
182 Temp = 2.0 * ET.Tx(iEnd - 1, 0, k) - ET.Tx(iEnd - 2, 0, k);
183 for (size_t ic = 0; ic < nSpecies; ic++)
184 {
185 MassFractions(i, j, k,{ic}) = 2.0 * f4[ic] - f3[ic];
186 }
187 }
188 else if (BCOrder==2)
189 {
190 Temp = ET.Tx(iEnd - 1, j, k) + (ET.Tx(iEnd - 1, j, k) - ET.Tx(iEnd - 2, j, k)) +
191 0.5 * (ET.Tx(iEnd - 1, j, k) - 2.0 * ET.Tx(iEnd - 2, j, k) + ET.Tx(iEnd - 3, j, k));
192 for (size_t ic = 0; ic < nSpecies; ic++)
193 {
194 MassFractions(i, j, k,{ic}) = f4[ic] + (f4[ic] - f3[ic]) +
195 0.5 * (f4[ic] - 2.0 * f3[ic] + f2[ic]);
196 }
197 }
198 else if (BCOrder==3)
199 {
200 Temp = ET.Lagrange_polynomial4(0.0, 1.0, 2.0, 3.0, 4.0,
201 ET.Tx(iEnd - 4, j, k), ET.Tx(iEnd - 3, j, k),
202 ET.Tx(iEnd - 2, j, k), ET.Tx(iEnd - 1, j, k));
203 for (size_t ic = 0; ic < nSpecies; ic++)
204 {
205 MassFractions(i, j, k,{ic}) = ET.Lagrange_polynomial4(0.0, 1.0, 2.0, 3.0, 4.0, f1[ic], f2[ic], f3[ic], f4[ic]);
206 }
207 }
208 ET.Tx(i, j, k) = Temp;
209 }
210 }
211 }

222double SpeciesTransport::CalculateMeanMolarMass(const int i, const int j, const int k, string whichtime)
223{
224 double MMWi=0.0;
225 for(size_t n = 0; n < nSpecies; n++)
226 {
227 if (whichtime == "old")
228 {
229 MMWi += MassFractionsOld(i,j,k,{n}) / MolecularWeight({n});
230 }
231 if (whichtime == "new")
232 {
233 MMWi += MassFractions(i,j,k,{n}) / MolecularWeight({n});
234 }
235 }
236 double MMW=1.0/MMWi;
237 return MMW;
238}

251void SpeciesTransport::CalculateGhostPointsConjugateHeatTransfer(PhaseField& Phase, EnergyTransport& ET, FlowSolverLBM& FL, SolidBody& SB)
252{
253 size_t Gasidx = 0;
254 size_t Solididx = 1;
255 double D = SB.nDist;
256 double ep=1e-8;
257
258 for(size_t idx = 0; idx < Phase.FieldsProperties.size(); idx++)
259 if(Phase.FieldsProperties[idx].State == AggregateStates::Gas)
260 {
261 Gasidx = idx;
262 }
263 int iBC=Grid.dNx*2.0;
264 int jBC=Grid.dNy*2.0;
265 int kBC=Grid.dNz*2.0;
266
267 int idi=Grid.dNx;
268 int jdi=Grid.dNy;
269 int kdi=Grid.dNz;
270
271 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,ET.Tx,0,)
272 {
273 bool Set_TsOut=false;
274 if(!FL.Obstacle(i,j,k))
275 for(int ii = -iBC; ii <= iBC; ii++)
276 for(int jj = -jBC; jj <= jBC; jj++)
277 for(int kk = -kBC; kk <= kBC; kk++)
278 {
279 if(FL.Obstacle(i+ii,j+jj,k+kk))
280 {
281 if(!Set_TsOut)
282 {
283 dVector3 Norm{0.0,0.0,0.0};
284 for (auto it = Phase.Fields(i,j,k).cbegin(); it != Phase.Fields(i,j,k).cend(); ++it)
285 {
286 if(Phase.FieldsProperties[it->index].State == AggregateStates::Solid)
287 {
288 Solididx=it->index;
289 NodeAB<dVector3,dVector3> Normals=Phase.Normals(i,j,k);
290 Norm=Normals.get_asym1(Solididx, Gasidx);
291 }
292 }
293 double Phif = Phase.Fractions(i,j,k,{Gasidx});
294 double X = fabs(Phase.Grid.iWidth/Pi * asin(1.0-2.0*Phif));
295 dVector3 Xf = {double(i), double(j), double(k)};
296 dVector3 Xb = Xf - Norm*X;
297 dVector3 Xs = Xb - Norm*D;
298 double ic = Xs.getX();
299 double jc = Xs.getY();
300 double kc = Xs.getZ();
301 double i0 = floor(ic);
302 double i1 = i0 + 1.0;
303 double j0 = floor(jc);
304 double j1 = j0 + 1.0;
305 double k0 = floor(kc);
306 double k1 = k0 + 1.0;
307
308 double Tsi = SB.TrilinearInterPolation(ic,jc,kc,
309 ET.Ts(i0*idi,j0*jdi,k0*kdi),ET.Ts(i1*idi,j0*jdi,k0*kdi),
310 ET.Ts(i0*idi,j1*jdi,k0*kdi),ET.Ts(i1*idi,j1*jdi,k0*kdi),
311 ET.Ts(i0*idi,j0*jdi,k1*kdi),ET.Ts(i1*idi,j0*jdi,k1*kdi),
312 ET.Ts(i0*idi,j1*jdi,k1*kdi),ET.Ts(i1*idi,j1*jdi,k1*kdi));
313 double Ksi = SB.TrilinearInterPolation(ic,jc,kc,
314 ET.K_Solid(i0*idi,j0*jdi,k0*kdi),ET.K_Solid(i1*idi,j0*jdi,k0*kdi),
315 ET.K_Solid(i0*idi,j1*jdi,k0*kdi),ET.K_Solid(i1*idi,j1*jdi,k0*kdi),
316 ET.K_Solid(i0*idi,j0*jdi,k1*kdi),ET.K_Solid(i1*idi,j0*jdi,k1*kdi),
317 ET.K_Solid(i0*idi,j1*jdi,k1*kdi),ET.K_Solid(i1*idi,j1*jdi,k1*kdi));
318 double Tf = ET.Tx(i,j,k);
319 double Kf = ET.K_Mixture(i,j,k);
320
321 double Tb = (Kf/X*Tf+Ksi/D*Tsi)/(Kf/X+Ksi/D);
322
323 ET.Ts (i,j,k) = Tb * (1.0+X/D) - X/D*Tsi;
324 Set_TsOut=true;
325 }
326 }
327 }
328
329 bool Set_TfOut=false;
330 if(FL.Obstacle(i,j,k))
331 for(int ii = -iBC; ii <= iBC; ii++)
332 for(int jj = -jBC; jj <= jBC; jj++)
333 for(int kk = -kBC; kk <= kBC; kk++)
334 {
335 if(!FL.Obstacle(i+ii,j+jj,k+kk))
336 {
337 if(!Set_TfOut)
338 {
339 dVector3 Norm{0.0,0.0,0.0};
340 for (auto it = Phase.Fields(i,j,k).cbegin(); it != Phase.Fields(i,j,k).cend(); ++it)
341 {
342 if(Phase.FieldsProperties[it->index].State == AggregateStates::Solid)
343 {
344 Solididx=it->index;
345 NodeAB<dVector3,dVector3> Normals=Phase.Normals(i,j,k);
346 Norm=Normals.get_asym1(Solididx, Gasidx);
347 }
348 }
349 double Phif = Phase.Fractions(i,j,k,{Gasidx});
350 double X = fabs(Phase.Grid.iWidth/Pi * asin(1.0-2.0*Phif));
351 dVector3 Xs = {double(i), double(j), double(k)};
352 dVector3 Xb = Xs + Norm*X;
353 dVector3 Xf = Xb + Norm*D;
354 double ic = Xf.getX();
355 double jc = Xf.getY();
356 double kc = Xf.getZ();
357 double i0 = floor(ic);
358 double i1 = i0 + 1.0;
359 double j0 = floor(jc);
360 double j1 = j0 + 1.0;
361 double k0 = floor(kc);
362 double k1 = k0 + 1.0;
363
364 double Tfi = SB.TrilinearInterPolation(ic,jc,kc,
365 ET.Tx(i0*idi,j0*jdi,k0*kdi),ET.Tx(i1*idi,j0*jdi,k0*kdi),
366 ET.Tx(i0*idi,j1*jdi,k0*kdi),ET.Tx(i1*idi,j1*jdi,k0*kdi),
367 ET.Tx(i0*idi,j0*jdi,k1*kdi),ET.Tx(i1*idi,j0*jdi,k1*kdi),
368 ET.Tx(i0*idi,j1*jdi,k1*kdi),ET.Tx(i1*idi,j1*jdi,k1*kdi));
369 double Kfi = SB.TrilinearInterPolation(ic,jc,kc,
370 ET.K_Mixture(i0*idi,j0*jdi,k0*kdi),ET.K_Mixture(i1*idi,j0*jdi,k0*kdi),
371 ET.K_Mixture(i0*idi,j1*jdi,k0*kdi),ET.K_Mixture(i1*idi,j1*jdi,k0*kdi),
372 ET.K_Mixture(i0*idi,j0*jdi,k1*kdi),ET.K_Mixture(i1*idi,j0*jdi,k1*kdi),
373 ET.K_Mixture(i0*idi,j1*jdi,k1*kdi),ET.K_Mixture(i1*idi,j1*jdi,k1*kdi));
374 double Tsi = ET.Ts(i,j,k);
375 double Ksi = ET.K_Solid(i,j,k);
376
377 if(X<ep)
378 {
379 ET.Tx (i,j,k) = Tsi;
380 }
381 else
382 {
383 double Tb = (Kfi/D*Tfi+Ksi/X*Tsi)/(Kfi/D+Ksi/X);
384 ET.Tx(i,j,k) = Tb * (1.0+X/D) - X/D * Tfi;
385 }
386
387 double Cf[nSpecies];
388 for (size_t isp = 0; isp < nSpecies; isp++)
389 {
390 Cf[isp]= SB.TrilinearInterPolation(ic,jc,kc,
391 MassFractions(i0*idi,j0*jdi,k0*kdi,{isp}),MassFractions(i1*idi,j0*jdi,k0*kdi,{isp}),
392 MassFractions(i0*idi,j1*jdi,k0*kdi,{isp}),MassFractions(i1*idi,j1*jdi,k0*kdi,{isp}),
393 MassFractions(i0*idi,j0*jdi,k1*kdi,{isp}),MassFractions(i1*idi,j0*jdi,k1*kdi,{isp}),
394 MassFractions(i0*idi,j1*jdi,k1*kdi,{isp}),MassFractions(i1*idi,j1*jdi,k1*kdi,{isp}));
395 }
396 for (size_t isp=0; isp<nSpecies; isp++)
397 {
398 MassFractions(i,j,k,{isp}) = Cf[isp] ;
399 }
400 Set_TfOut=true;
401 }
402 }
403 }
404 }
405 OMP_PARALLEL_STORAGE_LOOP_END
406}

682void SpeciesTransport::CalculateAdvection(EnergyTransport& ET, FlowSolverLBM& FL, FlowMixture& FM, double dt)
683{
684 std::vector<int> dir(3);
685 double eps=1e-10;
686 double dx = Grid.dx;
687 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,MassFractions,0,)
688 {
689 if(!FL.Obstacle(i,j,k))
690 {
691 double Tc = ET.TxOld(i,j,k);
692 for (int direction=0; direction<3; ++direction)
693 {
694 dir[0] = (direction == 0)?1*Grid.dNx:0;
695 dir[1] = (direction == 1)?1*Grid.dNy:0;
696 dir[2] = (direction == 2)?1*Grid.dNz:0;
697 int ip=i+dir[0];
698 int jp=j+dir[1];
699 int kp=k+dir[2];
700 int im=i-dir[0];
701 int jm=j-dir[1];
702 int km=k-dir[2];
703
704 double Tp = ET.TxOld(ip,jp,kp);
705 double Tm = ET.TxOld(im,jm,km);
706 double Vc = FM.V_Mixture(i,j,k)[direction];
707
708 if(AdvectionUpwind)
709 {
710 if(Vc>=0)
711 {
712 ET.Tx(i,j,k) += -dt/dx * Vc * (Tc-Tm);
713 for(size_t n =0; n < nSpecies; n++)
714 {
715 double C = MassFractionsOld(i,j,k,{n});
716 double Cm = MassFractionsOld(im,jm,km,{n});
717 MassFractions(i, j, k,{n}) += ( -dt/dx * Vc * (C-Cm));
718 }
719 }
720 else
721 {
722 ET.Tx(i,j,k) += -dt/dx * Vc * (Tp-Tc);
723 for(size_t n =0; n < nSpecies; n++)
724 {
725 double C = MassFractionsOld(i,j,k,{n});
726 double Cp = MassFractionsOld(ip,jp,kp,{n});
727 MassFractions(i, j, k,{n}) += ( -dt/dx * Vc * (Cp-C));
728 }
729 }
730 }
731 else if(AdvectionVanLeer)
732 {
733 double Thp = (Tc+Tp)/2.0;
734 double Thm = (Tc+Tm)/2.0;
735 if(Vc>=0)
736 {
737 int imm=i-2.0*dir[0];
738 int jmm=j-2.0*dir[1];
739 int kmm=k-2.0*dir[2];
740 double Tmm = ET.TxOld(imm,jmm,kmm);
741 double r = (Tc-Tm)/(Tp-Tc+eps);
742 double phi = (r+fabs(r))/(1.0+fabs(r));
743 Thp = Tc+0.5*phi*(Tp-Tc);
744 double rm = (Tm-Tmm)/(Tc-Tm+eps);
745 double phim=(rm+fabs(rm))/(1.0+fabs(rm));
746 Thm=Tm+0.5*phim*(Tc-Tm);
747 ET.Tx(i,j,k) += -dt*Vc*(Thp-Thm)/dx;
748 for(size_t n =0; n < nSpecies; n++)
749 {
750 double C = MassFractionsOld(i,j,k,{n});
751 double Cp = MassFractionsOld(ip,jp,kp,{n});
752 double Cm = MassFractionsOld(im,jm,km,{n});
753 double Cmm = MassFractionsOld(imm,jmm,kmm,{n});
754 double r = (C-Cm)/(Cp-C+eps);
755 double phi = (r+fabs(r))/(1.0+fabs(r));
756 double Chp = C+0.5*phi*(Cp-C);
757 double rm = (Cm-Cmm)/(C-Cm+eps);
758 double phim = (rm+fabs(rm))/(1.0+fabs(rm));
759 double Chm = Cm+0.5*phim*(C-Cm);
760 MassFractions(i, j, k,{n}) += -dt * Vc * (Chp-Chm)/dx;
761 }
762 }
763 else
764 {
765 int ipp=i+2.0*dir[0];
766 int jpp=j+2.0*dir[1];
767 int kpp=k+2.0*dir[2];
768 double Tpp = ET.TxOld(ipp,jpp,kpp);
769 double r = (Tpp-Tp)/(Tp-Tc+eps);
770 double phi=(r+fabs(r))/(1.0+fabs(r));
771 Thp=Tp-0.5*phi*(Tp-Tc);
772 double rm = (Tp-Tc)/(Tc-Tm+eps);
773 double phim=(rm+fabs(rm))/(1.0+fabs(rm));
774 Thm=Tc-0.5*phim*(Tc-Tm);
775 ET.Tx(i,j,k) += -dt*Vc*(Thp-Thm)/dx;
776
777 for(size_t n =0; n < nSpecies; n++)
778 {
779 double C = MassFractionsOld(i,j,k,{n});
780 double Cm = MassFractionsOld(im,jm,km,{n});
781 double Cp = MassFractionsOld(ip,jp,kp,{n});
782 double Cpp = MassFractionsOld(ipp,jpp,kpp,{n});
783 double r = (Cpp-Cp)/(Cp-C+eps);
784 double phi = (r+fabs(r))/(1.0+fabs(r));
785 double Chp = Cp-0.5*phi*(Cp-C);
786 double rm = (Cp-C)/(C-Cm+eps);
787 double phim = (rm+fabs(rm))/(1.0+fabs(rm));
788 double Chm = C-0.5*phim*(C-Cm);
789 MassFractions(i, j, k,{n}) += -dt * Vc * (Chp-Chm)/dx;
790 }
791 }
792 }
793 else if (AdvectionCentral)
794 {
795 ET.Tx(i,j,k) += -dt/dx * Vc * (Tp-Tm)/2.0;
796 for(size_t n =0; n < nSpecies; n++)
797 {
798 double Cm = MassFractionsOld(im,jm,km,{n});
799 double Cp = MassFractionsOld(ip,jp,kp,{n});
800 MassFractions(i, j, k,{n}) += ( -dt/dx * Vc * (Cp-Cm)/2.0);
801 }
802 }
803 }
804 }
805 }
806 OMP_PARALLEL_STORAGE_LOOP_END
