228char* StrPair::ParseText( char* p, const char* endTag, int strFlags, int* curLineNumPtr )
229{
230 TIXMLASSERT( p );
231 TIXMLASSERT( endTag && *endTag );
232 TIXMLASSERT(curLineNumPtr);
233
234 char* start = p;
235 const char endChar = *endTag;
236 size_t length = strlen( endTag );
237
238 // Inner loop of text parsing.
239 while ( *p ) {
240 if ( *p == endChar && strncmp( p, endTag, length ) == 0 ) {
241 Set( start, p, strFlags );
242 return p + length;
243 } else if (*p == '\n') {
244 ++(*curLineNumPtr);
245 }
246 ++p;
247 TIXMLASSERT( p );
248 }
249 return 0;

273void StrPair::CollapseWhitespace()
274{
275 // Adjusting _start would cause undefined behavior on delete[]
276 TIXMLASSERT( ( _flags & NEEDS_DELETE ) == 0 );
277 // Trim leading space.
278 _start = XMLUtil::SkipWhiteSpace( _start, 0 );
279
280 if ( *_start ) {
281 const char* p = _start; // the read pointer
282 char* q = _start; // the write pointer
283
284 while( *p ) {
285 if ( XMLUtil::IsWhiteSpace( *p )) {
286 p = XMLUtil::SkipWhiteSpace( p, 0 );
287 if ( *p == 0 ) {
288 break; // don't write to q; this trims the trailing space.
289 }
290 *q = ' ';
291 ++q;
292 }
293 *q = *p;
294 ++q;
295 ++p;
296 }
297 *q = 0;
298 }

302const char* StrPair::GetStr()
303{
304 TIXMLASSERT( _start );
305 TIXMLASSERT( _end );
306 if ( _flags & NEEDS_FLUSH ) {
307 *_end = 0;
308 _flags ^= NEEDS_FLUSH;
309
310 if ( _flags ) {
311 const char* p = _start; // the read pointer
312 char* q = _start; // the write pointer
313
314 while( p < _end ) {
315 if ( (_flags & NEEDS_NEWLINE_NORMALIZATION) && *p == CR ) {
316 // CR-LF pair becomes LF
317 // CR alone becomes LF
318 // LF-CR becomes LF
319 if ( *(p+1) == LF ) {
320 p += 2;
321 }
322 else {
323 ++p;
324 }
325 *q = LF;
326 ++q;
327 }
328 else if ( (_flags & NEEDS_NEWLINE_NORMALIZATION) && *p == LF ) {
329 if ( *(p+1) == CR ) {
330 p += 2;
331 }
332 else {
333 ++p;
334 }
335 *q = LF;
336 ++q;
337 }
338 else if ( (_flags & NEEDS_ENTITY_PROCESSING) && *p == '&' ) {
339 // Entities handled by tinyXML2:
340 // - special entities in the entity table [in/out]
341 // - numeric character reference [in]
342 // &#20013; or &#x4e2d;
343
344 if ( *(p+1) == '#' ) {
345 const int buflen = 10;
346 char buf[buflen] = { 0 };
347 int len = 0;
348 const char* adjusted = const_cast<char*>( XMLUtil::GetCharacterRef( p, buf, &len ) );
349 if ( adjusted == 0 ) {
350 *q = *p;
351 ++p;
352 ++q;
353 }
354 else {
355 TIXMLASSERT( 0 <= len && len <= buflen );
356 TIXMLASSERT( q + len <= adjusted );
357 p = adjusted;
358 memcpy( q, buf, len );
359 q += len;
360 }
361 }
362 else {
363 bool entityFound = false;
364 for( int i = 0; i < NUM_ENTITIES; ++i ) {
365 const Entity& entity = entities[i];
366 if ( strncmp( p + 1, entity.pattern, entity.length ) == 0
367 && *( p + entity.length + 1 ) == ';' ) {
368 // Found an entity - convert.
369 *q = entity.value;
370 ++q;
371 p += entity.length + 2;
372 entityFound = true;
373 break;
374 }
375 }
376 if ( !entityFound ) {
377 // fixme: treat as error?
378 ++p;
379 ++q;
380 }
381 }
382 }
383 else {
384 *q = *p;
385 ++p;
386 ++q;
387 }
388 }
389 *q = 0;
390 }
391 // The loop below has plenty going on, and this
392 // is a less useful mode. Break it out.
393 if ( _flags & NEEDS_WHITESPACE_COLLAPSING ) {
394 CollapseWhitespace();
395 }
396 _flags = (_flags & NEEDS_DELETE);
397 }
398 TIXMLASSERT( _start );
399 return _start;
400}

438void XMLUtil::ConvertUTF32ToUTF8( unsigned long input, char* output, int* length )
439{
440 const unsigned long BYTE_MASK = 0xBF;
441 const unsigned long BYTE_MARK = 0x80;
442 const unsigned long FIRST_BYTE_MARK[7] = { 0x00, 0x00, 0xC0, 0xE0, 0xF0, 0xF8, 0xFC };
443
444 if (input < 0x80) {
445 *length = 1;
446 }
447 else if ( input < 0x800 ) {
448 *length = 2;
449 }
450 else if ( input < 0x10000 ) {
451 *length = 3;
452 }
453 else if ( input < 0x200000 ) {
454 *length = 4;
455 }
456 else {
457 *length = 0; // This code won't convert this correctly anyway.
458 return;
459 }
460
461 output += *length;
462
463 // Scary scary fall throughs are annotated with carefully designed comments
464 // to suppress compiler warnings such as -Wimplicit-fallthrough in gcc
465 switch (*length) {
466 case 4:
467 --output;
468 *output = static_cast<char>((input | BYTE_MARK) & BYTE_MASK);
469 input >>= 6;
470 TIXML_FALLTHROUGH;
471 case 3:
472 --output;
473 *output = static_cast<char>((input | BYTE_MARK) & BYTE_MASK);
474 input >>= 6;
475 TIXML_FALLTHROUGH;
476 case 2:
477 --output;
478 *output = static_cast<char>((input | BYTE_MARK) & BYTE_MASK);
479 input >>= 6;
480 TIXML_FALLTHROUGH;
481 case 1:
482 --output;
483 *output = static_cast<char>(input | FIRST_BYTE_MARK[*length]);
484 break;
485 default:
486 TIXMLASSERT( false );
487 }
488}

491const char* XMLUtil::GetCharacterRef(const char* p, char* value, int* length)
492{
493 // Assume an entity, and pull it out.
494 *length = 0;
495
496 static const uint32_t MAX_CODE_POINT = 0x10FFFF;
497
498 if (*(p + 1) == '#' && *(p + 2)) {
499 uint32_t ucs = 0;
500 ptrdiff_t delta = 0;
501 uint32_t mult = 1;
502 static const char SEMICOLON = ';';
503
504 bool hex = false;
505 uint32_t radix = 10;
506 const char* q = 0;
507 char terminator = '#';
508
509 if (*(p + 2) == 'x') {
510 // Hexadecimal.
511 hex = true;
512 radix = 16;
513 terminator = 'x';
514
515 q = p + 3;
516 }
517 else {
518 // Decimal.
519 q = p + 2;
520 }
521 if (!(*q)) {
522 return 0;
523 }
524
525 q = strchr(q, SEMICOLON);
526 if (!q) {
527 return 0;
528 }
529 TIXMLASSERT(*q == SEMICOLON);
530
531 delta = q - p;
532 --q;
533
534 while (*q != terminator) {
535 uint32_t digit = 0;
536
537 if (*q >= '0' && *q <= '9') {
538 digit = *q - '0';
539 }
540 else if (hex && (*q >= 'a' && *q <= 'f')) {
541 digit = *q - 'a' + 10;
542 }
543 else if (hex && (*q >= 'A' && *q <= 'F')) {
544 digit = *q - 'A' + 10;
545 }
546 else {
547 return 0;
548 }
549 TIXMLASSERT(digit < radix);
550
551 const unsigned int digitScaled = mult * digit;
552 ucs += digitScaled;
553 mult *= radix;
554
555 // Security check: could a value exist that is out of range?
556 // Easily; limit to the MAX_CODE_POINT, which also allows for a
557 // bunch of leading zeroes.
558 if (mult > MAX_CODE_POINT) {
559 mult = MAX_CODE_POINT;
560 }
561 --q;
562 }
563 // Out of range:
564 if (ucs > MAX_CODE_POINT) {
565 return 0;
566 }
567 // convert the UCS to UTF-8
568 ConvertUTF32ToUTF8(ucs, value, length);
569 if (length == 0) {
570 // If length is 0, there was an error. (Security? Bad input?)
571 // Fail safely.
572 return 0;
573 }
574 return p + delta + 1;
575 }
576 return p + 1;
577}

723char* XMLDocument::Identify( char* p, XMLNode** node, bool first )
724{
725 TIXMLASSERT( node );
726 TIXMLASSERT( p );
727 char* const start = p;
728 int const startLine = _parseCurLineNum;
729 p = XMLUtil::SkipWhiteSpace( p, &_parseCurLineNum );
730 if( !*p ) {
731 *node = 0;
732 TIXMLASSERT( p );
733 return p;
734 }
735
736 // These strings define the matching patterns:
737 static const char* xmlHeader = { "<?" };
738 static const char* commentHeader = { "<!--" };
739 static const char* cdataHeader = { "<![CDATA[" };
740 static const char* dtdHeader = { "<!" };
741 static const char* elementHeader = { "<" }; // and a header for everything else; check last.
742
743 static const int xmlHeaderLen = 2;
744 static const int commentHeaderLen = 4;
745 static const int cdataHeaderLen = 9;
746 static const int dtdHeaderLen = 2;
747 static const int elementHeaderLen = 1;
748
749 TIXMLASSERT( sizeof( XMLComment ) == sizeof( XMLUnknown ) ); // use same memory pool
750 TIXMLASSERT( sizeof( XMLComment ) == sizeof( XMLDeclaration ) ); // use same memory pool
751 XMLNode* returnNode = 0;
752 if ( XMLUtil::StringEqual( p, xmlHeader, xmlHeaderLen ) ) {
753 returnNode = CreateUnlinkedNode<XMLDeclaration>( _commentPool );
754 returnNode->_parseLineNum = _parseCurLineNum;
755 p += xmlHeaderLen;
756 }
757 else if ( XMLUtil::StringEqual( p, commentHeader, commentHeaderLen ) ) {
758 returnNode = CreateUnlinkedNode<XMLComment>( _commentPool );
759 returnNode->_parseLineNum = _parseCurLineNum;
760 p += commentHeaderLen;
761 }
762 else if ( XMLUtil::StringEqual( p, cdataHeader, cdataHeaderLen ) ) {
763 XMLText* text = CreateUnlinkedNode<XMLText>( _textPool );
764 returnNode = text;
765 returnNode->_parseLineNum = _parseCurLineNum;
766 p += cdataHeaderLen;
767 text->SetCData( true );
768 }
769 else if ( XMLUtil::StringEqual( p, dtdHeader, dtdHeaderLen ) ) {
770 returnNode = CreateUnlinkedNode<XMLUnknown>( _commentPool );
771 returnNode->_parseLineNum = _parseCurLineNum;
772 p += dtdHeaderLen;
773 }
774 else if ( XMLUtil::StringEqual( p, elementHeader, elementHeaderLen ) ) {
775
776 // Preserve whitespace pedantically before closing tag, when it's immediately after opening tag
777 if (WhitespaceMode() == PEDANTIC_WHITESPACE && first && p != start && *(p + elementHeaderLen) == '/') {
778 returnNode = CreateUnlinkedNode<XMLText>(_textPool);
779 returnNode->_parseLineNum = startLine;
780 p = start; // Back it up, all the text counts.
781 _parseCurLineNum = startLine;
782 }
783 else {
784 returnNode = CreateUnlinkedNode<XMLElement>(_elementPool);
785 returnNode->_parseLineNum = _parseCurLineNum;
786 p += elementHeaderLen;
787 }
788 }
789 else {

793 _parseCurLineNum = startLine;
794 }
795
796 TIXMLASSERT( returnNode );
797 TIXMLASSERT( p );
798 *node = returnNode;
799 return p;
800}

1093char* XMLNode::ParseDeep( char* p, StrPair* parentEndTag, int* curLineNumPtr )
1094{
1095 // This is a recursive method, but thinking about it "at the current level"
1096 // it is a pretty simple flat list:
1097 // <foo/>
1098 // <!-- comment -->
1099 //
1100 // With a special case:
1101 // <foo>
1102 // </foo>
1103 // <!-- comment -->
1104 //
1105 // Where the closing element (/foo) *must* be the next thing after the opening
1106 // element, and the names must match. BUT the tricky bit is that the closing
1107 // element will be read by the child.
1108 //
1109 // 'endTag' is the end tag for this node, it is returned by a call to a child.
1110 // 'parentEnd' is the end tag for the parent, which is filled in and returned.
1111
1112 XMLDocument::DepthTracker tracker(_document);
1113 if (_document->Error())
1114 return 0;
1115
1116 bool first = true;
1117 while( p && *p ) {
1118 XMLNode* node = 0;
1119
1120 p = _document->Identify( p, &node, first );
1121 TIXMLASSERT( p );
1122 if ( node == 0 ) {
1123 break;
1124 }
1125 first = false;
1126
1127 const int initialLineNum = node->_parseLineNum;
1128
1129 StrPair endTag;
1130 p = node->ParseDeep( p, &endTag, curLineNumPtr );
1131 if ( !p ) {
1132 _document->DeleteNode( node );
1133 if ( !_document->Error() ) {
1134 _document->SetError( XML_ERROR_PARSING, initialLineNum, 0);
1135 }
1136 break;
1137 }
1138
1139 const XMLDeclaration* const decl = node->ToDeclaration();
1140 if ( decl ) {
1141 // Declarations are only allowed at document level
1142 //
1143 // Multiple declarations are allowed but all declarations
1144 // must occur before anything else.
1145 //
1146 // Optimized due to a security test case. If the first node is
1147 // a declaration, and the last node is a declaration, then only
1148 // declarations have so far been added.
1149 bool wellLocated = false;
1150
1151 if (ToDocument()) {
1152 if (FirstChild()) {
1153 wellLocated =
1154 FirstChild() &&
1155 FirstChild()->ToDeclaration() &&
1156 LastChild() &&
1157 LastChild()->ToDeclaration();
1158 }
1159 else {
1160 wellLocated = true;
1161 }
1162 }
1163 if ( !wellLocated ) {
1164 _document->SetError( XML_ERROR_PARSING_DECLARATION, initialLineNum, "XMLDeclaration value=%s", decl->Value());
1165 _document->DeleteNode( node );
1166 break;
1167 }
1168 }
1169
1170 XMLElement* ele = node->ToElement();
1171 if ( ele ) {
1172 // We read the end tag. Return it to the parent.
1173 if ( ele->ClosingType() == XMLElement::CLOSING ) {
1174 if ( parentEndTag ) {
1175 ele->_value.TransferTo( parentEndTag );
1176 }
1177 node->_memPool->SetTracked(); // created and then immediately deleted.
1178 DeleteNode( node );
1179 return p;
1180 }
1181
1182 // Handle an end tag returned to this level.
1183 // And handle a bunch of annoying errors.
1184 bool mismatch = false;
1185 if ( endTag.Empty() ) {
1186 if ( ele->ClosingType() == XMLElement::OPEN ) {
1187 mismatch = true;
1188 }
1189 }
1190 else {
1191 if ( ele->ClosingType() != XMLElement::OPEN ) {
1192 mismatch = true;
1193 }
1194 else if ( !XMLUtil::StringEqual( endTag.GetStr(), ele->Name() ) ) {
1195 mismatch = true;
1196 }
1197 }
1198 if ( mismatch ) {
1199 _document->SetError( XML_ERROR_MISMATCHED_ELEMENT, initialLineNum, "XMLElement name=%s", ele->Name());
1200 _document->DeleteNode( node );
1201 break;
1202 }
1203 }
1204 InsertEndChild( node );
1205 }
1206 return 0;
1207}

1980char* XMLElement::ParseAttributes( char* p, int* curLineNumPtr )
1981{
1982 XMLAttribute* prevAttribute = 0;
1983
1984 // Read the attributes.
1985 while( p ) {
1986 p = XMLUtil::SkipWhiteSpace( p, curLineNumPtr );
1987 if ( !(*p) ) {
1988 _document->SetError( XML_ERROR_PARSING_ELEMENT, _parseLineNum, "XMLElement name=%s", Name() );
1989 return 0;
1990 }
1991
1992 // attribute.
1993 if (XMLUtil::IsNameStartChar( static_cast<unsigned char>(*p) ) ) {
1994 XMLAttribute* attrib = CreateAttribute();
1995 TIXMLASSERT( attrib );
1996 attrib->_parseLineNum = _document->_parseCurLineNum;
1997
1998 const int attrLineNum = attrib->_parseLineNum;
1999
2000 p = attrib->ParseDeep( p, _document->ProcessEntities(), curLineNumPtr );
2001 if ( !p || Attribute( attrib->Name() ) ) {
2002 DeleteAttribute( attrib );
2003 _document->SetError( XML_ERROR_PARSING_ATTRIBUTE, attrLineNum, "XMLElement name=%s", Name() );
2004 return 0;
2005 }
2006 // There is a minor bug here: if the attribute in the source xml
2007 // document is duplicated, it will not be detected and the
2008 // attribute will be doubly added. However, tracking the 'prevAttribute'
2009 // avoids re-scanning the attribute list. Preferring performance for
2010 // now, may reconsider in the future.
2011 if ( prevAttribute ) {
2012 TIXMLASSERT( prevAttribute->_next == 0 );
2013 prevAttribute->_next = attrib;
2014 }
2015 else {
2016 TIXMLASSERT( _rootAttribute == 0 );
2017 _rootAttribute = attrib;
2018 }
2019 prevAttribute = attrib;
2020 }
2021 // end of the tag
2022 else if ( *p == '>' ) {
2023 ++p;
2024 break;
2025 }
2026 // end of the tag
2027 else if ( *p == '/' && *(p+1) == '>' ) {
2028 _closingType = CLOSED;
2029 return p+2; // done; sealed element.
2030 }
2031 else {
2032 _document->SetError( XML_ERROR_PARSING_ELEMENT, _parseLineNum, 0 );
2033 return 0;
2034 }
2035 }
2036 return p;
2037}

2384XMLError XMLDocument::LoadFile( FILE* fp )
2385{

2387
2388 TIXML_FSEEK( fp, 0, SEEK_SET );
2389 if ( fgetc( fp ) == EOF && ferror( fp ) != 0 ) {
2390 SetError( XML_ERROR_FILE_READ_ERROR, 0, 0 );

2392 }
2393
2394 TIXML_FSEEK( fp, 0, SEEK_END );
2395
2396 unsigned long long filelength;
2397 {
2398 const long long fileLengthSigned = TIXML_FTELL( fp );
2399 TIXML_FSEEK( fp, 0, SEEK_SET );
2400 if ( fileLengthSigned == -1L ) {
2401 SetError( XML_ERROR_FILE_READ_ERROR, 0, 0 );
2402 return _errorID;
2403 }
2404 TIXMLASSERT( fileLengthSigned >= 0 );
2405 filelength = static_cast<unsigned long long>(fileLengthSigned);
2406 }
2407
2408 const size_t maxSizeT = static_cast<size_t>(-1);
2409 // We'll do the comparison as an unsigned long long, because that's guaranteed to be at
2410 // least 8 bytes, even on a 32-bit platform.
2411 if ( filelength >= static_cast<unsigned long long>(maxSizeT) ) {
2412 // Cannot handle files which won't fit in buffer together with null terminator
2413 SetError( XML_ERROR_FILE_READ_ERROR, 0, 0 );
2414 return _errorID;
2415 }

2687void XMLPrinter::PrintString( const char* p, bool restricted )
2688{
2689 // Look for runs of bytes between entities to print.
2690 const char* q = p;
2691
2692 if ( _processEntities ) {

2694 while ( *q ) {

2699 // the stream up until the entity, write the

2704 const int toPrint = ( INT_MAX < delta ) ? INT_MAX : static_cast<int>(delta);
2705 Write( p, toPrint );

2722 ++p;
2723 }
2724 }
2725 ++q;
2726 TIXMLASSERT( p <= q );
2727 }
2728 // Flush the remaining string. This will be the entire
2729 // string if an entity wasn't found.
2730 if ( p < q ) {
2731 const size_t delta = q - p;
2732 const int toPrint = ( INT_MAX < delta ) ? INT_MAX : static_cast<int>(delta);
