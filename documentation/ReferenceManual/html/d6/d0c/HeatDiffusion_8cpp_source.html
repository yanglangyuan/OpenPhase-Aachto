38inline pair<int, int> three_state_bounds_selector(int selector, int lower_bound, int upper_bound)
39{
40 pair<int, int> result;
41 switch (selector)
42 {
43 case -1: // lower bound only
44 {
45 result = make_pair(lower_bound, lower_bound + 1);
46 break;
47 }
48 case 0: // both bounds
49 {
50 result = make_pair(lower_bound, upper_bound);
51 break;
52 }
53 case 1: // upper bound only
54 {
55 result = make_pair(upper_bound - 1, upper_bound);
56 break;
57 }
58 }
59 return result;
60}

156void HeatDiffusion::SetLocalLatentHeat(const PhaseField& Phase, const Temperature& Tx)
157{
161 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Qdot,0,)
162 if(Phase.Fields(i,j,k).interface())
163 {
164 for(auto it = Phase.FieldsDot(i,j,k).cbegin();
165 it != Phase.FieldsDot(i,j,k).cend(); ++it)
166 if(fabs(it->value1) > DBL_EPSILON)
167 {
168 size_t PIdxA = Phase.FieldsProperties[it->indexA].Phase;
169 size_t PIdxB = Phase.FieldsProperties[it->indexB].Phase;
170
171 Qdot(i,j,k) += Tx.LatentHeat({PIdxA,PIdxB})*it->value1;
172 }
173 }
174 OMP_PARALLEL_STORAGE_LOOP_END
175}

177void HeatDiffusion::SetLocalLatentHeatAndApply(const PhaseField& Phase, Temperature& Tx, double dt)
178{
182 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Qdot,0,)
183 if(Phase.Fields(i,j,k).interface())
184 {
185 for(auto it = Phase.FieldsDot(i,j,k).cbegin();
186 it != Phase.FieldsDot(i,j,k).cend(); ++it)
187 if(fabs(it->value1) > DBL_EPSILON)
188 {
189 size_t PIdxA = Phase.FieldsProperties[it->indexA].Phase;
190 size_t PIdxB = Phase.FieldsProperties[it->indexB].Phase;
191
192 Qdot(i,j,k) += Tx.LatentHeat({PIdxA,PIdxB})*it->value1;
193 }
194 }
195 OMP_PARALLEL_STORAGE_LOOP_END
196 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Qdot,0,)
197 if(Phase.Fields(i,j,k).interface())
198 {
199 Tx(i,j,k) += dt*Qdot(i,j,k)/EffectiveHeatCapacity(i,j,k);
200 }
201 OMP_PARALLEL_STORAGE_LOOP_END
202 Qdot.Clear();
203}
204void HeatDiffusion::SetEffectiveProperties(const PhaseField& Phase, const Temperature& Tx)
205{
206 if(Tx.LatentHeatMode == LatentHeatModes::Local)
207 {
208 SetLocalLatentHeat(Phase, Tx);
209 }
210
211 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,EffectiveThermalConductivity,EffectiveThermalConductivity.Bcells(),)
212 {
213 EffectiveHeatCapacity(i,j,k) = 0.0;
214 EffectiveThermalConductivity(i,j,k) = 0.0;
215
216 for(size_t alpha = 0; alpha < Phase.Nphases; ++alpha)
217 if(Phase.Fractions(i,j,k,{alpha}) > 0.0)
218 {
219 EffectiveHeatCapacity(i,j,k)
220 += PhaseVolumetricHeatCapacity[alpha]*Phase.Fractions(i,j,k,{alpha});
221
222 EffectiveThermalConductivity(i,j,k)
223 += PhaseThermalConductivity[alpha]*Phase.Fractions(i,j,k,{alpha});
224 }
225 }
226 OMP_PARALLEL_STORAGE_LOOP_END
227}

229void HeatDiffusion::Perform1DIteration(Temperature& Tx,
230 Temperature1Dextension& TxExt,
231 BoundaryConditionTypes extBC,
232 double& residual, double dt)
233{
240 std::pair<int, int> Xbounds = three_state_bounds_selector(TxExt.Direction[0], 0, Tx.Grid.Nx);
241 std::pair<int, int> Ybounds = three_state_bounds_selector(TxExt.Direction[1], 0, Tx.Grid.Ny);
242 std::pair<int, int> Zbounds = three_state_bounds_selector(TxExt.Direction[2], 0, Tx.Grid.Nz);
243
244 double boundaryValue = 0.0;
245 double averageRhoCP = 0.0;
246 double averageLambda = 0.0;
247 double area = 0.0;
248
249 for (int i = Xbounds.first; i < Xbounds.second; ++i)
250 for (int j = Ybounds.first; j < Ybounds.second; ++j)
251 for (int k = Zbounds.first; k < Zbounds.second; ++k)
252 {
253 averageRhoCP += EffectiveHeatCapacity(i,j,k); // Average volumetric heat capacity [J/(m^3 K)]
254 averageLambda += EffectiveThermalConductivity(i,j,k); // Average thermal conductivity [J/(m s K)]
255 boundaryValue += Tx(i,j,k);
256 area++;
257 }
258
259#ifdef MPI_PARALLEL
260 if(TxExt.Direction[0] == 0)
261 {
262 OP_MPI_Allreduce(OP_MPI_IN_PLACE, &averageRhoCP, 1, OP_MPI_DOUBLE, OP_MPI_SUM, OP_MPI_COMM_WORLD);
263 OP_MPI_Allreduce(OP_MPI_IN_PLACE, &averageLambda, 1, OP_MPI_DOUBLE, OP_MPI_SUM, OP_MPI_COMM_WORLD);
264 OP_MPI_Allreduce(OP_MPI_IN_PLACE, &boundaryValue, 1, OP_MPI_DOUBLE, OP_MPI_SUM, OP_MPI_COMM_WORLD);
265 OP_MPI_Allreduce(OP_MPI_IN_PLACE, &area, 1, OP_MPI_DOUBLE, OP_MPI_SUM, OP_MPI_COMM_WORLD);
266 }
267#endif
268
269 averageRhoCP /= area;
270 averageLambda /= area;
271
272 TxExt.Data(-1) = boundaryValue/area;
273
274 double dt_dx2 = dt/(Grid.dx*Grid.dx);
275
276 for (size_t i = 0; i < TxExt.size(); ++i)
277 {
278 double locQ = 0.0;
279 if(i == TxExt.size()-1)
280 {
281 locQ = TxExt.Qdot;
282 }
283
284 double loc_deltaT = (averageRhoCP*TxExt.DataTMP(i) +
285 averageLambda*dt_dx2*(TxExt.Data(i+1) + TxExt.Data(i-1)) + locQ*dt)
286 /(averageRhoCP + 2.0*averageLambda*dt_dx2) - TxExt.Data(i);
287
288 residual = max(residual,loc_deltaT*loc_deltaT);
289 TxExt.Data(i) += loc_deltaT;
290 }
291
292 for (int i = Xbounds.first; i < Xbounds.second; ++i)
293 for (int j = Ybounds.first; j < Ybounds.second; ++j)
294 for (int k = Zbounds.first; k < Zbounds.second; ++k)
295 {
296 Tx(i+TxExt.Direction[0],j+TxExt.Direction[1],k+TxExt.Direction[2]) = TxExt.Data(0);
297 }
298
299 TxExt.setBC(extBC);
300}

302int HeatDiffusion::SolveImplicit(const PhaseField& Phase,
303 const BoundaryConditions& BC,
304 Temperature& Temp,
305 const double dt_in)
306{
307 int iteration = 0; // Iterations counter
308 double dt = dt_in;
309 if(SolverCallsCounter != 0) dt = SolverCallsInterval*dt_in;
310
311 if((SolverCallsCounter % SolverCallsInterval) == 0)
312 {
316 /* Populate "old" temperature for the iterations. */
317
318 if(Temp.ExtensionsActive)
319 {
320 if(Temp.ExtensionX0.isActive()) {Temp.ExtensionX0.store_temporary();}
321 if(Temp.ExtensionXN.isActive()) {Temp.ExtensionXN.store_temporary();}
322 if(Temp.ExtensionY0.isActive()) {Temp.ExtensionY0.store_temporary();}
323 if(Temp.ExtensionYN.isActive()) {Temp.ExtensionYN.store_temporary();}
324 if(Temp.ExtensionZ0.isActive()) {Temp.ExtensionZ0.store_temporary();}
325 if(Temp.ExtensionZN.isActive()) {Temp.ExtensionZN.store_temporary();}
326 }
327 Temp.CalculateMinMaxAvg();
328 TxOld = Temp.Tx;
329
330 double residual = 0.0; // Residual for calculating the accuracy of the converged solution
331 const double dt_dx2 = dt/(Grid.dx*Grid.dx);
332 const double dimension = 2.0*double(Grid.Active()); // Laplacian stencil dimension parameter
333
334 do
335 {
336 iteration++;
337 residual = 0.0;
338
339 if(Temp.ExtensionsActive)
340 {
341 /* Calculation of heat diffusion in 1D extension using GaussSeidel implicit method. */
342 #ifdef MPI_PARALLEL
343 if(MPI_RANK == 0)
344 #endif
345 if (Temp.ExtensionX0.isActive())
346 {
347 Perform1DIteration(Temp, Temp.ExtensionX0, BC.BC0X, residual, dt);
348 }
349 #ifdef MPI_PARALLEL
350 if(MPI_RANK == MPI_SIZE - 1)
351 #endif
352 if (Temp.ExtensionXN.isActive())
353 {
354 Perform1DIteration(Temp, Temp.ExtensionXN, BC.BCNX, residual, dt);
355 }
356 if (Temp.ExtensionY0.isActive())
357 {
358 Perform1DIteration(Temp, Temp.ExtensionY0, BC.BC0Y, residual, dt);
359 }
360 if (Temp.ExtensionYN.isActive())
361 {
362 Perform1DIteration(Temp, Temp.ExtensionYN, BC.BCNY, residual, dt);
363 }
364 if (Temp.ExtensionZ0.isActive())
365 {
366 Perform1DIteration(Temp, Temp.ExtensionZ0, BC.BC0Z, residual, dt);
367 }
368 if (Temp.ExtensionZN.isActive())
369 {
370 Perform1DIteration(Temp, Temp.ExtensionZN, BC.BCNZ, residual, dt);
371 }
372 }
373
374 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Temp.Tx,0,reduction(max:residual))
375 {
376 /* Calculation of heat diffusion using Jacobi implicit method. */
377
378 double RhoCp = EffectiveHeatCapacity(i,j,k); // Volumetric heat capacity [J/(m^3 K)]
379 double Lambda = EffectiveThermalConductivity(i,j,k); // Thermal conductivity [J/(m s K)]
380 double locQdot = Qdot(i,j,k); // Heat source [J/(m^3 s)]
381 double locStencil = 0.0; // Temperature stencil from the updated field [K]
382
383 if (Grid.dNx > 0) locStencil += (Temp(i+1,j,k) + Temp(i-1,j,k));
384 if (Grid.dNy > 0) locStencil += (Temp(i,j+1,k) + Temp(i,j-1,k));
385 if (Grid.dNz > 0) locStencil += (Temp(i,j,k+1) + Temp(i,j,k-1));
386
387 dTx(i,j,k) = (RhoCp*TxOld(i,j,k) + Lambda*locStencil*dt_dx2 + locQdot*dt)
388 /(RhoCp + dimension*Lambda*dt_dx2) - Temp(i,j,k);
389
390 residual = max(residual,dTx(i,j,k)*dTx(i,j,k));
391 }
392 OMP_PARALLEL_STORAGE_LOOP_END
393
394 /* Updating the temperature with the calculated increment.*/
395
396 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,Temp.Tx,0,)
397 {
398 Temp(i,j,k) += dTx(i,j,k);
399 dTx(i,j,k) = 0.0;
400 }
401 OMP_PARALLEL_STORAGE_LOOP_END
402
403 /* Assigning values for the boundary cells. */
404 Temp.SetBoundaryConditions(BC);
405
406 /* Warning output, if number of iterations exceeds MaxIterations. */
407
408 residual = sqrt(residual)/(Temp.Tavg + Tolerance); // Added Tolerance in denominator in case if Tavg == 0
409
410 #ifdef MPI_PARALLEL
411 OP_MPI_Allreduce(OP_MPI_IN_PLACE, &residual, 1, OP_MPI_DOUBLE, OP_MPI_MAX, OP_MPI_COMM_WORLD);
412 #endif
413
414 if(VerboseIterations)
415 {
416 std::string message = ConsoleOutput::GetStandard("Solver call",
417 std::to_string(SolverCallsCounter));
418 message += ConsoleOutput::GetStandard("Iteration",
419 std::to_string(iteration));
420 message += ConsoleOutput::GetStandard("Residual",
421 ConsoleOutput::to_string_with_precision(residual));
422 message += ConsoleOutput::GetStandard("Tolerance",
423 ConsoleOutput::to_string_with_precision(Tolerance));
424 ConsoleOutput::WriteWithinMethod(message, thisclassname, "SolveImplicit()");
425 }
426
427 if(iteration == MaxIterations)
428 {
429 stringstream value;
430 value.precision(16);
431 value << residual;
432
433 string tmp = "Maximum number of " + to_string(MaxIterations) + " iterations reached.\n"
434 + "Solver failed to converge to the requested tolerance of " + to_string(Tolerance) + ".\n"
435 + "Current convergence residual is " + value.str() + ".\n"
436 + "Simulation will continue!\n";
437
438 ConsoleOutput::WriteWarning(tmp,thisclassname,"SolveImplicit()");
439 break;
440 }
441 }
442 while (residual >= Tolerance);
443
444 /* Evaluate temperature statistics. */
445 Temp.CalculateMinMaxAvg();
446
447 if(Temp.ExtensionsActive)
448 {
449 if(Temp.ExtensionX0.isActive()) {Temp.ExtensionX0.Qdot = 0.0;}
450 if(Temp.ExtensionXN.isActive()) {Temp.ExtensionXN.Qdot = 0.0;}
451 if(Temp.ExtensionY0.isActive()) {Temp.ExtensionY0.Qdot = 0.0;}
452 if(Temp.ExtensionYN.isActive()) {Temp.ExtensionYN.Qdot = 0.0;}
453 if(Temp.ExtensionZ0.isActive()) {Temp.ExtensionZ0.Qdot = 0.0;}
454 if(Temp.ExtensionZN.isActive()) {Temp.ExtensionZN.Qdot = 0.0;}
455 }
456 Qdot.Clear();
457 }
458 SolverCallsCounter++;
459 return iteration;
460}
