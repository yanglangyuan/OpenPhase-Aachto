141 template <typename T>
142 void writeData(const std::string& directory, const std::string& definition, const T& parameter)
143 {
144 namespace fs = std::filesystem;
145
146 // Create directory if it doesn't exist
147 if (!fs::exists(directory))
148 {
149 if (!fs::create_directories(directory)) {
150 std::cerr << "Failed to create directory: " << directory << std::endl;
151 return;
152 }
153 }
154
155 std::string filePath = directory + "/simData.txt";
156 std::ofstream outFile(filePath, std::ios::app); // append mode
157
158 if (!outFile)
159 {
160 std::cerr << "Failed to open file for writing: " << filePath << std::endl;
161 return;
162 }
163
164 constexpr int columnWidth = 50; // Adjust as needed
165 outFile << std::left << std::setw(columnWidth) << (definition + ":") << parameter << std::endl;
166 outFile.close();
167 }
