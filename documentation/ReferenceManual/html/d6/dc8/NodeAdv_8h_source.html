92inline void NodeAdv::set_dXY(const size_t n, const double value)
93{
94 for (iterator i = Fields.begin(); i < Fields.end(); ++i)
95 if (i->index == n)
96 {
97 i->dXY = value;
98 return;
99 }
100
101 AdvEntry NewEntry;
102 NewEntry.index = n;
103 NewEntry.dXY = value;
104 NewEntry.dXZ = 0.0;
105 NewEntry.dYZ = 0.0;
106 NewEntry.dXYZ = 0.0;
107 Fields.push_back(NewEntry);
108}
