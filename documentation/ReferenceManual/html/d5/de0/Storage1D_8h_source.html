534 void Remesh(const long int nx)
535 {
536 T* tempData = new T [(nx + 2*Bcells)*Size_D] ();
537 Tensor<T, Rank>* tempTensors = new Tensor<T, Rank>[(nx + 2*Bcells)] ();
538
539 for(long int i = 0; i < (nx + 2*Bcells); i++)
540 {
541 tempTensors[i].Assign(&tempData[i],TensorDimensions);
542 }
543
544 double Xscale = double(Size_X)/double(nx);
545 for(long int x = 0; x < nx; x++)
546 {
547 long int x0 = floor((x - nx*0.5)*Xscale + Size_X * 0.5);
548 double dx = (x*Xscale - x0);
549
550 tempTensors[(x + Bcells)] =
551 locTensors[Index(x0)]*(1.0 - dx) +
552 locTensors[Index(x0+1)]*dx;
553 }
554 delete[] locTensors;
555 delete[] locData;
556
557 Size_X = nx;
558
559 Size_X_BC = Size_X + 2*Bcells;
560
561 locTensors = tempTensors;
562 locData = tempData;
563 }
