61void RunTimeControl::ReadInput(const string InputFileName)
62{
63 ConsoleOutput::WriteLineInsert("RunTimeControl input");
64 ConsoleOutput::WriteStandard("Source", InputFileName);
65
66 std::string filetype = FileInterface::getFileExtension(InputFileName);
67 if (filetype == "opi")
68 {
69 fstream inp(InputFileName.c_str(), ios::in | ios_base::binary);
70
71 if (!inp)
72 {
73 ConsoleOutput::WriteExit("File \"" + InputFileName + "\" could not be opened", thisclassname, "ReadInput()");
74 OP_Exit(EXIT_FAILURE);
75 };
76
77 std::stringstream data;
78 data << inp.rdbuf();
79 ReadInput(data);
80
81 inp.close();
82 }
83 else
84 if (filetype == "json")
85 {
86 ReadJSON(InputFileName);
87 }
88 else
89 {
90 std::cerr << "Filetype " << filetype << " not recognized. Filetype must be opi or json." << std::endl;
91 OP_Exit(1);
92 }
93}

95void RunTimeControl::ReadInput(stringstream& inp)
96{
97 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
98
99 SimulationTitle = FileInterface::ReadParameterF(inp, moduleLocation, {string("SimulationTitle"), string("SimTtl")});
100
101 UnitsOfLength = FileInterface::ReadParameterS(inp, moduleLocation, {string("UnitsOfLength"), string("LUnits")}, false, "m");
102 UnitsOfMass = FileInterface::ReadParameterS(inp, moduleLocation, {string("UnitsOfMass"), string("MUnits")}, false, "kg");
103 UnitsOfTime = FileInterface::ReadParameterS(inp, moduleLocation, {string("UnitsOfTime"), string("TUnits")}, false, "s");
104 UnitsOfEnergy = FileInterface::ReadParameterS(inp, moduleLocation, {string("UnitsOfEnergy"), string("EUnits")}, false, "J");
105
106#ifdef _OPENMP
107 OpenMPThreads = FileInterface::ReadParameterI(inp, moduleLocation, {string("OpenMPThreads"), string("nOMP")});
108 omp_set_num_threads(OpenMPThreads);
109#endif
110
111 MaxTimeStep = FileInterface::ReadParameterI(inp, moduleLocation, {string("MaxTimeStep"), string("nSteps")});
112 dt = FileInterface::ReadParameterD(inp, moduleLocation, string("dt"));
113
114 RestartSwitch = FileInterface::ReadParameterB(inp, moduleLocation, {string("RestartSwitch"), string("Restrt")}, false, false);
115 if(RestartSwitch)
116 {
117 StartTimeStep = FileInterface::ReadParameterI(inp, moduleLocation, {string("StartTimeStep"), string("RestartTimeStep"), string("tStart")});
118 }
119 else
120 {
121 StartTimeStep = 0;
122 }
123 CheckpointInterval = FileInterface::ReadParameterI(inp, moduleLocation, {string("CheckpointInterval"), string("tRstrt")});
124 VTKOutputInterval = FileInterface::ReadParameterI(inp, moduleLocation, {string("VTKOutputInterval"), string("FTime")});
125 ConsoleOutputInterval = FileInterface::ReadParameterI(inp, moduleLocation, {string("ConsoleOutputInterval"), string("STime")});
126 StopCheckInterval = FileInterface::ReadParameterD(inp, moduleLocation, string("StopCheckInterval"), false, 0.0);
127
128 LogModeScreen = FileInterface::ReadParameterB(inp, moduleLocation, string("LogScreen"), false, false);
129 LogModeVTK = FileInterface::ReadParameterB(inp, moduleLocation, string("LogVTK"), false, false);
130 LogScreenFactor = FileInterface::ReadParameterI(inp, moduleLocation, string("LogScreenF"), false, 1);
131 LogVTKFactor = FileInterface::ReadParameterI(inp, moduleLocation, string("LogVTKF"), false, 1);
132
133 ConsoleOutput::WriteLine();
134 ConsoleOutput::WriteLine();
135
136 if (VTKOutputInterval < 1)
137 {
138 ConsoleOutput::WriteSimple(string("Illegal value for VTKOutputInterval => ") + to_string(VTKOutputInterval) + string(" ! VTKOutputInterval is set to 1"));
139 VTKOutputInterval = 1;
140 }
141 if (ConsoleOutputInterval < 1)
142 {
143 ConsoleOutput::WriteSimple(string("Illegal value for ConsoleOutputInterval => ") + to_string(ConsoleOutputInterval) + string(" ! ConsoleOutputInterval is set to 1"));
144 ConsoleOutputInterval = 1;
145 }
146
147 if (std::filesystem::create_directories(VTKDir))
148 {
149 ConsoleOutput::WriteStandard("Created directory", VTKDir);
150 }
151
152 if (std::filesystem::create_directories(RawDataDir))
153 {
154 ConsoleOutput::WriteStandard("Created directory", RawDataDir);
155 }
156
157 if (std::filesystem::create_directories(TextDir))
158 {
159 ConsoleOutput::WriteStandard("Created directory", TextDir);
160 }
161
162 ConsoleOutput::WriteLine();
163 ConsoleOutput::WriteBlankLine();
164}

166void RunTimeControl::ReadJSON(const string InputFileName)
167{
168
169 std::ifstream f(InputFileName);
170 json data = json::parse(f);
171 if (data.contains(thisclassname))
172 {
173 json RTC = data[thisclassname];
174
175 SimulationTitle = FileInterface::ReadParameter<std::string>(RTC, {"SimulationTitle"}, "NoTitle");
176
177 UnitsOfLength = FileInterface::ReadParameter<std::string>(RTC, {"UnitsOfLength"}, "m");
178 UnitsOfMass = FileInterface::ReadParameter<std::string>(RTC, {"UnitsOfMass"}, "kg");
179 UnitsOfTime = FileInterface::ReadParameter<std::string>(RTC, {"UnitsOfTime"}, "s");
180 UnitsOfEnergy = FileInterface::ReadParameter<std::string>(RTC, {"UnitsOfTime"}, "J");
181
182 #ifdef _OPENMP
183 OpenMPThreads = FileInterface::ReadParameter<int>(RTC, {"OpenMPThreads"}, 1);
184 omp_set_num_threads(OpenMPThreads);
185 #endif
186
187 MaxTimeStep = FileInterface::ReadParameter<int>(RTC, {"MaxTimeStep"});
188 dt = FileInterface::ReadParameter<double>(RTC, {"dt"});
189
190 RestartSwitch = FileInterface::ReadParameter<bool>(RTC, {"RestartSwitch"}, false);
191 if(RestartSwitch)
192 {
193 StartTimeStep = FileInterface::ReadParameter<int>(RTC, {"StartTimeStep"}, 0);
194 }
195 else
196 {
197 StartTimeStep = 0;
198 }
199 CheckpointInterval = FileInterface::ReadParameter<int>(RTC, {"CheckpointInterval"}, 1);
200 VTKOutputInterval = FileInterface::ReadParameter<int>(RTC, {"VTKOutputInterval"}, 1);
201 ConsoleOutputInterval = FileInterface::ReadParameter<int>(RTC, {"ConsoleOutputInterval"}, 1);
202 StopCheckInterval = FileInterface::ReadParameter<double>(RTC, {"StopCheckInterval"}, 0.0);
203
204 LogModeScreen = FileInterface::ReadParameter<bool>(RTC, {"LogScreen"}, false);
205 LogModeVTK = FileInterface::ReadParameter<bool>(RTC, {"LogVTK"}, false);
206 LogScreenFactor = FileInterface::ReadParameter<double>(RTC, {"LogScreenF"}, 1);
207 LogVTKFactor = FileInterface::ReadParameter<double>(RTC, {"LogVTKF"}, 1);
208
209 ConsoleOutput::WriteLine();
210 ConsoleOutput::WriteLine();
211
212 if (VTKOutputInterval < 1)
213 {
214 ConsoleOutput::WriteSimple(string("Illegal value for VTKOutputInterval => ") + to_string(VTKOutputInterval) + string(" ! VTKOutputInterval is set to 1"));
215 VTKOutputInterval = 1;
216 }
217 if (ConsoleOutputInterval < 1)
218 {
219 ConsoleOutput::WriteSimple(string("Illegal value for ConsoleOutputInterval => ") + to_string(ConsoleOutputInterval) + string(" ! ConsoleOutputInterval is set to 1"));
220 ConsoleOutputInterval = 1;
221 }
222
223 if (std::filesystem::create_directories(VTKDir))
224 {
225 ConsoleOutput::WriteStandard("Created directory", VTKDir);
226 }
227
228 if (std::filesystem::create_directories(RawDataDir))
229 {
230 ConsoleOutput::WriteStandard("Created directory", RawDataDir);
231 }
232
233 if (std::filesystem::create_directories(TextDir))
234 {
235 ConsoleOutput::WriteStandard("Created directory", TextDir);
236 }
237 }
238 ConsoleOutput::WriteLine();
239 ConsoleOutput::WriteBlankLine();
240}

286bool RunTimeControl::RestartPossible()
287{
288 std::ifstream inp(RawDataDir + thisobjectname + ".dat");
289 if (inp)
290 {
291 inp >> StartTimeStep;
292 RestartSwitch = true;
293 }
294 else
295 {
296 RestartSwitch = false;
297 }
298 return RestartSwitch;
