28 * OPObject is a parent virtual class for most of the stand-alone classes in
29 * the OpenPhase. It defines the names of common methods and parameters which
30 * unifies their syntax across the library. The common base also allows creating
31 * lists of different objects and allows batch execution of common methods,
32 * e.g. Remesh() and Advect() on a list of different objects.
33 */

130 static OPObject* FindObject(std::vector<OPObject*> ObjectList,
131 std::string ObjectName, std::string thisclassname,
132 std::string thisfunctionname, bool necessary, bool verbose = true)
133 /* Returns a reference to the first OPObject in the list whose class name
134 * starts with ObjectName. I.e.: if ObjectName is "Elasticity", the first
135 * object in the list which has either the name ElasticitySteinbach or
136 * ElasticityKhachaturyan will be returned.
137 */
138 {
139 for(unsigned int i = 0; i < ObjectList.size(); i++)
140 {
141 if(ObjectList[i]->thisclassname.size() >= ObjectName.size())
142 {
143 std::string locObjectName = ObjectList[i]->thisclassname;
144 locObjectName.resize(ObjectName.size());
145
146 if(locObjectName == ObjectName)
147 {
148 return ObjectList[i];
149 }
150 }
151 }
152 if (verbose)
153 {
154 std::cout << "No " << ObjectName << " object found! " << thisclassname
155 << "." << thisfunctionname << "()" << std::endl;
156 }
157
158 if(necessary)
159 {
160 OP_Exit(13);
161 }
162
163 return nullptr;
164 }
165
166 static std::vector<OPObject*> FindObjects(std::vector<OPObject*> ObjectList,
167 std::string ObjectName, std::string thisclassname,
168 std::string thisfunctionname, bool necessary, bool verbose = true)
169 /* Returns a list of OPObject references found in the passed ObjectList
170 * whose class name start with ObjectName. I.e.: if ObjectName is
171 * "Elasticity", all objects in the ObjectList which have a name starting
172 * with "Elasticity", e.g. ElasticitySteinbach or ElasticityKhachaturyan
173 * will be returned in the result list.
174 */
175 {
176 std::vector<OPObject*> result;
177
178 for(unsigned int i = 0; i < ObjectList.size(); i++)
179 {
180 if(ObjectList[i]->thisclassname.size() >= ObjectName.size())
181 {
182 std::string locObjectName = ObjectList[i]->thisclassname;
183 locObjectName.resize(ObjectName.size());
184
185 if(locObjectName == ObjectName)
186 {
187 result.push_back(ObjectList[i]);
188 }
189 }
190 }
191
192 if(result.size() == 0)
193 {
194 if (verbose)
195 {
196 std::cout << "No " << ObjectName << " objects found! " << thisclassname
197 << thisfunctionname << std::endl;
198 }
199 if(necessary)
200 {
201 OP_Exit(13);
202 }
203 }
204 return result;
205 }
