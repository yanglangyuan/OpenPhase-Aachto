175 Matrix<T> operator*(const Matrix<T>& rhs) const
176 {
177 assert(Size_N == rhs.sizeM() && "Matrices have incompatible dimensions!");
178 assert(Size_M == rhs.sizeN() && "Matrices have incompatible dimensions!");
179
180 size_t sizeN = rhs.sizeM();
181 size_t sizeM = Size_N;

183 Matrix<T> Out(sizeN, sizeM);
184
185 for (size_t n = 0; n < sizeN; n++)
186 for (size_t m = 0; m < sizeM; m++)
187 for (size_t p = 0; p < sizeM; p++)
188 {
189 Out(n,m) += get(n,p)*rhs.get(p,m);
190 }
