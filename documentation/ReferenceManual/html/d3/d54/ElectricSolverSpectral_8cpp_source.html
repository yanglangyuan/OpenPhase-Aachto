42void ElectricSolverSpectral::Initialize(Settings& locSettings, std::string ObjectNameSuffix)
43{
44 thisclassname = "ElectricSolverSpectral";
45 thisobjectname = thisclassname + ObjectNameSuffix;
46
47 Grid = locSettings.Grid;
48 Nz2 = locSettings.Grid.Nz/2+1;
49 rlSize = Grid.Nx*Grid.Ny*Grid.Nz;
50 ftSize = Grid.Nx*Grid.Ny*Nz2;
51 freq = new fftw_complex [ftSize];
52 rhs = new double [rlSize];
53
54 ForwardPlan = fftw_plan_dft_r2c_3d(Grid.Nx, Grid.Ny, Grid.Nz, rhs, freq, FFTW_ESTIMATE);
55 BackwardPlan = fftw_plan_dft_c2r_3d(Grid.Nx, Grid.Ny, Grid.Nz, freq, rhs, FFTW_ESTIMATE);
56
57 initialized = true;
58 ConsoleOutput::WriteStandard(thisclassname, "Initialized");
59}

95void ElectricSolverSpectral::Solve(ElectricProperties& EP, BoundaryConditions& BC)
96{
97 const double Lx = Grid.TotalNx*Grid.dx;
98 const double Ly = Grid.TotalNy*Grid.dx;
99 const double Lz = Grid.TotalNz*Grid.dx;
100
101 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,EP.ChargeDensity,0,)
102 {
103 rhs[rlidx(i,j,k)] = - EP.ChargeDensity(i,j,k)/PhysicalConstants::epsilon_0;
104 }
105 OMP_PARALLEL_STORAGE_LOOP_END
106
107 fftw_execute(ForwardPlan);
108
109 for (long int i = 0; i < Grid.Nx; i++)
110 for (long int j = 0; j < Grid.Ny; j++)
111 for (long int k = 0; k < Nz2; k++)
112 {
113 const long int ii = Grid.OffsetX + i;
114 const long int jj = Grid.OffsetY + j;
115 const long int kk = Grid.OffsetZ + k;
116
117 assert(ii < Grid.TotalNx);
118 assert(jj < Grid.TotalNy);
119 assert(kk < Grid.TotalNz);
120
121 const double ni = (ii <= Grid.TotalNx/2) ? ii : ii - Grid.TotalNx;
122 const double nj = (jj <= Grid.TotalNy/2) ? jj : jj - Grid.TotalNy;
123 const double nk = kk;
124
125 const double Qx = 2.0*Pi*ni/Lx;
126 const double Qy = 2.0*Pi*nj/Ly;
127 const double Qz = 2.0*Pi*nk/Lz;
128
129 const double QQ = Qx*Qx + Qy*Qy + Qz*Qz;
130
131 const size_t XYZ = k + Nz2*(j + Grid.Ny*i);
132 assert( XYZ < ftSize );
133 if (QQ == 0.0)
134 {
135 freq[XYZ][0] = 0.0;
136 freq[XYZ][1] = 0.0;
137 }
138 else
139 {
140 freq[XYZ][0] /= -QQ;
141 freq[XYZ][1] /= -QQ;
142 }
143 }
144
145 fftw_execute(BackwardPlan);
146
147 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,EP.ElectricPotential,0,)
148 {
149 EP.ElectricPotential(i,j,k) = rhs[rlidx(i,j,k)]/Grid.LocalNumberOfCells();
150 }
151 OMP_PARALLEL_STORAGE_LOOP_END
152
153 BC.SetX(EP.ElectricPotential);
154 BC.SetY(EP.ElectricPotential);
155 BC.SetZ(EP.ElectricPotential);
156}
