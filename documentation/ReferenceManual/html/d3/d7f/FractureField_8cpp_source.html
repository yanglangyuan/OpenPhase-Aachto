360void FractureField::CreatePF(PhaseField& Phase, size_t PhaseIndex, BoundaryConditions& BC)
361{
362 size_t index = Phase.AddGrainInfo(PhaseIndex);
363 STORAGE_LOOP_BEGIN(i, j, k, Fields, Fields.Bcells())
364 {
365 if(Fields(i, j, k) > PFcutOff)
366 {
367 Phase.Fields(i,j,k).clear();
368 Phase.Fields(i, j, k).set_value(index, 1.0);
369 }
370 }
371 STORAGE_LOOP_END
372 STORAGE_LOOP_BEGIN(i,j,k,Phase.Fields,0)
373 if(Phase.Fields(i,j,k).size() == 1)
374 {
375 size_t alpha_index = Phase.Fields(i,j,k).begin()->index;
376
377 for(int di = -1; di <= 1; di++)
378 for(int dj = -1; dj <= 1; dj++)
379 for(int dk = -1; dk <= 1; dk++)
380 if(i+di >= 0 and i+di < Grid.Nx and
381 j+dj >= 0 and j+dj < Grid.Ny and
382 k+dk >= 0 and k+dk < Grid.Nz and
383 (di != 0 or dj != 0 or dk != 0))
384 {
385 bool grain_not_present = true;
386 for(auto beta = Phase.Fields(i+di,j+dj,k+dk).begin();
387 beta != Phase.Fields(i+di,j+dj,k+dk).end(); ++beta)
388 if(beta->index == alpha_index)
389 {
390 grain_not_present = false;
391 break;
392 }
393 else
394 {
395 break;
396 }
397 if(grain_not_present)
398 {
399 for(auto beta = Phase.Fields(i+di,j+dj,k+dk).begin();
400 beta != Phase.Fields(i+di,j+dj,k+dk).end(); ++beta)
401 {
402 Phase.Fields(i,j,k).add_value(beta->index, 1.0);
403 }
404 Phase.Fields(i+di,j+dj,k+dk).add_value(alpha_index, 1.0);
405 }
406 }
407 }
408 STORAGE_LOOP_END
409 Phase.FinalizeInitialization(BC);
410}

412void FractureField::CreateCrack(dVector3& CrackStart, dVector3& CrackEnd, Settings& OP, BoundaryConditions& BC)
413{
414 double CrackLine[3] = {-CrackEnd[1] + CrackStart[1],
415 CrackEnd[0] - CrackStart[0],
416 (CrackEnd[1] - CrackStart[1]) * CrackStart[0] -
417 (CrackEnd[0] - CrackStart[0]) * CrackStart[1]};
418
419 double CrackLength =
420 sqrt((CrackEnd[1] - CrackStart[1]) * (CrackEnd[1] - CrackStart[1]) +
421 (CrackEnd[0] - CrackStart[0]) * (CrackEnd[0] - CrackStart[0]));
422
423 STORAGE_LOOP_BEGIN(i, j, k, Fields, Fields.Bcells())
424 {
425 Fields(i, j, k) = 0;
426 Flag(i, j, k) = 2;
427 double Distance = sqrt((CrackLine[0] * i + CrackLine[1] * j + CrackLine[2]) *
428 (CrackLine[0] * i + CrackLine[1] * j + CrackLine[2]) /
429 (CrackLine[0] * CrackLine[0] + CrackLine[1] * CrackLine[1]));
430
431 if (Distance < (sWidth * 1))
432 {
433 double DistanceStart = (i - CrackStart[0]) * (i - CrackStart[0]) + (j - CrackStart[1]) * (j - CrackStart[1]);
434 double DistanceEnd = (i - CrackEnd[0]) * (i - CrackEnd[0]) + (j - CrackEnd[1]) * (j - CrackEnd[1]);
435 double DistanceSq = DistanceStart + DistanceEnd;
436
437 if (DistanceSq < 2 * Distance * Distance + CrackLength * CrackLength)
438 {
439 Fields(i, j, k) = pow((1 - Distance / (sWidth * 1)), 2);
440 }
441 else if (DistanceStart < DistanceEnd and sqrt(DistanceStart) < (sWidth * 1))
442 {
443 Fields(i, j, k) = pow((1 - sqrt(DistanceStart) / (sWidth * 1)), 2);
444 }
445 else if (DistanceStart > DistanceEnd and sqrt(DistanceEnd) < (sWidth * 1))
446 {
447 Fields(i, j, k) = pow((1 - sqrt(DistanceEnd) / (sWidth * 1)), 2);
448 }
449 }
450 }
451 STORAGE_LOOP_END
452 Finalize(BC);
453 CorrectFractureProfile(BC, 1e-8, 1000);
454}

456void FractureField::CorrectFractureProfile( BoundaryConditions& BC, double locMobility, size_t nSteps)
457{
458 auto CalculateFielddot = [&](int i, int j, int k) -> void
459 {
460 double Eta = sWidth * Grid.dx;
461 switch(FractureModel)
462 {
463 case FractureModels::Well:
464 {
465 if (Flag(i, j, k) == 2)
466 {
467 Fields_dot(i, j, k) =
468 locMobility * ((0.708 * Laplacian(i, j, k) * Eta - 0.5 * (Fields(i, j, k) + 0.5) / Eta));
469 }
470 break;
471 }
472 case FractureModels::Obstacle:
473 {
474 double epsilon = 3 / 16.0 * Eta;
475 double K = 9 / 64.0;
476 if (Flag(i, j, k) == 2)
477 {
478 Fields_dot(i, j, k) = locMobility * ((2 * epsilon * Laplacian(i, j, k) - K / epsilon));
479 }
480 break;
481 }
482 default:
483 {
484 break;
485 }
486 }
487 };
488 for (size_t i = 0; i < nSteps; ++i)
489 {
490 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i, j, k, Fields, 0, )
491 {
492 CalculateFielddot(i, j, k);
493 }
494 OMP_PARALLEL_STORAGE_LOOP_END
495 MergeIncrements(BC, 0.01);
496 }

499void FractureField::SetSurfaceEnergy(Composition& Cx, Temperature& Tx)
500{
501 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i, j, k, Fields, Fields.Bcells(), )
502 {
503 double expx = exp(-delta_g_0 / (PhysicalConstants::R * Tx(i, j, k)));
504 double theta = Cx.MoleFractions(i, j, k, {Comp}) / (expx + Cx.MoleFractions(i, j, k, {Comp}));
505 SurfaceEnergy(i, j, k) = InitialSurfaceEnergy * (1.0 - (Xi * theta));
506 }
507 OMP_PARALLEL_STORAGE_LOOP_END
508}

539void FractureField::CalculateIncrementsWell(const PhaseField& Phase, InterfaceProperties& IP, ElasticProperties& EP, DoubleObstacle& DO)
540{
541 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i, j, k, Fields, 0, )
542 {
543 if (Flag(i, j, k))
544 {
545 Fields_dot(i, j, k) =
546 Mobility *
547 (SurfaceEnergy(i, j, k) * (0.708 * Laplacian(i, j, k) * Eta - 0.5 * (Fields(i, j, k) + 0.5) / Eta) +
548 6.0 * Fields(i, j, k) * (1.0 - Fields(i, j, k)) * DO.PointEnergy(Phase, IP, i, j, k));
549 }
550
551 if (Flag(i, j, k) == 2 and Fields(i, j, k) != 1.0)
552 {
553 Fields_dot(i, j, k) += 2.0 * Mobility * EP.EnergyDensity(Phase,i, j, k) / (1.0 - Fields(i, j, k));
554 }
555 }
556 OMP_PARALLEL_STORAGE_LOOP_END
557}

635void FractureField::FixAdvectionFractureProfile(ElasticProperties &EP)
636{
637 auto maxDisplacement = [&](int i, int j, int k)
638 {
639 double Displacement = 0.0;
640 for (int ii = -Grid.dNx; ii <= +Grid.dNx; ii += 2)
641 for (int jj = -Grid.dNy; jj <= +Grid.dNy; jj += 2)
642 for (int kk = -Grid.dNz; kk <= +Grid.dNz; kk += 2)
643 {
644 Displacement = max(EP.Displacements(i + ii, j + jj, k + kk).abs(), Displacement);
645 }
646 return Displacement;
647 };
648 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i, j, k, Fields, 0, )
649 {
650 if (Flag(i, j, k))
651 {
652 if (Fields(i, j, k) > 0.96)
653 Fields(i, j, k) = 1;
654 if (Fields(i, j, k) > 0.5)
655 {
656 Fields(i, j, k) += maxDisplacement(i, j, k);
657 }
658 }
659 }
660 OMP_PARALLEL_STORAGE_LOOP_END

1079{
