34BoundaryConditionTypes BoundaryConditions::TranslateBoundaryConditions(string Key)
35{
36 BoundaryConditionTypes myBC;
37 bool check = false;
38 if(Key == "PERIODIC")
39 {
40 myBC = BoundaryConditionTypes::Periodic;
41 check = true;
42 }
43 if(Key == "NOFLUX")
44 {
45 myBC = BoundaryConditionTypes::NoFlux;
46 check = true;
47 }
48 if(Key == "FREE")
49 {
50 myBC = BoundaryConditionTypes::Free;
51 check = true;
52 }
53 if(Key == "FIXED")
54 {
55 myBC = BoundaryConditionTypes::Fixed;
56 check = true;
57 }
58 if(Key == "MIRROR")
59 {
60 myBC = BoundaryConditionTypes::Mirror;
61 check = true;
62 }
63
64 if (!check)
65 {
66 ConsoleOutput::WriteExit(string("Illegal Boundary Conditions Input. ") +
67 string("Legal boundary conditions are ") +
68 string("Periodic, NoFlux, Free, Fixed or Mirror."),
69 thisclassname, "ReadInput()");
70 OP_Exit(EXIT_FAILURE);
71 }
72 return myBC;
73}

97void BoundaryConditions::ReadInput(const string InputFileName)
98{
99 ConsoleOutput::WriteLineInsert("BoundaryConditions input");
100 ConsoleOutput::WriteStandard("Source", InputFileName);
101
102 std::string filetype = FileInterface::getFileExtension(InputFileName);
103 if (filetype == "opi")
104 {
105 fstream inp(InputFileName.c_str(), ios::in | ios_base::binary);
106
107 if (!inp)
108 {
109 ConsoleOutput::WriteExit("File \"" + InputFileName + "\" could not be opened", thisclassname, "ReadInput()");
110 OP_Exit(EXIT_FAILURE);
111 };
112
113 std::stringstream data;
114 data << inp.rdbuf();
115 ReadInput(data);
116
117 inp.close();
118 }
119 else
120 if (filetype == "json")
121 {
122 ReadJSON(InputFileName);
123 }
124 else
125 {
126 std::cerr << "Filetype " << filetype << " not recognized. Filetype must be opi or json." << std::endl;
127 OP_Exit(1);
128 }
129}

131void BoundaryConditions::ReadInput(stringstream& inp)
132{
133 int moduleLocation = FileInterface::FindModuleLocation(inp, thisclassname);
134
135 string BC0Xstring = FileInterface::ReadParameterK(inp, moduleLocation, string("BC0X"));
136 string BCNXstring = FileInterface::ReadParameterK(inp, moduleLocation, string("BCNX"));
137 string BC0Ystring = FileInterface::ReadParameterK(inp, moduleLocation, string("BC0Y"));
138 string BCNYstring = FileInterface::ReadParameterK(inp, moduleLocation, string("BCNY"));
139 string BC0Zstring = FileInterface::ReadParameterK(inp, moduleLocation, string("BC0Z"));
140 string BCNZstring = FileInterface::ReadParameterK(inp, moduleLocation, string("BCNZ"));
141
142 BC0X = TranslateBoundaryConditions(BC0Xstring);
143 BCNX = TranslateBoundaryConditions(BCNXstring);
144 BC0Y = TranslateBoundaryConditions(BC0Ystring);
145 BCNY = TranslateBoundaryConditions(BCNYstring);
146 BC0Z = TranslateBoundaryConditions(BC0Zstring);
147 BCNZ = TranslateBoundaryConditions(BCNZstring);
148
149#ifdef MPI_PARALLEL
150 if(MPI_3D_DECOMPOSITION)
151 {
152 Setup_MPIX();
153 Setup_MPIY();
154 Setup_MPIZ();
155 }
156 else
157 {
158 Setup_MPI();
159 }
160#endif
161 ConsoleOutput::WriteLine();
162 ConsoleOutput::WriteBlankLine();
163}

166{

168 json data = json::parse(f);
169 if (data.contains(thisclassname))
170 {
171 json bc = data[thisclassname];

211bool BoundaryConditions::Setup_MPI()
212{
213 if(BC0X == BoundaryConditionTypes::Periodic or
214 BCNX == BoundaryConditionTypes::Periodic)
215 {
216 MPIperiodicX = true;
217 BC0X = BoundaryConditionTypes::MPIcomm;
218 BCNX = BoundaryConditionTypes::MPIcomm;
219 return true;
220 }
221 if (MPI_RANK > 0) BC0X = BoundaryConditionTypes::MPIcomm;
222 if (MPI_RANK < MPI_SIZE - 1) BCNX = BoundaryConditionTypes::MPIcomm;
223 MPIperiodicX = false;
224 return false;
225}
