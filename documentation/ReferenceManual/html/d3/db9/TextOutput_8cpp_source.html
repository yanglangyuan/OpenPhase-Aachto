121void TextOutput::AveragePlasticStrain(ElasticProperties& EP, string filename,
122 double timeOrStrain, size_t precision)
123{
130 vStrain avgStrain = EP.AveragePlasticStrain();
131
132 static vStrain OldStrain;
133 static double OldTimeStep = 0.0;
134 vStrain Rate;
135 if (timeOrStrain)
136 {
137 Rate = (avgStrain - OldStrain)/(timeOrStrain - OldTimeStep);
138 }
139 OldStrain = avgStrain;
140 OldTimeStep = timeOrStrain;
141
142 double averagePEEQ = 0.0;
143 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i,j,k,EP.DeformationGradientsTotal,0,reduction(+:averagePEEQ))
144 {
145 averagePEEQ += EP.PlasticStrains(i,j,k).PEEQ();
146 }
147 OMP_PARALLEL_STORAGE_LOOP_END
148#ifdef MPI_PARALLEL
149 double tempaverage = averagePEEQ ;
150 OP_MPI_Reduce(&tempaverage,&averagePEEQ ,1, OP_MPI_DOUBLE, OP_MPI_SUM,0, OP_MPI_COMM_WORLD);
151 if(MPI_RANK == 0)
152#endif
153 {
154 averagePEEQ /= double(EP.Grid.TotalNumberOfCells());
155 if (!std::filesystem::exists(filename))
156 {
157 ofstream file(filename, ios::out);
158 file << "time"
159 << separator << "Epsilon_0"
160 << separator << "Epsilon_1"
161 << separator << "Epsilon_2"
162 << separator << "Epsilon_3"
163 << separator << "Epsilon_4"
164 << separator << "Epsilon_5"
165 << separator << "PEEQ"
166 << separator << "Rate_0"
167 << separator << "Rate_1"
168 << separator << "Rate_2"
169 << separator << "Rate_3"
170 << separator << "Rate_4"
171 << separator << "Rate_5\n";
172 file.close();
173 }
174
175 ofstream file(filename, ios::app);
176 file.precision(precision);
177 file << timeOrStrain << scientific
178 << separator << avgStrain[0]
179 << separator << avgStrain[1]
180 << separator << avgStrain[2]
181 << separator << avgStrain[3]
182 << separator << avgStrain[4]

194void TextOutput::maxElasticRotation(ElasticProperties& EP, PhaseField& Phase,Orientations& OR, int tStep, string Filename)
195{
196 double maxRotation = 0.0;
197 OMP_PARALLEL_STORAGE_LOOP_BEGIN(i, j, k, Phase.Fields,0, reduction(max:maxRotation))
198 {
199 double locangle = 0;
200 dVector3 locaxis;
201 Tools::getAxisAngle(EP.DeformationGradientsTotal(i, j, k), locaxis, locangle);
202 locangle *= 180.0 / Pi;
203 maxRotation = std::max(maxRotation, locangle);
204 }
205 OMP_PARALLEL_STORAGE_LOOP_END
206#ifdef MPI_PARALLEL
207 double tempmaxRotation = maxRotation;
208 OP_MPI_Allreduce(&tempmaxRotation,&maxRotation, 1, OP_MPI_DOUBLE, OP_MPI_MAX, OP_MPI_COMM_WORLD);
209#endif
210 WriteValue(maxRotation, Filename, tStep);
211}

448NodeAB<double,double> TextOutput::IntegrateDG(PhaseField& Phase, DrivingForce& dG, std::string filename, double RealTime)
449{
450 NodeAB<double,double> dg;
451 NodeAB<double,double> num_point;
452 STORAGE_LOOP_BEGIN(i, j, k, Phase.Fields, 0)
453 {
454 if (Phase.Fields(i,j,k).interface())
455 {
456 for(auto it = dG.Force(i,j,k).begin();
457 it != dG.Force(i,j,k).end(); ++it)
458 {
459 size_t indexA = it->indexA;
460 size_t indexB = it->indexB;
461
462 size_t pIndexA = Phase.FieldsProperties[indexA].Phase;
463 size_t pIndexB = Phase.FieldsProperties[indexB].Phase;
464 double phase = 1.; //sqrt(Phase.Fields(i,j,k).get_value(indexA)*Phase.Fields(i,j,k).get_value(indexB));
465 dg.add_asym1(pIndexA, pIndexB, phase*it->raw);
466 num_point.add_sym1(pIndexA, pIndexB,phase);
467 }
468 }
469 }
470 STORAGE_LOOP_END
471 #ifdef MPI_PARALLEL
472 for (size_t i = 0; i < Phase.Nphases; ++i)
473 for (size_t j = 0; j < i; ++j)
474 {
475 double value = dg.get_asym1(i,j);
476 double rvalue = 0;
477 OP_MPI_Allreduce(&value,&rvalue, 1, OP_MPI_DOUBLE, OP_MPI_SUM, OP_MPI_COMM_WORLD);
478 dg.set_asym1(i,j,rvalue);
479 double numpoints = num_point.get_sym1(i,j);
480 double rnumpoints = 0;
481 OP_MPI_Allreduce(&numpoints,&rnumpoints, 1, OP_MPI_DOUBLE, OP_MPI_SUM, OP_MPI_COMM_WORLD);
482 num_point.set_asym1(i,j,rnumpoints);
483 }
484 #endif
485 for (size_t i = 0; i < Phase.Nphases; ++i)
486 for (size_t j = 0; j < i; ++j)
487 {
488 double value = dg.get_asym1(i,j);
489 double numpoints = std::abs(num_point.get_sym1(i,j));
490 if (numpoints > 0.)
491 dg.set_asym1(i,j,value/numpoints);
492 else
493 dg.set_asym1(i,j,0.);
494 }
495 #ifdef MPI_PARALLEL
496 if(MPI_RANK == 0)
497 #endif
498 {
499 if (!std::filesystem::exists(filename) or RealTime == 0.)
500 {
501 ofstream file(filename, ios::out | ios::trunc);
502 file << "Time";
503 for (size_t i = 0; i < Phase.Nphases; ++i)
504 for (size_t j = i+1; j < Phase.Nphases; ++j)
505 {
506 file << separator << "dG_" << i << "_" << j;
507 }
508 file << "\n";
509 file.close();
510 }
511
512 ofstream file(filename, ios::app);
513 file << RealTime;
514 for (size_t i = 0; i < Phase.Nphases; ++i)
515 for (size_t j = i+1; j < Phase.Nphases; ++j)
516 {
517 file << separator << dg.get_asym1(i,j);
518 }
519 file << "\n";
520 file.close();
521 }
522 return dg;
523}
