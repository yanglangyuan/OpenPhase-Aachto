165 void ReadInputFaceted(std::stringstream& inp, int moduleLocation, std::string counter)
166 {
167 bool FacetFamilies = true;
168 Nfacets = FileInterface::ReadParameterI(inp, moduleLocation, std::string("NfacetFamilies") + counter, false, 0);
169 if(Nfacets == 0)
170 {
171 Nfacets = FileInterface::ReadParameterI(inp, moduleLocation, std::string("Nfacets") + counter);
172 FacetFamilies = false;
173 }
174
175 FacetVectors.resize(Nfacets);
176 FacetMobility.resize(Nfacets,0.0);
177 FacetEpsilon.resize(Nfacets,0.0);
178
179 for(size_t n = 0 ; n < Nfacets; n++)
180 {
181 std::stringstream converter;
182 converter << "_" << n;
183 std::string counter1 = converter.str();
184
185 iVector3 locFacet = FileInterface::ReadParameterV3I(inp, moduleLocation, std::string("Facet") + counter + counter1, true);
186
187 if(FacetFamilies)
188 {
189 FacetVectors[n] = Tools::findIndexPermutations(locFacet);
190 }
191 else
192 {
193 FacetVectors[n].push_back(locFacet);
194 }
195
196 FacetMobility[n] = FileInterface::ReadParameterD(inp, moduleLocation, std::string("MuF") + counter + counter1);
197 FacetEpsilon[n] = FileInterface::ReadParameterD(inp, moduleLocation, std::string("EpsilonMF") + counter + counter1);
198 }
199
200 for(size_t n = 0; n < Nfacets; n++)
201 {
202 if (FacetMobility[n] > MaxMobility)
203 {
204 MaxMobility = FacetMobility[n];
205 }
206 }
207 Model = InterfaceMobilityModels::Faceted;
208 };
209
210 double Calculate(dVector3& locNormal)
211 {
212 double locMobility = 0.0;
213 switch(Model)
214 {
215 case InterfaceMobilityModels::Ext:
216 {
217 break;
218 }
219 case InterfaceMobilityModels::Iso:
220 {
221 locMobility = MaxMobility;
222 break;
223 }
224 case InterfaceMobilityModels::Cubic:
225 {
226 locMobility = Mobility*(1.0 - Epsilon1 * (1.5 - 2.5*(pow(locNormal[0], 4) +
227 pow(locNormal[1], 4) +
228 pow(locNormal[2], 4))));
229 break;
230 }
231 case InterfaceMobilityModels::HexBoettger:
232 {
233 locMobility = Mobility*(1.0 + Epsilon1 *
234 ( std::pow(locNormal[0], 6) -
235 std::pow(locNormal[1], 6) -
236 15.0 * std::pow(locNormal[0], 4) * locNormal[1]*locNormal[1] +
237 15.0 * std::pow(locNormal[1], 4) * locNormal[0]*locNormal[0] +
238 (5.0 * std::pow(locNormal[2], 4) -
239 5.0 * std::pow(locNormal[2], 2) +
240 std::pow(locNormal[2], 6))));
241 break;
242 }
243 case InterfaceMobilityModels::HexSun:
244 {
245 locMobility = Mobility*(1.0
246 -Epsilon1*sqrt(5.0/16.0/Pi)*(3.0*locNormal[2]*locNormal[2] - 1.0)
247 -Epsilon2*3.0/16.0/sqrt(Pi)*(35.0*pow(locNormal[2],4) -
248 30.0*locNormal[2]*locNormal[2] + 3.0)
249 -Epsilon3*sqrt(13.0/Pi)/32.0*(231.0*pow(locNormal[2],6) -
250 315.0*pow(locNormal[2],4) +
251 105.0*locNormal[2]*locNormal[2] - 5.0)
252 -Epsilon4*sqrt(6006.0/Pi)/64.0*(pow(locNormal[0],6) -
253 15.0*pow(locNormal[0],4)*locNormal[1]*locNormal[1] +
254 15.0*locNormal[0]*locNormal[0]*pow(locNormal[1],4) -
255 pow(locNormal[1],6)));
256 break;
257 }
258 case InterfaceMobilityModels::HexYang:
259 {
260 locMobility = Mobility*(1.0
261 - Epsilon1 * pow((3.0*locNormal[2]*locNormal[2] - 1.0), 2)
262 - Epsilon2 * pow((locNormal[0]*locNormal[0]*locNormal[0] - 3.0*locNormal[0]*locNormal[1]*locNormal[1]),2)
263 * pow((9.0*locNormal[2]*locNormal[2] - 1.0 + Epsilon3),2));
264 break;
265 }
266 case InterfaceMobilityModels::Faceted:
267 {
268 double Inclination = Pi;
269 double MobilityA = 0.0;
270 double EpsilonMA = 0.0;
271
272 // Calculating inclination angle between facet normal and interface normal
273 for(size_t n = 0; n < Nfacets; n++)
274 for(size_t j = 0; j < FacetVectors[n].size(); j++)
275 {
276 dVector3 locFacet = FacetVectors[n][j].normalized();
277 double dotVec = acos(locNormal * locFacet);
278
279 if(fabs(dotVec) <= fabs(Inclination))
280 {
281 Inclination = dotVec;
282 MobilityA = FacetMobility[n];
283 EpsilonMA = FacetEpsilon[n];
284 }
285 }
286
287 locMobility = MobilityA * (EpsilonMA + (1.0 - EpsilonMA) * fabs(tan(Inclination)) * tanh(pow(fabs(tan(Inclination)),-1)));
288 break;
289 }
290 }
291 return locMobility;
292 };
